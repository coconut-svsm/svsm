73#[derive(Debug)]
74pub struct KernelPageTablePage<'a> {
75    entries: &'a mut [PTEntry],
76}
77
78impl KernelPageTablePage<'_> {
79    /// # Safety
80    /// The caller is required to supply a virtual address that is known to map
81    /// a full page of page table or page directory entries.
82    unsafe fn new(vaddr: VirtAddr) -> Self {
83        // SAFETY: the caller ensures the correctness of the virtual address.
84        let entries = unsafe {
85            let pte_ptr = vaddr.as_mut_ptr::<PTEntry>();
86            slice::from_raw_parts_mut(pte_ptr, svsm::mm::pagetable::ENTRY_COUNT)
87        };
88        Self { entries }
89    }
90
91    fn entry_mut(&mut self, index: usize) -> &mut PTEntry {
92        &mut self.entries[index]
93    }
94}

96struct Stage2BootLoader<'a> {
97    map_vaddr: VirtAddr,
98    kernel_region_base: PhysAddr,
99    platform: &'a dyn SvsmPlatform,
100    boot_params: &'a BootParams<'a>,
101}
102
103impl<'a> Stage2BootLoader<'a> {
104    fn new(
105        kernel_region: &MemoryRegion<PhysAddr>,
106        platform: &'a dyn SvsmPlatform,
107        boot_params: &'a BootParams<'_>,
108    ) -> Self {
109        // The temporary mapping address is initialized as the per-task area.
110        // The address range is arbitrary since it is only used for temporary
111        // mappings while stage2 is running, and selecting the per-task
112        // address space permits temporary mappings without requiring a
113        // TLB invalidation after the mappings are used.  The global address
114        // range must not be used since that range will be used after
115        // preparation of the boot image is complete, and that use must not
116        // conflict with temporary mappings that were created during boot
117        // image preparation.
118        Self {
119            kernel_region_base: kernel_region.start(),
120            map_vaddr: SVSM_PERTASK_BASE,
121            platform,
122            boot_params,
123        }
124    }
125
126    fn phys_to_virt(&self, paddr: u64) -> VirtAddr {
127        // The kernel physical memory region is mapped as a contiguous range
128        // in the local address space beginning at the mapping base address.
129        // This simplifies the task of looking up the virtual addres
130        // corresponding to a given physical address in the mapping area.
131        let offset = paddr - u64::from(self.kernel_region_base);
132        self.map_vaddr + (offset as usize)
133    }
134
135    fn add_page_data(
136        &mut self,
137        paddr: u64,
138        data: Option<&[u8]>,
139        total_size: u64,
140    ) -> Result<(), BootImageError> {
141        let total_size = total_size as usize;
142        assert_eq!((total_size & (PAGE_SIZE - 1)), 0);
143        // Create a local mapping at a virtual address based on the target
144        // physical address.
145        let map_region = MemoryRegion::new(self.phys_to_virt(paddr), total_size);
146        map_page_range(map_region, PhysAddr::from(paddr)).map_err(|e| {
147            log::error!("Failed to map kernel memory: {e:?}");
148            BootImageError::Host
149        })?;
150        // SAFETY: the virtual address used for mapping is in a portion of the
151        // virtual address space unused anywhere else, so it can safely be
152        // used for mapping here.
153        unsafe {
154            validate_mapped_region(self.platform, self.boot_params, map_region).map_err(|e| {
155                log::error!("Failed to validate kernel memory: {e:?}");
156                BootImageError::Host
157            })?;
158
159            let target_ptr = map_region.start().as_mut_ptr::<u8>();
160            let data_len = match data {
161                Some(data_slice) => {
162                    assert!(data_slice.len() <= total_size);
163                    core::ptr::copy_nonoverlapping(
164                        data_slice.as_ptr(),
165                        target_ptr,
166                        data_slice.len(),
167                    );
168                    data_slice.len()
169                }
170                None => 0,
171            };
172
173            // Zero the tail end of any partial page.
174            if data_len < total_size {
175                core::ptr::write_bytes(target_ptr.add(data_len), 0, total_size - data_len);
176            }
177        }
178
179        Ok(())
180    }
181}

183fn setup_stage2_allocator(heap_start: u64, heap_end: u64) {
184    let vstart = VirtAddr::from(heap_start);
185    let vend = VirtAddr::from(heap_end);
186    let pstart = PhysAddr::from(vstart.bits()); // Identity mapping
187    let nr_pages = (vend - vstart) / PAGE_SIZE;
188
189    root_mem_init(pstart, vstart, nr_pages, 0);
190}

192fn init_percpu(platform: &mut dyn SvsmPlatform) -> Result<(), SvsmError> {
193    // SAFETY: this is the first CPU, so there can be no other dependencies
194    // on multi-threaded access to the per-cpu areas.
195    let percpu_shared = unsafe { PERCPU_AREAS.create_new(0) };
196    let bsp_percpu = PerCpu::alloc(percpu_shared)?;
197    bsp_percpu.set_current_stack(MemoryRegion::from_addresses(
198        VirtAddr::from(STAGE2_STACK_END as u64),
199        VirtAddr::from(STAGE2_STACK as u64),
200    ));
201    // SAFETY: pgtable is properly aligned and is never freed within the
202    // lifetime of stage2. We go through a raw pointer to promote it to a
203    // static mut. Only the BSP is able to get a reference to it so no
204    // aliasing can occur.
205    let init_pgtable = unsafe { (&raw mut pgtable).as_mut().unwrap() };
206    bsp_percpu.set_pgtable(init_pgtable);
207    bsp_percpu.map_self_stage2()?;
208    platform.setup_guest_host_comm(bsp_percpu, true);
209    Ok(())
210}

212/// Release all resources in the `PerCpu` instance associated with the current
213/// CPU.
214///
215/// # Safety
216///
217/// The caller must ensure that the `PerCpu` is never used again.
218unsafe fn shutdown_percpu() {
219    let ptr = SVSM_PERCPU_BASE.as_mut_ptr::<PerCpu>();
220    // SAFETY: ptr is properly aligned but the caller must ensure the PerCpu
221    // structure is valid and not aliased.
222    unsafe {
223        core::ptr::drop_in_place(ptr);
224    }
225    // SAFETY: pgtable is properly aligned and is never freed within the
226    // lifetime of stage2. We go through a raw pointer to promote it to a
227    // static mut. Only the BSP is able to get a reference to it so no
228    // aliasing can occur.
229    let init_pgtable = unsafe { (&raw mut pgtable).as_mut().unwrap() };
230    init_pgtable.unmap_4k(SVSM_PERCPU_BASE);
231    flush_tlb_percpu();
232}

234// SAFETY: the caller must guarantee that the IDT specified here will remain
235// in scope until a new IDT is loaded.
236unsafe fn setup_env(
237    boot_params: &BootParams<'_>,
238    platform: &mut dyn SvsmPlatform,
239    launch_info: &Stage2LaunchInfo,
240    cpuid_vaddr: Option<VirtAddr>,
241    idt: &mut IDT<'_>,
242) {
243    GLOBAL_GDT.load_selectors();
244    // SAFETY: the caller guarantees that the lifetime of this IDT is suitable.
245    unsafe {
246        early_idt_init_no_ghcb(idt);
247    }
248
249    let debug_serial_port = boot_params.debug_serial_port();
250    install_console_logger("Stage2").expect("Console logger already initialized");
251    platform
252        .env_setup(debug_serial_port, launch_info.vtom.try_into().unwrap())
253        .expect("Early environment setup failed");
254
255    let kernel_mapping = FixedAddressMappingRange::new(
256        VirtAddr::from(u64::from(STAGE2_START)),
257        VirtAddr::from(u64::from(launch_info.stage2_end)),
258        PhysAddr::from(u64::from(STAGE2_START)),
259    );
260
261    if let Some(cpuid_addr) = cpuid_vaddr {
262        // SAFETY: the CPUID page address specified in the launch info was
263        // mapped by the loader, which promises to supply a correctly formed
264        // CPUID page at that address.
265        let cpuid_page = unsafe { &*cpuid_addr.as_ptr::<SnpCpuidTable>() };
266        register_cpuid_table(cpuid_page);
267    }
268
269    paging_init(platform, true).expect("Failed to initialize early paging");
270
271    // Use the low 640 KB of memory as the heap.
272    let lowmem_region =
273        MemoryRegion::from_addresses(VirtAddr::from(0u64), VirtAddr::from(u64::from(LOWMEM_END)));
274    let heap_mapping = FixedAddressMappingRange::new(
275        lowmem_region.start(),
276        lowmem_region.end(),
277        PhysAddr::from(0u64),
278    );
279    init_kernel_mapping_info(kernel_mapping, Some(heap_mapping));
280
281    // Now that the heap virtual-to-physical mapping has been established,
282    // validate the first 640 KB of memory so it can be used if necessary.
283    // SAFETY: the low memory region is known not to overlap any memory in use.
284    unsafe {
285        platform
286            .validate_low_memory(lowmem_region.end().into())
287            .expect("failed to validate low 640 KB");
288    }
289
290    // Configure the heap.
291    setup_stage2_allocator(STAGE2_HEAP_START.into(), STAGE2_HEAP_END.into());
292
293    init_percpu(platform).expect("Failed to initialize per-cpu area");
294
295    // Init IDT again with handlers requiring GHCB (eg. #VC handler)
296    // Must be done after init_percpu() to catch early #PFs
297    //
298    // SAFETY: the caller guarantees that the lifetime of this IDT is suitable.
299    unsafe {
300        early_idt_init(idt);
301    }
302
303    // Complete initializtion of the platform.  After that point, the console
304    // will be fully working and any unsupported configuration can be properly
305    // reported.
306    platform
307        .env_setup_late(debug_serial_port)
308        .expect("Late environment setup failed");
309
310    if cpuid_vaddr.is_some() {
311        dump_cpuid_table();
312    }
313}

315/// Map the specified virtual memory region at the given physical address.
316/// This will fail if the caller specifies a virtual address region that is
317/// already mapped.
318fn map_page_range(vregion: MemoryRegion<VirtAddr>, paddr: PhysAddr) -> Result<(), SvsmError> {
319    let flags = PTEntryFlags::PRESENT
320        | PTEntryFlags::WRITABLE
321        | PTEntryFlags::ACCESSED
322        | PTEntryFlags::DIRTY;
323
324    let mut pgtbl = this_cpu().get_pgtable();
325    pgtbl.map_region(vregion, paddr, flags)?;
326
327    Ok(())
328}

330fn get_kernel_elf(launch_info: &Stage2LaunchInfo) -> &'static [u8] {
331    // Find the bounds of the kernel ELF and load it into the ELF parser
332    let elf_start = PhysAddr::from(launch_info.kernel_elf_start as u64);
333    let elf_end = PhysAddr::from(launch_info.kernel_elf_end as u64);
334    let elf_len = elf_end - elf_start;
335    // SAFETY: the base address of the ELF image was selected by the loader and
336    // is known not to conflict with any other virtual address mappings.
337    unsafe { slice::from_raw_parts(elf_start.bits() as *const u8, elf_len) }
338}

340fn prepare_kernel_image(
341    stage2_platform: &dyn Stage2Platform,
342    launch_info: &Stage2LaunchInfo,
343    boot_params: &BootParams<'_>,
344    boot_image_params: &BootImageParams<'_>,
345    boot_loader: &mut Stage2BootLoader<'_>,
346) -> Result<BootImageInfo, BootImageError> {
347    // Obtain a reference to the kernel image data.
348    let kernel_image_bytes = get_kernel_elf(launch_info);
349
350    // Prepare the boot image using the common library.
351    let boot_image_info = prepare_boot_image(
352        boot_image_params,
353        kernel_image_bytes,
354        &mut |paddr, data, len| boot_loader.add_page_data(paddr, data, len),
355    )?;
356
357    // No confidentiality bits are present in the kernel page table portion of
358    // the boot image.  Therefore, the page tables need to be walked now to
359    // insert confidentiality bits as defined by the current platform.  This
360    // is only necessary if there is a non-zero confidentiailty mask.
361    if private_pte_mask() != 0 {
362        for pt_index in 0..boot_image_info.total_pt_pages {
363            let paddr = boot_image_info.kernel_page_tables_base + (pt_index * PAGE_SIZE as u64);
364            // SAFETY: the boot image has been fully mapped and the address
365            // translation from physical to virtual is known to be correct by
366            // construction.
367            let mut page_table_page =
368                unsafe { KernelPageTablePage::new(boot_loader.phys_to_virt(paddr)) };
369            for entry in 0..svsm::mm::pagetable::ENTRY_COUNT {
370                page_table_page.entry_mut(entry).make_private_if_present();
371            }
372        }
373    }
374
375    // Preparing the boot image only mapped those portion of the boot
376    // parameters that are measured.  The unmeasured contents need to be loaded
377    // now as the loader would load them.  Load them into the kernel area
378    // as if they had been prepared as part of the boot image.  The unmeasured
379    // boot parameters follow the measured boot parameters, so capture the
380    // unmeasured data as the portion of the boot data byte slice that follows
381    // the measured parameters.
382    let boot_params_measured_size = page_align_up(mem::size_of::<BootParamBlock>());
383    let boot_params_unmeasured_size = boot_params.size() - boot_params_measured_size;
384    let (_, unmeasured_slice) = boot_params
385        .as_byte_slice()
386        .split_at(boot_params_measured_size);
387
388    // Copy the data into the kernel image as if it had been prepared as part
389    // of the boot image.
390    boot_loader
391        .add_page_data(
392            boot_image_info.boot_params_paddr + boot_params_measured_size as u64,
393            Some(unmeasured_slice),
394            boot_params_unmeasured_size as u64,
395        )
396        .inspect_err(|_| {
397            log::error!("Failed to add unmeasured data to boot image");
398        })?;
399
400    // Copy the CPUID and secrets pages into the kernel image as if they had
401    // been prepared as part of the boot image.
402    // SAFETY: the platform guarantees the correctness of the CPUID and secrets
403    // page virtual addresses from the stage2 launch info.
404    let (cpuid_slice, secrets_slice) = unsafe {
405        (
406            stage2_platform
407                .get_cpuid_page(launch_info)
408                .map(|vaddr| slice::from_raw_parts(vaddr.as_ptr::<u8>(), PAGE_SIZE)),
409            stage2_platform
410                .get_secrets_page(launch_info)
411                .map(|vaddr| slice::from_raw_parts(vaddr.as_ptr::<u8>(), PAGE_SIZE)),
412        )
413    };
414
415    boot_loader
416        .add_page_data(boot_image_info.cpuid_paddr, cpuid_slice, PAGE_SIZE as u64)
417        .and_then(|_| {
418            boot_loader.add_page_data(
419                boot_image_info.secrets_paddr,
420                secrets_slice,
421                PAGE_SIZE as u64,
422            )
423        })
424        .inspect_err(|_| {
425            log::error!("Failed to add platform data pages to boot image");
426        })?;
427
428    Ok(boot_image_info)
429}

431#[unsafe(no_mangle)]
432pub extern "C" fn stage2_main(launch_info: &Stage2LaunchInfo) -> ! {
433    let platform_type = SvsmPlatformType::from(launch_info.platform_type);
434
435    init_platform_type(platform_type);
436    let mut platform_cell = SvsmPlatformCell::new(true);
437    let platform = platform_cell.platform_mut();
438    let stage2_platform_cell = Stage2PlatformCell::new(platform_type);
439    let stage2_platform = stage2_platform_cell.platform();
440
441    // SAFETY: the address in the launch info is known to be correct.
442    let boot_params = unsafe { BootParams::new(VirtAddr::from(launch_info.boot_params as u64)) }
443        .expect("Failed to get boot parameters");
444
445    // Set up space for an early IDT.  This will remain in scope as long as
446    // stage2 is in memory.
447    let mut early_idt = [IdtEntry::no_handler(); EARLY_IDT_ENTRIES];
448    let mut idt = IDT::new(&mut early_idt);
449
450    // Get a reference to the CPUID page if this platform requires it.
451    let cpuid_page = stage2_platform.get_cpuid_page(launch_info);
452
453    // SAFETY: the IDT here will remain in scope until the full IDT is
454    // initialized later, and thus can safely be used as the early IDT.
455    unsafe {
456        setup_env(&boot_params, platform, launch_info, cpuid_page, &mut idt);
457    }
458
459    // Get the available physical memory region for the kernel
460    let kernel_region = boot_params
461        .find_kernel_region()
462        .expect("Failed to find memory region for SVSM kernel");
463
464    log::info!("SVSM memory region: {kernel_region:#018x}");
465
466    // Invoke the boot image builder to assemble the kernel image.
467    let mut boot_image_loader = Stage2BootLoader::new(&kernel_region, platform, &boot_params);
468    let boot_image_params = BootImageParams {
469        boot_params: boot_params.param_block(),
470        kernel_region_start: u64::from(kernel_region.start()),
471        kernel_region_page_count: round_to_pages(kernel_region.len()).try_into().unwrap(),
472        kernel_fs_start: u64::from(launch_info.kernel_fs_start),
473        kernel_fs_end: u64::from(launch_info.kernel_fs_end),
474        stage2_start: 0x800000u64,
475        vtom: launch_info.vtom,
476    };
477    let boot_image_info = prepare_kernel_image(
478        stage2_platform,
479        launch_info,
480        &boot_params,
481        &boot_image_params,
482        &mut boot_image_loader,
483    )
484    .expect("Failed to load kernel image");
485
486    // Set the PML4E of the new kernel page tables in the current page table so
487    // the kernel address space is also visible in the current address space.
488    // SAFETY: the physical address of the current paging root is known to be
489    // identity-mapped in the current address space and therefore that address
490    // can be used to obtain a page table view.
491    unsafe {
492        let vaddr = VirtAddr::from(this_cpu().get_pgtable().cr3_value().bits());
493        let cur_pgtable = slice::from_raw_parts_mut(
494            vaddr.as_mut_ptr::<PTEntry>(),
495            svsm::mm::pagetable::ENTRY_COUNT,
496        );
497        let pxe_flags = PTEntryFlags::PRESENT | PTEntryFlags::WRITABLE | PTEntryFlags::ACCESSED;
498        cur_pgtable[boot_image_info.kernel_pml4e_index as usize].set_unrestricted(
499            make_private_address(PhysAddr::from(boot_image_info.kernel_pdpt_paddr)),
500            pxe_flags,
501        );
502    };
503
504    let mem_info = memory_info();
505    print_memory_info(&mem_info);
506
507    log::info!(
508        "  kernel_region_phys_start = {:#018x}",
509        kernel_region.start()
510    );
511    log::info!("  kernel_region_phys_end   = {:#018x}", kernel_region.end());
512
513    log::info!("Starting SVSM kernel...");
514
515    // SAFETY: the addreses used to invoke the kernel have been calculated
516    // correctly for use in the assembly trampoline.
517    unsafe {
518        // Shut down the PerCpu instance
519        shutdown_percpu();
520
521        switch_to_kernel(
522            boot_image_info.context.entry_point,
523            boot_image_info.context.initial_stack,
524            platform_type as u64,
525        );
526    };
527}
528
529global_asm!(
530    r#"
531        .section .text
532        .globl switch_to_kernel
533        switch_to_kernel:
534
535        /* Switch to the kernel stack. */
536        movq %rsi, %rsp
537
538        /* Load the platform type into rax as expected by the kernel */
539        movq %rdx, %rax
540
541        /* Enter the kernel. */
542        push %rdi
543        ret
544        "#,
545    options(att_syntax)
546);

548#[panic_handler]
549fn panic(info: &PanicInfo<'_>) -> ! {
550    log::error!("Panic! COCONUT-SVSM Version: {COCONUT_VERSION}");
551    log::error!("Info: {info}");
552
553    print_stack(3);
554
555    platform::terminate();
