55fn setup_stage2_allocator(heap_start: u64, heap_end: u64) {
56    let vstart = VirtAddr::from(heap_start);
57    let vend = VirtAddr::from(heap_end);
58    let pstart = PhysAddr::from(vstart.bits()); // Identity mapping
59    let nr_pages = (vend - vstart) / PAGE_SIZE;
60
61    root_mem_init(pstart, vstart, nr_pages);
62}

64fn init_percpu(platform: &mut dyn SvsmPlatform) -> Result<(), SvsmError> {
65    // SAFETY: this is the first CPU, so there can be no other dependencies
66    // on multi-threaded access to the per-cpu areas.
67    let percpu_shared = unsafe { PERCPU_AREAS.create_new(0) };
68    let bsp_percpu = PerCpu::alloc(percpu_shared)?;
69    bsp_percpu.set_current_stack(MemoryRegion::from_addresses(
70        VirtAddr::from(STAGE2_STACK_END as u64),
71        VirtAddr::from(STAGE2_STACK as u64),
72    ));
73    // SAFETY: pgtable is properly aligned and is never freed within the
74    // lifetime of stage2. We go through a raw pointer to promote it to a
75    // static mut. Only the BSP is able to get a reference to it so no
76    // aliasing can occur.
77    let init_pgtable = unsafe { (&raw mut pgtable).as_mut().unwrap() };
78    bsp_percpu.set_pgtable(init_pgtable);
79    bsp_percpu.map_self_stage2()?;
80    platform.setup_guest_host_comm(bsp_percpu, true);
81    Ok(())
82}

84/// Release all resources in the `PerCpu` instance associated with the current
85/// CPU.
86///
87/// # Safety
88///
89/// The caller must ensure that the `PerCpu` is never used again.
90unsafe fn shutdown_percpu() {
91    let ptr = SVSM_PERCPU_BASE.as_mut_ptr::<PerCpu>();
92    // SAFETY: ptr is properly aligned but the caller must ensure the PerCpu
93    // structure is valid and not aliased.
94    unsafe {
95        core::ptr::drop_in_place(ptr);
96    }
97    // SAFETY: pgtable is properly aligned and is never freed within the
98    // lifetime of stage2. We go through a raw pointer to promote it to a
99    // static mut. Only the BSP is able to get a reference to it so no
100    // aliasing can occur.
101    let init_pgtable = unsafe { (&raw mut pgtable).as_mut().unwrap() };
102    init_pgtable.unmap_4k(SVSM_PERCPU_BASE);
103    flush_tlb_percpu();
104}

106// SAFETY: the caller must guarantee that the IDT specified here will remain
107// in scope until a new IDT is loaded.
108unsafe fn setup_env(
109    config: &SvsmConfig<'_>,
110    platform: &mut dyn SvsmPlatform,
111    launch_info: &Stage2LaunchInfo,
112    idt: &mut IDT<'_>,
113) {
114    GLOBAL_GDT.load_selectors();
115    // SAFETY: the caller guarantees that the lifetime of this IDT is suitable.
116    unsafe {
117        early_idt_init_no_ghcb(idt);
118    }
119
120    let debug_serial_port = config.debug_serial_port();
121    install_console_logger("Stage2").expect("Console logger already initialized");
122    platform
123        .env_setup(debug_serial_port, launch_info.vtom.try_into().unwrap())
124        .expect("Early environment setup failed");
125
126    let kernel_mapping = FixedAddressMappingRange::new(
127        VirtAddr::from(u64::from(STAGE2_START)),
128        VirtAddr::from(u64::from(launch_info.stage2_end)),
129        PhysAddr::from(u64::from(STAGE2_START)),
130    );
131
132    // SAFETY: the CPUID page address specified in the launch info was mapped
133    // by the loader, which promises to supply a correctly formed CPUID page
134    // at that address.
135    let cpuid_page = unsafe { &*(launch_info.cpuid_page as *const SnpCpuidTable) };
136    register_cpuid_table(cpuid_page);
137    paging_init(platform, true).expect("Failed to initialize early paging");
138
139    // Use the low 640 KB of memory as the heap.
140    let lowmem_region =
141        MemoryRegion::from_addresses(VirtAddr::from(0u64), VirtAddr::from(u64::from(LOWMEM_END)));
142    let heap_mapping = FixedAddressMappingRange::new(
143        lowmem_region.start(),
144        lowmem_region.end(),
145        PhysAddr::from(0u64),
146    );
147    init_kernel_mapping_info(kernel_mapping, Some(heap_mapping));
148
149    // Now that the heap virtual-to-physical mapping has been established,
150    // validate the first 640 KB of memory so it can be used if necessary.
151    // SAFETY: the low memory region is known not to overlap any memory in use.
152    unsafe {
153        platform
154            .validate_virtual_page_range(lowmem_region, PageValidateOp::Validate)
155            .expect("failed to validate low 640 KB");
156    }
157
158    // SAFETY: ap_flag is an extern static and this is the only place where we
159    // get a reference to it.
160    unsafe {
161        // Allow APs to proceed as the environment is now ready.
162        //
163        // Although APs use non-atomic loads in the ap_wait_for_env spin loop,
164        // the language and architectural guarantees of this atomic store (e.g.
165        // the compiler cannot move the previous stores past this atomic
166        // store-release, and x86 is a strongly-ordered system) make setting
167        // this flag more deterministic.
168        ap_flag.store(1, Ordering::Release);
169    }
170
171    // Configure the heap.
172    setup_stage2_allocator(STAGE2_HEAP_START.into(), STAGE2_HEAP_END.into());
173
174    init_percpu(platform).expect("Failed to initialize per-cpu area");
175
176    // Init IDT again with handlers requiring GHCB (eg. #VC handler)
177    // Must be done after init_percpu() to catch early #PFs
178    //
179    // SAFETY: the caller guarantees that the lifetime of this IDT is suitable.
180    unsafe {
181        early_idt_init(idt);
182    }
183
184    // Complete initializtion of the platform.  After that point, the console
185    // will be fully working and any unsupported configuration can be properly
186    // reported.
187    platform
188        .env_setup_late(debug_serial_port)
189        .expect("Late environment setup failed");
190
191    dump_cpuid_table();
192}

194/// Map and validate the specified virtual memory region at the given physical
195/// address.
196/// # Safety
197/// The caller is required to ensure the safety of the virtual address range.
198unsafe fn map_and_validate(
199    platform: &dyn SvsmPlatform,
200    config: &SvsmConfig<'_>,
201    vregion: MemoryRegion<VirtAddr>,
202    paddr: PhysAddr,
203) -> Result<(), SvsmError> {
204    let flags = PTEntryFlags::PRESENT
205        | PTEntryFlags::WRITABLE
206        | PTEntryFlags::ACCESSED
207        | PTEntryFlags::DIRTY;
208
209    let mut pgtbl = this_cpu().get_pgtable();
210    pgtbl.map_region(vregion, paddr, flags)?;
211
212    if config.page_state_change_required() {
213        platform.page_state_change(
214            MemoryRegion::new(paddr, vregion.len()),
215            PageSize::Huge,
216            PageStateChangeOp::Private,
217        )?;
218    }
219    // SAFETY: the caller has ensured the safety of the virtual address range.
220    unsafe {
221        platform.validate_virtual_page_range(vregion, PageValidateOp::Validate)?;
222    }
223    valid_bitmap_set_valid_range(paddr, paddr + vregion.len());
224    Ok(())
225}

227#[inline]
228fn check_launch_info(launch_info: &KernelLaunchInfo) {
229    let offset: u64 = launch_info.heap_area_virt_start - launch_info.heap_area_phys_start;
230    let align: u64 = PAGE_SIZE_2M.try_into().unwrap();
231
232    assert!(is_aligned(offset, align));
233}

244/// Loads a single ELF segment and returns its virtual memory region.
245/// # Safety
246/// The caller is required to supply an appropriate virtual address for this
247/// ELF segment.
248unsafe fn load_elf_segment(
249    segment: elf::Elf64ImageLoadSegment<'_>,
250    paddr: PhysAddr,
251    platform: &dyn SvsmPlatform,
252    config: &SvsmConfig<'_>,
253) -> Result<MemoryRegion<VirtAddr>, SvsmError> {
254    // Find the segment's bounds
255    let segment_start = VirtAddr::from(segment.vaddr_range.vaddr_begin);
256    let segment_end = VirtAddr::from(segment.vaddr_range.vaddr_end).page_align_up();
257    let segment_len = segment_end - segment_start;
258    let segment_region = MemoryRegion::new(segment_start, segment_len);
259
260    // All ELF segments should be aligned to the page size. If not, there's
261    // the risk of pvalidating a page twice, bail out if so. Note that the
262    // ELF reading code had already verified that the individual segments,
263    // with bounds specified as in the ELF file, are non-overlapping.
264    if !segment_start.is_page_aligned() {
265        return Err(SvsmError::Elf(ElfError::UnalignedSegmentAddress));
266    }
267
268    // Map and validate the segment at the next contiguous physical address
269    // SAFETY: the caller has verified the safety of this virtual address
270    // region.
271    unsafe {
272        map_and_validate(platform, config, segment_region, paddr)?;
273    }
274
275    // Copy the segment contents and pad with zeroes
276    // SAFETY: the caller guarantees the correctness of the ELF segment's
277    // virtual address range.
278    let segment_buf =
279        unsafe { slice::from_raw_parts_mut(segment_start.as_mut_ptr::<u8>(), segment_len) };
280    let contents_len = segment.file_contents.len();
281    segment_buf[..contents_len].copy_from_slice(segment.file_contents);
282    segment_buf[contents_len..].fill(0);
283
284    Ok(segment_region)
285}

287/// Loads the kernel ELF and returns the virtual memory region where it
288/// resides, as well as its entry point. Updates the used physical memory
289/// region accordingly.
290fn load_kernel_elf(
291    launch_info: &Stage2LaunchInfo,
292    loaded_phys: &mut MemoryRegion<PhysAddr>,
293    platform: &dyn SvsmPlatform,
294    config: &SvsmConfig<'_>,
295) -> Result<(VirtAddr, MemoryRegion<VirtAddr>), SvsmError> {
296    // Find the bounds of the kernel ELF and load it into the ELF parser
297    let elf_start = PhysAddr::from(launch_info.kernel_elf_start as u64);
298    let elf_end = PhysAddr::from(launch_info.kernel_elf_end as u64);
299    let elf_len = elf_end - elf_start;
300    // SAFETY: the base address of the ELF image was selected by the loader and
301    // is known not to conflict with any other virtual address mappings.
302    let bytes = unsafe { slice::from_raw_parts(elf_start.bits() as *const u8, elf_len) };
303    let elf = elf::Elf64File::read(bytes)?;
304
305    let vaddr_alloc_info = elf.image_load_vaddr_alloc_info();
306    let vaddr_alloc_base = vaddr_alloc_info.range.vaddr_begin;
307
308    // Map, validate and populate the SVSM kernel ELF's PT_LOAD segments. The
309    // segments' virtual address range might not necessarily be contiguous,
310    // track their total extent along the way. Physical memory is successively
311    // being taken from the physical memory region, the remaining space will be
312    // available as heap space for the SVSM kernel. Remember the end of all
313    // physical memory occupied by the loaded ELF image.
314    let mut load_virt_start = None;
315    let mut load_virt_end = VirtAddr::null();
316    for segment in elf.image_load_segment_iter(vaddr_alloc_base) {
317        // SAFETY: the virtual address is calculated based on the base address
318        // of the image and the previously loaded segments, so it is correct
319        // for use.
320        let region = unsafe { load_elf_segment(segment, loaded_phys.end(), platform, config)? };
321        // Remember the mapping range's lower and upper bounds to pass it on
322        // the kernel later. Note that the segments are being iterated over
323        // here in increasing load order.
324        if load_virt_start.is_none() {
325            load_virt_start = Some(region.start());
326        }
327        load_virt_end = region.end();
328
329        // Update to the next contiguous physical address
330        *loaded_phys = loaded_phys.expand(region.len());
331    }
332
333    let Some(load_virt_start) = load_virt_start else {
334        log::error!("No loadable segment found in kernel ELF");
335        return Err(SvsmError::Mem);
336    };
337
338    // Apply relocations, if any
339    if let Some(dyn_relocs) =
340        elf.apply_dyn_relas(elf::Elf64X86RelocProcessor::new(), vaddr_alloc_base)?
341    {
342        for reloc in dyn_relocs {
343            let Some(reloc) = reloc? else {
344                continue;
345            };
346            // TODO: ensure that the ELF package rejects illegal relocations
347            // the point outside the image.
348            // SAFETY: the relocation address is known to be correct.
349            let dst = unsafe { slice::from_raw_parts_mut(reloc.dst as *mut u8, reloc.value_len) };
350            let src = &reloc.value[..reloc.value_len];
351            dst.copy_from_slice(src)
352        }
353    }
354
355    let entry = VirtAddr::from(elf.get_entry(vaddr_alloc_base));
356    let region = MemoryRegion::from_addresses(load_virt_start, load_virt_end);
357    Ok((entry, region))
358}

360/// Loads the IGVM params at the next contiguous location from the loaded
361/// kernel image. Returns the virtual and physical memory regions hosting the
362/// loaded data.
363/// # Safety
364/// Ther caller is required to specify the correct virtual address for the
365/// kernel virtual region.
366unsafe fn load_igvm_params(
367    launch_info: &Stage2LaunchInfo,
368    params: &IgvmParams<'_>,
369    loaded_kernel_vregion: &MemoryRegion<VirtAddr>,
370    loaded_kernel_pregion: &MemoryRegion<PhysAddr>,
371    platform: &dyn SvsmPlatform,
372    config: &SvsmConfig<'_>,
373) -> Result<(MemoryRegion<VirtAddr>, MemoryRegion<PhysAddr>), SvsmError> {
374    // Map and validate destination region
375    let igvm_vregion = MemoryRegion::new(loaded_kernel_vregion.end(), params.size());
376    let igvm_pregion = MemoryRegion::new(loaded_kernel_pregion.end(), params.size());
377    // SAFETY: the virtual address region was computed not to overlap the
378    // kernel image.
379    unsafe {
380        map_and_validate(platform, config, igvm_vregion, igvm_pregion.start())?;
381    }
382
383    // Copy the contents over
384    let src_addr = VirtAddr::from(launch_info.igvm_params as u64);
385    // SAFETY: the source address specified in the launch info was mapped
386    // by the loader, which promises to supply a correctly formed IGRM
387    // parameter block
388    let igvm_src = unsafe { slice::from_raw_parts(src_addr.as_ptr::<u8>(), igvm_vregion.len()) };
389    // SAFETY: the destination address is calculated to follow the kernel ELF
390    // image, which the caller is required to calculate correctly.
391    let igvm_dst = unsafe {
392        slice::from_raw_parts_mut(igvm_vregion.start().as_mut_ptr::<u8>(), igvm_vregion.len())
393    };
394    igvm_dst.copy_from_slice(igvm_src);
395
396    Ok((igvm_vregion, igvm_pregion))
397}

399/// Maps any remaining memory between the end of the kernel image and the end
400/// of the allocated kernel memory region as heap space. Exclude any memory
401/// reserved by the configuration.
402///
403/// # Panics
404///
405/// Panics if the allocated kernel region (`kernel_region`) is not sufficient
406/// to host the loaded kernel region (`loaded_kernel_pregion`) plus memory
407/// reserved for configuration.
408fn prepare_heap(
409    kernel_region: MemoryRegion<PhysAddr>,
410    loaded_kernel_pregion: MemoryRegion<PhysAddr>,
411    loaded_kernel_vregion: MemoryRegion<VirtAddr>,
412    platform: &dyn SvsmPlatform,
413    config: &SvsmConfig<'_>,
414) -> Result<(MemoryRegion<VirtAddr>, MemoryRegion<PhysAddr>), SvsmError> {
415    // Heap starts after kernel
416    let heap_pstart = loaded_kernel_pregion.end();
417    let heap_vstart = loaded_kernel_vregion.end();
418
419    // Compute size, excluding any memory reserved by the configuration.
420    let heap_size = kernel_region
421        .end()
422        .checked_sub(heap_pstart.into())
423        .and_then(|r| r.checked_sub(config.reserved_kernel_area_size()))
424        .expect("Insufficient physical space for kernel image")
425        .into();
426    let heap_pregion = MemoryRegion::new(heap_pstart, heap_size);
427    let heap_vregion = MemoryRegion::new(heap_vstart, heap_size);
428
429    // SAFETY: the virtual address range was computed so it is within the valid
430    // region and does not collide with the kernel.
431    unsafe {
432        map_and_validate(platform, config, heap_vregion, heap_pregion.start())?;
433    }
434
435    Ok((heap_vregion, heap_pregion))
436}

438#[no_mangle]
439pub extern "C" fn stage2_main(launch_info: &Stage2LaunchInfo) -> ! {
440    let platform_type = SvsmPlatformType::from(launch_info.platform_type);
441
442    init_platform_type(platform_type);
443    let mut platform_cell = SvsmPlatformCell::new(true);
444    let platform = platform_cell.platform_mut();
445
446    let config = get_svsm_config(launch_info, platform).expect("Failed to get SVSM configuration");
447
448    // Set up space for an early IDT.  This will remain in scope as long as
449    // stage2 is in memory.
450    let mut early_idt = [IdtEntry::no_handler(); EARLY_IDT_ENTRIES];
451    let mut idt = IDT::new(&mut early_idt);
452
453    // SAFETY: the IDT here will remain in scope until the full IDT is
454    // initialized later, and thus can safely be used as the early IDT.
455    unsafe {
456        setup_env(&config, platform, launch_info, &mut idt);
457    }
458
459    // Get the available physical memory region for the kernel
460    let kernel_region = config
461        .find_kernel_region()
462        .expect("Failed to find memory region for SVSM kernel");
463
464    log::info!("SVSM memory region: {kernel_region:#018x}");
465
466    init_valid_bitmap_alloc(kernel_region).expect("Failed to allocate valid-bitmap");
467
468    // The physical memory region we've loaded so far
469    let mut loaded_kernel_pregion = MemoryRegion::new(kernel_region.start(), 0);
470
471    // Load first the kernel ELF and update the loaded physical region
472    let (kernel_entry, mut loaded_kernel_vregion) =
473        load_kernel_elf(launch_info, &mut loaded_kernel_pregion, platform, &config)
474            .expect("Failed to load kernel ELF");
475
476    // Load the IGVM params, if present. Update loaded region accordingly.
477    // SAFETY: The loaded kernel region was correctly calculated above and
478    // is sized appropriately to include a copy of the IGVM parameters.
479    let (igvm_vregion, igvm_pregion) = unsafe {
480        load_igvm_params(
481            launch_info,
482            config.get_igvm_params(),
483            &loaded_kernel_vregion,
484            &loaded_kernel_pregion,
485            platform,
486            &config,
487        )
488    }
489    .expect("Failed to load IGVM params");
490
491    // Update the loaded kernel region
492    loaded_kernel_pregion = loaded_kernel_pregion.expand(igvm_vregion.len());
493    loaded_kernel_vregion = loaded_kernel_vregion.expand(igvm_pregion.len());
494
495    // Use remaining space after kernel image as heap space.
496    let (heap_vregion, heap_pregion) = prepare_heap(
497        kernel_region,
498        loaded_kernel_pregion,
499        loaded_kernel_vregion,
500        platform,
501        &config,
502    )
503    .expect("Failed to map and validate heap");
504
505    // Determine whether use of interrupts on the SVSM should be suppressed.
506    // This is required when running SNP under KVM/QEMU.
507    let suppress_svsm_interrupts = match platform_type {
508        SvsmPlatformType::Snp => config.suppress_svsm_interrupts_on_snp(),
509        _ => false,
510    };
511
512    // Build the handover information describing the memory layout and hand
513    // control to the SVSM kernel.
514    let launch_info = KernelLaunchInfo {
515        kernel_region_phys_start: u64::from(kernel_region.start()),
516        kernel_region_phys_end: u64::from(kernel_region.end()),
517        heap_area_phys_start: u64::from(heap_pregion.start()),
518        heap_area_virt_start: u64::from(heap_vregion.start()),
519        heap_area_size: heap_vregion.len() as u64,
520        kernel_region_virt_start: u64::from(loaded_kernel_vregion.start()),
521        kernel_elf_stage2_virt_start: u64::from(launch_info.kernel_elf_start),
522        kernel_elf_stage2_virt_end: u64::from(launch_info.kernel_elf_end),
523        kernel_fs_start: u64::from(launch_info.kernel_fs_start),
524        kernel_fs_end: u64::from(launch_info.kernel_fs_end),
525        stage2_start: 0x800000u64,
526        stage2_end: launch_info.stage2_end as u64,
527        cpuid_page: launch_info.cpuid_page as u64,
528        secrets_page: launch_info.secrets_page as u64,
529        stage2_igvm_params_phys_addr: u64::from(launch_info.igvm_params),
530        stage2_igvm_params_size: igvm_pregion.len() as u64,
531        igvm_params_phys_addr: u64::from(igvm_pregion.start()),
532        igvm_params_virt_addr: u64::from(igvm_vregion.start()),
533        vtom: launch_info.vtom,
534        debug_serial_port: config.debug_serial_port(),
535        use_alternate_injection: config.use_alternate_injection(),
536        suppress_svsm_interrupts,
537        platform_type,
538    };
539
540    check_launch_info(&launch_info);
541
542    let mem_info = memory_info();
543    print_memory_info(&mem_info);
544
545    log::info!(
546        "  kernel_region_phys_start = {:#018x}",
547        kernel_region.start()
548    );
549    log::info!("  kernel_region_phys_end   = {:#018x}", kernel_region.end());
550    log::info!(
551        "  kernel_virtual_base      = {:#018x}",
552        loaded_kernel_vregion.start()
553    );
554
555    let valid_bitmap = valid_bitmap_addr();
556
557    log::info!("Starting SVSM kernel...");
558
559    // SAFETY: the addreses used to invoke the kernel have been calculated
560    // correctly for use in the assembly trampoline.
561    unsafe {
562        // Shut down the PerCpu instance
563        shutdown_percpu();
564
565        asm!("jmp *%rax",
566             in("rax") u64::from(kernel_entry),
567             in("rdi") &launch_info,
568             in("rsi") valid_bitmap.bits(),
569             options(att_syntax))
570    };
571
572    unreachable!("Road ends here!");
573}
