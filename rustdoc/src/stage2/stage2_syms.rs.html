19/// Parses out the `.symtab` and `.strtab` sections in the ELF binary to create
20/// symbol information for the kernel. Returns two memory regions allocated
21/// from the `KernelHeap`, each containing the information for the `.symtab` and
22/// `.strtab` respectively.
23pub fn load_kernel_symbols(
24    elf: &elf::Elf64File<'_>,
25    heap: &mut KernelHeap,
26) -> (MemoryRegion<VirtAddr>, MemoryRegion<VirtAddr>) {
27    let empty = MemoryRegion::new(VirtAddr::null(), 0);
28
29    let Some((symtab, strtab)) = elf.symtab.as_ref().zip(elf.strtab.as_ref()) else {
30        log::info!("No kernel symbols!");
31        return (empty, empty);
32    };
33
34    allocate_kernel_symbols(symtab, strtab, heap).unwrap_or_else(|e| {
35        log::warn!("Failed to allocate memory for kernel symbols: {e:?}");
36        (empty, empty)
37    })
38}
39
40fn allocate_kernel_symbols(
41    symtab: &Elf64Symtab<'_>,
42    strtab: &Elf64Strtab<'_>,
43    heap: &mut KernelHeap,
44) -> Result<(MemoryRegion<VirtAddr>, MemoryRegion<VirtAddr>), SvsmError> {
45    let mut dst1 = PageCursor::new(heap);
46    for i in 0..symtab.syms_num() {
47        let sym = symtab.read_sym(i)?;
48        // `KSym` stores the size in a u32 since there should be no symbols
49        // spanning more than 4GiB, and this helps shave off 8 bytes from
50        // the struct (there is one per defined symbol, so this adds up to a
51        // reduction of ~100KiB in memory footprint as of writing). Thus, in
52        // the strange case that there is a symbol of this size, simply use the
53        // maximum value of a u32.
54        let size = u32::try_from(sym.st_size).unwrap_or_else(|_| {
55            log::warn!(
56                "Symbol at {:#x} is larger than 4GiB, symbol resolution may be inaccurate.",
57                sym.st_size
58            );
59            u32::MAX
60        });
61        dst1.write(KSym::new(sym.st_value as usize, size, sym.st_name).as_bytes())?;
62    }
63    let symtab_region = dst1.region();
64
65    let mut dst2 = PageCursor::new(heap);
66    dst2.write(strtab.buf())?;
67    let strtab_region = dst2.region();
68
69    Ok((symtab_region, strtab_region))
70}

72/// A type that allows progressively reserving a dynamically-sized portion of
73/// memory.
74struct PageCursor<'a> {
75    region: MemoryRegion<VirtAddr>,
76    pos: usize,
77    heap: &'a mut KernelHeap,
78}
79
80impl<'a> PageCursor<'a> {
81    /// Create an empty `PageCursor`. This does not allocate.
82    fn new(heap: &'a mut KernelHeap) -> Self {
83        Self {
84            region: MemoryRegion::new(VirtAddr::null(), 0),
85            pos: 0,
86            heap,
87        }
88    }
89
90    /// Write some bytes into the cursor, allocating more memory in the process
91    /// if needed.
92    fn write(&mut self, val: &[u8]) -> Result<(), SvsmError> {
93        let new_pos = self.pos + val.len();
94        if let Some(need) = new_pos.checked_sub(self.region.len()) {
95            self.get_pages(need)?;
96        };
97
98        let dst = self.region.start() + self.pos;
99        // SAFETY: we trust the allocator to reserve memory properly.
100        // The source and destination are byte slices, so there are
101        // no invalid representations nor alignment requirements.
102        unsafe {
103            dst.as_mut_ptr::<u8>()
104                .copy_from_nonoverlapping(val.as_ptr(), val.len())
105        };
106
107        self.pos = new_pos;
108        Ok(())
109    }
110
111    /// Extends the cursor's allocation by requesting pages to satisfy the
112    /// requested additional length.
113    fn get_pages(&mut self, len: usize) -> Result<(), SvsmError> {
114        if len == 0 {
115            return Ok(());
116        }
117        let pages = round_to_pages(len);
118        let page = self.heap.allocate_pages(pages)?.0;
119        let region = MemoryRegion::new(page, pages * PAGE_SIZE);
120        if self.region.is_empty() {
121            self.region = region;
122            return Ok(());
123        }
124
125        // This should never happen, as the allocator returns contiguous
126        // pages, and we have an exclusive reference to it. Leave this to
127        // catch a potential change in the allocator.
128        assert!(self.region.contiguous(&region));
129
130        self.region = self.region.expand(region.len());
131        Ok(())
132    }
133
134    /// Returns the contiguous memory region that was allocated by the `PageCursor`.
135    fn region(&self) -> MemoryRegion<VirtAddr> {
136        MemoryRegion::new(self.region.start(), self.pos)
137    }
138}
