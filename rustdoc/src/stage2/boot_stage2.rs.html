21const MAILBOX_ADDR: u32 = bootlib::kernel_launch::SIPI_STUB_GPA;
22const AP_CTXT_ADDR: u32 = MAILBOX_ADDR + PAGE_SIZE as u32 - size_of::<ApStartContext>() as u32;
23
24// The mailbox page must be within the first 640 KB of memory so it can be
25// validated by the BSP in stage2. It also can't be in stage2's heap area.
26const _: () = assert!(MAILBOX_ADDR + (PAGE_SIZE as u32) <= STAGE2_HEAP_START);
27
28global_asm!(
29    r#"
30        .text
31        .section ".startup.text","ax"
32        .code32
33
34        .org 0
35        .globl startup_32
36        startup_32:
37
38        /* Save pointer to startup structure in EBP */
39        movl %esp, %ebp
40
41        /*
42         * Load a GDT. Despite the naming, it contains valid
43         * entries for both, "legacy" 32bit and long mode each.
44         */
45        movl $gdt64_desc, %eax
46        lgdt (%eax)
47
48        movw $0x10, %ax
49        movw %ax, %ds
50        movw %ax, %es
51        movw %ax, %fs
52        movw %ax, %gs
53        movw %ax, %ss
54
55        ljmpl $0x8, $.Lon_svsm32_cs
56
57    .Lon_svsm32_cs:
58        /*
59         * SEV: %esi is always 0, only BSP running
60         * TDX: %esi is the TD CPU index
61         */
62        test %esi, %esi
63        jnz .Lskip_paging_setup
64
65        /* Clear out the static page table pages. */
66        movl $pgtable_end, %ecx
67        subl $pgtable, %ecx
68        shrl $2, %ecx
69        xorl %eax, %eax
70        movl $pgtable, %edi
71        rep stosl
72
73        /* Determine the C-bit position within PTEs. */
74        call get_pte_c_bit
75        movl %eax, %edx
76
77        /* Populate the static page table pages with an identity mapping. */
78        movl $pgtable, %edi
79        leal 0x1007(%edi), %eax
80        movl %eax, 0(%edi)
81        addl %edx, 4(%edi)
82
83        addl $0x1000, %edi
84        leal 0x1007(%edi), %eax
85        movl $4, %ecx
86        1: movl %eax, 0(%edi)
87        addl %edx, 4(%edi)
88        addl $0x1000, %eax
89        addl $8, %edi
90        decl %ecx
91        jnz 1b
92        andl $0xfffff000, %edi
93
94        addl $0x1000, %edi
95        movl $0x00000183, %eax
96        movl $2048, %ecx
97        1: movl %eax, 0(%edi)
98        addl %edx, 4(%edi)
99        addl $0x00200000, %eax
100        addl $8, %edi
101        decl %ecx
102        jnz 1b
103
104        /* Insert a self-map entry */
105        movl $pgtable, %edi
106        movl %edi, %eax
107        orl $0x63, %eax
108        movl %eax, 8*{PGTABLE_LVL3_IDX_PTE_SELFMAP}(%edi)
109        movl $0x80000000, %eax
110        orl %edx, %eax
111        movl %eax, 0xF6C(%edi)
112
113        jmp 2f
114
115    .Lskip_paging_setup:
116        movl $ap_flag, %edi
117    .Lap_wait_for_env:
118        cmpl $0, (%edi)
119        pause
120        je .Lap_wait_for_env
121
122    2:
123        /* Enable 64bit PTEs, CR4.PAE. */
124        movl %cr4, %eax
125        bts $5, %eax
126        movl %eax, %cr4
127
128        /* Enable long mode, EFER.LME. Also ensure NXE is set. */
129        movl ${EFER}, %ecx
130        rdmsr
131        movl %eax, %ebx
132        orl $({LME} | {NXE}), %eax
133        cmp %eax, %ebx
134        jz 3f
135        wrmsr
136    3:
137        /* Load the static page table root. */
138        movl $pgtable, %eax
139        movl %eax, %cr3
140
141        /* Enable paging, CR0.PG. */
142        movl %cr0, %eax
143        bts $31, %eax
144        movl %eax, %cr0
145
146        ljmpl $0x18, $startup_64
147
148    get_pte_c_bit:
149        /*
150         * Check if this is an SNP platform.  If not, there is no C bit.
151         */
152        cmpl $1, 8(%ebp)
153        jnz .Lvtom
154
155        /*
156         * Check that the SNP_Active bit in the SEV_STATUS MSR is set.
157         */
158        movl ${SEV_STATUS}, %ecx
159        rdmsr
160
161        testl $0x04, %eax
162        jz .Lno_sev_snp
163
164        /*
165         * Check whether VTOM is selected
166         */
167        testl $0x08, %eax
168        jnz .Lvtom
169
170        /* Determine the PTE C-bit position from the CPUID page. */
171
172        /* Locate the table.  The pointer to the CPUID page is 12 bytes into
173         * the stage2 startup structure. */
174        movl 12(%ebp), %ecx
175        /* Read the number of entries. */
176        movl (%ecx), %eax
177        /* Create a pointer to the first entry. */
178        leal 16(%ecx), %ecx
179
180    .Lcheck_entry:
181        /* Check that there is another entry. */
182        test %eax, %eax
183        je .Lno_sev_snp
184
185        /* Check the input parameters of the current entry. */
186        cmpl $0x8000001f, (%ecx) /* EAX_IN */
187        jne .Lwrong_entry
188        cmpl $0, 4(%ecx) /* ECX_IN */
189        jne .Lwrong_entry
190        cmpl $0, 8(%ecx) /* XCR0_IN (lower half) */
191        jne .Lwrong_entry
192        cmpl $0, 12(%ecx) /* XCR0_IN (upper half) */
193        jne .Lwrong_entry
194        cmpl $0, 16(%ecx) /* XSS_IN (lower half) */
195        jne .Lwrong_entry
196        cmpl $0, 20(%ecx) /* XSS_IN (upper half) */
197        jne .Lwrong_entry
198
199        /* All parameters were correct. */
200        jmp .Lfound_entry
201
202    .Lwrong_entry:
203        /*
204         * The current entry doesn't contain the correct input
205         * parameters. Try the next one.
206         */
207        decl %eax
208        addl $0x30, %ecx
209        jmp .Lcheck_entry
210
211    .Lfound_entry:
212        /* Extract the c-bit location from the cpuid entry. */
213        movl 28(%ecx), %ebx
214        andl $0x3f, %ebx
215
216        /*
217         * Verify that the C-bit position is within reasonable bounds:
218         * >= 32 and < 64.
219         */
220        cmpl $32, %ebx
221        jl .Lno_sev_snp
222        cmpl $64, %ebx
223        jae .Lno_sev_snp
224
225        subl $32, %ebx
226        xorl %eax, %eax
227        btsl %ebx, %eax
228        ret
229
230    .Lvtom:
231        xorl %eax, %eax
232        ret
233
234    .Lno_sev_snp:
235        hlt
236        jmp .Lno_sev_snp
237
238        .code64
239
240    startup_64:
241        /* Reload the data segments with 64bit descriptors. */
242        movw $0x20, %ax
243        movw %ax, %ds
244        movw %ax, %es
245        movw %ax, %fs
246        movw %ax, %gs
247        movw %ax, %ss
248
249        test %esi, %esi
250        jz .Lbsp_main
251
252        movl ${MAILBOX_VPIDX_ADDR}, %eax
253
254        /*
255         * The following wait-for-signal code must be written in asm because
256         * APs run stacklessly here.
257         */
258    .Lap_wait_for_signal:
259        /* vCPU index */
260        cmpl %esi, (%eax)
261        pause
262        jne .Lap_wait_for_signal
263
264        /* See smp.rs for prerequisites. */
265        movl ${AP_CTXT_ADDR}, %edi
266        movl ${AP_STARTUP_RIP_ADDR}, %eax
267        movq (%rax), %rax
268        jmp *%rax
269
270    .Lbsp_main:
271        /* Clear out .bss and transfer control to the main stage2 code. */
272        xorq %rax, %rax
273        leaq _bss(%rip), %rdi
274        leaq _ebss(%rip), %rcx
275        subq %rdi, %rcx
276        shrq $3, %rcx
277        rep stosq
278
279        /*
280         * Follow the C calling convention for x86-64:
281         *
282         * - Pass &Stage2LaunchInfo as the first argument (%rdi)
283         * - Make sure (%rsp + 8) is 16b-aligned when control is transferred
284         *   to stage2_main
285         */
286        movl %ebp, %edi
287        andq $~0xf, %rsp
288
289        /* Mark the next stack frame as the bottom frame */
290        xor %rbp, %rbp
291
292        call stage2_main
293
294        .data
295
296        .align 4
297        .globl ap_flag
298    ap_flag:
299        .long 0
300
301    idt32:
302        .rept 32
303        .quad 0
304        .endr
305    idt32_end:
306
307    idt32_desc:
308        .word idt32_end - idt32 - 1
309        .long idt32
310
311    idt64:
312        .rept 32
313        .octa 0
314        .endr
315    idt64_end:
316
317    idt64_desc:
318        .word idt64_end - idt64 - 1
319        .quad idt64
320
321        .align 256
322    gdt64:
323        .quad 0
324        .quad 0x00cf9a000000ffff /* 32 bit code segment */
325        .quad 0x00cf93000000ffff /* 32 bit data segment */
326        .quad 0x00af9a000000ffff /* 64 bit code segment */
327        .quad 0x00cf92000000ffff /* 64 bit data segment */
328    gdt64_end:
329
330    gdt64_desc:
331        .word gdt64_end - gdt64 - 1
332        .quad gdt64
333
334        .align {PAGE_SIZE}
335        .globl pgtable
336    pgtable:
337        .fill 7*{PAGE_SIZE}, 1, 0
338    pgtable_end:"#,
339    PAGE_SIZE = const PAGE_SIZE,
340    PGTABLE_LVL3_IDX_PTE_SELFMAP = const PGTABLE_LVL3_IDX_PTE_SELFMAP,
341    EFER = const EFER,
342    LME = const EFERFlags::LME.bits(),
343    NXE = const EFERFlags::NXE.bits(),
344    SEV_STATUS = const SEV_STATUS,
345    MAILBOX_VPIDX_ADDR = const MAILBOX_ADDR + offset_of!(TdMailbox, vcpu_index) as u32,
346    AP_CTXT_ADDR = const AP_CTXT_ADDR,
347    AP_STARTUP_RIP_ADDR = const AP_CTXT_ADDR + offset_of!(ApStartContext, initial_rip) as u32,
348    options(att_syntax)
349);
