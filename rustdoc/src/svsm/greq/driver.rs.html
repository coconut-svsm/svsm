7//! Driver to send `SNP_GUEST_REQUEST` commands to the PSP. It can be any of the
8//! request or response command types defined in the SEV-SNP spec, regardless if it's
9//! a regular or an extended command.
10

30/// Global `SNP_GUEST_REQUEST` driver instance
31static GREQ_DRIVER: SpinLock<OnceCell<SnpGuestRequestDriver>> = SpinLock::new(OnceCell::new());
32
33// Hypervisor error codes
34
35/// Buffer provided is too small
36const SNP_GUEST_REQ_INVALID_LEN: u64 = BIT!(32);
37/// Hypervisor busy, try again
38const SNP_GUEST_REQ_ERR_BUSY: u64 = BIT!(33);
39
40/// Class of the `SNP_GUEST_REQUEST` command: Regular or Extended
41#[derive(Clone, Copy, Debug, PartialEq)]
42#[repr(u8)]
43enum SnpGuestRequestClass {
44    Regular = 0,
45    Extended = 1,
46}
47
48/// `SNP_GUEST_REQUEST` driver
49#[derive(Debug)]
50struct SnpGuestRequestDriver {
51    /// Shared page used for the `SNP_GUEST_REQUEST` request
52    request: SharedBox<SnpGuestRequestMsg>,
53    /// Shared page used for the `SNP_GUEST_REQUEST` response
54    response: SharedBox<SnpGuestRequestMsg>,
55    /// Encrypted page where we perform crypto operations
56    staging: Box<SnpGuestRequestMsg>,
57    /// Extended data buffer that will be provided to the hypervisor
58    /// to store the SEV-SNP certificates
59    ext_data: SharedBox<SnpGuestRequestExtData>,
60    /// Extended data size (`certs` size) provided by the user in [`super::services::get_extended_report`].
61    /// It will be provided to the hypervisor.
62    user_extdata_size: usize,
63    /// Each `SNP_GUEST_REQUEST` message contains a sequence number per VMPCK.
64    /// The sequence number is incremented with each message sent. Messages
65    /// sent by the guest to the PSP and by the PSP to the guest must be
66    /// delivered in order. If not, the PSP will reject subsequent messages
67    /// by the guest when it detects that the sequence numbers are out of sync.
68    ///
69    /// NOTE: If the vmpl field of a `SNP_GUEST_REQUEST` message is set to VMPL0,
70    /// then it must contain the VMPL0 sequence number and be protected (encrypted)
71    /// with the VMPCK0 key; additionally, if this message fails, the VMPCK0 key
72    /// must be disabled. The same idea applies to the other VMPL levels.
73    ///
74    /// The SVSM needs to support only VMPL0 `SNP_GUEST_REQUEST` commands because
75    /// other layers in the software stack (e.g. OVMF and guest kernel) can send
76    /// non-VMPL0 commands directly to PSP. Therefore, the SVSM needs to maintain
77    /// the sequence number and the VMPCK only for VMPL0.
78    vmpck0_seqno: u64,
79}
80
81impl SnpGuestRequestDriver {
82    /// Create a new [`SnpGuestRequestDriver`]
83    pub fn new() -> Result<Self, SvsmReqError> {
84        let request = SharedBox::try_new_zeroed()?;
85        let response = SharedBox::try_new_zeroed()?;
86        let staging = SnpGuestRequestMsg::new_box_zeroed()
87            .map_err(|_| SvsmError::Alloc(AllocError::OutOfMemory))?;
88        let ext_data = SharedBox::try_new_zeroed()?;
89
90        Ok(Self {
91            request,
92            response,
93            staging,
94            ext_data,
95            user_extdata_size: size_of::<SnpGuestRequestExtData>(),
96            vmpck0_seqno: 0,
97        })
98    }
99
100    /// Get the last VMPCK0 sequence number accounted
101    fn seqno_last_used(&self) -> u64 {
102        self.vmpck0_seqno
103    }
104
105    /// Increase the VMPCK0 sequence number by two. In order to keep the
106    /// sequence number in-sync with the PSP, this is called only when the
107    /// `SNP_GUEST_REQUEST` response is received.
108    fn seqno_add_two(&mut self) {
109        self.vmpck0_seqno += 2;
110    }
111
112    /// Set the user_extdata_size to `n` and clear the first `n` bytes from `ext_data`
113    pub fn set_user_extdata_size(&mut self, n: usize) -> Result<(), SvsmReqError> {
114        // At least one page
115        if (n >> PAGE_SHIFT) == 0 {
116            return Err(SvsmReqError::invalid_parameter());
117        }
118        self.ext_data.nclear(n)?;
119        self.user_extdata_size = n;
120
121        Ok(())
122    }
123
124    /// Call the GHCB layer to send the encrypted SNP_GUEST_REQUEST message
125    /// to the PSP.
126    fn send(&mut self, req_class: SnpGuestRequestClass) -> Result<(), SvsmReqError> {
127        let req_page = self.request.addr();
128        let resp_page = self.response.addr();
129        let data_pages = self.ext_data.addr();
130        let ghcb = current_ghcb();
131
132        if req_class == SnpGuestRequestClass::Extended {
133            let num_user_pages = (self.user_extdata_size >> PAGE_SHIFT) as u64;
134            ghcb.guest_ext_request(req_page, resp_page, data_pages, num_user_pages)?;
135        } else {
136            ghcb.guest_request(req_page, resp_page)?;
137        }
138
139        self.seqno_add_two();
140
141        Ok(())
142    }
143
144    // Encrypt the request message from encrypted memory
145    fn encrypt_request(
146        &mut self,
147        msg_type: SnpGuestRequestMsgType,
148        msg_seqno: u64,
149        buffer: &[u8],
150        command_len: usize,
151    ) -> Result<(), SvsmReqError> {
152        // VMPL0 `SNP_GUEST_REQUEST` commands are encrypted with the VMPCK0 key
153        let vmpck0: [u8; VMPCK_SIZE] = secrets_page().get_vmpck(0);
154
155        let inbuf = buffer
156            .get(..command_len)
157            .ok_or_else(SvsmReqError::invalid_parameter)?;
158
159        // For security reasons, encrypt the message in protected memory (staging)
160        // and then copy the result to shared memory (request)
161        self.staging
162            .encrypt_set(msg_type, msg_seqno, &vmpck0, inbuf)?;
163        self.request.write_from(&self.staging);
164        Ok(())
165    }
166
167    // Decrypt the response message from encrypted memory
168    fn decrypt_response(
169        &mut self,
170        msg_seqno: u64,
171        msg_type: SnpGuestRequestMsgType,
172        buffer: &mut [u8],
173    ) -> Result<usize, SvsmReqError> {
174        let vmpck0: [u8; VMPCK_SIZE] = secrets_page().get_vmpck(0);
175
176        // For security reasons, decrypt the message in protected memory (staging)
177        self.response.read_into(&mut self.staging);
178        let result = self
179            .staging
180            .decrypt_get(msg_type, msg_seqno, &vmpck0, buffer);
181
182        if let Err(e) = result {
183            match e {
184                // The buffer provided is too small to store the unwrapped response.
185                // There is no need to clear the VMPCK0, just report it as invalid parameter.
186                SvsmReqError::RequestError(SvsmResultCode::INVALID_PARAMETER) => (),
187                _ => secrets_page_mut().clear_vmpck(0),
188            }
189        }
190
191        result
192    }
193
194    /// Send the provided VMPL0 `SNP_GUEST_REQUEST` command to the PSP.
195    ///
196    /// The command will be encrypted using AES-256 GCM.
197    ///
198    /// # Arguments
199    ///
200    /// * `req_class`: whether this is a regular or extended `SNP_GUEST_REQUEST` command
201    /// * `msg_type`: type of the command stored in `buffer`, e.g. SNP_MSG_REPORT_REQ
202    /// * `buffer`: buffer with the `SNP_GUEST_REQUEST` command to be sent.
203    ///   The same buffer will also be used to store the response.
204    /// * `command_len`: Size (in bytes) of the command stored in `buffer`
205    ///
206    /// # Returns
207    ///
208    /// * Success:
209    ///     * `usize`: Size (in bytes) of the response stored in `buffer`
210    /// * Error:
211    ///     * [`SvsmReqError`]
212    fn send_request(
213        &mut self,
214        req_class: SnpGuestRequestClass,
215        msg_type: SnpGuestRequestMsgType,
216        buffer: &mut [u8],
217        command_len: usize,
218    ) -> Result<usize, SvsmReqError> {
219        if secrets_page().is_vmpck_clear(0) {
220            return Err(SvsmReqError::invalid_request());
221        }
222
223        // Message sequence number overflow, the driver will not able
224        // to send subsequent `SNP_GUEST_REQUEST` messages to the PSP.
225        // The sequence number is restored only when the guest is rebooted.
226        let Some(msg_seqno) = self.seqno_last_used().checked_add(1) else {
227            log::error!("SNP_GUEST_REQUEST: sequence number overflow");
228            secrets_page_mut().clear_vmpck(0);
229            return Err(SvsmReqError::invalid_request());
230        };
231
232        self.encrypt_request(msg_type, msg_seqno, buffer, command_len)?;
233
234        if let Err(e) = self.send(req_class) {
235            if let SvsmReqError::FatalError(SvsmError::Ghcb(GhcbError::VmgexitError(_rbx, info2))) =
236                e
237            {
238                // For some reason the hypervisor did not forward the request to the PSP.
239                //
240                // Because the message sequence number is used as part of the AES-GCM IV, it is important that the
241                // guest retry the request before allowing another request to be performed so that the IV cannot be
242                // reused on a new message payload.
243                match info2 & 0xffff_ffff_0000_0000u64 {
244                    // The certificate buffer provided is too small.
245                    SNP_GUEST_REQ_INVALID_LEN => {
246                        if req_class == SnpGuestRequestClass::Extended {
247                            if let Err(e1) = self.send(SnpGuestRequestClass::Regular) {
248                                log::error!(
249                                    "SNP_GUEST_REQ_INVALID_LEN. Aborting, request resend failed"
250                                );
251                                secrets_page_mut().clear_vmpck(0);
252                                return Err(e1);
253                            }
254                            return Err(e);
255                        } else {
256                            // We sent a regular SNP_GUEST_REQUEST, but the hypervisor returned
257                            // an error code that is exclusive for extended SNP_GUEST_REQUEST
258                            secrets_page_mut().clear_vmpck(0);
259                            return Err(SvsmReqError::invalid_request());
260                        }
261                    }
262                    // The hypervisor is busy.
263                    SNP_GUEST_REQ_ERR_BUSY => {
264                        if let Err(e2) = self.send(req_class) {
265                            log::error!("SNP_GUEST_REQ_ERR_BUSY. Aborting, request resend failed");
266                            secrets_page_mut().clear_vmpck(0);
267                            return Err(e2);
268                        }
269                        // ... request resend worked, continue normally.
270                    }
271                    // Failed for unknown reason. Status codes can be found in
272                    // the AMD SEV-SNP spec or in the linux kernel include/uapi/linux/psp-sev.h
273                    _ => {
274                        log::error!("SNP_GUEST_REQUEST failed, unknown error code={}\n", info2);
275                        secrets_page_mut().clear_vmpck(0);
276                        return Err(e);
277                    }
278                }
279            }
280        }
281
282        let msg_seqno = self.seqno_last_used();
283        let resp_msg_type = SnpGuestRequestMsgType::try_from(msg_type as u8 + 1)?;
284
285        self.decrypt_response(msg_seqno, resp_msg_type, buffer)
286    }
287
288    /// Send the provided regular `SNP_GUEST_REQUEST` command to the PSP
289    pub fn send_regular_guest_request(
290        &mut self,
291        msg_type: SnpGuestRequestMsgType,
292        buffer: &mut [u8],
293        command_len: usize,
294    ) -> Result<usize, SvsmReqError> {
295        self.send_request(SnpGuestRequestClass::Regular, msg_type, buffer, command_len)
296    }
297
298    /// Send the provided extended `SNP_GUEST_REQUEST` command to the PSP
299    pub fn send_extended_guest_request(
300        &mut self,
301        msg_type: SnpGuestRequestMsgType,
302        buffer: &mut [u8],
303        command_len: usize,
304        certs: &mut [u8],
305    ) -> Result<usize, SvsmReqError> {
306        self.set_user_extdata_size(certs.len())?;
307
308        let outbuf_len: usize = self.send_request(
309            SnpGuestRequestClass::Extended,
310            msg_type,
311            buffer,
312            command_len,
313        )?;
314
315        // The SEV-SNP certificates can be used to verify the attestation report.
316        self.ext_data.copy_to_slice(certs)?;
317        // At this point, a zeroed ext_data buffer indicates that the
318        // certificates were not imported. The VM owner can import them from the
319        // host using the virtee/snphost project
320        if certs[..24] == [0; 24] {
321            log::warn!("SEV-SNP certificates not found. Make sure they were loaded from the host.");
322        }
323
324        Ok(outbuf_len)
325    }
326}

328/// Initialize the global `SnpGuestRequestDriver`
329///
330/// # Panics
331///
332/// This function panics if we fail to initialize any of the `SnpGuestRequestDriver` fields.
333pub fn guest_request_driver_init() {
334    let cell = GREQ_DRIVER.lock();
335    let _ = cell.get_or_init(|| {
336        SnpGuestRequestDriver::new().expect("SnpGuestRequestDriver failed to initialize")
337    });
338}
339
340/// Send the provided regular `SNP_GUEST_REQUEST` command to the PSP.
341/// Further details can be found in the `SnpGuestRequestDriver.send_request()` documentation.
342pub fn send_regular_guest_request(
343    msg_type: SnpGuestRequestMsgType,
344    buffer: &mut [u8],
345    request_len: usize,
346) -> Result<usize, SvsmReqError> {
347    let mut cell = GREQ_DRIVER.lock();
348    let driver: &mut SnpGuestRequestDriver =
349        cell.get_mut().ok_or_else(SvsmReqError::invalid_request)?;
350    driver.send_regular_guest_request(msg_type, buffer, request_len)
351}
