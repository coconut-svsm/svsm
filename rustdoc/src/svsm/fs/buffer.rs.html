13pub trait Buffer {
14    /// Copy data from the buffer into a slice
15    ///
16    /// # Arguments
17    ///
18    /// - `buf`: Destination slice for data.
19    /// - `offset`: Offset into the buffer to start copying from.
20    ///
21    /// # Returns
22    ///
23    /// A `usize` representing the number of bytes copied on success, or
24    /// [`SvsmError`] on failure. Not that the content of `buf` is undefined on
25    /// failure.
26    fn read_buffer(&self, buf: &mut [u8], offset: usize) -> Result<usize, SvsmError>;
27
28    /// Copy data from a slice into the Buffer
29    ///
30    /// # Arguments
31    ///
32    /// - `buf`: Source slice for data.
33    /// - `offset`: Offset into the buffer to start copying to.
34    ///
35    /// # Returns
36    ///
37    /// A `usize` representing the number of bytes copied on success, or
38    /// [`SvsmError`] on failure.
39    fn write_buffer(&mut self, _buf: &[u8], _offset: usize) -> Result<usize, SvsmError> {
40        Err(SvsmError::FileSystem(FsError::not_supported()))
41    }
42
43    /// Total number of bytes represented by this buffer.
44    ///
45    /// # Returns
46    ///
47    /// Total number of bytes that can be copied from/to the buffer.
48    fn size(&self) -> usize;
49}

51/// Struct to add a [`Buffer`] interface to a mutable `&[u8]` slice
52#[derive(Debug)]
53pub struct SliceMutRefBuffer<'a> {
54    slice: &'a mut [u8],
55}
56
57impl<'a> SliceMutRefBuffer<'a> {
58    pub fn new(slice: &'a mut [u8]) -> Self {
59        Self { slice }
60    }
61}
62
63impl Buffer for SliceMutRefBuffer<'_> {
64    fn read_buffer(&self, buf: &mut [u8], offset: usize) -> Result<usize, SvsmError> {
65        let size = cmp::min(buf.len(), self.slice.len() - offset);
66        buf[..size].clone_from_slice(&self.slice[offset..offset + size]);
67        Ok(size)
68    }
69
70    fn write_buffer(&mut self, buf: &[u8], offset: usize) -> Result<usize, SvsmError> {
71        let size = cmp::min(buf.len(), self.slice.len() - offset);
72        self.slice[offset..offset + size].clone_from_slice(&buf[..size]);
73        Ok(size)
74    }
75
76    fn size(&self) -> usize {
77        self.slice.len()
78    }
79}

105#[derive(Debug)]
106pub struct UserBuffer {
107    addr: VirtAddr,
108    size: usize,
109}
110
111impl UserBuffer {
112    pub fn new(addr: VirtAddr, size: usize) -> Self {
113        Self { addr, size }
114    }
115}
116
117impl Buffer for UserBuffer {
118    fn read_buffer(&self, buf: &mut [u8], offset: usize) -> Result<usize, SvsmError> {
119        let size = cmp::min(buf.len(), self.size.checked_sub(offset).unwrap());
120        if size > 0 {
121            copy_from_user(self.addr + offset, &mut buf[..size])?;
122        }
123        Ok(size)
124    }
125
126    fn write_buffer(&mut self, buf: &[u8], offset: usize) -> Result<usize, SvsmError> {
127        let size = cmp::min(buf.len(), self.size.checked_sub(offset).unwrap());
128        if size > 0 {
129            copy_to_user(&buf[..size], self.addr + offset)?;
130        }
131        Ok(size)
132    }
133
134    fn size(&self) -> usize {
135        self.size
136    }
137}
