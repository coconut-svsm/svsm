21fn get_and_clear_caa_request_flag(vmsa_ref: &GuestVmsaRef) -> Result<bool, SvsmReqError> {
22    if let Some(caa_addr) = vmsa_ref.caa_addr() {
23        let calling_area = GuestPtr::<SvsmCaa>::new(caa_addr);
24        // SAFETY: guest vmsa and ca are always validated before beeing updated
25        // (core_remap_ca(), core_create_vcpu() or prepare_fw_launch()) so
26        // they're safe to use.
27        let caa = unsafe { calling_area.read()? };
28
29        let caa_serviced = caa.serviced();
30
31        // SAFETY: guest vmsa is always validated before beeing updated
32        // (core_remap_ca() or core_create_vcpu()) so it's safe to use.
33        unsafe {
34            calling_area.write(caa_serviced)?;
35        }
36
37        Ok(caa.call_pending())
38    } else {
39        Ok(false)
40    }
41}

43fn get_svsm_request_message(vmsa_ref: &mut GuestVmsaRef) -> Option<GuestExitMessage> {
44    let vmsa = vmsa_ref.vmsa();
45
46    // Ignore guest exits that were not initiated by VMFEXIT.
47    if !matches!(vmsa.guest_exit_code, GuestVMExit::VMGEXIT) {
48        return None;
49    }
50    let rax = vmsa.rax;
51    let protocol = (rax >> 32) as u32;
52    let request = (rax & 0xffff_ffff) as u32;
53    let params = RequestParams::from_vmsa(vmsa);
54
55    match get_and_clear_caa_request_flag(vmsa_ref) {
56        Ok(pending) => {
57            if pending {
58                return Some(GuestExitMessage::Svsm((protocol, request, params)));
59            }
60        }
61        Err(SvsmReqError::RequestError(code)) => {
62            log::debug!(
63                "Soft error handling protocol {} request {}: {:?}",
64                protocol,
65                request,
66                code
67            );
68        }
69        Err(SvsmReqError::FatalError(err)) => {
70            panic!(
71                "Fatal error handling core protocol request {}: {:?}",
72                request, err
73            );
74        }
75    }
76
77    None
78}

80pub fn enter_guest(mut regs: &[GuestRegister]) -> GuestExitMessage {
81    let cpu = this_cpu();
82
83    // If no VMSA or CAA are configured, then the guest cannot be entered.
84    if cpu.update_guest_mappings().is_err() {
85        return GuestExitMessage::NoMappings;
86    }
87
88    loop {
89        // Modify guest registers before disabling interrupts.
90        let mut vmsa_ref = cpu.guest_vmsa_ref();
91        let caa_addr = vmsa_ref.caa_addr();
92        let vmsa = vmsa_ref.vmsa();
93
94        for reg in regs {
95            set_guest_register(vmsa, reg);
96        }
97
98        // Ensure that no further register modification occurs if the loop
99        // restarts.
100        regs = &[];
101
102        // No interrupts may be processed once guest APIC state is updated,
103        // since handling an interrupt may modify the guest APIC state
104        // calculations, which could cause state corruption.  If interrupts are
105        // disabled, then any additional guest APIC updates generated by the
106        // host will block the VMPL transition and permit reevaluation of
107        // guest APIC state.
108        let guard = IrqGuard::new();
109
110        // Update APIC interrupt emulation state if required.
111        cpu.update_apic_emulation(vmsa, caa_addr);
112
113        // Make VMSA runnable again by setting EFER.SVME.
114        vmsa.enable();
115
116        // The VMSA reference must not be held when the guest is running.
117        drop(vmsa_ref);
118
119        flush_tlb_global_sync();
120
121        switch_to_vmpl(GUEST_VMPL as u32);
122
123        // Interrupts can safely be reenabled once the guest has returned to the
124        // SVSM.
125        drop(guard);
126
127        // If no mapping exists, then indicate to the caller that the guest
128        // exited with no valid mappings.
129        if cpu.update_guest_mappings().is_err() {
130            return GuestExitMessage::NoMappings;
131        }
132
133        // Obtain a reference to the VMSA just long enough to extract the
134        // request parameters.
135        {
136            let mut vmsa_ref = cpu.guest_vmsa_ref();
137            let vmsa = vmsa_ref.vmsa();
138
139            // Clear EFER.SVME in guest VMSA.
140            vmsa.disable();
141
142            if let Some(msg) = get_svsm_request_message(vmsa_ref.deref_mut()) {
143                return msg;
144            }
145        }
146    }
147}
