7use crate::cpu::control_regs::{cr0_sse_enable, cr4_osfxsr_enable, cr4_xsave_enable};
8use crate::cpu::cpuid::CpuidResult;
9use core::arch::asm;
10use core::arch::x86_64::{_xgetbv, _xsetbv};
11use core::sync::atomic::{AtomicU64, Ordering};
12
13const CPUID_EDX_SSE1: u32 = 25;
14const CPUID_ECX_XSAVE: u32 = 26;
15const CPUID_EAX_XSAVEOPT: u32 = 0;
16const XCR0_X87_ENABLE: u64 = 0x1;
17const XCR0_SSE_ENABLE: u64 = 0x2;
18const XCR0_YMM_ENABLE: u64 = 0x4;
19
20static SVSM_XCR0: AtomicU64 = AtomicU64::new(XCR0_X87_ENABLE | XCR0_SSE_ENABLE);
21
22fn legacy_sse_supported() -> bool {
23    let res = CpuidResult::get(1, 0);
24    (res.edx & (1 << CPUID_EDX_SSE1)) != 0
25}
26
27fn legacy_sse_enable() {
28    if legacy_sse_supported() {
29        cr4_osfxsr_enable();
30        cr0_sse_enable();
31    } else {
32        panic!("Legacy SSE unsupported");
33    }
34}
35
36fn extended_sse_supported() -> bool {
37    let res = CpuidResult::get(0xD, 0);
38    (res.eax & 0x7) == 0x7
39}
40
41fn xsave_supported() -> bool {
42    let res = CpuidResult::get(1, 0);
43    (res.ecx & (1 << CPUID_ECX_XSAVE)) != 0
44}
45
46fn xsaveopt_supported() -> bool {
47    let res = CpuidResult::get(0xD, 1);
48    (res.eax & (1 << CPUID_EAX_XSAVEOPT)) != 0
49}
50
51fn xcr0_set() {
52    // SAFETY: No impact on memory safety, enables FPU87 and SSE XSAVE features
53    unsafe {
54        // set bits [0-2] in XCR0 to enable extended SSE
55        let xr0 = _xgetbv(0) | SVSM_XCR0.load(Ordering::Relaxed);
56        _xsetbv(0, xr0);
57    }
58}
59
60pub fn get_xsave_area_size() -> u32 {
61    let res = CpuidResult::get(0xD, 0);
62    res.ecx
63}
64
65fn xsave_enable() {
66    if xsave_supported() && xsaveopt_supported() {
67        if extended_sse_supported() {
68            SVSM_XCR0.fetch_or(XCR0_YMM_ENABLE, Ordering::Relaxed);
69        }
70        cr4_xsave_enable();
71        xcr0_set();
72    } else {
73        panic!("xsave unsupported");
74    }
75}
76
77// Enable media and x87 instructions
78pub fn sse_init() {
79    legacy_sse_enable();
80    xsave_enable();
81}

83/// # Safety
84/// inline assembly here is used to save the SSE/FPU
85/// context. This context store is specific to a task and
86/// no other part of the code is accessing this memory at the same time.
87pub unsafe fn sse_save_context(addr: u64) {
88    let save_bits = SVSM_XCR0.load(Ordering::Relaxed);
89    // SAFETY: Inline assembly used to save the SSE/FPU context. This context
90    // store is specific to a task and no other part of the code is accessing
91    // this memory at the same time.
92    unsafe {
93        asm!(
94            r#"
95            xsaveopt (%rsi)
96            "#,
97            in("rsi") addr,
98            in("rax") save_bits,
99            in("rdx") 0,
100            options(att_syntax));
101    }
102}
103
104/// # Safety
105/// inline assembly here is used to restore the SSE/FPU
106/// context. This context store is specific to a task and
107/// no other part of the code is accessing this memory at the same time.
108pub unsafe fn sse_restore_context(addr: u64) {
109    let save_bits = SVSM_XCR0.load(Ordering::Relaxed);
110    // SAFETY: Inline assembly used to restore the SSE/FPU context. This context
111    // store is specific to a task and no other part of the code is accessing
112    // this memory at the same time.
113    unsafe {
114        asm!(
115            r#"
116            xrstor (%rsi)
117            "#,
118            in("rsi") addr,
119            in("rax") save_bits,
120            in("rdx") 0,
121            options(att_syntax));
122    }
123}
