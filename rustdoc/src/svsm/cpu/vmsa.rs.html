15pub fn svsm_code_segment() -> hyperv::HvSegmentRegister {
16    hyperv::HvSegmentRegister {
17        selector: SVSM_CS,

19        limit: 0xffff_ffff,
20        base: 0,

51impl From<hyperv::HvSegmentRegister> for VMSASegment {
52    fn from(segment: hyperv::HvSegmentRegister) -> Self {
53        Self {
54            selector: segment.selector,
55            flags: (segment.attributes & 0xFF) | ((segment.attributes & 0xF000) >> 4),
56            limit: segment.limit,
57            base: segment.base,
58        }
59    }
60}
61
62impl From<hyperv::HvTableRegister> for VMSASegment {
63    fn from(table: hyperv::HvTableRegister) -> Self {
64        Self {
65            selector: 0,
66            flags: 0,
67            limit: table.limit as u32,
68            base: table.base,
69        }
70    }
71}

73pub fn init_svsm_vmsa(vmsa: &mut VMSA, vtom: u64, context: &hyperv::HvInitialVpContext) {
74    vmsa.es = context.es.into();
75    vmsa.cs = context.cs.into();
76    vmsa.ss = context.ss.into();
77    vmsa.ds = context.ds.into();
78    vmsa.fs = context.fs.into();
79    vmsa.gs = context.gs.into();
80    vmsa.tr = context.tr.into();
81
82    vmsa.gdt = context.gdtr.into();
83    vmsa.idt = context.idtr.into();
84
85    vmsa.rip = context.rip;
86    vmsa.rsp = context.rsp;
87    vmsa.rflags = context.rflags;
88
89    vmsa.cr0 = context.cr0;
90    vmsa.cr3 = context.cr3;
91    vmsa.cr4 = context.cr4;
92    vmsa.efer = context.efer;
93
94    vmsa.g_pat = context.pat;
95
96    vmsa.dr6 = 0xffff0ff0;
97    vmsa.dr7 = 0x400;
98    vmsa.xcr0 = 1;
99    vmsa.mxcsr = 0x1f80;
100    vmsa.x87_ftw = 0x5555;
101    vmsa.x87_fcw = 0x0040;
102    vmsa.vmpl = 0;
103    vmsa.vtom = vtom;
104
105    vmsa.sev_features = sev_flags().as_sev_features();
106}

108fn real_mode_code_segment(rip: u64) -> VMSASegment {
109    VMSASegment {
110        selector: 0xf000,
111        base: rip & 0xffff_0000u64,
112        limit: 0xffff,
113        flags: 0x9b,
114    }
115}
116fn real_mode_data_segment() -> VMSASegment {
117    VMSASegment {
118        selector: 0,
119        flags: 0x93,
120        limit: 0xFFFF,
121        base: 0,
122    }
123}

125fn real_mode_sys_seg(flags: u16) -> VMSASegment {
126    VMSASegment {
127        selector: 0,
128        base: 0,
129        limit: 0xffff,
130        flags,
131    }
132}

134pub fn init_guest_vmsa(v: &mut VMSA, rip: u64, alternate_injection: bool) {
135    v.cr0 = 0x6000_0010;
136    v.rflags = 0x2;
137    v.rip = rip & 0xffff;
138    v.cs = real_mode_code_segment(rip);
139    v.ds = real_mode_data_segment();
140    v.es = real_mode_data_segment();
141    v.fs = real_mode_data_segment();
142    v.gs = real_mode_data_segment();
143    v.ss = real_mode_data_segment();
144    v.gdt = real_mode_sys_seg(0);
145    v.idt = real_mode_sys_seg(0);
146    v.ldt = real_mode_sys_seg(0x82);
147    v.tr = real_mode_sys_seg(0x8b);
148    v.dr6 = 0xffff_0ff0;
149    v.dr7 = 0x0400;
150    v.g_pat = 0x0007040600070406u64;
151    v.xcr0 = 1;
152    v.mxcsr = 0x1f80;
153    v.x87_ftw = 0x5555;
154    v.x87_fcw = 0x0040;
155
156    v.vmpl = GUEST_VMPL as u8;
157
158    let mut sev_status = sev_flags();
159
160    // Ensure that guest VMSAs do not enable restricted injection.
161    sev_status.remove(SEVStatusFlags::REST_INJ);
162
163    // Also ensure that guest VMSAs do not use debug register virtualization,
164    // since they may expect to make use of GHCB calls.
165    // TODO: find a way to make this optional so guests that expect debug
166    // register protection can achieve it.
167    sev_status.remove(SEVStatusFlags::DBGSWP);
168
169    // Enable alternate injection if requested.
170    if alternate_injection {
171        sev_status.insert(SEVStatusFlags::ALT_INJ);
172    }
173
174    v.sev_features = sev_status.as_sev_features();
175}
