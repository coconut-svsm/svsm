18pub const S_CET: u32 = 0x6a2;
19pub const PL0_SSP: u32 = 0x6a4;
20pub const ISST_ADDR: u32 = 0x6a8;
21
22pub const MODE_64BIT: usize = 1;
23pub const BUSY: usize = 1;
24
25pub static IS_CET_SUPPORTED: AtomicBool = AtomicBool::new(false);
26
27// Try to enable the CET feature in CR4 and set `IS_CET_SUPPORTED` if successful.
28pub fn determine_cet_support(platform: &dyn SvsmPlatform) {
29    if platform.determine_cet_support() {
30        IS_CET_SUPPORTED.store(true, Ordering::Relaxed);
31    }
32}
33
34pub fn determine_cet_support_from_cpuid() -> bool {
35    let cpuid = CpuidResult::get(7, 0);
36    (cpuid.ecx & 0x80) != 0
37}
38
39/// Returns whether shadow stacks are supported by the CPU and the kernel.
40#[inline(always)]
41pub fn is_cet_ss_supported() -> bool {
42    // In theory CPUs can have support for CET, but not CET_SS, but in practice
43    // no such CPUs exist. Treat CET being supported as CET_SS being supported.
44    IS_CET_SUPPORTED.load(Ordering::Relaxed)
45}
46
47pub fn shadow_stack_info() {
48    log::info!(
49        "Kernel shadow stacks {}",
50        match is_cet_ss_supported() {
51            true => "enabled",
52            false => "not supported",
53        }
54    );
55}

57/// Enable shadow stacks.
58///
59/// This code is placed in a macro instead of a function so that we don't have
60/// to set up the shadow stack to return from this code.
61#[macro_export]
62macro_rules! enable_shadow_stacks {
63    ($bsp_percpu:ident) => {{
64        use core::arch::asm;
65
66        let token_addr = $bsp_percpu.get_top_of_shadow_stack().unwrap();
67
68        // SAFETY: This assembly enables shadow-stacks and does not impact Rust
69        // memory safety.
70        unsafe {
71            asm!(
72                // Enable shadow stacks.
73                "wrmsr",
74                // Write a shadow stack restore token onto the stack.
75                "wrssq [{token_addr}], {token_val}",
76                // Load the shadow stack.
77                "rstorssp [{token_addr}]",
78                in("ecx") S_CET,
79                in("edx") 0,
80                in("eax") SCetFlags::SH_STK_EN.bits() | SCetFlags::WR_SHSTK_EN.bits(),
81                token_addr = in(reg) token_addr.bits(),
82                token_val = in(reg) token_addr.bits() + 8 + MODE_64BIT,
83                options(nostack, readonly),
84            );
85        }
86    }};
87}

89pub fn read_s_cet() -> SCetFlags {
90    SCetFlags::from_bits_retain(read_msr(S_CET))
91}

93bitflags! {
94    pub struct SCetFlags: u64 {
95        const SH_STK_EN = 1 << 0; // Enables the shadow stacks
96        const WR_SHSTK_EN = 1 << 1; // Enables the WRSS instruction
97    }
98}

100#[derive(Debug)]
101pub enum ShadowStackInit {
102    /// The initial shadow stack used by a CPU.
103    ///
104    /// This won't place any tokens on the shadow stack.
105    Init,
106    /// A shadow stack to be used during normal execution of a task.
107    ///
108    /// This will create a shadow stack with a shadow stack restore token.
109    Normal {
110        /// The address of the first instruction that will be executed by the task.
111        entry_return: usize,
112        /// The address of the function that's executed when the task exits.
113        exit_return: Option<usize>,
114    },
115    /// A shadow stack to be used during context switches.
116    ///
117    /// This will create a shadow stack with a shadow stack restore token.
118    ContextSwitch,
119    /// A shadow stack to be used for exception handling (either in PL0_SSP or
120    /// in the ISST).
121    ///
122    /// This will create a shadow stack with a supervisor shadow stack token.
123    Exception,
124}
125
126pub fn init_shadow_stack(
127    page: &PageRef,
128    top_of_sstack: VirtAddr,
129    init: ShadowStackInit,
130) -> (Option<VirtAddr>, VirtAddr) {
131    // Initialize the shadow stack.
132    let mut chunk = [0; 24];
133    let (base_token_addr, ssp) = match init {
134        ShadowStackInit::Normal {
135            entry_return,
136            exit_return,
137        } => {
138            // If exit return is empty, then this thread will be used as a
139            // user task stack.  In that case, place a busy token at the
140            // base of the shadow stack.
141            let base_token_addr = top_of_sstack - 8;
142            let base_token = match exit_return {
143                Some(addr) => addr,
144                None => base_token_addr.bits() + BUSY,
145            };
146
147            let (token_bytes, rip_bytes) = chunk.split_at_mut(8);
148
149            // Create a shadow stack restore token.
150            let token_addr = top_of_sstack - 24;
151            let token = (token_addr + 8).bits() + MODE_64BIT;
152            token_bytes.copy_from_slice(&token.to_ne_bytes());
153
154            let (entry_bytes, base_bytes) = rip_bytes.split_at_mut(8);
155            entry_bytes.copy_from_slice(&entry_return.to_ne_bytes());
156            base_bytes.copy_from_slice(&base_token.to_ne_bytes());
157
158            (Some(base_token_addr), token_addr)
159        }
160        ShadowStackInit::ContextSwitch => {
161            let (_, token_bytes) = chunk.split_at_mut(16);
162
163            // Create a shadow stack restore token.
164            let token_addr = top_of_sstack - 8;
165            let token = (token_addr + 8).bits() + MODE_64BIT;
166            token_bytes.copy_from_slice(&token.to_ne_bytes());
167
168            (None, token_addr)
169        }
170        ShadowStackInit::Exception => {
171            let (_, token_bytes) = chunk.split_at_mut(16);
172
173            // Create a supervisor shadow stack token.
174            let token_addr = top_of_sstack - 8;
175            let token = token_addr.bits();
176            token_bytes.copy_from_slice(&token.to_ne_bytes());
177
178            (None, token_addr)
179        }
180        ShadowStackInit::Init => (None, top_of_sstack - 8),
181    };
182
183    page.write(PAGE_SIZE - chunk.len(), &chunk);
184
185    (base_token_addr, ssp)
186}
