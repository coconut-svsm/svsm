7use super::tss::X86Tss;
8use crate::address::VirtAddr;
9use crate::ro_after_init_section;
10use crate::types::{SVSM_CS, SVSM_DS, SVSM_TSS};
11use core::arch::asm;
12use core::mem;
13
14#[repr(C, packed(2))]
15#[derive(Clone, Copy, Debug, Default)]
16struct GDTDesc {
17    size: u16,
18    addr: VirtAddr,
19}
20
21// The base address of the GDT should be aligned on an 8-byte boundary
22// to yield the best processor performance.
23#[derive(Copy, Clone, Debug, Default)]
24pub struct GDTEntry(u64);
25
26impl GDTEntry {
27    pub const fn from_raw(entry: u64) -> Self {
28        Self(entry)
29    }
30
31    pub fn to_raw(&self) -> u64 {
32        self.0
33    }
34
35    pub const fn null() -> Self {
36        Self(0u64)
37    }
38
39    pub const fn code_64_kernel() -> Self {
40        Self(0x00af9b000000ffffu64)
41    }
42
43    pub const fn data_64_kernel() -> Self {
44        Self(0x00cf93000000ffffu64)
45    }
46
47    pub const fn code_64_user() -> Self {
48        Self(0x00affb000000ffffu64)
49    }
50
51    pub const fn data_64_user() -> Self {
52        Self(0x00cff3000000ffffu64)
53    }
54}
55
56const GDT_SIZE: u16 = 8;
57
58// SAFETY: after this section's pages are made read-only, a write to this section
59// is a memory safety violation triggering a #PF, which is the intended
60// behavior.
61#[unsafe(link_section = ro_after_init_section!())]
62pub static GLOBAL_GDT: GDT = GDT::new();
63
64#[derive(Clone, Debug, Default)]
65pub struct GDT {
66    entries: [GDTEntry; GDT_SIZE as usize],
67}
68
69impl GDT {
70    pub const fn new() -> Self {
71        Self {
72            entries: [
73                GDTEntry::null(),
74                GDTEntry::code_64_kernel(),
75                GDTEntry::data_64_kernel(),
76                GDTEntry::code_64_user(),
77                GDTEntry::data_64_user(),
78                GDTEntry::null(),
79                GDTEntry::null(),
80                GDTEntry::null(),
81            ],
82        }
83    }
84
85    fn set_tss_entry(&mut self, desc0: GDTEntry, desc1: GDTEntry) {
86        let idx = (SVSM_TSS / 8) as usize;
87
88        let tss_entries = &self.entries[idx..idx + 1].as_mut_ptr();
89
90        // SAFETY:
91        // For add():
92        //   - idx and idx + size_of::<GDTEntry>() don't overflow isize.
93        //   - the borrow checker guarantees that self is still allocated
94        //   - self.entries[6:8] fits in self's allocation.
95        // For write_volatile():
96        //   - the borrow checker guarantees that self.entries is allocated
97        //   - alignment is checked inside
98        unsafe {
99            assert_eq!(align_of_val(&tss_entries.add(0)), align_of::<GDTEntry>());
100            assert_eq!(align_of_val(&tss_entries.add(1)), align_of::<GDTEntry>());
101
102            tss_entries.add(0).write_volatile(desc0);
103            tss_entries.add(1).write_volatile(desc1);
104        }
105    }
106
107    fn clear_tss_entry(&mut self) {
108        self.set_tss_entry(GDTEntry::null(), GDTEntry::null());
109    }
110
111    pub fn load_tss(&mut self, tss: &'static X86Tss) {
112        let (desc0, desc1) = tss.to_gdt_entry();
113
114        self.set_tss_entry(desc0, desc1);
115        // SAFETY: loading task register must me done in assembly.
116        // tss is ensured to have a static lifetime so this is safe.
117        unsafe { asm!("ltr %ax", in("ax") SVSM_TSS, options(att_syntax)) };
118        self.clear_tss_entry()
119    }
120
121    pub fn kernel_cs(&self) -> GDTEntry {
122        self.entries[(SVSM_CS / 8) as usize]
123    }
124
125    pub fn kernel_ds(&self) -> GDTEntry {
126        self.entries[(SVSM_DS / 8) as usize]
127    }
128
129    /// Makes this GDT the active GDT.
130    pub fn load(&self) {
131        let gdt_desc = self.descriptor();
132        // SAFETY: loading the GDT must be done in assembly.  Use of the GDT
133        // descriptor is safe because it describes a valid objct which
134        // implements Drop to clean up if the GDT object ever ceases to be
135        // valid.
136        unsafe {
137            asm!("lgdt ({0})",
138                 in(reg) &gdt_desc,
139                 options(att_syntax));
140        }
141    }
142
143    /// Loads all selectors from the current GDT.
144    pub fn load_selectors(&self) {
145        self.load();
146        // SAFETY: assembly is required to load segments from the GDT.  In the
147        // x86-64 architecture, the chosen selector values do not affect the
148        // validity of any memory addresses and thus cannot impact memory
149        // safety.
150        unsafe {
151            asm!(r#" /* Load GDT */
152
153                 /* Reload data segments */
154                 movw   %cx, %ds
155                 movw   %cx, %es
156                 movw   %cx, %fs
157                 movw   %cx, %gs
158                 movw   %cx, %ss
159
160                 /* Reload code segment */
161                 pushq  %rdx
162                 leaq   1f(%rip), %rax
163                 pushq  %rax
164                 lretq
165            1:
166                 "#,
167                in("rdx") SVSM_CS,
168                in("rcx") SVSM_DS,
169                options(att_syntax));
170        }
171    }
172
173    fn descriptor(&self) -> GDTDesc {
174        GDTDesc {
175            size: (GDT_SIZE * 8) - 1,
176            addr: VirtAddr::from(self.entries.as_ptr()),
177        }
178    }
179
180    pub fn base_limit(&self) -> (u64, u16) {
181        let gdt_entries = GDT_SIZE as usize;
182        let base: *const GDT = core::ptr::from_ref(self);
183        let limit = ((mem::size_of::<u64>() * gdt_entries) - 1) as u16;
184        (base as u64, limit)
185    }
186}
187
188impl Drop for GDT {
189    fn drop(&mut self) {
190        // Check to see whether the GDT being dropped is the one currently
191        // loaded on this CPU.  If so, reload the global GDT.
192        let gdt_desc: GDTDesc = Default::default();
193        // SAFETY: assembly is required to obtain the current GDT descriptor.
194        // The address of the returned descriptor is only used as a comparison
195        // to `self` and not for data access, so memory safety is not affected
196        // by the returned address.
197        unsafe {
198            asm!("sgdt ({0})",
199                 in(reg) &gdt_desc,
200                 options(att_syntax));
201        }
202
203        let gdt_addr = gdt_desc.addr;
204        if gdt_addr == VirtAddr::from(self as *const GDT) {
205            GLOBAL_GDT.load();
206        }
207    }
208}
