7use super::gdt::GDTEntry;
8use crate::address::VirtAddr;
9use core::arch::asm;
10use core::num::NonZeroU8;
11use core::ptr::addr_of;
12
13// IST offsets
14pub const IST_DF: NonZeroU8 = NonZeroU8::new(1).unwrap();
15
16#[derive(Debug, Default, Clone, Copy)]
17#[repr(C, packed(4))]
18pub struct X86Tss {
19    reserved0: u32,
20    stacks: [VirtAddr; 3],
21    reserved1: u64,
22    ist_stacks: [VirtAddr; 7],
23    reserved2: u64,
24    reserved3: u16,
25    io_bmp_base: u16,
26}
27
28pub const TSS_LIMIT: u64 = core::mem::size_of::<X86Tss>() as u64;
29
30impl X86Tss {
31    pub const fn new() -> Self {
32        X86Tss {
33            reserved0: 0,
34            stacks: [VirtAddr::null(); 3],
35            reserved1: 0,
36            ist_stacks: [VirtAddr::null(); 7],
37            reserved2: 0,
38            reserved3: 0,
39            io_bmp_base: (TSS_LIMIT + 1) as u16,
40        }
41    }
42
43    /// # Safety
44    /// No checks are performed on the stack address.  The caller must
45    /// ensure that the address is valid for stack usage.
46    pub unsafe fn set_ist_stack(&self, index: NonZeroU8, addr: VirtAddr) {
47        // IST entries start at index 1
48        let index = usize::from(index.get() - 1);
49        let stack_ptr = addr_of!(self.ist_stacks[index]);
50        // SAFETY: this move must be done in assembly because the target
51        // address is unaligned, and is not a candidate for using `Cell` to
52        // provide interior mutability on the stack pointer.
53        unsafe {
54            asm!("movq {0}, ({1})",
55                 in(reg) u64::from(addr),
56                 in(reg) stack_ptr,
57                 options(att_syntax));
58        }
59    }
60
61    /// # Safety
62    /// No checks are performed on the stack address.  The caller must
63    /// ensure that the address is valid for stack usage.
64    pub unsafe fn set_rsp0(&self, addr: VirtAddr) {
65        let stack_ptr = addr_of!(self.stacks[0]);
66        // SAFETY: this move must be done in assembly because the target
67        // address is unaligned, and is not a candidate for using `Cell` to
68        // provide interior mutability on the stack pointer.
69        unsafe {
70            asm!("movq {0}, ({1})",
71                 in(reg) u64::from(addr),
72                 in(reg) stack_ptr,
73                 options(att_syntax));
74        }
75    }
76
77    pub fn to_gdt_entry(&self) -> (GDTEntry, GDTEntry) {
78        let addr = (self as *const X86Tss) as u64;
79
80        let mut desc0: u64 = 0;
81        let mut desc1: u64 = 0;
82
83        // Limit
84        desc0 |= TSS_LIMIT & 0xffffu64;
85        desc0 |= ((TSS_LIMIT >> 16) & 0xfu64) << 48;
86
87        // Address
88        desc0 |= (addr & 0x00ff_ffffu64) << 16;
89        desc0 |= (addr & 0xff00_0000u64) << 32;
90        desc1 |= addr >> 32;
91
92        // Present
93        desc0 |= 1u64 << 47;
94
95        // Type
96        desc0 |= 0x9u64 << 40;
97
98        (GDTEntry::from_raw(desc0), GDTEntry::from_raw(desc1))
99    }
100}

102#[cfg(test)]
103mod test {
104    use super::*;
105    use core::mem::offset_of;
106
107    #[test]
108    fn test_tss_offsets() {
109        assert_eq!(offset_of!(X86Tss, reserved0), 0x0);
110        assert_eq!(offset_of!(X86Tss, stacks), 0x4);
111        assert_eq!(offset_of!(X86Tss, reserved1), 0x1c);
112        assert_eq!(offset_of!(X86Tss, ist_stacks), 0x24);
113        assert_eq!(offset_of!(X86Tss, reserved2), 0x5c);
114        assert_eq!(offset_of!(X86Tss, reserved3), 0x64);
115        assert_eq!(offset_of!(X86Tss, io_bmp_base), 0x66);
116    }
117}
