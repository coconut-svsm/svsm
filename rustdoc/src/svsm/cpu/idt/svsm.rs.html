114/// # Safety
115/// The caller must guarantee that the IDT object passed in does not go out of
116/// scope before the IDT is reloaded with a different object.
117pub unsafe fn early_idt_init(idt: &mut IDT<'_>) {
118    // Initialize the exception portion of the IDT.
119    init_idt_exceptions(idt);
120
121    // Load IDT
122    // SAFETY: the caller guarantees that the lifetime of the IDT object is
123    // appropriate for use here.
124    unsafe {
125        idt.load();
126    }
127}
128
129pub fn idt_init() -> Result<(), SvsmError> {
130    // Allocate a page of memory to use as the IDT.
131    let count = NonZero::new(PAGE_SIZE / mem::size_of::<IdtEntry>()).unwrap();
132    let idt_page = PageBox::<[IdtEntry]>::try_new_slice(IdtEntry::no_handler(), count)?;
133    let mut idt = IDT::new_from_page(idt_page);
134
135    // Configure the exception vectors
136    init_idt_exceptions(&mut idt);
137
138    // Switch #PF handler to the default one
139    idt.set_entry(PF_VECTOR, IdtEntry::entry(asm_entry_pf));
140
141    // Interupts
142    idt.set_entry(INT_INJ_VECTOR, IdtEntry::entry(asm_entry_irq_int_inj));
143    idt.set_entry(0x80, IdtEntry::user_entry(asm_entry_int80));
144    idt.set_entry(IPI_VECTOR, IdtEntry::entry(asm_entry_irq_ipi));
145
146    // Set IST vectors
147    init_ist_vectors(&mut idt);
148
149    // SAFETY:
150    // Capture an address that can be used by assembly code to read the #HV
151    // doorbell page.  The address of each CPU's doorbell page may be
152    // different, but the address of the field in the PerCpu structure that
153    // holds the actual pointer is constant across all CPUs, so that is the
154    // pointer that is actually captured.  The address that is captured is
155    // stored as a usize instead of a typed value, because the declarations
156    // required for type safety here are cumbersome, and the assembly code
157    // that uses the value is not type safe in any case, so enforcing type
158    // safety on the pointer would offer no meaningful value.
159    unsafe {
160        HV_DOORBELL_ADDR = this_cpu().hv_doorbell_addr() as usize;
161    };
162
163    // SAFETY: the IDT page was allocated above and is permanently associated
164    // with the IDT, so it can safely be loaded now.
165    unsafe {
166        idt.load();
167    }
168
169    // SAFETY: `idt` is the global static IDT, and it's populated.
170    unsafe {
171        idt.make_ro()?;
172    }
173
174    GLOBAL_IDT.init(idt).map_err(|_| SvsmError::PlatformInit)
175}

189// Doube-Fault handler
190#[no_mangle]
191extern "C" fn ex_handler_double_fault(ctxt: &mut X86ExceptionContext) {
192    let cr2 = read_cr2();
193    let rip = ctxt.frame.rip;
194    let rsp = ctxt.frame.rsp;
195
196    if user_mode(ctxt) {
197        log::error!(
198            "Double-Fault at RIP {:#018x} RSP: {:#018x} CR2: {:#018x} - Terminating task",
199            rip,
200            rsp,
201            cr2
202        );
203        terminate();
204    } else {
205        panic!(
206            "Double-Fault at RIP {:#018x} RSP: {:#018x} CR2: {:#018x}",
207            rip, rsp, cr2
208        );
209    }
210}

247// Page-Fault handler
248#[no_mangle]
249extern "C" fn ex_handler_page_fault(ctxt: &mut X86ExceptionContext, vector: usize) {
250    let cr2 = read_cr2();
251    let rip = ctxt.frame.rip;
252    let err = ctxt.error_code;
253    let vaddr = VirtAddr::from(cr2);
254
255    if user_mode(ctxt) {
256        let kill_task: bool = if is_task_fault(vaddr) {
257            current_task()
258                .fault(vaddr, (err & PageFaultError::W.bits() as usize) != 0)
259                .is_err()
260        } else {
261            true
262        };
263
264        if kill_task {
265            log::error!("Unexpected user-mode page-fault at RIP {:#018x} CR2: {:#018x} error code: {:#018x} - Terminating task",
266                    rip, cr2, err);
267            terminate();
268        }
269    } else if this_cpu()
270        .handle_pf(
271            VirtAddr::from(cr2),
272            (err & PageFaultError::W.bits() as usize) != 0,
273        )
274        .is_err()
275        && !handle_exception_table(ctxt)
276    {
277        handle_debug_exception(ctxt, vector);
278        panic!(
279            "Unhandled Page-Fault at RIP {:#018x} CR2: {:#018x} error code: {:#018x}",
280            rip, cr2, err
281        );
282    }
283}

285// Control-Protection handler
286#[no_mangle]
287extern "C" fn ex_handler_control_protection(ctxt: &mut X86ExceptionContext, _vector: usize) {
288    // From AMD64 Architecture Programmer's Manual, Volume 2, 8.4.3
289    // Control-Protection Error Code:
290    /// A RET (near) instruction encountered a return address mismatch.
291    const NEAR_RET: usize = 1;
292    /// A RET (far) or IRET instruction encountered a return address mismatch.
293    const FAR_RET_IRET: usize = 2;
294    /// An RSTORSSP instruction encountered an invalid shadow stack restore
295    /// token.
296    const RSTORSSP: usize = 4;
297    /// A SETSSBSY instruction encountered an invalid supervisor shadow stack
298    /// token.
299    const SETSSBSY: usize = 5;
300
301    let rip = ctxt.frame.rip;
302    match ctxt.error_code & 0x7fff {
303        code @ (NEAR_RET | FAR_RET_IRET) => {
304            // Read the return address on the normal stack.
305            let ret_ptr: GuestPtr<u64> = GuestPtr::new(VirtAddr::from(ctxt.frame.rsp));
306            // SAFETY: `rsp` is a valid guest address filled by the CPU in the
307            // X86InterruptFrame
308            let ret = unsafe { ret_ptr.read() }.expect("Failed to read return address");
309
310            // Read the return address on the shadow stack.
311            let prev_rssp_ptr: GuestPtr<u64> = GuestPtr::new(VirtAddr::from(ctxt.ssp));
312            // SAFETY: `ssp` is a valid guest address filled by the CPU in the
313            // X86ExceptionContext
314            let prev_rssp = unsafe { prev_rssp_ptr.read() }
315                .expect("Failed to read address of previous shadow stack pointer");
316            // The offset to the return pointer is different for RET and IRET.
317            let offset = if code == NEAR_RET { 0 } else { 8 };
318            let ret_ptr: GuestPtr<u64> = GuestPtr::new(VirtAddr::from(prev_rssp + offset));
319            let ret_on_ssp =
320                // SAFETY: `ssp` is a valid guest address filled by the CPU in the
321                // X86ExceptionContext
322                unsafe { ret_ptr.read() }.expect("Failed to read return address on shadow stack");
323
324            panic!("thread at {rip:#018x} tried to return to {ret:#x}, but return address on shadow stack was {ret_on_ssp:#x}!");
325        }
326        RSTORSSP => {
327            panic!("rstorssp instruction encountered an unexpected shadow stack restore token at RIP {rip:#018x}");
328        }
329        SETSSBSY => {
330            panic!("setssbsy instruction encountered an unexpected supervisor shadow stack token at RIP {rip:#018x}");
331        }
332        code => unreachable!("unexpected code for #CP exception: {code}"),
333    }
334}

336// Virtualization Exception handler
337#[no_mangle]
338extern "C" fn ex_handler_ve(ctxt: &mut X86ExceptionContext) {
339    let rip = ctxt.frame.rip;
340    let code = ctxt.error_code;
341
342    if let Err(err) = handle_virtualization_exception(ctxt) {
343        log::error!("#VE handling error: {:?}", err);
344        if user_mode(ctxt) {
345            log::error!("Failed to handle #VE from user-mode at RIP {:#018x} code: {:#018x} - Terminating task", rip, code);
346            terminate();
347        } else {
348            panic!(
349                "Failed to handle #VE from kernel-mode at RIP {:#018x} code: {:#018x}",
350                rip, code
351            );
352        }
353    }
354}

376// System Call SoftIRQ handler
377#[no_mangle]
378extern "C" fn ex_handler_system_call(
379    ctxt: &mut X86ExceptionContext,
380    vector: usize,
381    event_type: IdtEventType,
382) {
383    // Ensure that this vector was not invoked as a hardware interrupt vector.
384    if event_type.is_external_interrupt(vector) {
385        panic!("Syscall handler invoked as external interrupt!");
386    }
387
388    if !user_mode(ctxt) {
389        panic!("Syscall handler called from kernel mode!");
390    }
391
392    let Ok(input) = TryInto::<u64>::try_into(ctxt.regs.rax) else {
393        ctxt.regs.rax = !0;
394        return;
395    };
396
397    ctxt.regs.rax = match input {
398        // Class 0 SysCalls.
399        SYS_EXIT => sys_exit(ctxt.regs.rdi as u32),
400        SYS_EXEC => sys_exec(ctxt.regs.rdi, ctxt.regs.rsi, ctxt.regs.r8),
401        SYS_CLOSE => sys_close(ctxt.regs.rdi as u32),
402        // Class 1 SysCalls.
403        SYS_OPEN => sys_open(ctxt.regs.rdi, ctxt.regs.rsi, ctxt.regs.r8),
404        SYS_READ => sys_read(ctxt.regs.rdi as u32, ctxt.regs.rsi, ctxt.regs.r8),
405        SYS_WRITE => sys_write(ctxt.regs.rdi as u32, ctxt.regs.rsi, ctxt.regs.r8),
406        SYS_SEEK => sys_seek(ctxt.regs.rdi as u32, ctxt.regs.rsi, ctxt.regs.r8),
407        SYS_UNLINK => sys_unlink(ctxt.regs.rdi),
408        SYS_TRUNCATE => sys_truncate(ctxt.regs.rdi as u32, ctxt.regs.rsi),
409        SYS_OPENDIR => sys_opendir(ctxt.regs.rdi),
410        SYS_READDIR => sys_readdir(ctxt.regs.rdi as u32, ctxt.regs.rsi, ctxt.regs.r8),
411        SYS_MKDIR => sys_mkdir(ctxt.regs.rdi),
412        SYS_RMDIR => sys_rmdir(ctxt.regs.rdi),
413        // Class 3 SysCalls.
414        SYS_CAPABILITIES => sys_capabilities(ctxt.regs.rdi as u32),
415        _ => Err(SysCallError::EINVAL),
416    }
417    .map_or_else(|e| e as usize, |v| v as usize);
418}

432#[no_mangle]
433pub extern "C" fn common_isr_handler_entry(vector: usize) {
434    // Since interrupt handlers execute with interrupts disabled, it is
435    // necessary to increment the per-CPU interrupt disable nesting state
436    // while the handler is running in case common code attempts to disable
437    // interrupts temporarily.  The fact that this interrupt was received
438    // means that the previous state must have had interrupts enabled.
439    let cpu = this_cpu();
440    cpu.irqs_push_nesting(true);
441
442    common_isr_handler(vector);
443
444    // Decrement the interrupt disable nesting count, but do not permit
445    // interrupts to be reenabled.  They will be reenabled during the IRET
446    // flow.
447    cpu.irqs_pop_nesting();
448}
449
450pub fn common_isr_handler(vector: usize) {
451    // Set TPR based on the vector being handled and reenable interrupts to
452    // permit delivery of higher priority interrupts.  Because this routine
453    // dispatches interrupts which should only be observable if interrupts
454    // are enabled, the IRQ nesting count must be zero at this point.
455    let previous_tpr = raw_get_tpr();
456    raw_set_tpr(tpr_from_vector(vector));
457
458    let cpu = this_cpu();
459    cpu.irqs_enable();
460
461    // Process the requested interrupt vector.
462    match vector {
463        IPI_VECTOR => handle_ipi_interrupt(),
464        _ => {
465            // Ignore all unrecognized interrupt vectors and treat them as
466            // spurious interrupts.
467        }
468    }
469
470    // Disable interrupts before restoring TPR.
471    cpu.irqs_disable();
472    raw_set_tpr(previous_tpr);
473
474    // Perform the EOI cycle after the interrupt processing state has been
475    // restored so that recurrent interrupts will not introduce recursion at
476    // this point.
477    apic_eoi();
478}
