106/// # Safety
107/// The caller must guarantee that the IDT object passed in does not go out of
108/// scope before the IDT is reloaded with a different object.
109pub unsafe fn early_idt_init(idt: &mut IDT<'_>) {
110    // Initialize the exception portion of the IDT.
111    init_idt_exceptions(idt);
112
113    // Load IDT
114    // SAFETY: the caller guarantees that the lifetime of the IDT object is
115    // appropriate for use here.
116    unsafe {
117        idt.load();
118    }
119}

121pub fn idt_init() -> Result<(), SvsmError> {
122    // Allocate a page of memory to use as the IDT.
123    let count = NonZero::new(PAGE_SIZE / mem::size_of::<IdtEntry>()).unwrap();
124    let idt_page = PageBox::<[IdtEntry]>::try_new_slice(IdtEntry::no_handler(), count)?;
125    let mut idt = IDT::new_from_page(idt_page);
126
127    // Configure the exception vectors
128    init_idt_exceptions(&mut idt);
129
130    // Switch #PF handler to the default one
131    idt.set_entry(PF_VECTOR, IdtEntry::entry(asm_entry_pf));
132
133    // Interupts
134    idt.set_entry(INT_INJ_VECTOR, IdtEntry::entry(asm_entry_irq_int_inj));
135    idt.set_entry(0x80, IdtEntry::user_entry(asm_entry_int80));
136    idt.set_entry(IPI_VECTOR, IdtEntry::entry(asm_entry_irq_ipi));
137
138    // Set IST vectors
139    init_ist_vectors(&mut idt);
140
141    // SAFETY:
142    // Capture an address that can be used by assembly code to read the #HV
143    // doorbell page.  The address of each CPU's doorbell page may be
144    // different, but the address of the field in the PerCpu structure that
145    // holds the actual pointer is constant across all CPUs, so that is the
146    // pointer that is actually captured.  The address that is captured is
147    // stored as a usize instead of a typed value, because the declarations
148    // required for type safety here are cumbersome, and the assembly code
149    // that uses the value is not type safe in any case, so enforcing type
150    // safety on the pointer would offer no meaningful value.
151    unsafe {
152        HV_DOORBELL_ADDR = this_cpu().hv_doorbell_addr() as usize;
153    };
154
155    // SAFETY: the IDT page was allocated above and is permanently associated
156    // with the IDT, so it can safely be loaded now.
157    unsafe {
158        idt.load();
159    }
160
161    // SAFETY: `idt` is the global static IDT, and it's populated.
162    unsafe {
163        idt.make_ro()?;
164    }
165
166    GLOBAL_IDT.init(idt).map_err(|_| SvsmError::PlatformInit)
167}

169// General task termination handler
170#[unsafe(no_mangle)]
171extern "C" fn ex_handler_terminate(ctx: &mut X86ExceptionContext, vector: usize) {
172    if user_mode(ctx) {
173        // Terminate the current task if it is a user-mode task.
174        log::error!(
175            "Terminating task due to unhandled user mode exception {:#02x} at {:#018x}",
176            vector,
177            { ctx.frame.rip }
178        );
179        terminate();
180    } else {
181        // Kernel-mode tasks should never cause a termination class exception,
182        // so any such exception is grounds for panic.
183        ex_handler_panic(ctx, vector);
184    }
185}

199// Doube-Fault handler
200#[unsafe(no_mangle)]
201extern "C" fn ex_handler_double_fault(ctxt: &mut X86ExceptionContext) {
202    let cr2 = read_cr2();
203    let rip = ctxt.frame.rip;
204    let rsp = ctxt.frame.rsp;
205
206    if user_mode(ctxt) {
207        log::error!(
208            "Double-Fault at RIP {rip:#018x} RSP: {rsp:#018x} CR2: {cr2:#018x} - Terminating task"
209        );
210        terminate();
211    } else {
212        panic!(
213            "Double-Fault at RIP {:#018x} RSP: {:#018x} CR2: {:#018x}",
214            rip, rsp, cr2
215        );
216    }
217}

254// Page-Fault handler
255#[unsafe(no_mangle)]
256extern "C" fn ex_handler_page_fault(ctxt: &mut X86ExceptionContext, vector: usize) {
257    let cr2 = read_cr2();
258    let rip = ctxt.frame.rip;
259    let err = ctxt.error_code;
260    let vaddr = VirtAddr::from(cr2);
261
262    if user_mode(ctxt) {
263        let kill_task: bool = if is_task_fault(vaddr) {
264            current_task()
265                .fault(vaddr, (err & PageFaultError::W.bits() as usize) != 0)
266                .is_err()
267        } else {
268            true
269        };
270
271        if kill_task {
272            log::error!(
273                "Unexpected user-mode page-fault at RIP {rip:#018x} CR2: {cr2:#018x} error code: {err:#018x} - Terminating task"
274            );
275            terminate();
276        }
277    } else if this_cpu()
278        .handle_pf(
279            VirtAddr::from(cr2),
280            (err & PageFaultError::W.bits() as usize) != 0,
281        )
282        .is_err()
283        && !handle_exception_table(ctxt)
284    {
285        handle_debug_exception(ctxt, vector);
286        panic!(
287            "Unhandled Page-Fault at RIP {:#018x} CR2: {:#018x} error code: {:#018x}",
288            rip, cr2, err
289        );
290    }
291}

293// Control-Protection handler
294#[unsafe(no_mangle)]
295extern "C" fn ex_handler_control_protection(ctxt: &mut X86ExceptionContext, _vector: usize) {
296    // From AMD64 Architecture Programmer's Manual, Volume 2, 8.4.3
297    // Control-Protection Error Code:
298    /// A RET (near) instruction encountered a return address mismatch.
299    const NEAR_RET: usize = 1;
300    /// A RET (far) or IRET instruction encountered a return address mismatch.
301    const FAR_RET_IRET: usize = 2;
302    /// An RSTORSSP instruction encountered an invalid shadow stack restore
303    /// token.
304    const RSTORSSP: usize = 4;
305    /// A SETSSBSY instruction encountered an invalid supervisor shadow stack
306    /// token.
307    const SETSSBSY: usize = 5;
308
309    let rip = ctxt.frame.rip;
310    match ctxt.error_code & 0x7fff {
311        code @ (NEAR_RET | FAR_RET_IRET) => {
312            // Read the return address on the normal stack.
313            let ret_ptr: GuestPtr<u64> = GuestPtr::new(VirtAddr::from(ctxt.frame.rsp));
314            // SAFETY: `rsp` is a valid guest address filled by the CPU in the
315            // X86InterruptFrame
316            let ret = unsafe { ret_ptr.read() }.expect("Failed to read return address");
317
318            // Read the return address on the shadow stack.
319            let prev_rssp_ptr: GuestPtr<u64> = GuestPtr::new(VirtAddr::from(ctxt.ssp));
320            // SAFETY: `ssp` is a valid guest address filled by the CPU in the
321            // X86ExceptionContext
322            let prev_rssp = unsafe { prev_rssp_ptr.read() }
323                .expect("Failed to read address of previous shadow stack pointer");
324            // The offset to the return pointer is different for RET and IRET.
325            let offset = if code == NEAR_RET { 0 } else { 8 };
326            let ret_ptr: GuestPtr<u64> = GuestPtr::new(VirtAddr::from(prev_rssp + offset));
327            let ret_on_ssp =
328                // SAFETY: `ssp` is a valid guest address filled by the CPU in the
329                // X86ExceptionContext
330                unsafe { ret_ptr.read() }.expect("Failed to read return address on shadow stack");
331
332            panic!(
333                "thread at {rip:#018x} tried to return to {ret:#x}, but return address on shadow stack was {ret_on_ssp:#x}!"
334            );
335        }
336        RSTORSSP => {
337            panic!(
338                "rstorssp instruction encountered an unexpected shadow stack restore token at RIP {rip:#018x}"
339            );
340        }
341        SETSSBSY => {
342            panic!(
343                "setssbsy instruction encountered an unexpected supervisor shadow stack token at RIP {rip:#018x}"
344            );
345        }
346        code => unreachable!("unexpected code for #CP exception: {code}"),
347    }
348}

350// Virtualization Exception handler
351#[unsafe(no_mangle)]
352extern "C" fn ex_handler_ve(ctxt: &mut X86ExceptionContext) {
353    let rip = ctxt.frame.rip;
354    let code = ctxt.error_code;
355
356    if let Err(err) = handle_virtualization_exception(ctxt) {
357        log::error!("#VE handling error: {err:?}");
358        if user_mode(ctxt) {
359            log::error!(
360                "Failed to handle #VE from user-mode at RIP {rip:#018x} code: {code:#018x} - Terminating task"
361            );
362            terminate();
363        } else {
364            panic!(
365                "Failed to handle #VE from kernel-mode at RIP {:#018x} code: {:#018x}",
366                rip, code
367            );
368        }
369    }
370}

394// System Call SoftIRQ handler
395#[unsafe(no_mangle)]
396extern "C" fn ex_handler_system_call(
397    ctxt: &mut X86ExceptionContext,
398    vector: usize,
399    event_type: IdtEventType,
400) {
401    // Ensure that this vector was not invoked as a hardware interrupt vector.
402    if event_type.is_external_interrupt(vector) {
403        panic!("Syscall handler invoked as external interrupt!");
404    }
405
406    if !user_mode(ctxt) {
407        panic!("Syscall handler called from kernel mode!");
408    }
409
410    let Ok(input) = TryInto::<u64>::try_into(ctxt.regs.rax) else {
411        ctxt.regs.rax = !0;
412        return;
413    };
414
415    ctxt.regs.rax = match input {
416        // Class 0 SysCalls.
417        SYS_EXIT => sys_exit(ctxt.regs.rdi as u32),
418        SYS_EXEC => sys_exec(ctxt.regs.rdi, ctxt.regs.rsi, ctxt.regs.r8),
419        SYS_CLOSE => sys_close(ctxt.regs.rdi as u32),
420        // Class 1 SysCalls.
421        SYS_OPEN => sys_open(ctxt.regs.rdi, ctxt.regs.rsi, ctxt.regs.r8),
422        SYS_READ => sys_read(ctxt.regs.rdi as u32, ctxt.regs.rsi, ctxt.regs.r8),
423        SYS_WRITE => sys_write(ctxt.regs.rdi as u32, ctxt.regs.rsi, ctxt.regs.r8),
424        SYS_SEEK => sys_seek(ctxt.regs.rdi as u32, ctxt.regs.rsi, ctxt.regs.r8),
425        SYS_UNLINK => sys_unlink(ctxt.regs.rdi),
426        SYS_TRUNCATE => sys_truncate(ctxt.regs.rdi as u32, ctxt.regs.rsi),
427        SYS_OPENDIR => sys_opendir(ctxt.regs.rdi),
428        SYS_READDIR => sys_readdir(ctxt.regs.rdi as u32, ctxt.regs.rsi, ctxt.regs.r8),
429        SYS_MKDIR => sys_mkdir(ctxt.regs.rdi),
430        SYS_RMDIR => sys_rmdir(ctxt.regs.rdi),
431        // Class 3 SysCalls.
432        SYS_CAPABILITIES => sys_capabilities(ctxt.regs.rdi as u32),
433        _ => Err(SysCallError::EINVAL),
434    }
435    .map_or_else(|e| e as usize, |v| v as usize);
436}

450#[unsafe(no_mangle)]
451pub extern "C" fn common_isr_handler_entry(vector: usize) {
452    // Since interrupt handlers execute with interrupts disabled, it is
453    // necessary to increment the per-CPU interrupt disable nesting state
454    // while the handler is running in case common code attempts to disable
455    // interrupts temporarily.  The fact that this interrupt was received
456    // means that the previous state must have had interrupts enabled.
457    let cpu = this_cpu();
458    cpu.irqs_push_nesting(true);
459
460    common_isr_handler(vector);
461
462    // Decrement the interrupt disable nesting count, but do not permit
463    // interrupts to be reenabled.  They will be reenabled during the IRET
464    // flow.
465    cpu.irqs_pop_nesting();
466}
467
468pub fn common_isr_handler(vector: usize) {
469    // Set TPR based on the vector being handled and reenable interrupts to
470    // permit delivery of higher priority interrupts.  Because this routine
471    // dispatches interrupts which should only be observable if interrupts
472    // are enabled, the IRQ nesting count must be zero at this point.
473    let previous_tpr = raw_get_tpr();
474    raw_set_tpr(tpr_from_vector(vector));
475
476    let cpu = this_cpu();
477    cpu.irqs_enable();
478
479    // Process the requested interrupt vector.
480    match vector {
481        IPI_VECTOR => handle_ipi_interrupt(),
482        _ => {
483            // Ignore all unrecognized interrupt vectors and treat them as
484            // spurious interrupts.
485        }
486    }
487
488    // Disable interrupts before restoring TPR.
489    cpu.irqs_disable();
490    raw_set_tpr(previous_tpr);
491
492    // Perform the EOI cycle after the interrupt processing state has been
493    // restored so that recurrent interrupts will not introduce recursion at
494    // this point.
495    apic_eoi();
496}
