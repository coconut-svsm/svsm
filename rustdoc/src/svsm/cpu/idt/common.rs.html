73impl X86ExceptionContext {
74    /// # Safety
75    ///
76    /// The caller must ensure to update the rest of the execution state as
77    /// actual hardware would have done it (e.g. for MMIO emulation, CPUID,
78    /// MSR, etc.).
79    pub unsafe fn set_rip(&mut self, new_rip: usize) {
80        self.frame.rip = new_rip;
81
82        if is_cet_ss_enabled() {
83            let return_on_stack = (self.ssp + 8) as *const usize;
84            let return_on_stack_val = new_rip;
85            // SAFETY: Inline assembly to update the instruction pointer on
86            // the shadow stack. The safety of the RIP value is delegated to
87            // the caller of this function which is unsafe.
88            unsafe {
89                asm!(
90                    "wrssq [{}], {}",
91                    in(reg) return_on_stack,
92                    in(reg) return_on_stack_val
93                );
94            }
95        }
96    }
97}

218// The base addresses of the IDT should be aligned on an 8-byte boundary
219// to maximize performance of cache line fills.
220#[derive(Copy, Clone, Default, Debug)]
221#[repr(C, packed(8))]
222pub struct IdtEntry {
223    low: u64,
224    high: u64,
225}
226
227const IDT_TARGET_MASK_1: u64 = 0x0000_0000_0000_ffff;
228const IDT_TARGET_MASK_2: u64 = 0x0000_0000_ffff_0000;
229const IDT_TARGET_MASK_3: u64 = 0xffff_ffff_0000_0000;
230
231const IDT_TARGET_MASK_1_SHIFT: u64 = 0;
232const IDT_TARGET_MASK_2_SHIFT: u64 = 48 - 16;
233const IDT_TARGET_MASK_3_SHIFT: u64 = 32;
234
235const IDT_TYPE_MASK: u8 = 0x0f;
236const IDT_TYPE_SHIFT: u64 = 40;
237const IDT_TYPE_CALL: u8 = 0x0c;
238const IDT_TYPE_INT: u8 = 0x0e;
239const IDT_TYPE_TRAP: u8 = 0x0f;
240
241fn idt_type_mask(t: u8) -> u64 {
242    ((t & IDT_TYPE_MASK) as u64) << IDT_TYPE_SHIFT
243}
244
245const IDT_DPL_MASK: u8 = 0x03;
246const IDT_DPL_SHIFT: u64 = 45;
247
248fn idt_dpl_mask(dpl: u8) -> u64 {
249    ((dpl & IDT_DPL_MASK) as u64) << IDT_DPL_SHIFT
250}
251
252const IDT_PRESENT_MASK: u64 = 0x1u64 << 47;
253const IDT_CS_SHIFT: u64 = 16;
254
255const IDT_IST_MASK: u64 = 0x7;
256const IDT_IST_SHIFT: u64 = 32;
257
258impl IdtEntry {
259    fn create(target: VirtAddr, cs: u16, desc_type: u8, dpl: u8, ist: u8) -> Self {
260        let vaddr = target.bits() as u64;
261        let cs_mask = (cs as u64) << IDT_CS_SHIFT;
262        let ist_mask = ((ist as u64) & IDT_IST_MASK) << IDT_IST_SHIFT;
263        let low = (vaddr & IDT_TARGET_MASK_1) << IDT_TARGET_MASK_1_SHIFT
264            | (vaddr & IDT_TARGET_MASK_2) << IDT_TARGET_MASK_2_SHIFT
265            | idt_type_mask(desc_type)
266            | IDT_PRESENT_MASK
267            | idt_dpl_mask(dpl)
268            | cs_mask
269            | ist_mask;
270        let high = (vaddr & IDT_TARGET_MASK_3) >> IDT_TARGET_MASK_3_SHIFT;
271
272        IdtEntry { low, high }
273    }
274
275    pub fn raw_entry(target: VirtAddr) -> Self {
276        IdtEntry::create(target, SVSM_CS, IDT_TYPE_INT, 0, 0)
277    }
278
279    pub fn entry(handler: unsafe extern "C" fn()) -> Self {
280        let target = VirtAddr::from(handler as *const ());
281        IdtEntry::create(target, SVSM_CS, IDT_TYPE_INT, 0, 0)
282    }
283
284    pub fn user_entry(handler: unsafe extern "C" fn()) -> Self {
285        let target = VirtAddr::from(handler as *const ());
286        IdtEntry::create(target, SVSM_CS, IDT_TYPE_INT, 3, 0)
287    }
288
289    pub fn ist_entry(handler: unsafe extern "C" fn(), ist: u8) -> Self {
290        let target = VirtAddr::from(handler as *const ());
291        IdtEntry::create(target, SVSM_CS, IDT_TYPE_INT, 0, ist)
292    }
293
294    pub fn trap_entry(handler: unsafe extern "C" fn()) -> Self {
295        let target = VirtAddr::from(handler as *const ());
296        IdtEntry::create(target, SVSM_CS, IDT_TYPE_TRAP, 0, 0)
297    }
298
299    pub fn call_entry(handler: unsafe extern "C" fn()) -> Self {
300        let target = VirtAddr::from(handler as *const ());
301        IdtEntry::create(target, SVSM_CS, IDT_TYPE_CALL, 3, 0)
302    }
303
304    pub const fn no_handler() -> Self {
305        IdtEntry { low: 0, high: 0 }
306    }
307}

311#[repr(C, packed(2))]
312#[derive(Default, Clone, Copy, Debug)]
313struct IdtDesc {
314    limit: u16,
315    address: VirtAddr,
316}
317
318#[derive(Debug)]
319pub struct IDT<'a> {
320    entries: &'a mut [IdtEntry],
321}
322
323impl<'a> IDT<'a> {
324    pub const fn new(entries: &'a mut [IdtEntry]) -> Self {
325        Self { entries }
326    }
327
328    /// # Safety
329    /// The caller is required to supply the virtual address of a freshly
330    /// allocated page.
331    pub fn new_from_page(page: PageBox<[IdtEntry]>) -> Self {
332        Self {
333            entries: PageBox::leak(page),
334        }
335    }
336
337    pub fn init(&mut self, handler_array: *const u8, size: usize) -> &mut Self {
338        // Set IDT handlers
339        let handlers = VirtAddr::from(handler_array);
340
341        for idx in 0..size {
342            self.set_entry(idx, IdtEntry::raw_entry(handlers + (32 * idx)));
343        }
344
345        self
346    }
347
348    pub fn set_entry(&mut self, idx: usize, entry: IdtEntry) -> &mut Self {
349        self.entries[idx] = entry;
350
351        self
352    }
353
354    pub fn base_limit(&self) -> (u64, u16) {
355        let base: *const IdtEntry = self.entries.as_ptr();
356        let limit = mem::size_of_val(self.entries) - 1;
357        (base as u64, limit.try_into().unwrap())
358    }
359
360    /// Load an IDT.
361    /// # Safety
362    /// The caller must guarantee that the IDT lifetime must be static so that
363    /// its entries are always available to the CPU.
364    pub unsafe fn load(&self) {
365        let (base, limit) = self.base_limit();
366        let desc = IdtDesc {
367            limit,
368            address: VirtAddr::from(base),
369        };
370
371        // SAFETY: Inline assembly to load an IDT. `'static` lifetime ensures
372        // that address is always available for the CPU.
373        unsafe {
374            asm!("lidt (%rax)", in("rax") &raw const desc, options(att_syntax));
375        }
376    }
377
378    /// Make the IDT read-only.
379    /// This is supposed to be used for the global IDT only.
380    ///
381    /// # Safety
382    ///
383    /// The caller must ensure this method is called on the global IDT, and
384    /// after it is initialized.
385    pub unsafe fn make_ro(&self) -> Result<(), SvsmError> {
386        let idt_start = self.entries.as_ptr();
387
388        assert!(is_aligned(idt_start.addr(), PAGE_SIZE));
389
390        // Size of IDT is not supposed to be bigger than PAGE_SIZE.
391        let region = MemoryRegion::from_addresses(
392            VirtAddr::from(idt_start),
393            VirtAddr::from(idt_start) + PAGE_SIZE,
394        );
395
396        // SAFETY: 4k alignement on start address and size are verified.
397        // The check that this method is called on the initialized global IDT
398        // is delegated to the caller.
399        unsafe { make_ro(region) }
400    }
401}

403pub fn triple_fault() {
404    let desc: IdtDesc = IdtDesc {
405        limit: 0,
406        address: VirtAddr::from(0u64),
407    };
408
409    // SAFETY: This ends execution, this function will not return so memory
410    // safety is not an issue.
411    unsafe {
412        asm!("lidt (%rax)
413              int3", in("rax") &raw const desc, options(att_syntax));
414    }
415}

422pub fn is_exception_handler_return_site(rip: VirtAddr) -> bool {
423    let start = VirtAddr::from(&raw const entry_code_start);
424    let end = VirtAddr::from(&raw const entry_code_end);
425    (start..end).contains(&rip)
426}
