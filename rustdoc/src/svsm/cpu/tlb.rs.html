7use crate::address::{Address, VirtAddr};
8use crate::cpu::control_regs::{CR4Flags, read_cr3, read_cr4, write_cr3, write_cr4};
9use crate::cpu::ipi::{IpiMessage, IpiTarget, send_multicast_ipi};
10use crate::platform::SVSM_PLATFORM;
11
12use core::arch::asm;
13use core::sync::atomic::{AtomicBool, Ordering};
14
15static FLUSH_SMP: AtomicBool = AtomicBool::new(false);
16
17/// Defines the scope of a TLB flush.
18#[derive(Copy, Clone, Debug)]
19pub enum TlbFlushScope {
20    /// Indicates that all addresses must be flushed on all processors,
21    /// including global addresses.
22    AllGlobal,
23
24    /// Indicates that all addresses must be flushed on all processors,
25    /// excluding global addresses.
26    AllNonGlobal,
27}
28
29impl TlbFlushScope {
30    pub fn flush_percpu(&self) {
31        match self {
32            Self::AllGlobal => flush_tlb_global_percpu(),
33            Self::AllNonGlobal => flush_tlb_percpu(),
34        }
35    }
36
37    pub fn flush_all(&self) {
38        // If SMP has not yet been started, then perform all flushes as local only.
39        // Prior to SMP startup, there is no need to reach into other processors,
40        // and the SVSM platform object may not even exist when flushes are
41        // attempted prior to SMP startup.
42        if FLUSH_SMP.load(Ordering::Relaxed) {
43            SVSM_PLATFORM.flush_tlb(self);
44        } else {
45            self.flush_percpu();
46        }
47    }
48}
49
50// SAFETY: The TlbFlushScope structure contains no references and can safely
51// rely on the default implementation of the IPI message copy routines.
52unsafe impl IpiMessage for TlbFlushScope {
53    fn invoke(&self) {
54        self.flush_percpu();
55    }
56}
57
58pub fn flush_tlb(flush_scope: &TlbFlushScope) {
59    send_multicast_ipi(IpiTarget::All, flush_scope);
60}
61
62pub fn set_tlb_flush_smp() {
63    FLUSH_SMP.store(true, Ordering::Relaxed);
64}
65
66pub fn flush_tlb_global_sync() {
67    let flush_scope = TlbFlushScope::AllGlobal;
68    flush_scope.flush_all();
69}
70
71pub fn flush_tlb_global_percpu() {
72    let cr4 = read_cr4();
73
74    // SAFETY: we are not changing any execution-state relevant flags
75    unsafe {
76        write_cr4(cr4 ^ CR4Flags::PGE);
77        write_cr4(cr4);
78    }
79}
80
81pub fn flush_tlb_percpu() {
82    // SAFETY: reloading CR3 with its current value is always safe.
83    unsafe {
84        write_cr3(read_cr3());
85    }
86}
87
88pub fn flush_address_percpu(va: VirtAddr) {
89    let va: u64 = va.page_align().bits() as u64;
90    // SAFETY: Inline assembly to invalidate TLB Entries, which does not change
91    // any state related to memory safety.
92    unsafe {
93        asm!("invlpg (%rax)",
94             in("rax") va,
95             options(att_syntax));
96    }
97}
