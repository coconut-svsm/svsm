70pub static PERCPU_AREAS: PerCpuAreas = PerCpuAreas::new();

72// We use an UnsafeCell to allow for a static with interior mutability.
73// Normally, we would need to guarantee synchronization on the backing
74// datatype, but this is not needed because writes to the structure only occur
75// at initialization, from CPU 0, and reads of the Vec should only occur after
76// all writes are done.
77#[derive(Debug)]
78pub struct PerCpuAreas {
79    areas: UnsafeCell<Vec<&'static PerCpuShared>>,
80}
81
82// SAFETY: Any operation that can affect synchronization safety is declared as
83// unsafe, and therefore callers guarantee that synchronization is always
84// maintained. Also see comment above struct declaration.
85unsafe impl Sync for PerCpuAreas {}
86
87impl PerCpuAreas {
88    const fn new() -> Self {
89        Self {
90            areas: UnsafeCell::new(Vec::new()),
91        }
92    }
93
94    /// # Safety
95    /// The areas vector obtained here is not multi-thread safe, so the caller
96    /// must guarantee that is not used in a context that expects multi-thread
97    /// safety.
98    unsafe fn get_areas(&self) -> &Vec<&'static PerCpuShared> {
99        // SAFETY: the caller guarantees that accessing the unsafe cell is
100        // appropriate.
101        unsafe { &*self.areas.get() }
102    }
103
104    /// # Safety
105    /// This function can only be invoked before any other processors have
106    /// started.  Otherwise, accesses to the areas vector will not be safe.
107    pub unsafe fn create_new(&self, apic_id: u32) -> &'static PerCpuShared {
108        // SAFETY: the caller guarantees that it is safe to obtain a mutable
109        // reference to the areas vector.
110        let areas = unsafe { &mut *self.areas.get() };
111
112        // Allocate a new shared per-CPU area to describe the APIC ID being
113        // created.  The CPU index will be the next in sequence ased on the
114        // size of the vector.  The new shared per-CPU area is allocated on
115        // the heap so it is globally visible.
116        let cpu_index = areas.len();
117        let percpu_shared = Box::leak(Box::new(PerCpuShared::new(apic_id, cpu_index)));
118
119        // Leak the box so the allocation persists with a static lifetime,
120        // and install the allocated per-CPU area into the areas vector.
121        areas.push(percpu_shared);
122
123        percpu_shared
124    }
125
126    pub fn len(&self) -> usize {
127        // SAFETY: it is safe to obtain a shared reference to the areas
128        // vector because callers attempting to mutate the vector guarantee
129        // that they cannot race with iteration.
130        let areas = unsafe { self.get_areas() };
131        areas.len()
132    }
133
134    pub fn is_empty(&self) -> bool {
135        // SAFETY: it is safe to obtain a shared reference to the areas
136        // vector because callers attempting to mutate the vector guarantee
137        // that they cannot race with iteration.
138        let areas = unsafe { self.get_areas() };
139        areas.is_empty()
140    }
141
142    pub fn iter(&self) -> Iter<'_, &'static PerCpuShared> {
143        // SAFETY: it is safe to obtain a shared reference to the areas
144        // vector because callers attempting to mutate the vector guarantee
145        // that they cannot race with iteration.
146        let areas = unsafe { self.get_areas() };
147        areas.iter()
148    }
149
150    // Fails if no such area exists or its address is NULL
151    pub fn get_by_apic_id(&self, apic_id: u32) -> Option<&'static PerCpuShared> {
152        // For this to not produce UB the only invariant we must
153        // uphold is that there are no mutations or mutable aliases
154        // going on when casting via as_ref(). This only happens via
155        // Self::push(), which is intentionally unsafe and private.
156        self.iter()
157            .find(|shared| shared.apic_id() == apic_id)
158            .copied()
159    }
160
161    /// Callers are expected to specify a valid CPU index.
162    pub fn get_by_cpu_index(&self, index: usize) -> &'static PerCpuShared {
163        // SAFETY: it is safe to obtain a shared reference to the areas
164        // vector because callers attempting to mutate the vector guarantee
165        // that they cannot race with iteration.
166        let areas = unsafe { self.get_areas() };
167        areas[index]
168    }
169}

219        self.vmsa = vmsa;
220        self.caa = caa;
221        self.generation += 1;
222    }
223
224    pub fn set_updated(&mut self) {
225        self.gen_in_use = self.generation;
226    }
227
228    pub fn vmsa_phys(&self) -> Option<PhysAddr> {
229        self.vmsa
230    }
231
232    pub fn caa_phys(&self) -> Option<PhysAddr> {
233        self.caa
234    }
235
236    pub fn vmsa(&mut self) -> &mut VMSA {
237        assert!(self.vmsa.is_some());
238        // SAFETY: this function takes &mut self, so only one mutable
239        // reference to the underlying VMSA can exist.
240        unsafe { SVSM_PERCPU_VMSA_BASE.as_mut_ptr::<VMSA>().as_mut().unwrap() }
241    }
242
243    pub fn caa(&self) -> Option<NonNull<SvsmCaa>> {
244        let caa_phys = self.caa_phys()?;
245        let offset = caa_phys.page_offset();
246        let ptr = (SVSM_PERCPU_CAA_BASE + offset).as_mut_ptr();
247        // SAFETY: `SVSM_PERCPU_CAA_BASE` is defined at compile time to
248        // page-aligned and non-zero. Adding a page offset to a page-aligned
249        // address can never overflow.
250        unsafe { Some(NonNull::new_unchecked(ptr)) }
251    }

304    pub fn clear_guest_vmsa_if_match(&self, paddr: PhysAddr) {
305        let mut locked = self.guest_vmsa.lock();
306        if locked.vmsa.is_none() {
307            return;
308        }

310        let vmsa_phys = locked.vmsa_phys();

324    pub fn request_ipi(&self, vector: u8) {
325        let index = vector >> 5;
326        let bit = 1u32 << (vector & 31);
327        // Request the IPI via the IRR vector before signaling that an IPI has
328        // been requested.
329        self.ipi_irr[index as usize].fetch_or(bit, Ordering::Relaxed);
330        self.ipi_pending.store(true, Ordering::Release);
331    }
332
333    pub fn request_nmi(&self) {
334        self.nmi_pending.store(true, Ordering::Relaxed);
335        self.ipi_pending.store(true, Ordering::Release);
336    }
337
338    pub fn ipi_pending(&self) -> bool {
339        self.ipi_pending.swap(false, Ordering::Acquire)
340    }
341
342    pub fn ipi_irr_vector(&self, index: usize) -> u32 {
343        self.ipi_irr[index].swap(0, Ordering::Relaxed)
344    }
345
346    pub fn nmi_pending(&self) -> bool {
347        self.nmi_pending.swap(false, Ordering::Relaxed)
348    }

360/// CPU-local data.
361///
362/// While the contents of this struct are never accessed outside the local CPU
363/// (except for the [`PerCpuShared`] portion), reentrant access to the structure
364/// from preempted context essentially behaves as concurrent access, since
365/// the preemption may happen at any instruction boundary. If we use interior
366/// mutability types, the reentrant access may lead to undefined behavior.
367/// Thus, we require that the type is [`Sync`]. This is currently enforced by
368/// protecting mutable fields with a [`RWLock`]. To avoid deadlocks, only the
369/// non-blocking methods in the lock are used, causing an immediate panic when a
370/// reentrant access is attempted.
371///
372/// If the [`Sync`] requirement below breaks your build, it means you introduced
373/// interior mutability, which is not safe.
374#[derive(Debug)]
375pub struct PerCpu
376where
377    Self: Sync,
378{
379    /// Reference to the `PerCpuShared` that is valid in the global, shared
380    /// address space.
381    shared: &'static PerCpuShared,
382
383    /// APIC access object
384    apic: X86Apic,
385
386    /// PerCpu IRQ state tracking
387    irq_state: IrqState,
388
389    pgtbl: RWLock<Option<&'static mut PageTable>>,
390    tss: X86Tss,
391    isst: RWLock<Isst>,
392    svsm_vmsa: ImmutAfterInitCell<VmsaPage>,
393    reset_ip: AtomicU64,
394    /// PerCpu Virtual Memory Range
395    vm_range: VMR,
396    /// Address allocator for per-cpu 4k temporary mappings
397    vrange_4k: RWLock<VirtualRange>,
398    /// Address allocator for per-cpu 2m temporary mappings
399    vrange_2m: RWLock<VirtualRange>,
400    /// Task list that has been assigned for scheduling on this CPU
401    runqueue: RWLockIrqSafe<RunQueue>,
402    /// Local APIC state for APIC emulation if enabled
403    guest_apic: RWLock<Option<LocalApic>>,
404
405    /// GHCB page for this CPU.
406    ghcb: ImmutAfterInitCell<GhcbPage>,
407
408    /// Hypercall input/output pages for this CPU if running under Hyper-V.
409    hypercall_pages: RWLock<Option<(HypercallPage, HypercallPage)>>,
410
411    /// `#HV` doorbell page for this CPU.
412    hv_doorbell: ImmutAfterInitCell<SharedBox<HVDoorbell>>,
413
414    init_shadow_stack: ImmutAfterInitCell<VirtAddr>,
415    context_switch_stack: ImmutAfterInitCell<VirtAddr>,
416    ist: IstStacks,
417
418    /// Stack boundaries of the currently running task.
419    current_stack: RWLock<MemoryRegion<VirtAddr>>,
420}

650    fn allocate_stack(&self, base: VirtAddr) -> Result<VirtAddr, SvsmError> {
651        let stack = VMKernelStack::new()?;

660    fn allocate_shadow_stack(
661        &self,
662        base: VirtAddr,
663        init: ShadowStackInit,
664    ) -> Result<VirtAddr, SvsmError> {
665        let shadow_stack = VMKernelStack::new_shadow()?;
666        let offset = shadow_stack.top_of_stack();
667        let shadow_stack_page = shadow_stack.shadow_page();

672        Ok(ssp)
673    }

786    pub fn setup(
787        &self,
788        platform: &dyn SvsmPlatform,
789        pgtable: PageBox<PageTable>,
790    ) -> Result<(), SvsmError> {
791        self.init_page_table(pgtable)?;
792
793        // Map PerCpu data in own page-table
794        self.map_self()?;
795
796        // Reserve ranges for temporary mappings
797        self.initialize_vm_ranges()?;
798
799        if is_cet_ss_supported() {
800            self.allocate_init_shadow_stack()?;
801        }
802
803        // Allocate per-cpu context switch stack
804        self.allocate_context_switch_stack()?;
805
806        if is_cet_ss_supported() {
807            self.allocate_context_switch_shadow_stack()?;
808        }
809
810        // Allocate IST stacks
811        self.allocate_ist_stacks()?;
812
813        // Setup TSS
814        self.setup_tss();
815
816        if is_cet_ss_supported() {
817            // Allocate ISST shadow stacks
818            self.allocate_isst_shadow_stacks()?;
819
820            // Setup ISST
821            self.setup_isst();
822        }
823
824        // Initialize allocator for temporary mappings
825        self.virt_range_init();
826
827        self.finish_page_table();
828
829        // Complete platform-specific initialization.
830        platform.setup_percpu(self)?;
831
832        // Allocate hypercall pages if running on Hyper-V, unless this is the
833        // BSP (where they will be allocated later).
834        if self.shared.cpu_index() != 0 && *IS_HYPERV {
835            self.allocate_hypercall_pages()?;
836        }
837
838        Ok(())
839    }

898        let data_segment = svsm_data_segment();

968    pub fn alloc_guest_vmsa(&self) -> Result<(), SvsmError> {
969        // Enable alternate injection if the hypervisor supports it.
970        let use_alternate_injection = SVSM_PLATFORM.query_apic_registration_state();
971        if use_alternate_injection {
972            *self.guest_apic.write_noblock() = Some(LocalApic::new());
973
974            // Configure the interrupt injection vector.
975            let ghcb = self.ghcb().unwrap();
976            ghcb.configure_interrupt_injection(INT_INJ_VECTOR)?;
977        }
978
979        let mut vmsa = VmsaPage::new(RMPFlags::GUEST_VMPL)?;
980        let paddr = vmsa.paddr();
981
982        init_guest_vmsa(
983            &mut vmsa,
984            self.reset_ip.load(Ordering::Relaxed),
985            use_alternate_injection,
986        );
987
988        self.shared().update_guest_vmsa(paddr);
989        let _ = VmsaPage::leak(vmsa);
990
991        Ok(())
992    }

1022    pub fn update_guest_mappings(&self) -> Result<(), SvsmError> {
1023        let mut locked = self.guest_vmsa_ref();
1024        let mut ret = Ok(());
1025
1026        if !locked.needs_update() {
1027            // If there is no VMSA, then the update request must be considered a
1028            // failure even though no work was required.
1029            return match locked.vmsa_phys() {
1030                Some(_) => Ok(()),
1031                None => Err(SvsmError::MissingVMSA),
1032            };
1033        }
1034
1035        self.unmap_guest_vmsa();
1036        self.unmap_caa();
1037
1038        match locked.vmsa_phys() {
1039            Some(paddr) => self.map_guest_vmsa(paddr)?,
1040            None => ret = Err(SvsmError::MissingVMSA),
1041        }
1042
1043        if let Some(paddr) = locked.caa_phys() {
1044            self.map_guest_caa(paddr)?
1045        }
1046
1047        locked.set_updated();
1048
1049        ret
1050    }

1076    pub fn ai_handle_intercepts(&self, vmsa: &mut VMSA) {
1077        let use_alternate_injection = SVSM_PLATFORM.query_apic_registration_state();
1078        let g_eii = vmsa.guest_exitintinfo;
1079
1080        if !use_alternate_injection {
1081            return;
1082        }
1083
1084        // Re-inject events when intercept happens
1085        if g_eii.valid() {
1086            vmsa.event_inj = g_eii;
1087        }
1088
1089        // Clear busy bit
1090        let mut vintr_ctrl = vmsa.vintr_ctrl;
1091        if vintr_ctrl.busy() {
1092            vintr_ctrl.set_busy(false);
1093            vmsa.vintr_ctrl = vintr_ctrl;
1094        }
1095    }

1239pub fn this_cpu() -> &'static PerCpu {
1240    // SAFETY: The PerCPU area is always mapped at the same virtual address, so
1241    // dereferencing a pointer to that address is safe. The PerCPU area is also
1242    // never freed, so using a static lifetime is safe as well.
1243    unsafe { &*SVSM_PERCPU_BASE.as_ptr::<PerCpu>() }
1244}
1245
1246pub fn try_this_cpu() -> Option<&'static PerCpu> {
1247    let mut rcx: u64;
1248
1249    // SAFETY: Inline assembly to test the validity of the PerCpu page.
1250    // Any #PF generated here is handled by the #PF handler and it will
1251    // either return an error in %rcx or crash the system. This however
1252    // does not impact any state related to memory safety.
1253    unsafe {
1254        asm!("1: movq ({0}), {1}",
1255             "   xorq %rcx, %rcx",
1256             "2:",
1257             ".pushsection \"__early_exception_table\",\"a\"",
1258             ".balign 16",
1259             ".quad (1b)",
1260             ".quad (2b)",
1261             ".popsection",
1262                in(reg) SVSM_PERCPU_BASE.bits(),
1263                out(reg) _,
1264                out("rcx") rcx,
1265                options(att_syntax, nostack));
1266    }
1267    if rcx == 0 { Some(this_cpu()) } else { None }
1268}

1355pub static PERCPU_VMSAS: PerCpuVmsas = PerCpuVmsas::new();

1357#[derive(Debug)]
1358pub struct PerCpuVmsas {
1359    vmsas: RWLock<Vec<VmsaRegistryEntry>>,
1360}
1361
1362impl PerCpuVmsas {
1363    const fn new() -> Self {
1364        Self {
1365            vmsas: RWLock::new(Vec::new()),
1366        }
1367    }
1368
1369    pub fn exists(&self, paddr: PhysAddr) -> bool {
1370        self.vmsas
1371            .lock_read()
1372            .iter()
1373            .any(|vmsa| vmsa.paddr == paddr)
1374    }
1375
1376    pub fn overlaps(&self, region: &MemoryRegion<PhysAddr>) -> bool {
1377        self.vmsas
1378            .lock_read()
1379            .iter()
1380            .any(|vmsa| region.overlap(&MemoryRegion::new(vmsa.paddr, PAGE_SIZE)))
1381    }
1382
1383    pub fn register(
1384        &self,
1385        paddr: PhysAddr,
1386        cpu_index: usize,
1387        guest_owned: bool,
1388    ) -> Result<(), SvsmError> {
1389        let mut guard = self.vmsas.lock_write();
1390        if guard.iter().any(|vmsa| vmsa.paddr == paddr) {
1391            return Err(SvsmError::InvalidAddress);
1392        }
1393
1394        guard.push(VmsaRegistryEntry::new(paddr, cpu_index, guest_owned));
1395        Ok(())
1396    }
1397
1398    pub fn set_used(&self, paddr: PhysAddr) -> Option<usize> {
1399        self.vmsas
1400            .lock_write()
1401            .iter_mut()
1402            .find(|vmsa| vmsa.paddr == paddr && !vmsa.in_use)
1403            .map(|vmsa| {
1404                vmsa.in_use = true;
1405                vmsa.cpu_index
1406            })
1407    }
1408
1409    pub fn unregister(&self, paddr: PhysAddr, in_use: bool) -> Result<VmsaRegistryEntry, u64> {
1410        let mut guard = self.vmsas.lock_write();
1411        let index = guard
1412            .iter()
1413            .position(|vmsa| vmsa.paddr == paddr && vmsa.in_use == in_use)
1414            .ok_or(0u64)?;
1415
1416        if in_use {
1417            let vmsa = &guard[index];
1418
1419            let target_cpu = PERCPU_AREAS.get_by_cpu_index(vmsa.cpu_index);
1420            target_cpu.clear_guest_vmsa_if_match(paddr);
1421        }
1422
1423        Ok(guard.swap_remove(index))
1424    }
1425}

1431pub fn cpu_idle_loop(cpu_index: usize) {
1432    debug_assert_eq!(cpu_index, this_cpu().get_cpu_index());
1433
1434    loop {
1435        // Go idle
1436        scheduler_idle();
1437
1438        // If idle was explicitly requested by another task, then schedule that
1439        // task to execute again in case it wants to perform processing as a
1440        // result of the wake from idle.
1441        let maybe_task = this_cpu().runqueue_mut().wake_from_idle();
1442        if let Some(task) = maybe_task {
1443            schedule_task(task);
1444        }
1445
1446        // Execute any tasks that are currently runnable.
1447        schedule();
1448    }
