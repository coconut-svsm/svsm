63// PERCPU areas virtual addresses into shared memory
64pub static PERCPU_AREAS: PerCpuAreas = PerCpuAreas::new();
65
66// We use an UnsafeCell to allow for a static with interior mutability.
67// Normally, we would need to guarantee synchronization on the backing
68// datatype, but this is not needed because writes to the structure only occur
69// at initialization, from CPU 0, and reads of the Vec should only occur after
70// all writes are done.
71#[derive(Debug)]
72pub struct PerCpuAreas {
73    areas: UnsafeCell<Vec<&'static PerCpuShared>>,
74}
75
76// SAFETY: Any operation that can affect synchronization safety is declared as
77// unsafe, and therefore callers guarantee that synchronization is always
78// maintained. Also see comment above struct declaration.
79unsafe impl Sync for PerCpuAreas {}
80
81impl PerCpuAreas {
82    const fn new() -> Self {
83        Self {
84            areas: UnsafeCell::new(Vec::new()),
85        }
86    }
87
88    /// # Safety
89    /// The areas vector obtained here is not multi-thread safe, so the caller
90    /// must guarantee that is not used in a context that expects multi-thread
91    /// safety.
92    unsafe fn get_areas(&self) -> &Vec<&'static PerCpuShared> {
93        // SAFETY: the caller guarantees that accessing the unsafe cell is
94        // appropriate.
95        unsafe { &*self.areas.get() }
96    }
97
98    /// # Safety
99    /// This function can only be invoked before any other processors have
100    /// started.  Otherwise, accesses to the areas vector will not be safe.
101    pub unsafe fn create_new(&self, apic_id: u32) -> &'static PerCpuShared {
102        // SAFETY: the caller guarantees that it is safe to obtain a mutable
103        // reference to the areas vector.
104        let areas = unsafe { &mut *self.areas.get() };
105
106        // Allocate a new shared per-CPU area to describe the APIC ID being
107        // created.  The CPU index will be the next in sequence ased on the
108        // size of the vector.  The new shared per-CPU area is allocated on
109        // the heap so it is globally visible.
110        let cpu_index = areas.len();
111        let percpu_shared = Box::leak(Box::new(PerCpuShared::new(apic_id, cpu_index)));
112
113        // Leak the box so the allocation persists with a static lifetime,
114        // and install the allocated per-CPU area into the areas vector.
115        areas.push(percpu_shared);
116
117        percpu_shared
118    }
119
120    pub fn len(&self) -> usize {
121        // SAFETY: it is safe to obtain a shared reference to the areas
122        // vector because callers attempting to mutate the vector guarantee
123        // that they cannot race with iteration.
124        let areas = unsafe { self.get_areas() };
125        areas.len()
126    }
127
128    pub fn is_empty(&self) -> bool {
129        // SAFETY: it is safe to obtain a shared reference to the areas
130        // vector because callers attempting to mutate the vector guarantee
131        // that they cannot race with iteration.
132        let areas = unsafe { self.get_areas() };
133        areas.is_empty()
134    }
135
136    pub fn iter(&self) -> Iter<'_, &'static PerCpuShared> {
137        // SAFETY: it is safe to obtain a shared reference to the areas
138        // vector because callers attempting to mutate the vector guarantee
139        // that they cannot race with iteration.
140        let areas = unsafe { self.get_areas() };
141        areas.iter()
142    }
143
144    // Fails if no such area exists or its address is NULL
145    pub fn get_by_apic_id(&self, apic_id: u32) -> Option<&'static PerCpuShared> {
146        // For this to not produce UB the only invariant we must
147        // uphold is that there are no mutations or mutable aliases
148        // going on when casting via as_ref(). This only happens via
149        // Self::push(), which is intentionally unsafe and private.
150        self.iter()
151            .find(|shared| shared.apic_id() == apic_id)
152            .copied()
153    }
154
155    /// Callers are expected to specify a valid CPU index.
156    pub fn get_by_cpu_index(&self, index: usize) -> &'static PerCpuShared {
157        // SAFETY: it is safe to obtain a shared reference to the areas
158        // vector because callers attempting to mutate the vector guarantee
159        // that they cannot race with iteration.
160        let areas = unsafe { self.get_areas() };
161        areas[index]
162    }
163}

214        self.caa = caa;
215        self.generation += 1;
216    }

218    pub fn set_updated(&mut self) {
219        self.gen_in_use = self.generation;
220    }

222    pub fn vmsa_phys(&self) -> Option<PhysAddr> {
223        self.vmsa
224    }
225
226    pub fn caa_phys(&self) -> Option<PhysAddr> {
227        self.caa
228    }
229
230    pub fn vmsa(&mut self) -> &mut VMSA {
231        assert!(self.vmsa.is_some());
232        // SAFETY: this function takes &mut self, so only one mutable
233        // reference to the underlying VMSA can exist.
234        unsafe { SVSM_PERCPU_VMSA_BASE.as_mut_ptr::<VMSA>().as_mut().unwrap() }
235    }
236
237    pub fn caa_addr(&self) -> Option<VirtAddr> {
238        let caa_phys = self.caa_phys()?;
239        let offset = caa_phys.page_offset();
240
241        Some(SVSM_PERCPU_CAA_BASE + offset)
242    }

295    pub fn clear_guest_vmsa_if_match(&self, paddr: PhysAddr) {
296        let mut locked = self.guest_vmsa.lock();

315    pub fn request_ipi(&self, vector: u8) {
316        let index = vector >> 5;
317        let bit = 1u32 << (vector & 31);
318        // Request the IPI via the IRR vector before signaling that an IPI has
319        // been requested.
320        self.ipi_irr[index as usize].fetch_or(bit, Ordering::Relaxed);
321        self.ipi_pending.store(true, Ordering::Release);
322    }
323
324    pub fn request_nmi(&self) {
325        self.nmi_pending.store(true, Ordering::Relaxed);
326        self.ipi_pending.store(true, Ordering::Release);
327    }
328
329    pub fn ipi_pending(&self) -> bool {
330        self.ipi_pending.swap(false, Ordering::Acquire)
331    }
332
333    pub fn ipi_irr_vector(&self, index: usize) -> u32 {
334        self.ipi_irr[index].swap(0, Ordering::Relaxed)
335    }
336
337    pub fn nmi_pending(&self) -> bool {
338        self.nmi_pending.swap(false, Ordering::Relaxed)
339    }

351/// CPU-local data.
352///
353/// This type is not [`Sync`], as its contents will only be accessed from the
354/// local CPU, much like thread-local data in an std environment. The only
355/// part of the struct that may be accessed from a different CPU is the
356/// `shared` field, a reference to which will be stored in [`PERCPU_AREAS`].
357#[derive(Debug)]
358pub struct PerCpu {
359    /// Reference to the `PerCpuShared` that is valid in the global, shared
360    /// address space.
361    shared: &'static PerCpuShared,
362
363    /// APIC access object
364    apic: X86Apic,
365
366    /// PerCpu IRQ state tracking
367    irq_state: IrqState,
368
369    pgtbl: RefCell<Option<&'static mut PageTable>>,
370    tss: X86Tss,
371    isst: Cell<Isst>,
372    svsm_vmsa: OnceCell<VmsaPage>,
373    reset_ip: Cell<u64>,
374    /// PerCpu Virtual Memory Range
375    vm_range: VMR,
376    /// Address allocator for per-cpu 4k temporary mappings
377    pub vrange_4k: RefCell<VirtualRange>,
378    /// Address allocator for per-cpu 2m temporary mappings
379    pub vrange_2m: RefCell<VirtualRange>,
380    /// Task list that has been assigned for scheduling on this CPU
381    runqueue: RWLockIrqSafe<RunQueue>,
382    /// Local APIC state for APIC emulation if enabled
383    guest_apic: RefCell<Option<LocalApic>>,
384
385    /// GHCB page for this CPU.
386    ghcb: OnceCell<GhcbPage>,
387
388    /// Hypercall input/output pages for this CPU if running under Hyper-V.
389    hypercall_pages: RefCell<Option<(HypercallPage, HypercallPage)>>,
390
391    /// `#HV` doorbell page for this CPU.
392    hv_doorbell: OnceCell<SharedBox<HVDoorbell>>,
393
394    init_shadow_stack: Cell<Option<VirtAddr>>,
395    context_switch_stack: Cell<Option<VirtAddr>>,
396    ist: IstStacks,
397
398    /// Stack boundaries of the currently running task.
399    current_stack: Cell<MemoryRegion<VirtAddr>>,
400}

630        let stack = VMKernelStack::new()?;

640        &self,
641        base: VirtAddr,
642        init: ShadowStackInit,
643    ) -> Result<VirtAddr, SvsmError> {
644        let shadow_stack = VMKernelStack::new_shadow()?;
645        let offset = shadow_stack.top_of_stack();
646        let shadow_stack_page = shadow_stack.shadow_page();

651        Ok(ssp)
652    }

769    pub fn setup(
770        &self,
771        platform: &dyn SvsmPlatform,
772        pgtable: PageBox<PageTable>,
773    ) -> Result<(), SvsmError> {
774        self.init_page_table(pgtable)?;
775
776        // Map PerCpu data in own page-table
777        self.map_self()?;
778
779        // Reserve ranges for temporary mappings
780        self.initialize_vm_ranges()?;
781
782        if is_cet_ss_supported() {
783            self.allocate_init_shadow_stack()?;
784        }
785
786        // Allocate per-cpu context switch stack
787        self.allocate_context_switch_stack()?;
788
789        if is_cet_ss_supported() {
790            self.allocate_context_switch_shadow_stack()?;
791        }
792
793        // Allocate IST stacks
794        self.allocate_ist_stacks()?;
795
796        // Setup TSS
797        self.setup_tss();
798
799        if is_cet_ss_supported() {
800            // Allocate ISST shadow stacks
801            self.allocate_isst_shadow_stacks()?;
802
803            // Setup ISST
804            self.setup_isst();
805        }
806
807        // Initialize allocator for temporary mappings
808        self.virt_range_init();
809
810        self.finish_page_table();
811
812        // Complete platform-specific initialization.
813        platform.setup_percpu(self)?;
814
815        // Allocate hypercall pages if running on Hyper-V, unless this is the
816        // BSP (where they will be allocated later).
817        if self.shared.cpu_index() != 0 && *IS_HYPERV {
818            self.allocate_hypercall_pages()?;
819        }
820
821        Ok(())
822    }

908    /// Allocates and initializes a new VMSA for this CPU. Returns its
909    /// physical address and SEV features. Returns an error if allocation
910    /// fails of this CPU's VMSA was already initialized.
911    pub fn alloc_svsm_vmsa(&self, vtom: u64, start_rip: u64) -> Result<(PhysAddr, u64), SvsmError> {
912        if self.svsm_vmsa.get().is_some() {
913            // FIXME: add a more explicit error variant for this condition
914            return Err(SvsmError::Mem);
915        }
916
917        let mut vmsa = VmsaPage::new(RMPFlags::VMPL1)?;
918        let paddr = vmsa.paddr();
919
920        // Initialize VMSA
921        let context = self.get_initial_context(start_rip);
922        init_svsm_vmsa(&mut vmsa, vtom, &context);
923        vmsa.enable();
924
925        let sev_features = vmsa.sev_features;
926
927        // We already checked that the VMSA is unset
928        self.svsm_vmsa.set(vmsa).unwrap();
929
930        Ok((paddr, sev_features))
931    }

952    pub fn alloc_guest_vmsa(&self) -> Result<(), SvsmError> {
953        // Enable alternate injection if the hypervisor supports it.
954        let use_alternate_injection = SVSM_PLATFORM.query_apic_registration_state();
955        if use_alternate_injection {
956            self.guest_apic.replace(Some(LocalApic::new()));
957
958            // Configure the interrupt injection vector.
959            let ghcb = self.ghcb().unwrap();
960            ghcb.configure_interrupt_injection(INT_INJ_VECTOR)?;
961        }
962
963        let mut vmsa = VmsaPage::new(RMPFlags::GUEST_VMPL)?;
964        let paddr = vmsa.paddr();
965
966        init_guest_vmsa(&mut vmsa, self.reset_ip.get(), use_alternate_injection);
967
968        self.shared().update_guest_vmsa(paddr);
969        let _ = VmsaPage::leak(vmsa);
970
971        Ok(())
972    }

1002    pub fn update_guest_mappings(&self) -> Result<(), SvsmError> {
1003        let mut locked = self.guest_vmsa_ref();
1004        let mut ret = Ok(());
1005
1006        if !locked.needs_update() {
1007            // If there is no VMSA, then the update request must be considered a
1008            // failure even though no work was required.
1009            return match locked.vmsa_phys() {
1010                Some(_) => Ok(()),
1011                None => Err(SvsmError::MissingVMSA),
1012            };
1013        }
1014
1015        self.unmap_guest_vmsa();
1016        self.unmap_caa();
1017
1018        match locked.vmsa_phys() {
1019            Some(paddr) => self.map_guest_vmsa(paddr)?,
1020            None => ret = Err(SvsmError::MissingVMSA),
1021        }
1022
1023        if let Some(paddr) = locked.caa_phys() {
1024            self.map_guest_caa(paddr)?
1025        }
1026
1027        locked.set_updated();
1028
1029        ret
1030    }

1180pub fn this_cpu() -> &'static PerCpu {
1181    // SAFETY: The PerCPU area is always mapped at the same virtual address, so
1182    // dereferencing a pointer to that address is safe. The PerCPU area is also
1183    // never freed, so using a static lifetime is safe as well.
1184    unsafe { &*SVSM_PERCPU_BASE.as_ptr::<PerCpu>() }
1185}

1187pub fn try_this_cpu() -> Option<&'static PerCpu> {
1188    let mut rcx: u64;
1189
1190    // SAFETY: Inline assembly to test the validity of the PerCpu page.
1191    // Any #PF generated here is handled by the #PF handler and it will
1192    // either return an error in %rcx or crash the system. This however
1193    // does not impact any state related to memory safety.
1194    unsafe {
1195        asm!("1: movq ({0}), {1}",
1196             "   xorq %rcx, %rcx",
1197             "2:",
1198             ".pushsection \"__early_exception_table\",\"a\"",
1199             ".balign 16",
1200             ".quad (1b)",
1201             ".quad (2b)",
1202             ".popsection",
1203                in(reg) SVSM_PERCPU_BASE.bits(),
1204                out(reg) _,
1205                out("rcx") rcx,
1206                options(att_syntax, nostack));
1207    }
1208    if rcx == 0 {
1209        Some(this_cpu())
1210    } else {
1211        None
1212    }
1213}

1302#[derive(Debug)]
1303pub struct PerCpuVmsas {
1304    vmsas: RWLock<Vec<VmsaRegistryEntry>>,
1305}
1306
1307impl PerCpuVmsas {
1308    const fn new() -> Self {
1309        Self {
1310            vmsas: RWLock::new(Vec::new()),
1311        }
1312    }
1313
1314    pub fn exists(&self, paddr: PhysAddr) -> bool {
1315        self.vmsas
1316            .lock_read()
1317            .iter()
1318            .any(|vmsa| vmsa.paddr == paddr)
1319    }
1320
1321    pub fn overlaps(&self, region: &MemoryRegion<PhysAddr>) -> bool {
1322        self.vmsas
1323            .lock_read()
1324            .iter()
1325            .any(|vmsa| region.overlap(&MemoryRegion::new(vmsa.paddr, PAGE_SIZE)))
1326    }
1327
1328    pub fn register(
1329        &self,
1330        paddr: PhysAddr,
1331        cpu_index: usize,
1332        guest_owned: bool,
1333    ) -> Result<(), SvsmError> {
1334        let mut guard = self.vmsas.lock_write();
1335        if guard.iter().any(|vmsa| vmsa.paddr == paddr) {
1336            return Err(SvsmError::InvalidAddress);
1337        }
1338
1339        guard.push(VmsaRegistryEntry::new(paddr, cpu_index, guest_owned));
1340        Ok(())
1341    }
1342
1343    pub fn set_used(&self, paddr: PhysAddr) -> Option<usize> {
1344        self.vmsas
1345            .lock_write()
1346            .iter_mut()
1347            .find(|vmsa| vmsa.paddr == paddr && !vmsa.in_use)
1348            .map(|vmsa| {
1349                vmsa.in_use = true;
1350                vmsa.cpu_index
1351            })
1352    }
1353
1354    pub fn unregister(&self, paddr: PhysAddr, in_use: bool) -> Result<VmsaRegistryEntry, u64> {
1355        let mut guard = self.vmsas.lock_write();
1356        let index = guard
1357            .iter()
1358            .position(|vmsa| vmsa.paddr == paddr && vmsa.in_use == in_use)
1359            .ok_or(0u64)?;
1360
1361        if in_use {
1362            let vmsa = &guard[index];
1363
1364            let target_cpu = PERCPU_AREAS.get_by_cpu_index(vmsa.cpu_index);
1365            target_cpu.clear_guest_vmsa_if_match(paddr);
1366        }
1367
1368        Ok(guard.swap_remove(index))
1369    }
1370}

1376pub fn cpu_idle_loop(cpu_index: usize) {
1377    debug_assert_eq!(cpu_index, this_cpu().get_cpu_index());
1378
1379    loop {
1380        // Go idle
1381        halt();
1382
1383        // If idle was explicitly requested by another task, then schedule that
1384        // task to execute again in case it wants to perform processing as a
1385        // result of the wake from idle.
1386        let maybe_task = this_cpu().runqueue().lock_write().wake_from_idle();
1387        if let Some(task) = maybe_task {
1388            schedule_task(task);
1389        }
1390
1391        // Execute any tasks that are currently runnable.
1392        schedule();
1393    }
