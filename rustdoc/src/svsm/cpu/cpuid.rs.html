14static CPUID_PAGE: ImmutAfterInitCell<&SnpCpuidTable> = ImmutAfterInitCell::uninit();

16const _: () = assert!(size_of::<SnpCpuidTable>() <= PAGE_SIZE);
17
18pub fn register_cpuid_table(table: &'static SnpCpuidTable) {
19    CPUID_PAGE
20        .init(table)
21        .expect("Could not initialize CPUID page");
22}

24/// # Safety
25/// The caller must specify a valid virtual address to use for CPUID table
26/// initialization.
27///
28/// # Panics
29///
30/// Panics if the provided address is not aligned to a [`SnpCpuidTable`].
31pub unsafe fn init_cpuid_table(addr: VirtAddr) {
32    // SAFETY: the caller takes responsibility for the correctness of the
33    // virtual address.
34    let table = unsafe {
35        addr.aligned_mut::<SnpCpuidTable>()
36            .expect("Misaligned SNP CPUID table address")
37    };
38
39    for func in table.func.iter_mut().take(table.count as usize) {
40        if func.eax_in == 0x8000001f {
41            func.eax_out |= 1 << 28;
42        }
43    }
44
45    register_cpuid_table(table);
46}

48/// Copy a CPUID page's content to memory pointed to by a [`VirtAddr`]
49///
50/// # Safety
51///
52/// The caller should verify that `dst` points to mapped memory whose size is
53/// at least 4K. We assert above at compile time that SnpCpuidTable fits within
54/// a page, so the write is safe.
55///
56/// The caller should verify not to corrupt arbitrary memory, as this function
57/// doesn't make any checks in that regard.
58pub unsafe fn copy_cpuid_table_to(dst: VirtAddr) {
59    let start = dst.as_mut_ptr::<u8>();
60    // SAFETY: caller must ensure the address is valid and not aliased.
61    unsafe {
62        // Zero target and copy data
63        start.write_bytes(0, PAGE_SIZE);
64        start
65            .cast::<SnpCpuidTable>()
66            .copy_from_nonoverlapping(*CPUID_PAGE, 1);
67    }
68}

94#[derive(Clone, Copy, Debug)]
95pub struct CpuidResult {
96    pub eax: u32,
97    pub ebx: u32,
98    pub ecx: u32,
99    pub edx: u32,
100}
101
102impl CpuidResult {
103    pub fn get(cpuid_fn: u32, cpuid_subfn: u32) -> Self {
104        let mut result_eax: u32;
105        let mut result_ebx: u32;
106        let mut result_ecx: u32;
107        let mut result_edx: u32;
108        // SAFETY: Inline assembly to execute the CPUID instruction which does
109        // not change any state. Input registers (EAX, ECX) and output
110        // registers (EAX, EBX, ECX, EDX) are safely managed.
111        unsafe {
112            asm!("push %rbx",
113                 "cpuid",
114                 "movl %ebx, %edi",
115                 "pop %rbx",
116                 in("eax") cpuid_fn,
117                 in("ecx") cpuid_subfn,
118                 lateout("eax") result_eax,
119                 lateout("edi") result_ebx,
120                 lateout("ecx") result_ecx,
121                 lateout("edx") result_edx,
122                 options(att_syntax));
123        }
124        Self {
125            eax: result_eax,
126            ebx: result_ebx,
127            ecx: result_ecx,
128            edx: result_edx,
129        }
130    }
131}

133pub fn cpuid_table_raw(eax: u32, ecx: u32, xcr0: u64, xss: u64) -> Option<CpuidResult> {
134    let count: usize = CPUID_PAGE.count as usize;
135
136    for i in 0..count {
137        if eax == CPUID_PAGE.func[i].eax_in
138            && ecx == CPUID_PAGE.func[i].ecx_in
139            && xcr0 == CPUID_PAGE.func[i].xcr0_in
140            && xss == CPUID_PAGE.func[i].xss_in
141        {
142            return Some(CpuidResult {
143                eax: CPUID_PAGE.func[i].eax_out,
144                ebx: CPUID_PAGE.func[i].ebx_out,
145                ecx: CPUID_PAGE.func[i].ecx_out,
146                edx: CPUID_PAGE.func[i].edx_out,
147            });
148        }
149    }
150
151    None
152}
153
154pub fn cpuid_table(eax: u32, ecx: u32) -> Option<CpuidResult> {
155    cpuid_table_raw(eax, ecx, 0, 0)
156}

158pub fn dump_cpuid_table() {
159    if let Ok(table) = CPUID_PAGE.try_get_inner() {
160        let count = table.count as usize;
161
162        log::trace!("CPUID Table entry count: {count}");
163
164        for i in 0..count {
165            let eax_in = table.func[i].eax_in;
166            let ecx_in = table.func[i].ecx_in;
167            let xcr0_in = table.func[i].xcr0_in;
168            let xss_in = table.func[i].xss_in;
169            let eax_out = table.func[i].eax_out;
170            let ebx_out = table.func[i].ebx_out;
171            let ecx_out = table.func[i].ecx_out;
172            let edx_out = table.func[i].edx_out;
173            log::trace!(
174                "EAX_IN: {eax_in:#010x} ECX_IN: {ecx_in:#010x} XCR0_IN: {xcr0_in:#010x} XSS_IN: {xss_in:#010x} EAX_OUT: {eax_out:#010x} EBX_OUT: {ebx_out:#010x} ECX_OUT: {ecx_out:#010x} EDX_OUT: {edx_out:#010x}"
175            );
176        }
177    }
178}
