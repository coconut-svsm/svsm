25/// Remove XSAVE feature bits from leaf 0xD if the corresponding size
26/// subleaf (2-63) is not present in the CPUID page. This prevents
27/// enabling a feature in XCR0/XSS whose XSAVE area size cannot be
28/// determined.
29///
30/// Updates as well ECX in subleaf 0, which reports maximum XSAVE
31/// area size.
32fn filter_xsave_features(table: &mut SnpCpuidTable) {
33    // Legacy x87 + SSE are always present
34    let mut avail = 3u64;
35    let mut max_size = XSAVE_LEGACY_SIZE;
36
37    // Gather available features from ECX={2-63} subleaves
38    for e in table.entries() {
39        // If the size subleaf is present and reports a non-zero size, the feature
40        // can be enabled.
41        if e.eax_in == 0xd && e.ecx_in >= 2 && e.ecx_in < 64 && e.eax_out != 0 {
42            avail |= 1u64 << e.ecx_in;
43            max_size = max_size.max(e.ebx_out + e.eax_out);
44        }
45    }
46
47    // Now remove the available feature bits from ECX={0,1} subleaves
48    for e in table.entries_mut().iter_mut().filter(|e| e.eax_in == 0xd) {
49        let (hi, lo) = match e.ecx_in {
50            // Subleaf 0: EAX/EDX contain XCR0 features
51            0 => (&mut e.edx_out, &mut e.eax_out),
52            // Subleaf 1: ECX/EDX contain XSS features
53            1 => (&mut e.edx_out, &mut e.ecx_out),
54            _ => continue,
55        };
56        *lo &= avail as u32;
57        *hi &= (avail >> 32) as u32;
58    }
59
60    // ECX in subleaf ECX=0 indicates maximum XSAVE area size, update it
61    // as well
62    if let Some(e) = table
63        .entries_mut()
64        .iter_mut()
65        .find(|e| e.eax_in == 0xd && e.ecx_in == 0)
66    {
67        e.ecx_out = max_size;
68    }
69}

71/// # Safety
72/// The caller must specify a valid virtual address to use for CPUID table
73/// initialization.
74///
75/// # Panics
76///
77/// Panics if the provided address is not aligned to a [`SnpCpuidTable`].
78pub unsafe fn init_cpuid_table(addr: VirtAddr) {
79    // SAFETY: the caller takes responsibility for the correctness of the
80    // virtual address.
81    let table = unsafe {
82        addr.aligned_mut::<SnpCpuidTable>()
83            .expect("Misaligned SNP CPUID table address")
84    };
85
86    for func in table.entries_mut() {
87        if func.eax_in == 0x8000001f {
88            func.eax_out |= 1 << 28;
89        }
90    }
91
92    filter_xsave_features(table);
93    register_cpuid_table(table);
94}

96/// Copy a CPUID page's content to memory pointed to by a [`VirtAddr`]
97///
98/// # Safety
99///
100/// The caller should verify that `dst` points to mapped memory whose size is
101/// at least 4K. We assert above at compile time that SnpCpuidTable fits within
102/// a page, so the write is safe.
103///
104/// The caller should verify not to corrupt arbitrary memory, as this function
105/// doesn't make any checks in that regard.
106pub unsafe fn copy_cpuid_table_to(dst: VirtAddr) {
107    let start = dst.as_mut_ptr::<u8>();
108    // SAFETY: caller must ensure the address is valid and not aliased.
109    unsafe {
110        // Zero target and copy data
111        start.write_bytes(0, PAGE_SIZE);
112        start
113            .cast::<SnpCpuidTable>()
114            .copy_from_nonoverlapping(*CPUID_PAGE, 1);
115    }
116}

119#[repr(C, packed)]
120pub struct CpuidLeaf {
121    pub cpuid_fn: u32,
122    pub cpuid_subfn: u32,
123    pub eax: u32,
124    pub ebx: u32,
125    pub ecx: u32,
126    pub edx: u32,
127}

129impl CpuidLeaf {
130    pub fn new(cpuid_fn: u32, cpuid_subfn: u32) -> Self {
131        CpuidLeaf {
132            cpuid_fn,
133            cpuid_subfn,
134            eax: 0,
135            ebx: 0,
136            ecx: 0,
137            edx: 0,
138        }
139    }
140}

142pub fn cpuid_table(eax: u32, ecx: u32) -> Option<CpuidResult> {
143    CPUID_PAGE
144        .entries()
145        .iter()
146        .find(|f| eax == f.eax_in && ecx == f.ecx_in)
147        .map(|f| CpuidResult {
148            eax: f.eax_out,
149            ebx: f.ebx_out,
150            ecx: f.ecx_out,
151            edx: f.edx_out,
152        })
153}

155pub fn dump_cpuid_table() {
156    if let Ok(table) = CPUID_PAGE.try_get_inner() {
157        let entries = table.entries();
158        log::trace!("CPUID Table entry count: {}", entries.len());
159
160        for func in entries {
161            let eax_in = func.eax_in;
162            let ecx_in = func.ecx_in;
163            let xcr0_in = func.xcr0_in;
164            let xss_in = func.xss_in;
165            let eax_out = func.eax_out;
166            let ebx_out = func.ebx_out;
167            let ecx_out = func.ecx_out;
168            let edx_out = func.edx_out;
169            log::trace!(
170                "EAX_IN: {eax_in:#010x} ECX_IN: {ecx_in:#010x} XCR0_IN: {xcr0_in:#010x} XSS_IN: {xss_in:#010x} EAX_OUT: {eax_out:#010x} EBX_OUT: {ebx_out:#010x} ECX_OUT: {ecx_out:#010x} EDX_OUT: {edx_out:#010x}"
171            );
172        }
173    }
174}
