39pub struct VcError {

41    pub code: usize,
42    pub error_type: VcErrorType,
43}

45impl VcError {
46    const fn new(ctx: &X86ExceptionContext, error_type: VcErrorType) -> Self {
47        Self {
48            rip: ctx.frame.rip,
49            code: ctx.error_code,
50            error_type,
51        }
52    }

55#[derive(Clone, Copy, Debug)]
56pub enum VcErrorType {
57    Unsupported,
58    DecodeFailed,
59    UnknownCpuidLeaf,
60}

62impl From<VcError> for SvsmError {
63    fn from(e: VcError) -> Self {
64        Self::Vc(e)
65    }

68impl fmt::Display for VcError {
69    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
70        write!(f, "Unhandled #VC exception ")?;
71        match self.error_type {
72            VcErrorType::Unsupported => {
73                write!(f, "unsupported #VC exception")?;
74            }
75            VcErrorType::DecodeFailed => {
76                write!(f, "invalid instruction")?;
77            }
78            VcErrorType::UnknownCpuidLeaf => {
79                write!(f, "unknown CPUID leaf")?;
80            }
81        }
82        write!(
83            f,
84            " RIP: {:#018x}: error code: {:#018x}",
85            self.rip, self.code
86        )
87    }
88}

90/// Handles a #VC exception in stage 2 before the GHCB is set up.
91///
92/// # Parameters
93/// - registers state before the exception was raise through a [`X86ExceptionContext`].
94///
95/// # Returns
96/// - a `SvsmError` if the handling went wrong.
97pub fn stage2_handle_vc_exception_no_ghcb(ctx: &mut X86ExceptionContext) -> Result<(), SvsmError> {
98    let err = ctx.error_code;
99    let insn_ctx = vc_decode_insn(ctx)?;
100
101    match err {
102        SVM_EXIT_CPUID => handle_cpuid(ctx),
103        _ => Err(VcError::new(ctx, VcErrorType::Unsupported).into()),
104    }?;
105
106    vc_finish_insn(ctx, &insn_ctx);
107    Ok(())
108}

110/// Handles a #VC exception in stage 2 using GHCB features, requiring the GHCB to
111/// be set up beforehand.
112///
113/// # Parameters
114/// - registers state before the exception was raise through a [`X86ExceptionContext`].
115///
116/// # Returns
117/// - a `SvsmError` if the handling went wrong.
118pub fn stage2_handle_vc_exception(ctx: &mut X86ExceptionContext) -> Result<(), SvsmError> {
119    let err = ctx.error_code;
120
121    // To handle NAE events, we're supposed to reset the VALID_BITMAP field of
122    // the GHCB. This is currently only relevant for IOIO, RDTSC and RDTSCP
123    // handling. This field is currently reset in the relevant GHCB methods
124    // but it would be better to move the reset out of the different
125    // handlers.
126    let ghcb = current_ghcb();
127
128    let insn_ctx = vc_decode_insn(ctx)?;
129
130    match (err, insn_ctx.and_then(|d| d.insn())) {
131        (SVM_EXIT_CPUID, Some(DecodedInsn::Cpuid)) => handle_cpuid(ctx),
132        (SVM_EXIT_IOIO, Some(ins)) => early_handle_ioio(ctx, ghcb, ins),
133        (SVM_EXIT_MSR, Some(ins)) => handle_msr(ctx, ghcb, ins),
134        (SVM_EXIT_RDTSC, Some(DecodedInsn::Rdtsc)) => ghcb.rdtsc_regs(&mut ctx.regs),
135        (SVM_EXIT_RDTSCP, Some(DecodedInsn::Rdtsc)) => ghcb.rdtscp_regs(&mut ctx.regs),
136        _ => Err(VcError::new(ctx, VcErrorType::Unsupported).into()),
137    }?;
138
139    vc_finish_insn(ctx, &insn_ctx);
140    Ok(())
141}

143/// Handles a runtime #VC exception.
144///
145/// # Parameters
146/// - registers state before the exception was raise through a [`X86ExceptionContext`].
147/// - the exception vector code.
148///
149/// # Returns
150/// - a `SvsmError` if the handling went wrong.
151pub fn handle_vc_exception(ctx: &mut X86ExceptionContext, vector: usize) -> Result<(), SvsmError> {
152    let error_code = ctx.error_code;
153
154    // To handle NAE events, we're supposed to reset the VALID_BITMAP field of
155    // the GHCB. This is currently only relevant for IOIO, RDTSC and RDTSCP
156    // handling. This field is currently reset in the relevant GHCB methods
157    // but it would be better to move the reset out of the different
158    // handlers.
159    let ghcb = current_ghcb();
160
161    let insn_ctx = vc_decode_insn(ctx)?;
162
163    match (error_code, insn_ctx.as_ref().and_then(|d| d.insn())) {
164        // If the gdb stub is enabled then debugging operations such as single stepping
165        // will cause either an exception via DB_VECTOR if the DEBUG_SWAP sev_feature is
166        // clear, or a VC exception with an error code of X86_TRAP if set.
167        (X86_TRAP, _) => {
168            handle_debug_exception(ctx, vector);
169            Ok(())
170        }
171        (SVM_EXIT_CPUID, Some(DecodedInsn::Cpuid)) => handle_cpuid(ctx),
172        (SVM_EXIT_IOIO, Some(_)) => handle_ioio(ctx, &insn_ctx.unwrap()),
173        (SVM_EXIT_MSR, Some(ins)) => handle_msr(ctx, ghcb, ins),
174        (SVM_EXIT_RDTSC, Some(DecodedInsn::Rdtsc)) => ghcb.rdtsc_regs(&mut ctx.regs),
175        (SVM_EXIT_RDTSCP, Some(DecodedInsn::Rdtsc)) => ghcb.rdtscp_regs(&mut ctx.regs),
176        _ => Err(VcError::new(ctx, VcErrorType::Unsupported).into()),
177    }?;
178
179    vc_finish_insn(ctx, &insn_ctx);
180    Ok(())
181}

183#[inline]
184const fn get_msr(regs: &X86GeneralRegs) -> u64 {
185    ((regs.rdx as u64) << 32) | regs.rax as u64 & u32::MAX as u64
186}
187
188/// Handles a read from the SVSM-specific MSR defined the in SVSM spec.
189fn handle_svsm_caa_rdmsr(ctx: &mut X86ExceptionContext) -> Result<(), SvsmError> {
190    let caa = this_cpu()
191        .guest_vmsa_ref()
192        .caa_phys()
193        .ok_or(SvsmError::MissingCAA)?
194        .bits();
195    ctx.regs.rdx = (caa >> 32) & 0xffffffff;
196    ctx.regs.rax = caa & 0xffffffff;
197    Ok(())
198}
199
200fn handle_msr(
201    ctx: &mut X86ExceptionContext,
202    ghcb: &GHCB,
203    ins: DecodedInsn,
204) -> Result<(), SvsmError> {
205    match ins {
206        DecodedInsn::Wrmsr => {
207            if get_msr(&ctx.regs) == MSR_SVSM_CAA {
208                return Ok(());
209            }
210            ghcb.wrmsr_regs(&ctx.regs)
211        }
212        DecodedInsn::Rdmsr => {
213            if get_msr(&ctx.regs) == MSR_SVSM_CAA {
214                return handle_svsm_caa_rdmsr(ctx);
215            }
216            ghcb.rdmsr_regs(&mut ctx.regs)
217        }
218        _ => Err(VcError::new(ctx, VcErrorType::DecodeFailed).into()),
219    }
220}

222fn handle_cpuid(ctx: &mut X86ExceptionContext) -> Result<(), SvsmError> {
223    // Section 2.3.1 GHCB MSR Protocol in SEV-ES Guest-Hypervisor Communication Block
224    // Standardization Rev. 2.02.
225    // For SEV-ES/SEV-SNP, we can use the CPUID table already defined and populated with
226    // firmware information.
227    // We choose for now not to call the hypervisor to perform CPUID, since it's no trusted.
228    // Since GHCB is not needed to handle CPUID with the firmware table, we can call the handler
229    // very soon in stage 2.
230    snp_cpuid(ctx)
231}
232
233fn snp_cpuid(ctx: &mut X86ExceptionContext) -> Result<(), SvsmError> {
234    let mut leaf = CpuidLeaf::new(ctx.regs.rax as u32, ctx.regs.rcx as u32);
235    let xcr0_in = if leaf.cpuid_fn == 0xD && (leaf.cpuid_subfn == 1 || leaf.cpuid_subfn == 0) {
236        1
237    } else {
238        0
239    };
240
241    let Some(ret) = cpuid_table_raw(leaf.cpuid_fn, leaf.cpuid_subfn, xcr0_in, 0) else {
242        return Err(VcError::new(ctx, VcErrorType::UnknownCpuidLeaf).into());
243    };
244
245    leaf.eax = ret.eax;
246    leaf.ebx = ret.ebx;
247    leaf.ecx = ret.ecx;
248    leaf.edx = ret.edx;
249
250    ctx.regs.rax = leaf.eax as usize;
251    ctx.regs.rbx = leaf.ebx as usize;
252    ctx.regs.rcx = leaf.ecx as usize;
253    ctx.regs.rdx = leaf.edx as usize;
254
255    Ok(())
256}

258fn vc_finish_insn(ctx: &mut X86ExceptionContext, insn_ctx: &Option<DecodedInsnCtx>) {
259    let new_rip = ctx.frame.rip + insn_ctx.as_ref().map_or(0, |d| d.size());
260    // SAFETY: we are advancing the instruction pointer from the RIP generate by
261    // the exception plus the size of the decoded instruction.
262    unsafe {
263        ctx.set_rip(new_rip);
264    }
265}

267fn ioio_get_port(source: Operand, ctx: &X86ExceptionContext) -> u16 {
268    match source {
269        Operand::Reg(Register::Rdx) => ctx.regs.rdx as u16,
270        Operand::Reg(..) => unreachable!("Port value is always in DX"),
271        Operand::Imm(imm) => match imm {
272            Immediate::U8(val) => val as u16,
273            _ => unreachable!("Port value in immediate is always 1 byte"),
274        },
275    }
276}

278fn early_handle_ioio(
279    ctx: &mut X86ExceptionContext,
280    ghcb: &GHCB,
281    insn: DecodedInsn,
282) -> Result<(), SvsmError> {
283    match insn {
284        DecodedInsn::In(source, in_len) => {
285            let port = ioio_get_port(source, ctx);
286            ctx.regs.rax = (ghcb.ioio_in(port, in_len.try_into()?)? & in_len.mask()) as usize;
287            Ok(())
288        }
289        DecodedInsn::Out(source, out_len) => {
290            let out_value = ctx.regs.rax as u64;
291            let port = ioio_get_port(source, ctx);
292            ghcb.ioio_out(port, out_len.try_into()?, out_value)
293        }
294        _ => Err(VcError::new(ctx, VcErrorType::DecodeFailed).into()),
295    }
296}

298fn handle_ioio(ctx: &mut X86ExceptionContext, insn_ctx: &DecodedInsnCtx) -> Result<(), SvsmError> {
299    insn_ctx.emulate_ioio(ctx).map_err(SvsmError::from)
300}

302fn vc_decode_insn(ctx: &X86ExceptionContext) -> Result<Option<DecodedInsnCtx>, SvsmError> {
303    if !vc_decoding_needed(ctx.error_code) {
304        return Ok(None);
305    }
306
307    // TODO: the instruction fetch will likely to be handled differently when
308    // #VC exception will be raised from CPL > 0.
309
310    let rip: GuestPtr<[u8; MAX_INSN_SIZE]> = GuestPtr::new(VirtAddr::from(ctx.frame.rip));
311
312    // rip and rip+15 addresses should belong to a mapped page.
313    // To ensure this, we rely on GuestPtr::read() that uses the exception table
314    // to handle faults while fetching.
315    // SAFETY: we trust the CPU-provided register state to be valid. Thus, RIP
316    // will point to the instruction that caused #VC to be raised, so it can
317    // safely be read.
318    let insn_raw = unsafe { rip.read()? };
319
320    let insn = Instruction::new(insn_raw);
321    Ok(Some(insn.decode(ctx)?))
322}
323
324fn vc_decoding_needed(error_code: usize) -> bool {
325    !(SVM_EXIT_EXCP_BASE..=SVM_EXIT_LAST_EXCP).contains(&error_code)
326}
