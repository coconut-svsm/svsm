30fn start_cpu(
31    platform: &dyn SvsmPlatform,
32    percpu_shared: &'static PerCpuShared,
33) -> Result<(), SvsmError> {
34    let start_rip: u64 = (start_ap as *const u8) as u64;
35    let percpu = PerCpu::alloc(percpu_shared)?;
36    let pgtable = this_cpu().get_pgtable().clone_shared()?;
37    percpu.setup(platform, pgtable)?;
38
39    platform.start_cpu(percpu, start_rip)?;
40
41    let percpu_shared = percpu.shared();
42    while !percpu_shared.is_online() {}
43    Ok(())
44}

46pub fn start_secondary_cpus(platform: &dyn SvsmPlatform, cpus: &[ACPICPUInfo]) {
47    // Create the shared CPU structures for each application processor
48    // while still running single processor.  This ensures that the
49    // PERCPU_AREAS array is completely initialized before any additional
50    // processor starts running.
51    for c in cpus.iter().filter(|c| c.apic_id != 0 && c.enabled) {
52        // SAFETY: it is safe to create new shraed CPU areas because the system
53        // is still running single processor.
54        unsafe {
55            PERCPU_AREAS.create_new(c.apic_id);
56        }
57    }
58
59    // Now loop over all CPUs to start them.
60    for (cpu_index, percpu_shared) in PERCPU_AREAS.iter().enumerate() {
61        // CPU 0 is already running, so it doesn't require processing.
62        if cpu_index != 0 {
63            // If this is the first AP being started, then advise the TLB
64            // package that future TLB flushes will have to be done with SMP
65            // scope.
66            if cpu_index == 1 {
67                set_tlb_flush_smp();
68            }
69
70            log::info!(
71                "Launching AP {} with APIC-ID {}",
72                cpu_index,
73                percpu_shared.apic_id()
74            );
75            start_cpu(platform, percpu_shared).expect("failed");
76        }
77    }
78}

80#[no_mangle]
81extern "C" fn start_ap_setup(top_of_stack: u64) {
82    // Initialize the GDT, TSS, and IDT.
83    this_cpu().load_gdt_tss(true);
84    load_static_idt();
85    // Now the stack unwinder can be used
86    this_cpu().set_current_stack(MemoryRegion::new(
87        VirtAddr::from(top_of_stack)
88            .checked_sub(STACK_SIZE)
89            .unwrap(),
90        STACK_SIZE,
91    ));
92}

94extern "C" {
95    fn start_ap_indirect();
96}
97
98global_asm!(
99    r#"
100        .globl start_ap_indirect
101    start_ap_indirect:
102        /*
103         * %rdi stores the address of ApStartContext
104         * Load fields from the context structure
105         */
106        movq    (%rdi), %r8     /* CR0 */
107        movq    8(%rdi), %r9    /* CR3 */
108        movq    16(%rdi), %r10  /* CR4 */
109        movl    24(%rdi), %eax  /* Low bits of EFER */
110        movl    28(%rdi), %edx  /* High bits of EFER */
111        movq    32(%rdi), %r12  /* Start RIP */
112        movq    40(%rdi), %rsp  /* Initial RSP */
113
114        /* Switch to the target environment.  This will remove the transition
115         * environment and context structure from the address space. */
116        movq    %r8, %cr0
117        movq    %r10, %cr4
118
119        /* Check to see whether EFER.LME is specified.  If not, then EFER
120         * should not be reloaded. */
121        testl   ${LME}, %eax
122        je      1f
123        movl    $0xC0000080, %ecx   /* EFER */
124        wrmsr
125    1:
126        movq    %r9, %cr3
127
128        /* Make sure stack frames are 16b-aligned */
129        andq    $~0xf, %rsp
130
131        /* Mark the next stack frame as the bottom frame */
132        xor     %rbp, %rbp
133
134        /*
135         * Call a startup function to complete setup in the local
136         * environment.
137         *
138         * %r12 is preserved per x86-64 calling convention.
139         */
140        mov     %rsp, %rdi
141        call    start_ap_setup
142
143        /* Begin execution from the starting RIP */
144        call    *%r12
145        int3
146        "#,
147    LME = const EFERFlags::LME.bits(),
148    options(att_syntax)
149);

151pub fn create_ap_start_context(
152    initial_context: &hyperv::HvInitialVpContext,
153    transition_cr3: u32,
154) -> ApStartContext {
155    ApStartContext {
156        cr0: initial_context.cr0.try_into().unwrap(),
157        cr3: initial_context.cr3.try_into().unwrap(),
158        cr4: initial_context.cr4.try_into().unwrap(),
159        efer: initial_context.efer.try_into().unwrap(),
160        start_rip: initial_context.rip.try_into().unwrap(),
161        rsp: initial_context.rsp.try_into().unwrap(),
162        transition_cr3,
163        initial_rip: start_ap_indirect as usize,
164        context_size: mem::size_of::<ApStartContext>() as u32,
165    }
166}

168#[no_mangle]
169extern "C" fn start_ap() -> ! {
170    let percpu = this_cpu();
171
172    if is_cet_ss_supported() {
173        enable_shadow_stacks!(percpu);
174    }
175
176    percpu
177        .setup_on_cpu(&**SVSM_PLATFORM)
178        .expect("setup_on_cpu() failed");
179
180    percpu
181        .setup_idle_task(cpu_idle_loop)
182        .expect("Failed to allocate idle task for AP");
183
184    // Send a life-sign
185    log::info!("CPU {} is online", this_cpu().get_cpu_index());
186
187    // Mark this CPU as participating in IPI usage.
188    ipi_start_cpu();
189
190    // Set CPU online so that BSP can proceed
191    this_cpu_shared().set_online();
192
193    sse_init();
194
195    // SAFETY: there is no current task running on this processor yet, so
196    // initializing the scheduler is safe.
197    unsafe {
198        schedule_init();
199    }
200
201    unreachable!("Road ends here!");
202}
