9use crate::cpu::msr::{read_msr, write_msr};
10
11/// End-of-Interrupt register MSR offset
12pub const MSR_X2APIC_EOI: u32 = 0x80B;
13/// Spurious-Interrupt-Register MSR offset
14pub const MSR_X2APIC_SPIV: u32 = 0x80F;
15/// Interrupt-Service-Register base MSR offset
16pub const MSR_X2APIC_ISR: u32 = 0x810;
17/// Interrupt-Control-Register register MSR offset
18pub const MSR_X2APIC_ICR: u32 = 0x830;
19
20const MSR_APIC_BASE: u32 = 0x1B;
21const APIC_ENABLE_MASK: u64 = 0x800;
22const APIC_X2_ENABLE_MASK: u64 = 0x400;
23
24// SPIV bits
25const APIC_SPIV_VECTOR_MASK: u64 = (1u64 << 8) - 1;
26const APIC_SPIV_SW_ENABLE_MASK: u64 = 1 << 8;
27
28/// Get the MSR offset relative to a bitmap base MSR and the mask for the MSR
29/// value to check for a specific vector bit being set in IRR, ISR, or TMR.
30///
31/// # Returns
32///
33/// A `(u32, u32)` tuple with the MSR offset as the first and the vector
34/// bitmask as the second value.
35pub fn apic_register_bit(vector: usize) -> (u32, u32) {
36    let index: u8 = vector as u8;
37    ((index >> 5) as u32, 1 << (index & 0x1F))
38}
39
40/// Enables the X2APIC by setting the AE and EXTD bits in the APIC base address
41/// register.
42pub fn x2apic_enable() {
43    // Enable X2APIC mode.
44    let apic_base = read_msr(MSR_APIC_BASE);
45    let apic_base_x2_enabled = apic_base | APIC_ENABLE_MASK | APIC_X2_ENABLE_MASK;
46    if apic_base != apic_base_x2_enabled {
47        // SAFETY: enabling X2APIC mode allows accessing APIC's control
48        // registers through MSR accesses, so enabling it doesn't break
49        // memory safety itself.
50        unsafe {
51            write_msr(MSR_APIC_BASE, apic_base_x2_enabled);
52        }
53    }
54    // Set SW-enable in SPIV to enable IRQ delivery
55    x2apic_sw_enable();
56}
57
58/// Send an End-of-Interrupt notification to the X2APIC.
59pub fn x2apic_eoi() {
60    // SAFETY: writing to EOI MSR doesn't break memory safety.
61    unsafe { write_msr(MSR_X2APIC_EOI, 0) };
62}
63
64/// Check whether a give IRQ vector is currently being serviced by returning
65/// the value of its ISR bit from X2APIC.
66///
67/// # Arguments
68///
69/// - `vector` - The IRQ vector for which to check the ISR bit.
70///
71/// # Returns
72///
73/// Returns `True` when the ISR bit for the vector is 1, `False` otherwise.
74pub fn x2apic_in_service(vector: usize) -> bool {
75    // Examine the APIC ISR to determine whether this interrupt vector is
76    // active.  If so, it is assumed to be an external interrupt.
77    let (msr, mask) = apic_register_bit(vector);
78    (read_msr(MSR_X2APIC_ISR + msr) & mask as u64) != 0
79}

81/// Write a command to the Interrupt Command Register.
82///
83/// # Arguments
84///
85/// - `icr` - The 64-bit value describing the interrupt command.
86pub fn x2apic_icr_write(icr: u64) {
87    // SAFETY: writing to ICR MSR doesn't break memory safety.
88    unsafe { write_msr(MSR_X2APIC_ICR, icr) };
89}
90
91/// Set Spurious-Interrupt-Vector Register
92///
93/// # Arguments
94///
95/// - `vector` - The IRQ vector to deliver spurious interrupts to.
96/// - `enable` - Value of the APIC-Software-Enable bit.
97pub fn x2apic_spiv_write(vector: u8, enable: bool) {
98    let apic_spiv: u64 = if enable { APIC_SPIV_SW_ENABLE_MASK } else { 0 }
99        | ((vector as u64) & APIC_SPIV_VECTOR_MASK);
100    // SAFETY: Setting bits in SIPV does not break memory safety.
101    unsafe { write_msr(MSR_X2APIC_SPIV, apic_spiv) };
102}
103
104/// Enable the APIC-Software-Enable bit.
105pub fn x2apic_sw_enable() {
106    x2apic_spiv_write(0xff, true);
107}
