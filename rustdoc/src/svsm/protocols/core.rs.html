43// This lock prevents races around PVALIDATE and CREATE_VCPU
44//
45// Without the lock there is a possible attack where the error path of
46// core_create_vcpu() could give the guest OS access to a SVSM page.
47//
48// The PValidate path will take the lock for read, the create_vcpu path takes
49// the lock for write.
50static PVALIDATE_LOCK: RWLock<()> = RWLock::new(());
51
52#[repr(C, packed)]
53#[derive(Copy, Clone)]
54struct PValidateRequest {
55    entries: u16,
56    next: u16,
57    resv: u32,
58}

60/// # Safety
61/// The caller must only call this function on a page that was committed for
62/// use as a guest VMSA.
63unsafe fn core_create_vcpu_error_restore(paddr: Option<PhysAddr>, vaddr: Option<VirtAddr>) {
64    if let Some(v) = vaddr {
65        // SAFETY: the caller guarantees the safety of this address.
66        if let Err(err) = unsafe { rmp_clear_guest_vmsa(v) } {
67            log::error!("Failed to restore page permissions: {:#?}", err);
68        }
69    }
70    // In case mappings have been changed
71    flush_tlb_global_sync();
72
73    if let Some(p) = paddr {
74        // SAFETY: This can only fail if another CPU unregisters our
75        // unused VMSA. This is not possible, since unregistration of
76        // an unused VMSA only happens in the error path of core_create_vcpu(),
77        // with a physical address that only this CPU managed to register.
78        PERCPU_VMSAS.unregister(p, false).unwrap();
79    }
80}

82// VMSA validity checks according to SVSM spec
83fn check_vmsa(new: &VMSA, sev_features: u64, svme_mask: u64) -> bool {
84    new.vmpl == RMPFlags::GUEST_VMPL.bits() as u8
85        && new.efer & svme_mask == svme_mask
86        && new.sev_features == sev_features
87}

89/// per-cpu request mapping area size (1GB)
90fn core_create_vcpu(params: &RequestParams) -> Result<(), SvsmReqError> {
91    let paddr = PhysAddr::from(params.rcx);
92    let pcaa = PhysAddr::from(params.rdx);
93    let apic_id: u32 = (params.r8 & 0xffff_ffff) as u32;
94
95    // Check VMSA address
96    if !valid_phys_address(paddr) || !paddr.is_page_aligned() {
97        return Err(SvsmReqError::invalid_address());
98    }
99
100    // Check CAA address
101    if !valid_phys_address(pcaa) || !pcaa.is_page_aligned() {
102        return Err(SvsmReqError::invalid_address());
103    }
104
105    // Check whether VMSA page and CAA region overlap
106    //
107    // Since both areas are 4kb aligned and 4kb in size, and correct alignment
108    // was already checked, it is enough here to check whether VMSA and CAA
109    // page have the same starting address.
110    if paddr == pcaa {
111        return Err(SvsmReqError::invalid_address());
112    }
113
114    let target_cpu = PERCPU_AREAS
115        .get_by_apic_id(apic_id)
116        .ok_or_else(SvsmReqError::invalid_parameter)?;
117
118    // Got valid gPAs and APIC ID, register VMSA immediately to avoid races
119    PERCPU_VMSAS.register(paddr, target_cpu.cpu_index(), true)?;
120
121    // Time to map the VMSA. No need to clean up the registered VMSA on the
122    // error path since this is a fatal error anyway.
123    // SAFETY: the physical address is known to point to a VMSA which is not
124    // already mapped.
125    let new_vmsa = unsafe { PerCPUMapping::<VMSA>::create(paddr)? };
126    let vaddr = new_vmsa.virt_addr();
127
128    // Prevent any parallel PVALIDATE requests from being processed
129    let lock = PVALIDATE_LOCK.lock_write();
130
131    // Make sure the guest can't make modifications to the VMSA page
132    rmp_revoke_guest_access(vaddr, PageSize::Regular).inspect_err(|_| {
133        // SAFETY: this address has already been validated as a guest-owned
134        // address.
135        unsafe {
136            core_create_vcpu_error_restore(Some(paddr), None);
137        }
138    })?;
139
140    // TLB flush needed to propagate new permissions
141    flush_tlb_global_sync();
142
143    let svme_mask: u64 = 1u64 << 12;
144
145    // VMSA validity checks according to SVSM spec
146    if !check_vmsa(&new_vmsa, params.sev_features, svme_mask) {
147        // SAFETY: this address has already been validated as a guest-owned
148        // address.
149        unsafe {
150            core_create_vcpu_error_restore(Some(paddr), Some(vaddr));
151        }
152        return Err(SvsmReqError::invalid_parameter());
153    }
154
155    // Set the VMSA bit
156    // SAFETY: this page was already validated to be a guest-owned page.
157    unsafe {
158        rmp_set_guest_vmsa(vaddr).inspect_err(|_| {
159            core_create_vcpu_error_restore(Some(paddr), Some(vaddr));
160        })?;
161    }
162
163    drop(lock);
164
165    assert!(PERCPU_VMSAS.set_used(paddr) == Some(target_cpu.cpu_index()));
166    target_cpu.update_guest_vmsa_caa(paddr, pcaa);
167
168    Ok(())
169}

171fn core_delete_vcpu(params: &RequestParams) -> Result<(), SvsmReqError> {
172    let paddr = PhysAddr::from(params.rcx);
173
174    PERCPU_VMSAS
175        .unregister(paddr, true)
176        .map_err(|_| SvsmReqError::invalid_parameter())?;
177
178    // Map the VMSA
179    // SAFETY: the physical address is known to point to a VMSA which is not
180    // already mapped.
181    let mut del_vmsa = unsafe { PerCPUMapping::<VMSA>::create(paddr)? };
182
183    // Clear EFER.SVME on deleted VMSA. If the VMSA is executing
184    // disable() will loop until that is not the case
185    del_vmsa.disable();
186
187    // Do not return early here, as we need to do a TLB flush
188    // SAFETY: this page is known to already be in use as a guest VMSA.
189    let res = unsafe {
190        rmp_clear_guest_vmsa(del_vmsa.virt_addr()).map_err(|_| SvsmReqError::invalid_address())
191    };
192
193    // Unmap the page
194    drop(del_vmsa);
195
196    // Tell everyone the news and flush temporary mapping
197    flush_tlb_global_sync();
198
199    res
200}

212fn protocol_supported(version: u32, version_min: u32, version_max: u32) -> u64 {
213    if version >= version_min && version <= version_max {
214        let ret_low: u64 = version_min.into();
215        let ret_high: u64 = version_max.into();
216
217        ret_low | (ret_high << 32)
218    } else {
219        0
220    }
221}

223fn core_query_protocol(params: &mut RequestParams) -> Result<(), SvsmReqError> {
224    let rcx: u64 = params.rcx;
225    let protocol: u32 = (rcx >> 32).try_into().unwrap();
226    let version: u32 = (rcx & 0xffff_ffffu64).try_into().unwrap();
227
228    let ret_val = match protocol {
229        SVSM_CORE_PROTOCOL => protocol_supported(
230            version,
231            CORE_PROTOCOL_VERSION_MIN,
232            CORE_PROTOCOL_VERSION_MAX,
233        ),
234        SVSM_APIC_PROTOCOL => {
235            // The APIC protocol is only supported if the calling CPU supports
236            // alternate injection.
237            if this_cpu().use_apic_emulation() {
238                protocol_supported(
239                    version,
240                    APIC_PROTOCOL_VERSION_MIN,
241                    APIC_PROTOCOL_VERSION_MAX,
242                )
243            } else {
244                0
245            }
246        }
247        SVSM_ATTEST_PROTOCOL => protocol_supported(
248            version,
249            ATTEST_PROTOCOL_VERSION_MIN,
250            ATTEST_PROTOCOL_VERSION_MAX,
251        ),
252        _ => 0,
253    };
254
255    params.rcx = ret_val;
256
257    Ok(())
258}

260fn core_configure_vtom(params: &mut RequestParams) -> Result<(), SvsmReqError> {
261    let query: bool = (params.rcx & 1) == 1;
262
263    // Report that vTOM configuration is unsupported
264    if query {
265        params.rcx = 0;
266        Ok(())
267    } else {
268        Err(SvsmReqError::invalid_request())
269    }

272fn core_pvalidate_one(entry: u64, flush: &mut bool) -> Result<(), SvsmReqError> {
273    let (page_size_bytes, valign, huge) = match entry & 3 {
274        0 => (PAGE_SIZE, VIRT_ALIGN_4K, PageSize::Regular),
275        1 => (PAGE_SIZE_2M, VIRT_ALIGN_2M, PageSize::Huge),
276        _ => return Err(SvsmReqError::invalid_parameter()),
277    };
278
279    let valid = match (entry & 4) == 4 {
280        true => PvalidateOp::Valid,
281        false => PvalidateOp::Invalid,
282    };
283    let ign_cf = (entry & 8) == 8;
284
285    let paddr = PhysAddr::from(entry).page_align();
286
287    if !paddr.is_aligned(page_size_bytes) {
288        return Err(SvsmReqError::invalid_parameter());
289    }
290
291    if !valid_phys_address(paddr) {
292        log::debug!("Invalid phys address: {:#x}", paddr);
293        return Err(SvsmReqError::invalid_address());
294    }
295
296    let guard = PerCPUPageMappingGuard::create(paddr, paddr + page_size_bytes, valign)?;
297    let vaddr = guard.virt_addr();
298
299    // Take lock to prevent races with CREATE_VCPU calls
300    let lock = PVALIDATE_LOCK.lock_read();
301
302    if valid == PvalidateOp::Invalid {
303        *flush |= true;
304        rmp_revoke_guest_access(vaddr, huge)?;
305    }
306
307    // SAFETY: the physical address was guaranteed to be a guest address and
308    // cannot affect memory safety.
309    unsafe {
310        pvalidate(vaddr, huge, valid).or_else(|err| match err {
311            SvsmError::SevSnp(SevSnpError::FAIL_UNCHANGED(_)) if ign_cf => Ok(()),
312            _ => Err(err),
313        })?;
314    }
315
316    drop(lock);
317
318    if valid == PvalidateOp::Valid {
319        // Zero out a page when it is validated and before giving other VMPLs
320        // access to it. This is necessary to prevent a possible HV attack:
321        //
322        // Attack scenario:
323        //   1) SVSM stores secrets in VMPL0 memory at GPA A
324        //   2) HV invalidates GPA A and maps the SPA to GPA B, which is in the
325        //      OS range of GPAs
326        //   3) Guest OS asks SVSM to validate GPA B
327        //   4) SVSM validates page and gives OS access
328        //   5) OS can now read SVSM secrets from GPA B
329        //
330        // The SVSM will not notice the attack until it tries to access GPA A
331        // again. Prevent it by clearing every page before giving access to
332        // other VMPLs.
333        //
334        // Be careful to not clear GPAs which the HV might have mapped
335        // read-only, as the write operation might cause infinite #NPF loops.
336        //
337        // Special thanks to Tom Lendacky for reporting the issue and tracking
338        // down the #NPF loops.
339        //
340        if writable_phys_addr(paddr) {
341            // FIXME: This check leaves a window open for the attack described
342            // above. Remove the check once OVMF and Linux have been fixed and
343            // no longer try to pvalidate MMIO memory.
344
345            // SAFETY: paddr is validated at the beginning of the function, and
346            // we trust PerCPUPageMappingGuard::create() to return a valid
347            // vaddr pointing to a mapped region of at least page_size_bytes
348            // size.
349            unsafe {
350                zero_mem_region(vaddr, vaddr + page_size_bytes);
351            }
352        } else {
353            log::warn!("Not clearing possible read-only page at PA {:#x}", paddr);
354        }
355        // SAFETY: the address was validated earlier as a guest page and thus
356        // memory safety is not affected.
357        unsafe {
358            rmp_grant_guest_access(vaddr, huge)?;
359        }
360    }
361
362    Ok(())
363}

365fn core_pvalidate(params: &RequestParams) -> Result<(), SvsmReqError> {
366    let gpa = PhysAddr::from(params.rcx);
367
368    if !gpa.is_aligned(8) || !valid_phys_address(gpa) {
369        return Err(SvsmReqError::invalid_parameter());
370    }
371
372    let paddr = gpa.page_align();
373    let offset = gpa.page_offset();
374
375    let guard = PerCPUPageMappingGuard::create_4k(paddr)?;
376    let start = guard.virt_addr();
377
378    let guest_page = GuestPtr::<PValidateRequest>::new(start + offset);
379    // SAFETY: start is a new mapped page address, thus valid.
380    // offset can't exceed a page size, so guest_page belongs to mapped memory.
381    let mut request = unsafe { guest_page.read()? };
382
383    let entries = request.entries;
384    let next = request.next;
385
386    // Each entry is 8 bytes in size, 8 bytes for the request header
387    let max_entries: u16 = ((PAGE_SIZE - offset - 8) / 8).try_into().unwrap();
388
389    if entries == 0 || entries > max_entries || entries <= next {
390        return Err(SvsmReqError::invalid_parameter());
391    }
392
393    let mut loop_result = Ok(());
394    let mut flush = false;
395
396    let guest_entries = guest_page.offset(1).cast::<u64>();
397    for i in next..entries {
398        let index = i as isize;
399        // SAFETY: guest_entries comes from guest_page which is a new mapped
400        // page. index is between [next, entries) and both values have been
401        // validated.
402        let entry = match unsafe { guest_entries.offset(index).read() } {
403            Ok(v) => v,
404            Err(e) => {
405                loop_result = Err(e.into());
406                break;
407            }
408        };
409
410        loop_result = core_pvalidate_one(entry, &mut flush);
411        match loop_result {
412            Ok(()) => request.next += 1,
413            Err(SvsmReqError::RequestError(..)) => break,
414            Err(SvsmReqError::FatalError(..)) => return loop_result,
415        }
416    }
417
418    // SAFETY: guest_page is obtained from a guest-provided physical address
419    // (untrusted), so it needs to be valid (ie. belongs to the guest and only
420    // the guest). The physical address is validated by valid_phys_address()
421    // called at the beginning of SVSM_CORE_PVALIDATE handler (this one).
422    if let Err(e) = unsafe { guest_page.write_ref(&request) } {
423        loop_result = Err(e.into());
424    }
425
426    if flush {
427        flush_tlb_global_sync();
428    }
429
430    loop_result
431}

433fn core_remap_ca(params: &RequestParams) -> Result<(), SvsmReqError> {
434    let gpa = PhysAddr::from(params.rcx);
435
436    if !gpa.is_aligned(8) || !valid_phys_address(gpa) || gpa.crosses_page(8) {
437        return Err(SvsmReqError::invalid_parameter());
438    }
439
440    let offset = gpa.page_offset();
441    let paddr = gpa.page_align();
442
443    // Temporarily map new CAA to clear it
444    let mapping_guard = PerCPUPageMappingGuard::create_4k(paddr)?;
445    let vaddr = mapping_guard.virt_addr() + offset;
446
447    let pending = GuestPtr::<SvsmCaa>::new(vaddr);
448    // SAFETY: pending points to a new allocated page
449    unsafe { pending.write(SvsmCaa::zeroed())? };
450
451    // Clear any pending interrupt state before remapping the calling area to
452    // ensure that any pending lazy EOI has been processed.
453    this_cpu().clear_pending_interrupts();
454
455    this_cpu_shared().update_guest_caa(gpa);
456
457    Ok(())
458}

460pub fn core_protocol_request(request: u32, params: &mut RequestParams) -> Result<(), SvsmReqError> {
461    match request {
462        SVSM_REQ_CORE_REMAP_CA => core_remap_ca(params),
463        SVSM_REQ_CORE_PVALIDATE => core_pvalidate(params),
464        SVSM_REQ_CORE_CREATE_VCPU => core_create_vcpu(params),
465        SVSM_REQ_CORE_DELETE_VCPU => core_delete_vcpu(params),
466        SVSM_REQ_CORE_DEPOSIT_MEM => core_deposit_mem(params),
467        SVSM_REQ_CORE_WITHDRAW_MEM => core_withdraw_mem(params),
468        SVSM_REQ_CORE_QUERY_PROTOCOL => core_query_protocol(params),
469        SVSM_REQ_CORE_CONFIGURE_VTOM => core_configure_vtom(params),
470        _ => Err(SvsmReqError::unsupported_call()),
471    }
