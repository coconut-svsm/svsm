44// This lock prevents races around PVALIDATE and CREATE_VCPU
45//
46// Without the lock there is a possible attack where the error path of
47// core_create_vcpu() could give the guest OS access to a SVSM page.
48//
49// The PValidate path will take the lock for read, the create_vcpu path takes
50// the lock for write.
51static PVALIDATE_LOCK: RWLock<()> = RWLock::new(());
52
53#[repr(C, packed)]
54#[derive(Copy, Clone, FromBytes, IntoBytes)]
55struct PValidateRequest {
56    entries: u16,
57    next: u16,
58    resv: u32,
59}

61/// # Safety
62/// The caller must only call this function on a page that was committed for
63/// use as a guest VMSA.
64unsafe fn core_create_vcpu_error_restore(paddr: Option<PhysAddr>, vaddr: Option<VirtAddr>) {
65    if let Some(v) = vaddr {
66        // SAFETY: the caller guarantees the safety of this address.
67        if let Err(err) = unsafe { rmp_clear_guest_vmsa(v) } {
68            log::error!("Failed to restore page permissions: {err:#?}");
69        }
70    }
71    // In case mappings have been changed
72    flush_tlb_global_sync();
73
74    if let Some(p) = paddr {
75        // SAFETY: This can only fail if another CPU unregisters our
76        // unused VMSA. This is not possible, since unregistration of
77        // an unused VMSA only happens in the error path of core_create_vcpu(),
78        // with a physical address that only this CPU managed to register.
79        PERCPU_VMSAS.unregister(p, false).unwrap();
80    }
81}

83// VMSA validity checks according to SVSM spec
84fn check_vmsa(new: &VMSA, sev_features: u64, svme_mask: u64) -> bool {
85    new.vmpl == RMPFlags::GUEST_VMPL.bits() as u8
86        && new.efer & svme_mask == svme_mask
87        && new.sev_features == sev_features
88}

90/// per-cpu request mapping area size (1GB)
91fn core_create_vcpu(params: &RequestParams) -> Result<(), SvsmReqError> {
92    let paddr = PhysAddr::from(params.rcx);
93    let pcaa = PhysAddr::from(params.rdx);
94    let apic_id: u32 = (params.r8 & 0xffff_ffff) as u32;
95
96    // Check VMSA address
97    if !valid_phys_address(paddr) || !paddr.is_page_aligned() {
98        return Err(SvsmReqError::invalid_address());
99    }
100
101    // Check CAA address
102    if !valid_phys_address(pcaa) || !pcaa.is_page_aligned() {
103        return Err(SvsmReqError::invalid_address());
104    }
105
106    // Check whether VMSA page and CAA region overlap
107    //
108    // Since both areas are 4kb aligned and 4kb in size, and correct alignment
109    // was already checked, it is enough here to check whether VMSA and CAA
110    // page have the same starting address.
111    if paddr == pcaa {
112        return Err(SvsmReqError::invalid_address());
113    }
114
115    let target_cpu = PERCPU_AREAS
116        .get_by_apic_id(apic_id)
117        .ok_or_else(SvsmReqError::invalid_parameter)?;
118
119    // Got valid gPAs and APIC ID, register VMSA immediately to avoid races
120    PERCPU_VMSAS.register(paddr, target_cpu.cpu_index(), true)?;
121
122    // Time to map the VMSA. No need to clean up the registered VMSA on the
123    // error path since this is a fatal error anyway.
124    // SAFETY: the physical address is known to point to a VMSA which is not
125    // already mapped.
126    let new_vmsa = unsafe { PerCPUMapping::<VMSA>::create(paddr)? };
127    let vaddr = new_vmsa.virt_addr();
128
129    // Prevent any parallel PVALIDATE requests from being processed
130    let lock = PVALIDATE_LOCK.lock_write();
131
132    // Make sure the guest can't make modifications to the VMSA page
133    rmp_revoke_guest_access(vaddr, PageSize::Regular).inspect_err(|_| {
134        // SAFETY: this address has already been validated as a guest-owned
135        // address.
136        unsafe {
137            core_create_vcpu_error_restore(Some(paddr), None);
138        }
139    })?;
140
141    // TLB flush needed to propagate new permissions
142    flush_tlb_global_sync();
143
144    let svme_mask: u64 = 1u64 << 12;
145
146    // VMSA validity checks according to SVSM spec
147    if !check_vmsa(&new_vmsa, params.sev_features, svme_mask) {
148        // SAFETY: this address has already been validated as a guest-owned
149        // address.
150        unsafe {
151            core_create_vcpu_error_restore(Some(paddr), Some(vaddr));
152        }
153        return Err(SvsmReqError::invalid_parameter());
154    }
155
156    // Set the VMSA bit
157    // SAFETY: this page was already validated to be a guest-owned page.
158    unsafe {
159        rmp_set_guest_vmsa(vaddr).inspect_err(|_| {
160            core_create_vcpu_error_restore(Some(paddr), Some(vaddr));
161        })?;
162    }
163
164    drop(lock);
165
166    assert!(PERCPU_VMSAS.set_used(paddr) == Some(target_cpu.cpu_index()));
167    target_cpu.update_guest_vmsa_caa(paddr, pcaa);
168
169    Ok(())
170}

172fn core_delete_vcpu(params: &RequestParams) -> Result<(), SvsmReqError> {
173    let paddr = PhysAddr::from(params.rcx);
174
175    PERCPU_VMSAS
176        .unregister(paddr, true)
177        .map_err(|_| SvsmReqError::invalid_parameter())?;
178
179    // Map the VMSA
180    // SAFETY: the physical address is known to point to a VMSA which is not
181    // already mapped.
182    let mut del_vmsa = unsafe { PerCPUMapping::<VMSA>::create(paddr)? };
183
184    // Clear EFER.SVME on deleted VMSA. If the VMSA is executing
185    // disable() will loop until that is not the case
186    del_vmsa.disable();
187
188    // Do not return early here, as we need to do a TLB flush
189    // SAFETY: this page is known to already be in use as a guest VMSA.
190    let res = unsafe {
191        rmp_clear_guest_vmsa(del_vmsa.virt_addr()).map_err(|_| SvsmReqError::invalid_address())
192    };
193
194    // Unmap the page
195    drop(del_vmsa);
196
197    // Tell everyone the news and flush temporary mapping
198    flush_tlb_global_sync();
199
200    res
201}

213fn protocol_supported(version: u32, version_min: u32, version_max: u32) -> u64 {
214    if version >= version_min && version <= version_max {
215        let ret_low: u64 = version_min.into();
216        let ret_high: u64 = version_max.into();
217
218        ret_low | (ret_high << 32)
219    } else {
220        0
221    }
222}

224fn core_query_protocol(params: &mut RequestParams) -> Result<(), SvsmReqError> {
225    let rcx: u64 = params.rcx;
226    let protocol: u32 = (rcx >> 32).try_into().unwrap();
227    let version: u32 = (rcx & 0xffff_ffffu64).try_into().unwrap();
228
229    let ret_val = match protocol {
230        SVSM_CORE_PROTOCOL => protocol_supported(
231            version,
232            CORE_PROTOCOL_VERSION_MIN,
233            CORE_PROTOCOL_VERSION_MAX,
234        ),
235        SVSM_APIC_PROTOCOL => {
236            // The APIC protocol is only supported if the calling CPU supports
237            // alternate injection.
238            if this_cpu().use_apic_emulation() {
239                protocol_supported(
240                    version,
241                    APIC_PROTOCOL_VERSION_MIN,
242                    APIC_PROTOCOL_VERSION_MAX,
243                )
244            } else {
245                0
246            }
247        }
248        SVSM_ATTEST_PROTOCOL => protocol_supported(
249            version,
250            ATTEST_PROTOCOL_VERSION_MIN,
251            ATTEST_PROTOCOL_VERSION_MAX,
252        ),
253        _ => 0,
254    };
255
256    params.rcx = ret_val;
257
258    Ok(())
259}

261fn core_configure_vtom(params: &mut RequestParams) -> Result<(), SvsmReqError> {
262    let query: bool = (params.rcx & 1) == 1;
263
264    // Report that vTOM configuration is unsupported
265    if query {
266        params.rcx = 0;
267        Ok(())
268    } else {
269        Err(SvsmReqError::invalid_request())
270    }

273fn core_pvalidate_one(entry: u64, flush: &mut bool) -> Result<(), SvsmReqError> {
274    let (page_size_bytes, valign, huge) = match entry & 3 {
275        0 => (PAGE_SIZE, VIRT_ALIGN_4K, PageSize::Regular),
276        1 => (PAGE_SIZE_2M, VIRT_ALIGN_2M, PageSize::Huge),
277        _ => return Err(SvsmReqError::invalid_parameter()),
278    };
279
280    let valid = match (entry & 4) == 4 {
281        true => PvalidateOp::Valid,
282        false => PvalidateOp::Invalid,
283    };
284    let ign_cf = (entry & 8) == 8;
285
286    let paddr = PhysAddr::from(entry).page_align();
287
288    if !paddr.is_aligned(page_size_bytes) {
289        return Err(SvsmReqError::invalid_parameter());
290    }
291
292    if !valid_phys_address(paddr) {
293        log::debug!("Invalid phys address: {paddr:#x}");
294        return Err(SvsmReqError::invalid_address());
295    }
296
297    let guard = PerCPUPageMappingGuard::create(paddr, paddr + page_size_bytes, valign)?;
298    let vaddr = guard.virt_addr();
299
300    // Take lock to prevent races with CREATE_VCPU calls
301    let lock = PVALIDATE_LOCK.lock_read();
302
303    if valid == PvalidateOp::Invalid {
304        *flush |= true;
305        rmp_revoke_guest_access(vaddr, huge)?;
306    }
307
308    // SAFETY: the physical address was guaranteed to be a guest address and
309    // cannot affect memory safety.
310    unsafe {
311        pvalidate(vaddr, huge, valid).or_else(|err| match err {
312            SvsmError::SevSnp(SevSnpError::FAIL_UNCHANGED(_)) if ign_cf => Ok(()),
313            _ => Err(err),
314        })?;
315    }
316
317    drop(lock);
318
319    if valid == PvalidateOp::Valid {
320        // Zero out a page when it is validated and before giving other VMPLs
321        // access to it. This is necessary to prevent a possible HV attack:
322        //
323        // Attack scenario:
324        //   1) SVSM stores secrets in VMPL0 memory at GPA A
325        //   2) HV invalidates GPA A and maps the SPA to GPA B, which is in the
326        //      OS range of GPAs
327        //   3) Guest OS asks SVSM to validate GPA B
328        //   4) SVSM validates page and gives OS access
329        //   5) OS can now read SVSM secrets from GPA B
330        //
331        // The SVSM will not notice the attack until it tries to access GPA A
332        // again. Prevent it by clearing every page before giving access to
333        // other VMPLs.
334        //
335        // Be careful to not clear GPAs which the HV might have mapped
336        // read-only, as the write operation might cause infinite #NPF loops.
337        //
338        // Special thanks to Tom Lendacky for reporting the issue and tracking
339        // down the #NPF loops.
340        //
341        if writable_phys_addr(paddr) {
342            // FIXME: This check leaves a window open for the attack described
343            // above. Remove the check once OVMF and Linux have been fixed and
344            // no longer try to pvalidate MMIO memory.
345
346            // SAFETY: paddr is validated at the beginning of the function, and
347            // we trust PerCPUPageMappingGuard::create() to return a valid
348            // vaddr pointing to a mapped region of at least page_size_bytes
349            // size.
350            unsafe {
351                zero_mem_region(vaddr, vaddr + page_size_bytes);
352            }
353        } else {
354            log::warn!("Not clearing possible read-only page at PA {paddr:#x}");
355        }
356        // SAFETY: the address was validated earlier as a guest page and thus
357        // memory safety is not affected.
358        unsafe {
359            rmp_grant_guest_access(vaddr, huge)?;
360        }
361    }
362
363    Ok(())
364}

366fn core_pvalidate(params: &RequestParams) -> Result<(), SvsmReqError> {
367    let gpa = PhysAddr::from(params.rcx);
368
369    if !gpa.is_aligned(8) || !valid_phys_address(gpa) {
370        return Err(SvsmReqError::invalid_parameter());
371    }
372
373    let paddr = gpa.page_align();
374    let offset = gpa.page_offset();
375
376    let guard = PerCPUPageMappingGuard::create_4k(paddr)?;
377    let start = guard.virt_addr();
378
379    let guest_page = GuestPtr::<PValidateRequest>::new(start + offset);
380    // SAFETY: start is a new mapped page address, thus valid.
381    // offset can't exceed a page size, so guest_page belongs to mapped memory.
382    let mut request = unsafe { guest_page.read()? };
383
384    let entries = request.entries;
385    let next = request.next;
386
387    let max_entries: u16 = ((PAGE_SIZE - offset - size_of::<PValidateRequest>())
388        / size_of::<u64>())
389    .try_into()
390    .unwrap();
391
392    if entries == 0 || entries > max_entries || entries <= next {
393        return Err(SvsmReqError::invalid_parameter());
394    }
395
396    let mut loop_result = Ok(());
397    let mut flush = false;
398
399    let guest_entries = guest_page.offset(1).cast::<u64>();
400    for i in next..entries {
401        let index = i as isize;
402        // SAFETY: guest_entries comes from guest_page which is a new mapped
403        // page. index is between [next, entries) and both values have been
404        // validated.
405        let entry = match unsafe { guest_entries.offset(index).read() } {
406            Ok(v) => v,
407            Err(e) => {
408                loop_result = Err(e.into());
409                break;
410            }
411        };
412
413        loop_result = core_pvalidate_one(entry, &mut flush);
414        match loop_result {
415            Ok(()) => request.next += 1,
416            Err(SvsmReqError::RequestError(..)) => break,
417            Err(SvsmReqError::FatalError(..)) => return loop_result,
418        }
419    }
420
421    // SAFETY: guest_page is obtained from a guest-provided physical address
422    // (untrusted), so it needs to be valid (ie. belongs to the guest and only
423    // the guest). The physical address is validated by valid_phys_address()
424    // called at the beginning of SVSM_CORE_PVALIDATE handler (this one).
425    if let Err(e) = unsafe { guest_page.write_ref(&request) } {
426        loop_result = Err(e.into());
427    }
428
429    if flush {
430        flush_tlb_global_sync();
431    }
432
433    loop_result
434}

436fn core_remap_ca(params: &RequestParams) -> Result<(), SvsmReqError> {
437    let gpa = PhysAddr::from(params.rcx);
438
439    if !gpa.is_aligned(8) || !valid_phys_address(gpa) || gpa.crosses_page(8) {
440        return Err(SvsmReqError::invalid_parameter());
441    }
442
443    let offset = gpa.page_offset();
444    let paddr = gpa.page_align();
445
446    // Temporarily map new CAA to clear it
447    let mapping_guard = PerCPUPageMappingGuard::create_4k(paddr)?;
448    let vaddr = mapping_guard.virt_addr() + offset;
449
450    let pending = GuestPtr::<SvsmCaa>::new(vaddr);
451    // SAFETY: pending points to a new allocated page
452    unsafe { pending.write(SvsmCaa::zeroed())? };
453
454    // Clear any pending interrupt state before remapping the calling area to
455    // ensure that any pending lazy EOI has been processed.
456    this_cpu().clear_pending_interrupts();
457
458    this_cpu_shared().update_guest_caa(gpa);
459
460    Ok(())
461}

463pub fn core_protocol_request(request: u32, params: &mut RequestParams) -> Result<(), SvsmReqError> {
464    match request {
465        SVSM_REQ_CORE_REMAP_CA => core_remap_ca(params),
466        SVSM_REQ_CORE_PVALIDATE => core_pvalidate(params),
467        SVSM_REQ_CORE_CREATE_VCPU => core_create_vcpu(params),
468        SVSM_REQ_CORE_DELETE_VCPU => core_delete_vcpu(params),
469        SVSM_REQ_CORE_DEPOSIT_MEM => core_deposit_mem(params),
470        SVSM_REQ_CORE_WITHDRAW_MEM => core_withdraw_mem(params),
471        SVSM_REQ_CORE_QUERY_PROTOCOL => core_query_protocol(params),
472        SVSM_REQ_CORE_CONFIGURE_VTOM => core_configure_vtom(params),
473        _ => Err(SvsmReqError::unsupported_call()),
474    }
