22/// vTPM platform commands (SVSM spec, section 8.1 - SVSM_VTPM_QUERY)
23///
24/// The platform commmand values follow the values used by the
25/// Official TPM 2.0 Reference Implementation by Microsoft.
26///
27/// `tpm-20-ref/TPMCmd/Simulator/include/TpmTcpProtocol.h`
28#[repr(u32)]
29#[derive(PartialEq, Copy, Clone, Debug)]
30pub enum TpmPlatformCommand {
31    SendCommand = 8,
32}
33
34impl TryFrom<u32> for TpmPlatformCommand {
35    type Error = SvsmReqError;
36
37    fn try_from(value: u32) -> Result<Self, Self::Error> {
38        let cmd = match value {
39            x if x == TpmPlatformCommand::SendCommand as u32 => TpmPlatformCommand::SendCommand,
40            other => {
41                log::warn!("Failed to convert {} to a TPM platform command", other);
42                return Err(SvsmReqError::invalid_parameter());
43            }
44        };
45
46        Ok(cmd)
47    }
48}

50fn vtpm_platform_commands_supported_bitmap() -> u64 {
51    let mut bitmap: u64 = 0;
52    let vtpm = vtpm_get_locked();
53
54    for cmd in vtpm.get_supported_commands() {
55        bitmap |= 1u64 << *cmd as u32;
56    }
57
58    bitmap
59}
60
61fn is_vtpm_platform_command_supported(cmd: TpmPlatformCommand) -> bool {
62    let vtpm = vtpm_get_locked();
63    vtpm.get_supported_commands().iter().any(|x| *x == cmd)
64}

66const SEND_COMMAND_REQ_INBUF_SIZE: usize = PAGE_SIZE - 9;
67
68// vTPM protocol services (SVSM spec, table 14)
69const SVSM_VTPM_QUERY: u32 = 0;
70const SVSM_VTPM_COMMAND: u32 = 1;
71
72/// TPM_SEND_COMMAND request structure (SVSM spec, table 16)
73#[repr(C, packed)]
74#[derive(FromBytes, IntoBytes, Immutable, KnownLayout, Clone, Copy, Debug)]
75struct TpmSendCommandRequest {
76    /// MSSIM platform command ID
77    command: u32,
78    /// Locality usage for the vTPM is not defined yet (must be zero)
79    locality: u8,
80    /// Size of the input buffer
81    inbuf_size: u32,
82    /// Input buffer that contains the TPM command
83    inbuf: [u8; SEND_COMMAND_REQ_INBUF_SIZE],
84}
85
86impl TpmSendCommandRequest {
87    // Take as slice and return a reference for Self
88    pub fn try_from_as_ref(buffer: &[u8]) -> Result<&Self, SvsmReqError> {
89        let request =
90            Self::ref_from_bytes(buffer).map_err(|_| SvsmReqError::invalid_parameter())?;
91
92        if !request.validate() {
93            return Err(SvsmReqError::invalid_parameter());
94        }
95
96        Ok(request)
97    }
98
99    fn validate(&self) -> bool {
100        // TODO: Before implementing locality, we need to agree what it means
101        // to the platform
102        self.locality == 0
103            && self.command == TpmPlatformCommand::SendCommand as u32
104            && self.inbuf_size as usize <= SEND_COMMAND_REQ_INBUF_SIZE
105    }
106
107    pub fn send(&self) -> Result<Vec<u8>, SvsmReqError> {
108        let length = self.inbuf_size as usize;
109
110        let tpm_cmd = self
111            .inbuf
112            .get(..length)
113            .ok_or_else(SvsmReqError::invalid_parameter)?;
114
115        let vtpm = vtpm_get_locked();
116        let response = vtpm.send_tpm_command(tpm_cmd, self.locality)?;
117
118        Ok(response)
119    }
120}

122const SEND_COMMAND_RESP_OUTBUF_SIZE: usize = PAGE_SIZE - 4;
123
124/// TPM_SEND_COMMAND response structure (SVSM spec, table 17)
125#[repr(C, packed)]
126#[derive(FromBytes, IntoBytes, Immutable, KnownLayout, Clone, Copy, Debug)]
127struct TpmSendCommandResponse {
128    /// Size of the output buffer
129    outbuf_size: u32,
130    /// Output buffer that will hold the command response
131    outbuf: [u8; SEND_COMMAND_RESP_OUTBUF_SIZE],
132}
133
134impl TpmSendCommandResponse {
135    // Take as slice and return a &mut Self
136    pub fn try_from_as_mut_ref(buffer: &mut [u8]) -> Result<&mut Self, SvsmReqError> {
137        Self::mut_from_bytes(buffer).map_err(|_| SvsmReqError::invalid_parameter())
138    }
139
140    /// Write the response to the outbuf
141    ///
142    /// # Arguments
143    ///
144    /// * `response`: TPM_SEND_COMMAND response slice
145    pub fn set_outbuf(&mut self, response: &[u8]) -> Result<(), SvsmReqError> {
146        self.outbuf
147            .get_mut(..response.len())
148            .ok_or_else(SvsmReqError::invalid_request)?
149            .copy_from_slice(response);
150        self.outbuf_size = response.len() as u32;
151
152        Ok(())
153    }
154}

156fn vtpm_query_request(params: &mut RequestParams) -> Result<(), SvsmReqError> {
157    // Bitmap of the supported vTPM commands
158    params.rcx = vtpm_platform_commands_supported_bitmap();
159    // Supported vTPM features. Must-be-zero
160    params.rdx = 0;
161
162    Ok(())
163}

165/// Send a TpmSendCommandRequest to the vTPM
166///
167/// # Arguments
168///
169/// * `buffer`: Contains the TpmSendCommandRequest. It will also be
170///   used to store the TpmSendCommandResponse as a byte slice
171fn tpm_send_command_request(buffer: &mut [u8]) -> Result<(), SvsmReqError> {
172    let outbuf: Vec<u8> = {
173        let request = TpmSendCommandRequest::try_from_as_ref(buffer)?;
174        request.send()?
175    };
176    let response = TpmSendCommandResponse::try_from_as_mut_ref(buffer)?;
177    let _ = response.set_outbuf(outbuf.as_slice());
178
179    Ok(())
180}
181
182fn vtpm_command_request(params: &RequestParams) -> Result<(), SvsmReqError> {
183    let paddr = PhysAddr::from(params.rcx);
184
185    if paddr.is_null() {
186        return Err(SvsmReqError::invalid_parameter());
187    }
188
189    // vTPM common request/response structure (SVSM spec, table 15)
190    //
191    // First 4 bytes are used as input and output.
192    //     IN: platform command
193    //    OUT: platform command response size
194
195    let command = read_from_guest::<u32>(paddr).map_err(|_| SvsmReqError::invalid_parameter())?;
196
197    let cmd = TpmPlatformCommand::try_from(command)?;
198
199    if !is_vtpm_platform_command_supported(cmd) {
200        return Err(SvsmReqError::unsupported_call());
201    }
202
203    match cmd {
204        TpmPlatformCommand::SendCommand => {
205            // The vTPM buffer size is one page, but it not required to be page aligned.
206            let mut buffer = read_bytes_from_guest(paddr, PAGE_SIZE)?;
207            tpm_send_command_request(&mut buffer[..])?;
208            copy_slice_to_guest(&buffer[..], paddr)?;
209        }
210    };
211
212    Ok(())
213}
214
215pub fn vtpm_protocol_request(request: u32, params: &mut RequestParams) -> Result<(), SvsmReqError> {
216    match request {
217        SVSM_VTPM_QUERY => vtpm_query_request(params),
218        SVSM_VTPM_COMMAND => vtpm_command_request(params),
219        _ => Err(SvsmReqError::unsupported_call()),
220    }
