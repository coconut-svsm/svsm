43// According to
44// https://github.com/torvalds/linux/blob/155a3c003e555a7300d156a5252c004c392ec6b0/drivers/virt/coco/sev-guest/sev-guest.c#L370
45// https://github.com/torvalds/linux/blob/155a3c003e555a7300d156a5252c004c392ec6b0/include/linux/psp-sev.h#L17
46// the maximum size of the certificate buffer is 0x4000 bytes (16K).
47// Could not find the  maximum size of the certificate buffer in the
48// "Secure VM Service Module for SEV-SNP Guests, Revision 1.0",
49// "SEV Secure Nested Paging Firmware ABI Specification, Revision 1.58"
50// or "SEV-ES Guest-Hypervisor Communication Block Standardization, Revision 2.04"
51// If future specifications update the maximum certificate buffer size, update the
52// MAX_CERTIFICATE_SIZE constant below to match the new value.
53const MAX_CERTIFICATE_SIZE: usize = 0x4000;

55// Attest services operation structure, as defined in Table 11 of Secure VM Service Module for
56// SEV-SNP Guests 58019 Rev, 1.00 July 2023
57#[repr(C, packed)]
58#[derive(FromBytes, IntoBytes, Immutable, KnownLayout, Clone, Copy, Debug)]
59pub struct AttestServicesOp {
60    report_gpa: u64,
61    report_size: u32,
62    reserved_1: [u8; 4],
63    nonce_gpa: u64,
64    nonce_size: u16,
65    reserved_2: [u8; 6],
66    manifest_gpa: u64,
67    manifest_size: u32,
68    reserved_3: [u8; 4],
69    certificate_gpa: u64,
70    certificate_size: u32,
71    reserved_4: [u8; 4],
72}
73
74impl AttestServicesOp {
75    /// Take a slice and return a reference for Self
76    pub fn try_from_as_ref(buffer: &[u8]) -> Result<&Self, SvsmReqError> {
77        let ops: &Self =
78            Self::ref_from_bytes(buffer).map_err(|_| SvsmReqError::invalid_parameter())?;
79        if !ops.is_reserved_clear() {
80            return Err(SvsmReqError::invalid_parameter());
81        }
82
83        Ok(ops)
84    }
85
86    /// Checks if reserved fields are all set to zero
87    pub fn is_reserved_clear(&self) -> bool {
88        self.reserved_1.iter().all(|&x| x == 0)
89            && self.reserved_2.iter().all(|&x| x == 0)
90            && self.reserved_3.iter().all(|&x| x == 0)
91            && self.reserved_4.iter().all(|&x| x == 0)
92    }
93
94    /// Returns the nonce
95    pub fn get_nonce(&self) -> Result<Vec<u8>, SvsmReqError> {
96        // Check that nonce is the right size and doesn't cross a page boundary.
97        // Nonce i.e. REPORT_DATA is 64 bytes per Table 22 of
98        // "SEV Secure Nested Paging Firmware ABI Specification, Revision 1.57"
99        let gpa = PhysAddr::from(self.nonce_gpa);
100        let nonce_size = self.nonce_size.into();
101        if nonce_size != 64 || gpa.crosses_page(nonce_size) {
102            return Err(SvsmReqError::invalid_parameter());
103        }
104
105        read_bytes_from_guest(gpa, nonce_size).map_err(|_| SvsmReqError::invalid_parameter())
106    }
107
108    /// Returns the manifest buffer gpa and size
109    /// Checks if gpa is page aligned and valid.
110    /// Manifest buffer size can be greater than 4k, so it can cross page boundary.
111    pub fn get_manifest_region(&self) -> Result<MemoryRegion<PhysAddr>, SvsmReqError> {
112        let gpa = PhysAddr::from(self.manifest_gpa);
113        let size =
114            usize::try_from(self.manifest_size).map_err(|_| SvsmReqError::invalid_parameter())?;
115        if !gpa.is_page_aligned() {
116            return Err(SvsmReqError::invalid_parameter());
117        }
118
119        Ok(MemoryRegion::new(gpa, size))
120    }
121
122    /// Returns the report buffer gpa and size
123    /// Checks if gpa is page aligned and valid.
124    /// Report buffer size can be greater than 4k, so it can cross page boundary.
125    pub fn get_report_region(&self) -> Result<MemoryRegion<PhysAddr>, SvsmReqError> {
126        let gpa = PhysAddr::from(self.report_gpa);
127        let size =
128            usize::try_from(self.report_size).map_err(|_| SvsmReqError::invalid_parameter())?;
129        if !gpa.is_page_aligned() {
130            return Err(SvsmReqError::invalid_parameter());
131        }
132
133        Ok(MemoryRegion::new(gpa, size))
134    }
135
136    /// Returns an optional MemoryRegion describing the certificate buffer.
137    ///
138    /// Behaviour:
139    /// - certificate_size == 0 -> Ok(None)
140    ///   (caller did not request certificates / extended report)
141    /// - 0 < certificate_size <= MAX_CERTIFICATE_SIZE and certificate_gpa page aligned
142    ///   -> Ok(Some(MemoryRegion))
143    /// - Otherwise -> Err(SvsmReqError::invalid_parameter())
144    ///
145    /// Notes:
146    /// - Buffer may span multiple pages but must be physically contiguous.
147    /// - Buffer GPA must be page aligned.
148    /// - Size is capped at MAX_CERTIFICATE_SIZE (0x4000 bytes).
149    pub fn get_certificate_region(&self) -> Result<Option<MemoryRegion<PhysAddr>>, SvsmReqError> {
150        let size = usize::try_from(self.certificate_size)
151            .map_err(|_| SvsmReqError::invalid_parameter())?;
152
153        // If size is 0, the certificate buffer is not present.
154        // This is valid and shall not return an error.
155        // It is used to indicate that the user does not want an extended attestation
156        // that returns certificates.
157        if size == 0 {
158            return Ok(None);
159        }
160
161        // Ensure that size is not greater than MAX_CERTIFICATE_SIZE
162        if size > MAX_CERTIFICATE_SIZE {
163            return Err(SvsmReqError::invalid_parameter());
164        }
165
166        let gpa = PhysAddr::from(self.certificate_gpa);
167        if !gpa.is_page_aligned() {
168            return Err(SvsmReqError::invalid_parameter());
169        }
170
171        Ok(Some(MemoryRegion::new(gpa, size)))
172    }
173
174    /// Returns true if an extended report is requested
175    /// Extended report is requested by providing a certificate buffer gpa and
176    /// providing a non-zero certificate buffer size.
177    pub fn is_extended_report(&self) -> Result<bool, SvsmReqError> {
178        // If certificate buffer size is greater than 0, it is an extended report.
179        // else it is a regular report.
180        let size = usize::try_from(self.certificate_size)
181            .map_err(|_| SvsmReqError::invalid_parameter())?;
182        Ok(size > 0)
183    }
184}

186#[derive(Clone)]
187struct GuidTableEntry {
188    guid: uuid::Uuid,
189    data: Vec<u8>,
190}
191
192struct GuidTable {
193    entries: Vec<GuidTableEntry>,
194}
195
196impl GuidTable {
197    fn new() -> Self {
198        Self {
199            entries: Vec::new(),
200        }
201    }
202
203    // Only used when features are enabled.
204    #[allow(dead_code)]
205    fn push(&mut self, guid: uuid::Uuid, data: Vec<u8>) {
206        self.entries.push(GuidTableEntry { guid, data })
207    }
208
209    fn header_size(&self) -> usize {
210        // The services manifest header is 24 bytes, followed by 24 bytes for each entry.
211        (1 + self.entries.len()) * GUID_HEADER_ENTRY_SIZE
212    }
213
214    /// Returns how many bytes the wire ABI representation takes.
215    fn len(&self) -> usize {
216        let init = self.header_size();
217        self.entries
218            .iter()
219            .fold(init, |acc, entry| acc + entry.data.len())
220    }
221
222    // Writes the GuidTable contents at the end of `data` or returns an error.
223    fn to_vec(&self) -> Result<Vec<u8>, SvsmError> {
224        let length = self.len();
225        let mut data: Vec<u8> = Vec::with_capacity(length);
226        data.extend_from_slice(&SERVICES_MANIFEST_GUID.to_bytes_le());
227        data.extend_from_slice(&(length as u32).to_le_bytes());
228        data.extend_from_slice(&(self.entries.len() as u32).to_le_bytes());
229
230        let mut data_cursor = self.header_size() as u32;
231        self.entries.iter().for_each(|entry| {
232            data.extend_from_slice(&entry.guid.to_bytes_le());
233            data.extend_from_slice(&data_cursor.to_le_bytes());
234            data.extend_from_slice(&(entry.data.len() as u32).to_le_bytes());
235            data_cursor += entry.data.len() as u32;
236        });
237        self.entries
238            .iter()
239            .for_each(|entry| data.extend_from_slice(entry.data.as_slice()));
240        Ok(data)
241    }
242}

245// Table 13 of Secure VM Service Module for SEV-SNP Guests
246// 58019 Rev. 1.00 July 2023

249pub struct AttestSingleServiceOp {
250    op: AttestServicesOp,
251    guid: [u8; 16],
252    manifest_ver: u32,
253    reserved_5: [u8; 4],
254}
255
256impl AttestSingleServiceOp {
257    /// Take a slice and return a reference for Self
258    pub fn try_from_as_ref(buffer: &[u8]) -> Result<&Self, SvsmReqError> {
259        let ops: &Self =
260            Self::ref_from_bytes(buffer).map_err(|_| SvsmReqError::invalid_parameter())?;
261        if !ops.is_reserved_clear() || !ops.is_manifest_version_valid() {
262            return Err(SvsmReqError::invalid_parameter());
263        }
264
265        Ok(ops)
266    }
267
268    /// Checks if reserved fields are all set to zero
269    pub fn is_reserved_clear(&self) -> bool {
270        self.op.is_reserved_clear() && self.reserved_5.iter().all(|&x| x == 0)
271    }
272
273    /// Returns the nonce
274    pub fn get_nonce(&self) -> Result<Vec<u8>, SvsmReqError> {
275        self.op.get_nonce()
276    }

278    /// Returns the manifest buffer gpa and size
279    /// Checks if gpa is page aligned and valid.
280    /// Manifest buffer size can be greater than 4k, so it can cross page boundary.
281    pub fn get_manifest_region(&self) -> Result<MemoryRegion<PhysAddr>, SvsmReqError> {
282        self.op.get_manifest_region()
283    }
284
285    /// Returns the report buffer gpa and size
286    /// Checks if gpa is page aligned and valid.
287    /// Report buffer size can be greater than 4k, so it can cross page boundary.
288    pub fn get_report_region(&self) -> Result<MemoryRegion<PhysAddr>, SvsmReqError> {
289        self.op.get_report_region()
290    }
291
292    pub fn get_manifest_version(&self) -> u32 {
293        self.manifest_ver
294    }

297        // Currently only manifest version 0 is supported

299    }

303        Uuid::from_bytes_le(self.guid)

308        self.op.get_certificate_region()
309    }

312    pub fn is_extended_report(&self) -> Result<bool, SvsmReqError> {
313        self.op.is_extended_report()
314    }

317fn get_attestation_report_standard(nonce: &[u8]) -> Result<Box<SnpReportResponse>, SvsmReqError> {
318    let mut resp = SnpReportResponse::new_box_zeroed()
319        .map_err(|_| SvsmReqError::FatalError(SvsmError::Mem))?;
320    let resp_buffer = resp.as_mut_bytes();
321    // Cast error is infallibly discarded.
322    let (report_req, _) = SnpReportRequest::mut_from_prefix(resp_buffer)
323        .map_err(|_| SvsmReqError::invalid_parameter())?;
324    // Zero initialized, so
325    // vmpl=0
326    // flags=0: Use VLEK if installed, otherwise VCEK.
327    report_req.user_data = nonce
328        .try_into()
329        .map_err(|_| SvsmReqError::invalid_parameter())?;
330    let _response_size = get_regular_report(resp_buffer)?;
331
332    Ok(resp)
333}

335fn get_attestation_report_extended(
336    nonce: &[u8],
337) -> Result<(Box<SnpReportResponse>, Box<[u8; MAX_CERTIFICATE_SIZE]>), SvsmReqError> {
338    let mut resp = SnpReportResponse::new_box_zeroed()
339        .map_err(|_| SvsmReqError::FatalError(SvsmError::Mem))?;
340    let resp_buffer = resp.as_mut_bytes();
341    // Cast error is infallibly discarded.
342    let (report_req, _) = SnpReportRequest::mut_from_prefix(resp_buffer)
343        .map_err(|_| SvsmReqError::invalid_parameter())?;
344
345    // Extended report requires a certificate buffer.
346    // Allocate a zero initialized buffer for certificates.
347    // MAX_CERTIFICATE_SIZE is defined as 0x4000 bytes (16k).
348    let mut certs = <[u8; MAX_CERTIFICATE_SIZE]>::new_box_zeroed()
349        .map_err(|_| SvsmReqError::FatalError(SvsmError::Mem))?;
350    let certs_buffer = certs.as_mut_bytes();
351
352    // Zero initialized, so
353    // vmpl=0
354    // flags=0: Use VLEK if installed, otherwise VCEK.
355    report_req.user_data = nonce
356        .try_into()
357        .map_err(|_| SvsmReqError::invalid_parameter())?;
358
359    let _response_size = get_extended_report(resp_buffer, certs_buffer)?;
360
361    Ok((resp, certs))
362}

364fn write_report_and_manifest(
365    manifest: &[u8],
366    params: &mut RequestParams,
367    ops: &AttestServicesOp,
368    report: &[u8],
369) -> Result<(), SvsmReqError> {
370    // Get attestation report buffer's gPA from call's Attest Services Operation structure.
371    // The buffer is required to be page aligned but can be bigger than 4K so can cross pages.
372    // If it is bigger than 4K, it must be physically contiguous.
373    let report_region = ops.get_report_region()?;
374
375    // Get manifest buffer's GPA from call's Attest Services Operation structure
376    // The buffer is required to be page aligned but can be bigger than 4K so can cross pages.
377    // If it is bigger than 4K, it must be physically contiguous.
378    let manifest_region = ops.get_manifest_region()?;
379
380    // Check that the manifest will fit in the buffer by checking that the length of the manifest
381    // is less than the size of the buffer. The size of the buffer was used to create the guard,
382    // so can not be tricked into writing outside the buffer.
383    // If the manifest is larger than the buffer, it is either a malformed manifest or buffer too
384    // small. In either case, return an error.
385    if manifest.len() > manifest_region.len() {
386        return Err(SvsmError::Attestation(AttestError::Manifest).into());
387    }
388
389    // Check that the attestation report will fit in the buffer by checking that the length of the
390    // report is less than the size of the buffer. The size of the buffer was used to create the
391    // guard, so can not be tricked into writing outside the buffer.
392    // If the report is larger than the buffer, it is either a malformed report or buffer too small.
393    // In either case, return an error.
394    if report.len() > report_region.len() {
395        return Err(SvsmError::Attestation(AttestError::Report).into());
396    }
397
398    copy_slice_to_guest(report, report_region.start())?;
399
400    // Set report size in bytes in r8 register
401    params.r8 = report
402        .len()
403        .try_into()
404        .map_err(|_| SvsmError::Attestation(AttestError::Report))?;
405
406    copy_slice_to_guest(manifest, manifest_region.start())?;
407
408    // Set the manifest size in bytes in rcx register
409    params.rcx = manifest
410        .len()
411        .try_into()
412        .map_err(|_| SvsmError::Attestation(AttestError::Manifest))?;
413
414    Ok(())
415}

417fn write_certs(
418    certs: &[u8],
419    params: &mut RequestParams,
420    ops: &AttestServicesOp,
421) -> Result<(), SvsmReqError> {
422    // Get certificate buffer's gPA from call's Attest Services Operation structure.
423    // The buffer is required to be page aligned but can be bigger than 4K so can cross pages.
424    // If it is bigger than 4K, it must be physically contiguous.
425    let cert_region = ops.get_certificate_region()?;
426
427    // If certificate region is None, it means that the certificate buffer is not present.
428    // This is valid and shall not return an error.
429    let cert_region = match cert_region {
430        Some(region) => region,
431        None => {
432            // If the certificate region is None, it means that the certificate buffer is not present.
433            // This is valid and shall not return an error.
434            // It is used to indicate that the user does not want an extended attestation
435            // that returns certificates.
436            // Set certificate size to 0, so that the caller knows that there is no certificate.
437            params.rdx = 0;
438            return Ok(());
439        }
440    };
441
442    // According to "SEV-ES Guest-Hypervisor Communication Block Standardization, Revision 2.04"
443    // the initial 24 bytes of the certificate buffer is the certificate header and will be all zeros
444    // if there is no certificate returned.
445    // This is valid and shall not return an error.
446    // Set certificate size to 0, so that the caller knows that there is no certificate.
447    if certs[..24] == [0; 24] {
448        params.rdx = 0;
449        return Ok(());
450    }
451
452    // Check that certificates fit in the buffer. If too large,
453    // return an error indicating the buffer is too small .
454    if certs.len() > cert_region.len() {
455        return Err(SvsmError::Attestation(AttestError::CertificateSize).into());
456    }
457
458    copy_slice_to_guest(certs, cert_region.start())?;
459
460    // Set certificate size in bytes in rdx register
461    params.rdx = certs
462        .len()
463        .try_into()
464        .map_err(|_| SvsmError::Attestation(AttestError::Certificate))?;
465
466    Ok(())
467}

469fn get_attestation_report(
470    hash: &[u8],
471    manifest: &[u8],
472    params: &mut RequestParams,
473    ops: &AttestServicesOp,
474) -> Result<(), SvsmReqError> {
475    if ops.is_extended_report()? {
476        // If the report is an extended report, it means that the caller requested a certificate.
477        // Get extended attestation report from PSP with Sha512(nonce||manifest) as REPORT_DATA.
478        // Handle SvsmReqError::FatalError(SvsmError::Ghcb(GhcbError::VmgexitError(certs_buffer_size,psp_rc,)))
479        // from the PSP indicating that the certificate buffer is too small.
480        // The required size (in 4 KB pages) is in certs_buffer_size.
481        // Per SVSM spec, return required size (in bytes) in rdx register and raise an error.
482        let (resp, certs) = match get_attestation_report_extended(hash) {
483            Ok((resp, certs)) => (resp, certs),
484            Err(SvsmReqError::FatalError(SvsmError::Ghcb(GhcbError::VmgexitError(
485                certs_buffer_size,
486                _psp_rc,
487            )))) => {
488                // The PSP returned an error code indicating that the buffer size is too small.
489                // The required size is in certs_buffer_size but specified as number of 4 KB pages.
490                // Per SVSM spec, return required size (in bytes) in rdx and raise an error
491                params.rdx = certs_buffer_size << PAGE_SHIFT;
492                return Err(SvsmError::Attestation(AttestError::CertificateSize).into());
493            }
494            Err(e) => return Err(e),
495        };
496
497        write_report_and_manifest(manifest, params, ops, resp.report.as_bytes())?;
498        write_certs(certs.as_slice(), params, ops)
499    } else {
500        // Get attestation standard report from PSP with Sha512(nonce||manifest) as REPORT_DATA.
501        let resp = get_attestation_report_standard(hash)?;
502
503        write_report_and_manifest(manifest, params, ops, resp.report.as_bytes())
504    }
505}

507#[allow(dead_code)]
508fn attest_single_service(
509    manifest: &[u8],
510    params: &mut RequestParams,
511    ops: &AttestSingleServiceOp,
512) -> Result<(), SvsmReqError> {
513    let nonce = ops.get_nonce()?;
514
515    // Concatenate nonce and manifest and hash per page 29 of
516    // "Secure VM Service Module for SEV-SNP Guests 58019 Rev. 1.00".
517    let nonce_and_manifest = [&nonce[..], manifest].concat();
518    let hash = Sha512::digest(&nonce_and_manifest);
519
520    get_attestation_report(hash.as_slice(), manifest, params, &ops.op)
521}

531fn attest_multiple_services(params: &mut RequestParams) -> Result<(), SvsmReqError> {
532    let gpa = PhysAddr::from(params.rcx);
533
534    let attest_op =
535        read_from_guest::<AttestServicesOp>(gpa).map_err(|_| SvsmReqError::invalid_parameter())?;
536
537    // Attest multiple services is expected to return a GUID table (mixed endian ordering) of the
538    // enumerated active services' attestation manifest. A service that does not have its own
539    // manifest is still enumerated, but with an empty data blob.
540    #[allow(unused_mut)]
541    let mut services = GuidTable::new();
542
543    #[cfg(all(feature = "vtpm", not(test)))]
544    services.push(SVSM_ATTEST_VTPM_GUID, vtpm_get_manifest()?);
545    let manifest = services.to_vec()?;
546    let mut nonce_and_manifest = attest_op.get_nonce()?;
547    nonce_and_manifest.extend_from_slice(manifest.as_slice());
548
549    // Concatenate nonce and manifest and hash per page 29 of
550    // "Secure VM Service Module for SEV-SNP Guests 58019 Rev. 1.00".
551    let hash = Sha512::digest(&nonce_and_manifest);
552
553    // Get attestation report from PSP with Sha512(nonce||manifest) as REPORT_DATA.
554    get_attestation_report(hash.as_slice(), manifest.as_slice(), params, &attest_op)
555}

557#[allow(clippy::needless_pass_by_ref_mut)]
558fn attest_single_service_handler(params: &mut RequestParams) -> Result<(), SvsmReqError> {
559    // Get the gpa of Attest Single Service Operation structure
560    let gpa = PhysAddr::from(params.rcx);
561
562    let attest_op = read_from_guest::<AttestSingleServiceOp>(gpa)
563        .map_err(|_| SvsmReqError::invalid_parameter())?;
564
565    // Extract the GUID from the Attest Single Service Operation structure.
566    // The GUID is used to determine the specific service to be attested.
567    // Currently, only the VTPM service with the GUID 0xebf176c4_2301a545_9641b4e7_dde5bfe3
568    // is supported, see 8.3.1 of the spec "Secure VM Service Module for SEV-SNP Guests
569    // 58019 Rev. 1.00" for more details.
570    match attest_op.get_guid() {
571        #[cfg(all(feature = "vtpm", not(test)))]
572        SVSM_ATTEST_VTPM_GUID => attest_single_vtpm(params, &attest_op),
573        _ => Err(SvsmReqError::unsupported_protocol()),
574    }
575}

577pub fn attest_protocol_request(
578    request: u32,
579    params: &mut RequestParams,
580) -> Result<(), SvsmReqError> {
581    match request {
582        SVSM_ATTEST_SERVICES => attest_multiple_services(params),
583        SVSM_ATTEST_SINGLE_SERVICE => attest_single_service_handler(params),
584        _ => Err(SvsmReqError::unsupported_protocol()),
585    }
