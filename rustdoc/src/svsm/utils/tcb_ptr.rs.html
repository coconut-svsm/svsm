13use verus_stub::*;

15#[cfg(verus_keep_ghost)]
16use verify_proof::frac_ptr::FracTypedPerm;

18/// Trusted API to write a value at the pointer.
19/// Do not use this outside verified code.
20/// # Safety
21///
22/// without providing the tracked memory permission, this is unsafe.
23/// Must ensure that the pointer is valid and writable.
24/// It is safe with verification since it
25/// - checks the correct permission for accessing the memory is writable.
26/// - guarantees that there is no concurrent read/write access to the same memory location
27///   while the mutable permission reference is in use.
28#[verus_verify(external_body)]
29#[verus_spec(
30    with Tracked(perm): Tracked<&mut FracTypedPerm<T>>
31    requires
32        old(perm).ptr() == ptr,
33        old(perm).writable(),
34        old(perm).valid(),
35    ensures
36        perm@ == old(perm)@.update_value(vstd::simple_pptr::MemContents::Init(v)),
37    opens_invariants none
38    no_unwind
39)]
40#[inline(always)]
41pub(crate) unsafe fn ptr_write<T>(ptr: *mut T, v: T) {
42    // # Safety: guard by the tracked memory permission
43    unsafe {
44        ptr.write(v);
45    }
46}

48/// Trusted API to write a value at the pointer.
49/// Do not use this outside verified code.
50/// # Safety
51///
52/// without providing the tracked memory permission, this is unsafe.
53/// Must ensure that the pointer is valid and readable.
54/// It is safe with verification.
55#[verus_verify(external_body)]
56#[verus_spec(ret =>
57    with Tracked(perm): Tracked<&FracTypedPerm<T>>
58    requires
59        perm.readable(),
60        perm.valid(),
61    ensures
62        ret == perm.value(),
63    opens_invariants none
64    no_unwind
65)]
66#[inline(always)]
67pub(crate) unsafe fn ptr_read<T>(ptr: *const T) -> T {
68    // # Safety: guard by the tracked memory permission
69    unsafe { ptr.read() }
70}
