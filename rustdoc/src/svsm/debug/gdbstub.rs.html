47    // The static mutable reference to the stack is protected by the GDB_STATE lock.
48    #[allow(static_mut_refs)]
49    pub fn gdbstub_start(platform: &'static dyn SvsmPlatform) -> Result<(), u64> {
50        // Debugger initialization must only be attempted once.
51        assert!(
52            GDB_INIT_STATE
53                .compare_exchange(
54                    GdbInitState::Uninitialized as u32,
55                    GdbInitState::Initializing as u32,
56                    Ordering::SeqCst,
57                    Ordering::Relaxed
58                )
59                .is_ok()
60        );
61        // SAFETY: the use of mutable statics is acceptable because their use
62        // is limited to the debugger package, which uses them only during
63        // interactive debugging, at which time the runtime environment is
64        // stopped for single-threaded debugger activity.
65        unsafe {
66            let mut target = GdbStubTarget::new();
67            #[expect(static_mut_refs)]
68            let gdb = GdbStubBuilder::new(GdbStubConnection::new(platform))
69                .with_packet_buffer(&mut PACKET_BUFFER)
70                .build()
71                .expect("Failed to initialise GDB stub")
72                .run_state_machine(&mut target)
73                .expect("Failed to start GDB state machine");
74            *GDB_STATE.lock() = Some(SvsmGdbStub { gdb, target });
75            GDB_STACK_TOP = GDB_STACK.as_mut_ptr().offset(GDB_STACK.len() as isize - 1);
76        }
77        GDB_INIT_STATE.store(GdbInitState::Initialized as u32, Ordering::Relaxed);
78        Ok(())
79    }

99    pub fn handle_debug_exception(ctx: &mut X86ExceptionContext, exception: usize) {
100        let exception_type = ExceptionType::from(exception);
101        let id = this_cpu().runqueue().current_task_id();
102        let mut task_ctx = TaskContext {
103            regs: X86GeneralRegs {
104                r15: ctx.regs.r15,
105                r14: ctx.regs.r14,
106                r13: ctx.regs.r13,
107                r12: ctx.regs.r12,
108                r11: ctx.regs.r11,
109                r10: ctx.regs.r10,
110                r9: ctx.regs.r9,
111                r8: ctx.regs.r8,
112                rbp: ctx.regs.rbp,
113                rdi: ctx.regs.rdi,
114                rsi: ctx.regs.rsi,
115                rdx: ctx.regs.rdx,
116                rcx: ctx.regs.rcx,
117                rbx: ctx.regs.rbx,
118                rax: ctx.regs.rax,
119            },
120            rsp: ctx.frame.rsp as u64,
121            flags: ctx.frame.flags as u64,
122            ret_addr: ctx.frame.rip as u64,
123        };
124
125        // Locking the GDB state for the duration of the stop will cause any other
126        // APs that hit a breakpoint to busy-wait until the current CPU releases
127        // the GDB state. They will then resume and report the stop state
128        // to GDB.
129        // One thing to watch out for - if a breakpoint is inadvertently placed in
130        // the GDB handling code itself then this will cause a re-entrant state
131        // within the same CPU causing a deadlock.
132        loop {
133            let mut gdb_state = GDB_STATE.lock();
134            if let Some(stub) = gdb_state.as_ref() {
135                if stub.target.is_single_step != 0 && stub.target.is_single_step != id {
136                    continue;
137                }
138            }
139
140            // SAFETY: swaps the stack with GDB_STACK_TOP, which is held uniquely mutably
141            // under the GDB_STACK_LOCK, and then sets the stack to the result of handle_stop,
142            // which is the expected behavior.
143            unsafe {
144                asm!(
145                    r#"
146                        movq    %rsp, (%rax)
147                        movq    %rax, %rsp
148                        call    handle_stop
149                        popq    %rax
150                        movq    %rax, %rsp
151                    "#,
152                    in("rsi") exception_type as u64,
153                    in("rdi") &raw mut task_ctx,
154                    in("rdx") &raw mut gdb_state,
155                    in("rax") GDB_STACK_TOP.expose_provenance(),
156                    options(att_syntax));
157            }
158
159            ctx.frame.rip = task_ctx.ret_addr as usize;
160            ctx.frame.flags = task_ctx.flags as usize;
161            ctx.frame.rsp = task_ctx.rsp as usize;
162            ctx.regs.rax = task_ctx.regs.rax;
163            ctx.regs.rbx = task_ctx.regs.rbx;
164            ctx.regs.rcx = task_ctx.regs.rcx;
165            ctx.regs.rdx = task_ctx.regs.rdx;
166            ctx.regs.rsi = task_ctx.regs.rsi;
167            ctx.regs.rdi = task_ctx.regs.rdi;
168            ctx.regs.rbp = task_ctx.regs.rbp;
169            ctx.regs.r8 = task_ctx.regs.r8;
170            ctx.regs.r9 = task_ctx.regs.r9;
171            ctx.regs.r10 = task_ctx.regs.r10;
172            ctx.regs.r11 = task_ctx.regs.r11;
173            ctx.regs.r12 = task_ctx.regs.r12;
174            ctx.regs.r13 = task_ctx.regs.r13;
175            ctx.regs.r14 = task_ctx.regs.r14;
176            ctx.regs.r15 = task_ctx.regs.r15;
177
178            break;
179        }
180    }

182    pub fn debug_break() {
183        if GDB_INIT_STATE.load(Ordering::Acquire) == GdbInitState::Initialized as u32 {
184            log::info!("***********************************");
185            log::info!("* Waiting for connection from GDB *");
186            log::info!("***********************************");
187            // SAFETY: Inline assembly to generate a software interrupt,
188            // which does not change any state related to memory safety.
189            unsafe {
190                asm!("int3");
191            }
192        }
193    }

209    struct GdbTaskContext {
210        cr3: usize,
211        _task_pointer: Option<TaskPointer>,
212    }
213
214    impl GdbTaskContext {
215        /// # Safety
216        /// The caller is required to ensure that switching address spaces does
217        /// not violate memory safety.
218        #[must_use = "The task switch will have no effect if the context is dropped"]
219        unsafe fn switch_to_task(id: u32) -> Self {
220            let (cr3, task_pointer) = if is_current_task(id) {
221                (0, None)
222            } else {
223                let tl = TASKLIST.lock();
224                let cr3 = read_cr3();
225                let task = tl.get_task(id);
226                if let Some(task) = task {
227                    // SAFETY: the caller promises that it is safe to load the
228                    // target page table.  And since the task pointer that owns
229                    // the target page table is referenced and included in the
230                    // context return object, the task cannot be destroyed
231                    // (and thus the page tables remain valid) while the
232                    // task page table remains active.
233                    unsafe {
234                        task.page_table.lock().load();
235                    }
236                    (cr3.bits(), Some(task.clone()))
237                } else {
238                    (0, None)
239                }
240            };
241            Self {
242                cr3,
243                _task_pointer: task_pointer,
244            }
245        }
246    }

248    impl Drop for GdbTaskContext {
249        fn drop(&mut self) {
250            if self.cr3 != 0 {
251                // SAFETY: This is safe because it just restores the previous
252                // page-table, given that the percpu mappings is taken care of.
253                unsafe {
254                    asm!("mov %rax, %cr3",
255                         in("rax") self.cr3,
256                         options(att_syntax));
257                }
258            }
259        }
260    }

273    #[unsafe(no_mangle)]
274    fn handle_stop(
275        ctx: &mut TaskContext,
276        exception_type: ExceptionType,
277        gdb_state: &mut LockGuard<'_, Option<SvsmGdbStub<'_>>>,
278    ) {
279        let SvsmGdbStub { gdb, mut target } = gdb_state.take().expect("Invalid GDB state");
280
281        target.set_regs(ctx);
282
283        let hardcoded_bp = (exception_type == ExceptionType::SwBreakpoint)
284            && !target.is_breakpoint(ctx.ret_addr as usize - 1);
285
286        // If the current address is on a breakpoint then we need to
287        // move the IP back by one byte
288        if (exception_type == ExceptionType::SwBreakpoint)
289            && target.is_breakpoint(ctx.ret_addr as usize - 1)
290        {
291            ctx.ret_addr -= 1;
292        }
293
294        let tid = Tid::new(this_cpu().runqueue().current_task_id() as usize)
295            .expect("Current task has invalid ID");
296        let mut new_gdb = match gdb {
297            GdbStubStateMachine::Running(gdb_inner) => {
298                let reason = if hardcoded_bp {
299                    MultiThreadStopReason::SignalWithThread {
300                        tid,
301                        signal: Signal::SIGINT,
302                    }
303                } else if exception_type == ExceptionType::PageFault {
304                    MultiThreadStopReason::SignalWithThread {
305                        tid,
306                        signal: Signal::SIGSEGV,
307                    }
308                } else {
309                    MultiThreadStopReason::SwBreak(tid)
310                };
311                gdb_inner
312                    .report_stop(&mut target, reason)
313                    .expect("Failed to handle software breakpoint")
314            }
315            _ => gdb,
316        };
317
318        loop {
319            new_gdb = match new_gdb {
320                // The first entry into the debugger is via a forced breakpoint during
321                // initialisation. The state at this point will be Idle instead of
322                // Running.
323                GdbStubStateMachine::Idle(mut gdb_inner) => {
324                    let byte = gdb_inner
325                        .borrow_conn()
326                        .read()
327                        .expect("Failed to read from GDB port");
328                    gdb_inner.incoming_data(&mut target, byte)
329                        .expect("Could not open serial port for GDB connection. \
330                                Please ensure the virtual machine is configured to provide a second serial port.")
331                }
332                GdbStubStateMachine::Running(gdb_inner) => {
333                    new_gdb = gdb_inner.into();
334                    break;
335                }
336                _ => {
337                    panic!("Invalid GDB state when handling breakpoint interrupt");
338                }
339            };
340        }
341        if target.is_single_step == tid.get() as u32 {
342            ctx.flags |= 0x100;
343        } else {
344            ctx.flags &= !0x100;
345        }
346        **gdb_state = Some(SvsmGdbStub {
347            gdb: new_gdb,
348            target,
349        });
350    }

400    impl GdbStubTarget {
401        const fn new() -> Self {
402            Self {
403                ctx: core::ptr::null_mut(),
404                breakpoints: [GdbStubBreakpoint {
405                    addr: VirtAddr::null(),
406                    inst: 0,
407                }; MAX_BREAKPOINTS],
408                is_single_step: 0,
409            }
410        }
411
412        fn ctx(&self) -> Option<&TaskContext> {
413            // SAFETY: this is a pointer to the exception context on the
414            // stack, so it is not aliased from a different task. We trust
415            // the debug exception handler to pass a well-aligned pointer
416            // pointing to valid memory.
417            unsafe { self.ctx.as_ref() }
418        }
419
420        fn ctx_mut(&mut self) -> Option<&mut TaskContext> {
421            // SAFETY: this is a pointer to the exception context on the
422            // stack, so it is not aliased from a different task. We trust
423            // the debug exception handler to pass a well-aligned pointer
424            // pointing to valid memory.
425            unsafe { self.ctx.as_mut() }
426        }
427
428        fn set_regs(&mut self, ctx: &mut TaskContext) {
429            self.ctx = core::ptr::from_mut(ctx)
430        }
431
432        fn is_breakpoint(&self, rip: usize) -> bool {
433            self.breakpoints.iter().any(|b| b.addr.bits() == rip)
434        }
435
436        fn write_bp_address(addr: VirtAddr, value: u8) -> Result<(), SvsmError> {
437            // Virtual addresses in code are likely to be in read-only memory. If we
438            // can get the physical address for this VA then create a temporary
439            // mapping
440
441            let Ok(phys) = this_cpu().get_pgtable().phys_addr(addr) else {
442                // The virtual address is not one that SVSM has mapped.
443                // Try safely writing it to the original virtual address
444                // SAFETY: it is up to the user to ensure that the address we
445                // are writing a breakpoint to is valid.
446                return unsafe { write_u8(addr, value) };
447            };
448
449            let guard = PerCPUPageMappingGuard::create_4k(phys.page_align())?;
450            let dst = guard
451                .virt_addr()
452                .checked_add(phys.page_offset())
453                .ok_or(SvsmError::InvalidAddress)?;
454
455            // SAFETY: guard is a new mapped page, non controllable by user.
456            // We also checked that the destination address didn't overflow.
457            unsafe { write_u8(dst, value) }
458        }
459    }

505    impl MultiThreadBase for GdbStubTarget {
506        fn read_registers(
507            &mut self,
508            regs: &mut <Self::Arch as gdbstub::arch::Arch>::Registers,
509            tid: Tid,
510        ) -> gdbstub::target::TargetResult<(), Self> {
511            if is_current_task(tid.get() as u32) {
512                *regs = X86_64CoreRegs::from(self.ctx().unwrap());
513            } else {
514                let task = TASKLIST.lock().get_task(tid.get() as u32);
515                if let Some(task) = task {
516                    // The registers are stored in the top of the task stack as
517                    // part of the saved context. We need to switch to the task
518                    // pagetable to access them.
519                    // SAFETY: the debugger knows it is safe to operate in the
520                    // context of the specified task.
521                    let _task_context = unsafe { GdbTaskContext::switch_to_task(tid.get() as u32) };
522                    // SAFETY: the task stack is known to contain a register
523                    // context at the top of the stack, so the registers can
524                    // safely be copied through the stack pointer.
525                    unsafe {
526                        *regs = X86_64CoreRegs::from(&*(task.rsp as *const TaskContext));
527                    };
528                    regs.regs[7] = task.rsp;
529                } else {
530                    *regs = <Self::Arch as gdbstub::arch::Arch>::Registers::default();
531                }
532            }
533            Ok(())
534        }
535
536        fn write_registers(
537            &mut self,
538            regs: &<Self::Arch as gdbstub::arch::Arch>::Registers,
539            tid: Tid,
540        ) -> gdbstub::target::TargetResult<(), Self> {
541            if !is_current_task(tid.get() as u32) {
542                return Err(TargetError::NonFatal);
543            }
544
545            let context = self.ctx_mut().unwrap();
546
547            context.ret_addr = regs.rip;
548            context.regs.rax = regs.regs[0] as usize;
549            context.regs.rbx = regs.regs[1] as usize;
550            context.regs.rcx = regs.regs[2] as usize;
551            context.regs.rdx = regs.regs[3] as usize;
552            context.regs.rsi = regs.regs[4] as usize;
553            context.regs.rdi = regs.regs[5] as usize;
554            context.regs.rbp = regs.regs[6] as usize;
555            context.rsp = regs.regs[7];
556            context.regs.r8 = regs.regs[8] as usize;
557            context.regs.r9 = regs.regs[9] as usize;
558            context.regs.r10 = regs.regs[10] as usize;
559            context.regs.r11 = regs.regs[11] as usize;
560            context.regs.r12 = regs.regs[12] as usize;
561            context.regs.r13 = regs.regs[13] as usize;
562            context.regs.r14 = regs.regs[14] as usize;
563            context.regs.r15 = regs.regs[15] as usize;
564            context.flags = regs.eflags as u64;
565            Ok(())
566        }
567
568        fn read_addrs(
569            &mut self,
570            start_addr: <Self::Arch as gdbstub::arch::Arch>::Usize,
571            data: &mut [u8],
572            tid: Tid,
573        ) -> gdbstub::target::TargetResult<(), Self> {
574            // Switch to the task pagetable if necessary. The switch back will
575            // happen automatically when the variable falls out of scope
576            // SAFETY: the debugger knows it is safe to operate in the context
577            // of the specified task, if it exists.
578            let _task_context = unsafe { GdbTaskContext::switch_to_task(tid.get() as u32) };
579            let start_addr = VirtAddr::from(start_addr);
580            for (off, dst) in data.iter_mut().enumerate() {
581                // SAFETY: Unsafe but ok since this is a debug access
582                let Ok(val) = (unsafe { read_u8(start_addr + off) }) else {
583                    return Err(TargetError::NonFatal);
584                };
585                *dst = val;
586            }
587            Ok(())
588        }
589
590        fn write_addrs(
591            &mut self,
592            start_addr: <Self::Arch as gdbstub::arch::Arch>::Usize,
593            data: &[u8],
594            _tid: Tid,
595        ) -> gdbstub::target::TargetResult<(), Self> {
596            let start_addr = VirtAddr::from(start_addr);
597            for (off, src) in data.iter().enumerate() {
598                let dst = start_addr.checked_add(off).ok_or(TargetError::NonFatal)?;
599
600                // SAFETY: We trust the caller of this trait method to provide a valid address.
601                // We only cheked that start_adddr + off didn't overflow.
602                unsafe { write_u8(dst, *src).map_err(|_| TargetError::NonFatal)? }
603            }
604            Ok(())
605        }

612        fn list_active_threads(
613            &mut self,
614            thread_is_active: &mut dyn FnMut(Tid),
615        ) -> Result<(), Self::Error> {
616            let mut tl = TASKLIST.lock();
617
618            // Get the current task. If this is the first request after the remote
619            // GDB has connected then we need to report the current task first.
620            // There is no harm in doing this every time the thread list is requested.
621            let current_task = this_cpu().runqueue().current_task_id();
622            if current_task == INITIAL_TASK_ID {
623                thread_is_active(Tid::new(INITIAL_TASK_ID as usize).unwrap());
624            } else {
625                thread_is_active(Tid::new(current_task as usize).unwrap());
626
627                let mut cursor = tl.list().front_mut();
628                while cursor.get().is_some() {
629                    let this_task = cursor.get().unwrap().get_task_id();
630                    if this_task != current_task {
631                        thread_is_active(Tid::new(this_task as usize).unwrap());
632                    }
633                    cursor.move_next();
634                }
635            }
636            Ok(())
637        }
638
639        fn support_thread_extra_info(
640            &mut self,
641        ) -> Option<gdbstub::target::ext::thread_extra_info::ThreadExtraInfoOps<'_, Self>> {
642            Some(self)
643        }

646    impl ThreadExtraInfo for GdbStubTarget {
647        fn thread_extra_info(&self, tid: Tid, buf: &mut [u8]) -> Result<usize, Self::Error> {
648            // Get the current task from the stopped CPU so we can mark it as stopped
649            let tl = TASKLIST.lock();
650            let str = match tl.get_task(tid.get() as u32) {
651                Some(task) => {
652                    if task.is_running() {
653                        "Running".as_bytes()
654                    } else if task.is_terminated() {
655                        "Terminated".as_bytes()
656                    } else {
657                        "Blocked".as_bytes()
658                    }
659                }
660                None => "Stopped".as_bytes(),
661            };
662            let mut count = 0;
663            for (dst, src) in buf.iter_mut().zip(str) {
664                *dst = *src;
665                count += 1;
666            }
667            Ok(count)
668        }
669    }

729    impl SwBreakpoint for GdbStubTarget {
730        fn add_sw_breakpoint(
731            &mut self,
732            addr: <Self::Arch as gdbstub::arch::Arch>::Usize,
733            _kind: <Self::Arch as gdbstub::arch::Arch>::BreakpointKind,
734        ) -> gdbstub::target::TargetResult<bool, Self> {
735            // Find a free breakpoint slot
736            let Some(free_bp) = self.breakpoints.iter_mut().find(|b| b.addr.is_null()) else {
737                return Ok(false);
738            };
739            // The breakpoint works by taking the opcode at the bp address, storing
740            // it and replacing it with an INT3 instruction
741            let vaddr = VirtAddr::from(addr);
742            // SAFETY: Unsafe but ok since this is a debug access
743            let Ok(inst) = (unsafe { read_u8(vaddr) }) else {
744                return Ok(false);
745            };
746            let Ok(_) = GdbStubTarget::write_bp_address(vaddr, INT3_INSTR) else {
747                return Ok(false);
748            };
749            *free_bp = GdbStubBreakpoint { addr: vaddr, inst };
750            Ok(true)
751        }
752
753        fn remove_sw_breakpoint(
754            &mut self,
755            addr: <Self::Arch as gdbstub::arch::Arch>::Usize,
756            _kind: <Self::Arch as gdbstub::arch::Arch>::BreakpointKind,
757        ) -> gdbstub::target::TargetResult<bool, Self> {
758            let vaddr = VirtAddr::from(addr);
759            let Some(bp) = self.breakpoints.iter_mut().find(|b| b.addr == vaddr) else {
760                return Ok(false);
761            };
762            let Ok(_) = GdbStubTarget::write_bp_address(vaddr, bp.inst) else {
763                return Ok(false);
764            };
765            bp.addr = VirtAddr::null();
766            Ok(true)
767        }
768    }
