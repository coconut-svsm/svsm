47    // The static mutable reference to the stack is protected by the GDB_STATE lock.
48    #[allow(static_mut_refs)]
49    pub fn gdbstub_start(platform: &'static dyn SvsmPlatform) -> Result<(), u64> {
50        // Debugger initialization must only be attempted once.
51        assert!(GDB_INIT_STATE
52            .compare_exchange(
53                GdbInitState::Uninitialized as u32,
54                GdbInitState::Initializing as u32,
55                Ordering::SeqCst,
56                Ordering::Relaxed
57            )
58            .is_ok());
59        // SAFETY: the use of mutable statics is acceptable because their use
60        // is limited to the debugger package, which uses them only during
61        // interactive debugging, at which time the runtime environment is
62        // stopped for single-threaded debugger activity.
63        unsafe {
64            let mut target = GdbStubTarget::new();
65            #[expect(static_mut_refs)]
66            let gdb = GdbStubBuilder::new(GdbStubConnection::new(platform))
67                .with_packet_buffer(&mut PACKET_BUFFER)
68                .build()
69                .expect("Failed to initialise GDB stub")
70                .run_state_machine(&mut target)
71                .expect("Failed to start GDB state machine");
72            *GDB_STATE.lock() = Some(SvsmGdbStub { gdb, target });
73            GDB_STACK_TOP = GDB_STACK.as_mut_ptr().offset(GDB_STACK.len() as isize - 1);
74        }
75        GDB_INIT_STATE.store(GdbInitState::Initialized as u32, Ordering::Relaxed);
76        Ok(())
77    }

97    pub fn handle_debug_exception(ctx: &mut X86ExceptionContext, exception: usize) {
98        let exception_type = ExceptionType::from(exception);
99        let id = this_cpu().runqueue().lock_read().current_task_id();
100        let mut task_ctx = TaskContext {
101            regs: X86GeneralRegs {
102                r15: ctx.regs.r15,
103                r14: ctx.regs.r14,
104                r13: ctx.regs.r13,
105                r12: ctx.regs.r12,
106                r11: ctx.regs.r11,
107                r10: ctx.regs.r10,
108                r9: ctx.regs.r9,
109                r8: ctx.regs.r8,
110                rbp: ctx.regs.rbp,
111                rdi: ctx.regs.rdi,
112                rsi: ctx.regs.rsi,
113                rdx: ctx.regs.rdx,
114                rcx: ctx.regs.rcx,
115                rbx: ctx.regs.rbx,
116                rax: ctx.regs.rax,
117            },
118            rsp: ctx.frame.rsp as u64,
119            flags: ctx.frame.flags as u64,
120            ret_addr: ctx.frame.rip as u64,
121        };
122
123        // Locking the GDB state for the duration of the stop will cause any other
124        // APs that hit a breakpoint to busy-wait until the current CPU releases
125        // the GDB state. They will then resume and report the stop state
126        // to GDB.
127        // One thing to watch out for - if a breakpoint is inadvertently placed in
128        // the GDB handling code itself then this will cause a re-entrant state
129        // within the same CPU causing a deadlock.
130        loop {
131            let mut gdb_state = GDB_STATE.lock();
132            if let Some(stub) = gdb_state.as_ref() {
133                if stub.target.is_single_step != 0 && stub.target.is_single_step != id {
134                    continue;
135                }
136            }
137
138            // SAFETY: swaps the stack with GDB_STACK_TOP, which is held uniquely mutably
139            // under the GDB_STACK_LOCK, and then sets the stack to the result of handle_stop,
140            // which is the expected behavior.
141            unsafe {
142                asm!(
143                    r#"
144                        movq    %rsp, (%rax)
145                        movq    %rax, %rsp
146                        call    handle_stop
147                        popq    %rax
148                        movq    %rax, %rsp
149                    "#,
150                    in("rsi") exception_type as u64,
151                    in("rdi") &mut task_ctx,
152                    in("rdx") &mut gdb_state,
153                    in("rax") GDB_STACK_TOP.expose_provenance(),
154                    options(att_syntax));
155            }
156
157            ctx.frame.rip = task_ctx.ret_addr as usize;
158            ctx.frame.flags = task_ctx.flags as usize;
159            ctx.frame.rsp = task_ctx.rsp as usize;
160            ctx.regs.rax = task_ctx.regs.rax;
161            ctx.regs.rbx = task_ctx.regs.rbx;
162            ctx.regs.rcx = task_ctx.regs.rcx;
163            ctx.regs.rdx = task_ctx.regs.rdx;
164            ctx.regs.rsi = task_ctx.regs.rsi;
165            ctx.regs.rdi = task_ctx.regs.rdi;
166            ctx.regs.rbp = task_ctx.regs.rbp;
167            ctx.regs.r8 = task_ctx.regs.r8;
168            ctx.regs.r9 = task_ctx.regs.r9;
169            ctx.regs.r10 = task_ctx.regs.r10;
170            ctx.regs.r11 = task_ctx.regs.r11;
171            ctx.regs.r12 = task_ctx.regs.r12;
172            ctx.regs.r13 = task_ctx.regs.r13;
173            ctx.regs.r14 = task_ctx.regs.r14;
174            ctx.regs.r15 = task_ctx.regs.r15;
175
176            break;
177        }
178    }

180    pub fn debug_break() {
181        if GDB_INIT_STATE.load(Ordering::Acquire) == GdbInitState::Initialized as u32 {
182            log::info!("***********************************");
183            log::info!("* Waiting for connection from GDB *");
184            log::info!("***********************************");
185            // SAFETY: Inline assembly to generate a software interrupt,
186            // which does not change any state related to memory safety.
187            unsafe {
188                asm!("int3");
189            }
190        }
191    }

207    struct GdbTaskContext {
208        cr3: usize,
209        _task_pointer: Option<TaskPointer>,
210    }
211
212    impl GdbTaskContext {
213        /// # Safety
214        /// The caller is required to ensure that switching address spaces does
215        /// not violate memory safety.
216        #[must_use = "The task switch will have no effect if the context is dropped"]
217        unsafe fn switch_to_task(id: u32) -> Self {
218            let (cr3, task_pointer) = if is_current_task(id) {
219                (0, None)
220            } else {
221                let tl = TASKLIST.lock();
222                let cr3 = read_cr3();
223                let task = tl.get_task(id);
224                if let Some(task) = task {
225                    // SAFETY: the caller promises that it is safe to load the
226                    // target page table.  And since the task pointer that owns
227                    // the target page table is referenced and included in the
228                    // context return object, the task cannot be destroyed
229                    // (and thus the page tables remain valid) while the
230                    // task page table remains active.
231                    unsafe {
232                        task.page_table.lock().load();
233                    }
234                    (cr3.bits(), Some(task.clone()))
235                } else {
236                    (0, None)
237                }
238            };
239            Self {
240                cr3,
241                _task_pointer: task_pointer,
242            }
243        }
244    }

246    impl Drop for GdbTaskContext {
247        fn drop(&mut self) {
248            if self.cr3 != 0 {
249                // SAFETY: This is safe because it just restores the previous
250                // page-table, given that the percpu mappings is taken care of.
251                unsafe {
252                    asm!("mov %rax, %cr3",
253                         in("rax") self.cr3,
254                         options(att_syntax));
255                }
256            }
257        }
258    }

271    #[no_mangle]
272    fn handle_stop(
273        ctx: &mut TaskContext,
274        exception_type: ExceptionType,
275        gdb_state: &mut LockGuard<'_, Option<SvsmGdbStub<'_>>>,
276    ) {
277        let SvsmGdbStub { gdb, mut target } = gdb_state.take().expect("Invalid GDB state");
278
279        target.set_regs(ctx);
280
281        let hardcoded_bp = (exception_type == ExceptionType::SwBreakpoint)
282            && !target.is_breakpoint(ctx.ret_addr as usize - 1);
283
284        // If the current address is on a breakpoint then we need to
285        // move the IP back by one byte
286        if (exception_type == ExceptionType::SwBreakpoint)
287            && target.is_breakpoint(ctx.ret_addr as usize - 1)
288        {
289            ctx.ret_addr -= 1;
290        }
291
292        let tid = Tid::new(this_cpu().runqueue().lock_read().current_task_id() as usize)
293            .expect("Current task has invalid ID");
294        let mut new_gdb = match gdb {
295            GdbStubStateMachine::Running(gdb_inner) => {
296                let reason = if hardcoded_bp {
297                    MultiThreadStopReason::SignalWithThread {
298                        tid,
299                        signal: Signal::SIGINT,
300                    }
301                } else if exception_type == ExceptionType::PageFault {
302                    MultiThreadStopReason::SignalWithThread {
303                        tid,
304                        signal: Signal::SIGSEGV,
305                    }
306                } else {
307                    MultiThreadStopReason::SwBreak(tid)
308                };
309                gdb_inner
310                    .report_stop(&mut target, reason)
311                    .expect("Failed to handle software breakpoint")
312            }
313            _ => gdb,
314        };
315
316        loop {
317            new_gdb = match new_gdb {
318                // The first entry into the debugger is via a forced breakpoint during
319                // initialisation. The state at this point will be Idle instead of
320                // Running.
321                GdbStubStateMachine::Idle(mut gdb_inner) => {
322                    let byte = gdb_inner
323                        .borrow_conn()
324                        .read()
325                        .expect("Failed to read from GDB port");
326                    gdb_inner.incoming_data(&mut target, byte)
327                        .expect("Could not open serial port for GDB connection. \
328                                Please ensure the virtual machine is configured to provide a second serial port.")
329                }
330                GdbStubStateMachine::Running(gdb_inner) => {
331                    new_gdb = gdb_inner.into();
332                    break;
333                }
334                _ => {
335                    panic!("Invalid GDB state when handling breakpoint interrupt");
336                }
337            };
338        }
339        if target.is_single_step == tid.get() as u32 {
340            ctx.flags |= 0x100;
341        } else {
342            ctx.flags &= !0x100;
343        }
344        **gdb_state = Some(SvsmGdbStub {
345            gdb: new_gdb,
346            target,
347        });
348    }

398    impl GdbStubTarget {
399        const fn new() -> Self {
400            Self {
401                ctx: core::ptr::null_mut(),
402                breakpoints: [GdbStubBreakpoint {
403                    addr: VirtAddr::null(),
404                    inst: 0,
405                }; MAX_BREAKPOINTS],
406                is_single_step: 0,
407            }
408        }
409
410        fn ctx(&self) -> Option<&TaskContext> {
411            // SAFETY: this is a pointer to the exception context on the
412            // stack, so it is not aliased from a different task. We trust
413            // the debug exception handler to pass a well-aligned pointer
414            // pointing to valid memory.
415            unsafe { self.ctx.as_ref() }
416        }
417
418        fn ctx_mut(&mut self) -> Option<&mut TaskContext> {
419            // SAFETY: this is a pointer to the exception context on the
420            // stack, so it is not aliased from a different task. We trust
421            // the debug exception handler to pass a well-aligned pointer
422            // pointing to valid memory.
423            unsafe { self.ctx.as_mut() }
424        }
425
426        fn set_regs(&mut self, ctx: &mut TaskContext) {
427            self.ctx = core::ptr::from_mut(ctx)
428        }
429
430        fn is_breakpoint(&self, rip: usize) -> bool {
431            self.breakpoints.iter().any(|b| b.addr.bits() == rip)
432        }
433
434        fn write_bp_address(addr: VirtAddr, value: u8) -> Result<(), SvsmError> {
435            // Virtual addresses in code are likely to be in read-only memory. If we
436            // can get the physical address for this VA then create a temporary
437            // mapping
438
439            let Ok(phys) = this_cpu().get_pgtable().phys_addr(addr) else {
440                // The virtual address is not one that SVSM has mapped.
441                // Try safely writing it to the original virtual address
442                // SAFETY: it is up to the user to ensure that the address we
443                // are writing a breakpoint to is valid.
444                return unsafe { write_u8(addr, value) };
445            };
446
447            let guard = PerCPUPageMappingGuard::create_4k(phys.page_align())?;
448            let dst = guard
449                .virt_addr()
450                .checked_add(phys.page_offset())
451                .ok_or(SvsmError::InvalidAddress)?;
452
453            // SAFETY: guard is a new mapped page, non controllable by user.
454            // We also checked that the destination address didn't overflow.
455            unsafe { write_u8(dst, value) }
456        }
457    }

503    impl MultiThreadBase for GdbStubTarget {
504        fn read_registers(
505            &mut self,
506            regs: &mut <Self::Arch as gdbstub::arch::Arch>::Registers,
507            tid: Tid,
508        ) -> gdbstub::target::TargetResult<(), Self> {
509            if is_current_task(tid.get() as u32) {
510                *regs = X86_64CoreRegs::from(self.ctx().unwrap());
511            } else {
512                let task = TASKLIST.lock().get_task(tid.get() as u32);
513                if let Some(task) = task {
514                    // The registers are stored in the top of the task stack as
515                    // part of the saved context. We need to switch to the task
516                    // pagetable to access them.
517                    // SAFETY: the debugger knows it is safe to operate in the
518                    // context of the specified task.
519                    let _task_context = unsafe { GdbTaskContext::switch_to_task(tid.get() as u32) };
520                    // SAFETY: the task stack is known to contain a register
521                    // context at the top of the stack, so the registers can
522                    // safely be copied through the stack pointer.
523                    unsafe {
524                        *regs = X86_64CoreRegs::from(&*(task.rsp as *const TaskContext));
525                    };
526                    regs.regs[7] = task.rsp;
527                } else {
528                    *regs = <Self::Arch as gdbstub::arch::Arch>::Registers::default();
529                }
530            }
531            Ok(())
532        }
533
534        fn write_registers(
535            &mut self,
536            regs: &<Self::Arch as gdbstub::arch::Arch>::Registers,
537            tid: Tid,
538        ) -> gdbstub::target::TargetResult<(), Self> {
539            if !is_current_task(tid.get() as u32) {
540                return Err(TargetError::NonFatal);
541            }
542
543            let context = self.ctx_mut().unwrap();
544
545            context.ret_addr = regs.rip;
546            context.regs.rax = regs.regs[0] as usize;
547            context.regs.rbx = regs.regs[1] as usize;
548            context.regs.rcx = regs.regs[2] as usize;
549            context.regs.rdx = regs.regs[3] as usize;
550            context.regs.rsi = regs.regs[4] as usize;
551            context.regs.rdi = regs.regs[5] as usize;
552            context.regs.rbp = regs.regs[6] as usize;
553            context.rsp = regs.regs[7];
554            context.regs.r8 = regs.regs[8] as usize;
555            context.regs.r9 = regs.regs[9] as usize;
556            context.regs.r10 = regs.regs[10] as usize;
557            context.regs.r11 = regs.regs[11] as usize;
558            context.regs.r12 = regs.regs[12] as usize;
559            context.regs.r13 = regs.regs[13] as usize;
560            context.regs.r14 = regs.regs[14] as usize;
561            context.regs.r15 = regs.regs[15] as usize;
562            context.flags = regs.eflags as u64;
563            Ok(())
564        }
565
566        fn read_addrs(
567            &mut self,
568            start_addr: <Self::Arch as gdbstub::arch::Arch>::Usize,
569            data: &mut [u8],
570            tid: Tid,
571        ) -> gdbstub::target::TargetResult<(), Self> {
572            // Switch to the task pagetable if necessary. The switch back will
573            // happen automatically when the variable falls out of scope
574            // SAFETY: the debugger knows it is safe to operate in the context
575            // of the specified task, if it exists.
576            let _task_context = unsafe { GdbTaskContext::switch_to_task(tid.get() as u32) };
577            let start_addr = VirtAddr::from(start_addr);
578            for (off, dst) in data.iter_mut().enumerate() {
579                // SAFETY: Unsafe but ok since this is a debug access
580                let Ok(val) = (unsafe { read_u8(start_addr + off) }) else {
581                    return Err(TargetError::NonFatal);
582                };
583                *dst = val;
584            }
585            Ok(())
586        }
587
588        fn write_addrs(
589            &mut self,
590            start_addr: <Self::Arch as gdbstub::arch::Arch>::Usize,
591            data: &[u8],
592            _tid: Tid,
593        ) -> gdbstub::target::TargetResult<(), Self> {
594            let start_addr = VirtAddr::from(start_addr);
595            for (off, src) in data.iter().enumerate() {
596                let dst = start_addr.checked_add(off).ok_or(TargetError::NonFatal)?;
597
598                // SAFETY: We trust the caller of this trait method to provide a valid address.
599                // We only cheked that start_adddr + off didn't overflow.
600                unsafe { write_u8(dst, *src).map_err(|_| TargetError::NonFatal)? }
601            }
602            Ok(())
603        }

610        fn list_active_threads(
611            &mut self,
612            thread_is_active: &mut dyn FnMut(Tid),
613        ) -> Result<(), Self::Error> {
614            let mut tl = TASKLIST.lock();
615
616            // Get the current task. If this is the first request after the remote
617            // GDB has connected then we need to report the current task first.
618            // There is no harm in doing this every time the thread list is requested.
619            let current_task = this_cpu().runqueue().lock_read().current_task_id();
620            if current_task == INITIAL_TASK_ID {
621                thread_is_active(Tid::new(INITIAL_TASK_ID as usize).unwrap());
622            } else {
623                thread_is_active(Tid::new(current_task as usize).unwrap());
624
625                let mut cursor = tl.list().front_mut();
626                while cursor.get().is_some() {
627                    let this_task = cursor.get().unwrap().get_task_id();
628                    if this_task != current_task {
629                        thread_is_active(Tid::new(this_task as usize).unwrap());
630                    }
631                    cursor.move_next();
632                }
633            }
634            Ok(())
635        }
636
637        fn support_thread_extra_info(
638            &mut self,
639        ) -> Option<gdbstub::target::ext::thread_extra_info::ThreadExtraInfoOps<'_, Self>> {
640            Some(self)
641        }

644    impl ThreadExtraInfo for GdbStubTarget {
645        fn thread_extra_info(&self, tid: Tid, buf: &mut [u8]) -> Result<usize, Self::Error> {
646            // Get the current task from the stopped CPU so we can mark it as stopped
647            let tl = TASKLIST.lock();
648            let str = match tl.get_task(tid.get() as u32) {
649                Some(task) => {
650                    if task.is_running() {
651                        "Running".as_bytes()
652                    } else if task.is_terminated() {
653                        "Terminated".as_bytes()
654                    } else {
655                        "Blocked".as_bytes()
656                    }
657                }
658                None => "Stopped".as_bytes(),
659            };
660            let mut count = 0;
661            for (dst, src) in buf.iter_mut().zip(str) {
662                *dst = *src;
663                count += 1;
664            }
665            Ok(count)
666        }
667    }

727    impl SwBreakpoint for GdbStubTarget {
728        fn add_sw_breakpoint(
729            &mut self,
730            addr: <Self::Arch as gdbstub::arch::Arch>::Usize,
731            _kind: <Self::Arch as gdbstub::arch::Arch>::BreakpointKind,
732        ) -> gdbstub::target::TargetResult<bool, Self> {
733            // Find a free breakpoint slot
734            let Some(free_bp) = self.breakpoints.iter_mut().find(|b| b.addr.is_null()) else {
735                return Ok(false);
736            };
737            // The breakpoint works by taking the opcode at the bp address, storing
738            // it and replacing it with an INT3 instruction
739            let vaddr = VirtAddr::from(addr);
740            // SAFETY: Unsafe but ok since this is a debug access
741            let Ok(inst) = (unsafe { read_u8(vaddr) }) else {
742                return Ok(false);
743            };
744            let Ok(_) = GdbStubTarget::write_bp_address(vaddr, INT3_INSTR) else {
745                return Ok(false);
746            };
747            *free_bp = GdbStubBreakpoint { addr: vaddr, inst };
748            Ok(true)
749        }
750
751        fn remove_sw_breakpoint(
752            &mut self,
753            addr: <Self::Arch as gdbstub::arch::Arch>::Usize,
754            _kind: <Self::Arch as gdbstub::arch::Arch>::BreakpointKind,
755        ) -> gdbstub::target::TargetResult<bool, Self> {
756            let vaddr = VirtAddr::from(addr);
757            let Some(bp) = self.breakpoints.iter_mut().find(|b| b.addr == vaddr) else {
758                return Ok(false);
759            };
760            let Ok(_) = GdbStubTarget::write_bp_address(vaddr, bp.inst) else {
761                return Ok(false);
762            };
763            bp.addr = VirtAddr::null();
764            Ok(true)
765        }
766    }
