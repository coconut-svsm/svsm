19fn invalidate_boot_memory_region(
20    platform: &dyn SvsmPlatform,
21    boot_params: &BootParams<'_>,
22    region: MemoryRegion<PhysAddr>,
23) -> Result<(), SvsmError> {
24    // Caller must ensure the memory region's starting address is page-aligned
25    let aligned_region = MemoryRegion::new(region.start(), page_align_up(region.len()));
26    log::info!("Invalidating boot region {aligned_region:#018x}");
27
28    if !aligned_region.is_empty() {
29        // SAFETY: invalidating memory cannot cause UB.
30        unsafe {
31            platform.validate_physical_page_range(aligned_region, PageValidateOp::Invalidate)
32        }?;
33
34        if boot_params.page_state_change_required() {
35            platform.page_state_change(aligned_region, PageStateChangeOp::Shared)?;
36        }
37    }
38
39    Ok(())
40}

42pub fn enumerate_early_boot_regions(
43    boot_params: &BootParams<'_>,
44    launch_info: &KernelLaunchInfo,
45) -> Vec<MemoryRegion<PhysAddr>> {
46    let mut regions = Vec::new();
47
48    // Early boot memory must be invalidated after changing to the SVSM page
49    // page table to avoid invalidating page tables currently in use.  Always
50    // invalidate stage 2 memory, unless firmware is loaded into low memory.
51    // Also invalidate the boot data if required.
52    if !boot_params.fw_in_low_memory() {
53        regions.push(MemoryRegion::from_addresses(
54            PhysAddr::from(0u64),
55            PhysAddr::from(u64::from(LOWMEM_END)),
56        ));
57    }
58
59    // All stage2 memory is contiguous, and is bounded by the stage2 image
60    // at the base and the filesystem at the end.
61    let stage2_area_base = PhysAddr::from(launch_info.stage2_start);
62    let stage2_area_end = PhysAddr::new(launch_info.kernel_fs_end.try_into().unwrap());
63    regions.push(MemoryRegion::from_addresses(
64        stage2_area_base,
65        stage2_area_end,
66    ));
67
68    regions
69}
70
71pub fn invalidate_early_boot_memory(
72    platform: &dyn SvsmPlatform,
73    boot_params: &BootParams<'_>,
74    regions: &[MemoryRegion<PhysAddr>],
75) -> Result<(), SvsmError> {
76    for region in regions {
77        invalidate_boot_memory_region(platform, boot_params, *region)?;
78    }
79
80    Ok(())
