24pub fn init_page_table(
25    launch_info: &KernelLaunchInfo,
26    kernel_elf: &elf::Elf64File<'_>,
27) -> Result<PageBox<PageTable>, SvsmError> {
28    let mut pgtable = PageTable::allocate_new()?;
29
30    let igvm_param_info = if launch_info.igvm_params_virt_addr != 0 {
31        let addr = VirtAddr::from(launch_info.igvm_params_virt_addr);
32        IgvmParamInfo {
33            virt_addr: addr,
34            igvm_params: Some(IgvmParams::new(addr)?),
35        }
36    } else {
37        IgvmParamInfo {
38            virt_addr: VirtAddr::null(),
39            igvm_params: None,
40        }
41    };
42
43    // Install mappings for the kernel's ELF segments each.
44    // The memory backing the kernel ELF segments gets allocated back to back
45    // from the physical memory region by the Stage2 loader.
46    let mut phys = PhysAddr::from(launch_info.kernel_region_phys_start);
47    for segment in kernel_elf.image_load_segment_iter(launch_info.kernel_region_virt_start) {
48        let vaddr_start = VirtAddr::from(segment.vaddr_range.vaddr_begin);
49        let vaddr_end = VirtAddr::from(segment.vaddr_range.vaddr_end);
50        let aligned_vaddr_end = vaddr_end.page_align_up();
51        let segment_len = aligned_vaddr_end - vaddr_start;
52        let flags = if segment.flags.contains(elf::Elf64PhdrFlags::EXECUTE) {
53            PTEntryFlags::exec()
54        } else if segment.flags.contains(elf::Elf64PhdrFlags::WRITE) {
55            PTEntryFlags::data()
56        } else {
57            PTEntryFlags::data_ro()
58        };
59
60        let vregion = MemoryRegion::new(vaddr_start, segment_len);
61        pgtable
62            .map_region(vregion, phys, flags)
63            .expect("Failed to map kernel ELF segment");
64
65        phys = phys + segment_len;
66    }
67
68    // Map the IGVM parameters if present.
69    if let Some(ref igvm_params) = igvm_param_info.igvm_params {
70        let vregion = MemoryRegion::new(igvm_param_info.virt_addr, igvm_params.size());
71        pgtable
72            .map_region(
73                vregion,
74                PhysAddr::from(launch_info.igvm_params_phys_addr),
75                PTEntryFlags::data(),
76            )
77            .expect("Failed to map IGVM parameters");
78    }
79
80    // Map subsequent heap area.
81    let heap_vregion = MemoryRegion::new(
82        VirtAddr::from(launch_info.heap_area_virt_start),
83        launch_info.heap_area_size as usize,
84    );
85    pgtable
86        .map_region(
87            heap_vregion,
88            PhysAddr::from(launch_info.heap_area_phys_start),
89            PTEntryFlags::data(),
90        )
91        .expect("Failed to map heap");
92
93    init_global_ranges();
94
95    Ok(pgtable)
96}

98fn invalidate_boot_memory_region(
99    platform: &dyn SvsmPlatform,
100    config: &SvsmConfig<'_>,
101    region: MemoryRegion<PhysAddr>,
102) -> Result<(), SvsmError> {
103    // Caller must ensure the memory region's starting address is page-aligned
104    let aligned_region = MemoryRegion::new(region.start(), page_align_up(region.len()));
105    log::info!("Invalidating boot region {aligned_region:#018x}");
106
107    if !aligned_region.is_empty() {
108        platform.validate_physical_page_range(aligned_region, PageValidateOp::Invalidate)?;
109
110        if config.page_state_change_required() {
111            platform.page_state_change(
112                aligned_region,
113                PageSize::Regular,
114                PageStateChangeOp::Shared,
115            )?;
116        }
117    }
118
119    Ok(())
120}

122pub fn invalidate_early_boot_memory(
123    platform: &dyn SvsmPlatform,
124    config: &SvsmConfig<'_>,
125    launch_info: &KernelLaunchInfo,
126) -> Result<(), SvsmError> {
127    // Early boot memory must be invalidated after changing to the SVSM page
128    // page table to avoid invalidating page tables currently in use.  Always
129    // invalidate stage 2 memory, unless firmware is loaded into low memory.
130    // Also invalidate the boot data if required.
131    if !config.fw_in_low_memory() {
132        let lowmem_region = MemoryRegion::from_addresses(
133            PhysAddr::from(0u64),
134            PhysAddr::from(u64::from(LOWMEM_END)),
135        );
136        invalidate_boot_memory_region(platform, config, lowmem_region)?;
137    }
138
139    let stage2_base = PhysAddr::from(launch_info.stage2_start);
140    let stage2_end = PhysAddr::from(launch_info.stage2_end);
141    let stage2_region = MemoryRegion::from_addresses(stage2_base, stage2_end);
142    invalidate_boot_memory_region(platform, config, stage2_region)?;
143
144    let kernel_elf_size =
145        launch_info.kernel_elf_stage2_virt_end - launch_info.kernel_elf_stage2_virt_start;
146    let kernel_elf_region = MemoryRegion::new(
147        PhysAddr::new(launch_info.kernel_elf_stage2_virt_start.try_into().unwrap()),
148        kernel_elf_size.try_into().unwrap(),
149    );
150    invalidate_boot_memory_region(platform, config, kernel_elf_region)?;
151
152    let kernel_fs_size = launch_info.kernel_fs_end - launch_info.kernel_fs_start;
153    if kernel_fs_size > 0 {
154        let kernel_fs_region = MemoryRegion::new(
155            PhysAddr::new(launch_info.kernel_fs_start.try_into().unwrap()),
156            kernel_fs_size.try_into().unwrap(),
157        );
158        invalidate_boot_memory_region(platform, config, kernel_fs_region)?;
159    }
160
161    if launch_info.stage2_igvm_params_size > 0 {
162        let igvm_params_region = MemoryRegion::new(
163            PhysAddr::new(launch_info.stage2_igvm_params_phys_addr.try_into().unwrap()),
164            launch_info.stage2_igvm_params_size as usize,
165        );
166        invalidate_boot_memory_region(platform, config, igvm_params_region)?;
167    }
168
169    Ok(())
170}
