44/// # Safety
45///
46/// The caller must have verified that the given physical memory region is
47/// private to the CVM and that it does not alias SVSM memory.
48unsafe fn validate_fw_mem_region(
49    boot_params: &BootParams<'_>,
50    region: MemoryRegion<PhysAddr>,
51) -> Result<(), SvsmError> {
52    let pstart = region.start();
53    let pend = region.end();
54
55    log::info!("Validating {pstart:#018x}-{pend:#018x}");
56
57    if boot_params.page_state_change_required() {
58        current_ghcb()
59            .page_state_change(region, PageStateChangeOp::Private)
60            .expect("GHCB PSC call failed to validate firmware memory");
61    }
62
63    for paddr in region.iter_pages(PageSize::Regular) {
64        let guard = PerCPUPageMappingGuard::create_4k(paddr)?;
65        let vaddr = guard.virt_addr();
66
67        // SAFETY: the virtual address mapping is known to point to the guest
68        // physical address range supplied by the caller.
69        unsafe {
70            pvalidate(vaddr, PageSize::Regular, PvalidateOp::Valid)?;
71
72            // Make page accessible to guest VMPL
73            rmp_adjust(
74                vaddr,
75                RMPFlags::GUEST_VMPL | RMPFlags::RWX,
76                PageSize::Regular,
77            )?;
78
79            zero_mem_region(vaddr, vaddr + PAGE_SIZE);
80        }
81    }
82
83    Ok(())
84}

86/// # Safety
87///
88/// The caller must have verified that the given physical memory regions are
89/// private to the CVM and that they do not alias SVSM memory.
90unsafe fn validate_fw_memory_vec(
91    boot_params: &BootParams<'_>,
92    regions: Vec<MemoryRegion<PhysAddr>>,
93) -> Result<(), SvsmError> {
94    if regions.is_empty() {
95        return Ok(());
96    }
97
98    let mut next_vec = Vec::new();
99    let mut region = regions[0];
100
101    for next in regions.into_iter().skip(1) {
102        if region.contiguous(&next) {
103            region = region.merge(&next);
104        } else {
105            next_vec.push(next);
106        }
107    }
108    // SAFETY: the caller must uphold the safety requirements
109    unsafe {
110        validate_fw_mem_region(boot_params, region)?;
111        validate_fw_memory_vec(boot_params, next_vec)
112    }
113}

115pub fn validate_fw_memory(
116    boot_params: &BootParams<'_>,
117    fw_meta: &SevFWMetaData,
118    kernel_region: &MemoryRegion<PhysAddr>,
119) -> Result<(), SvsmError> {
120    // Initalize vector with regions from the FW
121    let mut regions = fw_meta.valid_mem.clone();
122
123    // Add region for CPUID page if present
124    if let Some(cpuid_paddr) = fw_meta.cpuid_page {
125        regions.push(MemoryRegion::new(cpuid_paddr, PAGE_SIZE));
126    }
127
128    // Add region for Secrets page if present
129    if let Some(secrets_paddr) = fw_meta.secrets_page {
130        regions.push(MemoryRegion::new(secrets_paddr, PAGE_SIZE));
131    }
132
133    // Add region for CAA page if present
134    if let Some(caa_paddr) = fw_meta.caa_page {
135        regions.push(MemoryRegion::new(caa_paddr, PAGE_SIZE));
136    }
137
138    // Sort regions by base address
139    regions.sort_unstable_by_key(|a| a.start());
140
141    for region in regions.iter() {
142        if region.overlap(kernel_region) {
143            log::error!("FwMeta region ovelaps with kernel");
144            return Err(SvsmError::Firmware);
145        }
146    }
147
148    // SAFETY: we've just checked that the firmware regions do not overlap with
149    // the SVSM kernel
150    unsafe { validate_fw_memory_vec(boot_params, regions) }
151}

153pub fn print_fw_meta(fw_meta: &SevFWMetaData) {
154    log::info!("FW Meta Data");
155
156    match fw_meta.cpuid_page {
157        Some(addr) => log::info!("  CPUID Page   : {addr:#010x}"),
158        None => log::info!("  CPUID Page   : None"),
159    };
160
161    match fw_meta.secrets_page {
162        Some(addr) => log::info!("  Secrets Page : {addr:#010x}"),
163        None => log::info!("  Secrets Page : None"),
164    };
165
166    match fw_meta.caa_page {
167        Some(addr) => log::info!("  CAA Page     : {addr:#010x}"),
168        None => log::info!("  CAA Page     : None"),
169    };
170
171    for region in &fw_meta.valid_mem {
172        log::info!("  Pre-Validated Region {region:#018x}");
173    }
174}

176/// # Safety
177///
178/// The caller must ensure that the given physical address is valid and writing
179/// to it does not violate Rust's memory model.
180unsafe fn copy_cpuid_table_to_fw(fw_addr: PhysAddr) -> Result<(), SvsmError> {
181    let guard = PerCPUPageMappingGuard::create_4k(fw_addr)?;
182
183    // SAFETY: the caller must uphold the safety requirements.
184    unsafe {
185        copy_cpuid_table_to(guard.virt_addr());
186    }
187
188    Ok(())
189}

191/// # Safety
192///
193/// The caller must ensure that the given physical address is valid and writing
194/// to it does not violate Rust's memory model.
195unsafe fn copy_secrets_page_to_fw(
196    fw_addr: PhysAddr,
197    caa_addr: PhysAddr,
198    kernel_region: &MemoryRegion<PhysAddr>,
199) -> Result<(), SvsmError> {
200    let guard = PerCPUPageMappingGuard::create_4k(fw_addr)?;
201    let start = guard.virt_addr();
202
203    // Zero target
204    // SAFETY: we trust PerCPUPageMappingGuard::create_4k() to return a
205    // valid pointer to a correctly mapped region of size PAGE_SIZE.
206    unsafe {
207        zero_mem_region(start, start + PAGE_SIZE);
208    }
209
210    // Copy secrets page
211    let mut fw_secrets_page = secrets_page().unwrap().copy_for_vmpl(GUEST_VMPL);
212
213    fw_secrets_page.set_svsm_data(
214        kernel_region.start().into(),
215        kernel_region.len().try_into().unwrap(),
216        u64::from(caa_addr),
217    );
218
219    // SAFETY: start points to a new allocated and zeroed page.
220    unsafe {
221        fw_secrets_page.copy_to(start);
222    }
223
224    Ok(())
225}

244/// # Safety
245///
246/// The caller must make sure that the physical addresses in `fw_meta` are safe to
247/// write to without breaking Rust's memory model.
248pub unsafe fn copy_tables_to_fw(
249    fw_meta: &SevFWMetaData,
250    kernel_region: &MemoryRegion<PhysAddr>,
251) -> Result<(), SvsmError> {
252    if let Some(addr) = fw_meta.cpuid_page {
253        // SAFETY: caller must uphold the safety requirements
254        unsafe { copy_cpuid_table_to_fw(addr) }?;
255    }
256
257    let secrets_page = fw_meta.secrets_page.ok_or(SvsmError::MissingSecrets)?;
258    let caa_page = fw_meta.caa_page.ok_or(SvsmError::MissingCAA)?;
259
260    // SAFETY: caller must uphold the safety requirements.
261    unsafe {
262        copy_secrets_page_to_fw(secrets_page, caa_page, kernel_region)?;
263        zero_caa_page(caa_page)?;
264    }
265
266    Ok(())
267}

269/// # Safety
270///
271/// The caller must have verified that the firmware physical memory regions in
272/// `boot_params` are private to the CVM and that they do not alias SVSM
273/// memory.
274pub unsafe fn validate_fw(boot_params: &BootParams<'_>) -> Result<(), SvsmError> {
275    let fw_regions = boot_params.get_fw_regions();
276
277    for (i, region) in fw_regions.into_iter().enumerate() {
278        log::info!("Firmware region {i} at {region:#018x}");
279
280        for paddr in region.iter_pages(PageSize::Regular) {
281            let guard = PerCPUPageMappingGuard::create_4k(paddr)?;
282            let vaddr = guard.virt_addr();
283            // SAFETY: the address is known to be a guest page.
284            if let Err(e) = unsafe {
285                rmp_adjust(
286                    vaddr,
287                    RMPFlags::GUEST_VMPL | RMPFlags::RWX,
288                    PageSize::Regular,
289                )
290            } {
291                log::info!("rmpadjust failed for addr {vaddr:#018x}");
292                return Err(e);
293            }
294        }
295    }
296
297    Ok(())
298}

300pub fn prepare_fw_launch(fw_meta: &SevFWMetaData) -> Result<(), SvsmError> {
301    if let Some(caa) = fw_meta.caa_page {
302        this_cpu_shared().update_guest_caa(caa);
303    }
304
305    this_cpu().alloc_guest_vmsa()?;
306    this_cpu().update_guest_mappings()?;
307
308    Ok(())
309}

311pub fn launch_fw(boot_params: &BootParams<'_>) -> Result<(), SvsmError> {
312    let cpu = this_cpu();
313    let mut vmsa_ref = cpu.guest_vmsa_ref();
314    let vmsa_pa = vmsa_ref.vmsa_phys().unwrap();
315    let vmsa = vmsa_ref.vmsa();
316
317    boot_params.initialize_guest_vmsa(vmsa)?;
318
319    log::info!("VMSA PA: {vmsa_pa:#x}");
320
321    let sev_features = vmsa.sev_features;
322
323    log::info!("Launching Firmware");
324    current_ghcb().register_guest_vmsa(vmsa_pa, 0, GUEST_VMPL as u64, sev_features)?;
325
326    Ok(())
