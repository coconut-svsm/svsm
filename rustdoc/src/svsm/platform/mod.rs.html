69/// This defines a platform abstraction to permit the SVSM to run on different
70/// underlying architectures.
71pub trait SvsmPlatform: Sync {
72    #[cfg(test)]
73    fn platform_type(&self) -> SvsmPlatformType;
74
75    /// Halts the system required by the platform.  Interrupt state is under
76    /// control of the caller.
77    fn halt()
78    where
79        Self: Sized,
80    {
81        // SAFETY: executing HLT in assembly is always safe.
82        unsafe {
83            asm!("hlt");
84        }
85    }
86
87    /// Halts the system with interrupts enabled as required by the platform.
88    /// # Arguments
89    /// _irq_guard: an IRQ guard structure.  This is not actually used, but
90    /// serves as proof that interrupts have been correctly disabled so that
91    /// interrupt state can be correctly manipulated as required by the idle
92    /// halt action.
93    fn idle_halt(&self, _irq_guard: &IrqGuard);
94
95    /// Performs basic early initialization of the runtime environment.
96    fn env_setup(&mut self, debug_serial_port: u16, vtom: usize) -> Result<(), SvsmError>;
97
98    /// Performs initialization of the platform runtime environment after
99    /// the core system environment has been initialized.
100    fn env_setup_late(&mut self, debug_serial_port: u16) -> Result<(), SvsmError>;
101
102    /// Performs initialiation of the environment specfic to the SVSM kernel
103    /// (for services not used by stage2).
104    fn env_setup_svsm(&self) -> Result<(), SvsmError>;
105
106    /// Performs the necessary preparations for launching guest boot firmware.
107    fn prepare_fw(
108        &self,
109        _config: &SvsmConfig<'_>,
110        _kernel_region: MemoryRegion<PhysAddr>,
111    ) -> Result<(), SvsmError> {
112        Ok(())
113    }
114
115    /// Launches guest boot firmware.
116    fn launch_fw(&self, _config: &SvsmConfig<'_>) -> Result<(), SvsmError> {
117        Ok(())
118    }
119
120    /// Completes initialization of a per-CPU object during construction.
121    fn setup_percpu(&self, cpu: &PerCpu) -> Result<(), SvsmError>;
122
123    /// Completes initialization of a per-CPU object on the target CPU.
124    fn setup_percpu_current(&self, cpu: &PerCpu) -> Result<(), SvsmError>;
125
126    /// Determines the paging encryption masks for the current architecture.
127    fn get_page_encryption_masks(&self) -> PageEncryptionMasks;
128
129    /// Determine whether shadow stacks are supported.
130    fn determine_cet_support(&self) -> bool {
131        determine_cet_support_from_cpuid()
132    }
133
134    /// Get the features and the capabilities of the platform.
135    fn capabilities(&self) -> Caps;
136
137    /// Enable platform-specific Hyper-V hypercall operations.
138    fn setup_hyperv_hypercalls(&self) -> Result<(), SvsmError> {
139        Ok(())
140    }
141
142    /// Perform a hypercall.
143    /// # Safety
144    /// Hypercalls may have side-effects that affect the integrity of the
145    /// system, and the caller must take responsibility for ensuring that the
146    /// hypercall operation is safe.
147    unsafe fn hypercall(
148        &self,
149        input_control: hyperv::HvHypercallInput,
150        hypercall_pages: &hyperv::HypercallPagesGuard<'_>,
151    ) -> hyperv::HvHypercallOutput;
152
153    /// Obtain CPUID using platform-specific tables.
154    fn cpuid(&self, eax: u32, ecx: u32) -> Option<CpuidResult>;
155
156    /// Write a host-owned MSR.
157    /// # Safety
158    /// The caller must ensure that the requested MSR modification does mot
159    /// affect memory safety.
160    unsafe fn write_host_msr(&self, msr: u32, value: u64);
161
162    /// Establishes state required for guest/host communication.
163    fn setup_guest_host_comm(&mut self, _cpu: &PerCpu, _is_bsp: bool) {}
164
165    /// Obtains a reference to an I/O port implemetation appropriate to the
166    /// platform.
167    fn get_io_port(&self) -> &'static dyn IOPort;
168
169    /// Validates low memory below the specified physical address, with the
170    /// exception of addresses reserved for use by the platform object which
171    /// may pre-validated.  Intended only for use during early boot.
172    /// # Safety
173    /// The caller is required to ensure that it is safe to validate low
174    /// memory.
175    unsafe fn validate_low_memory(&self, addr: u64) -> Result<(), SvsmError> {
176        // SAFETY: the caller takes responsibility for the safety of the
177        // validation operation.
178        unsafe {
179            self.validate_virtual_page_range(
180                MemoryRegion::new(VirtAddr::from(0u64), addr as usize),
181                PageValidateOp::Validate,
182            )
183        }
184    }
185
186    /// Performs a page state change between private and shared states.
187    fn page_state_change(
188        &self,
189        region: MemoryRegion<PhysAddr>,
190        size: PageSize,
191        op: PageStateChangeOp,
192    ) -> Result<(), SvsmError>;
193
194    /// Marks a physical range of pages as valid or invalid for use as private
195    /// pages.  Not usable in stage2.
196    ///
197    /// # Safety
198    ///
199    /// See the safety considerations for [SvsmPlatform::validate_virtual_page_range].
200    unsafe fn validate_physical_page_range(
201        &self,
202        region: MemoryRegion<PhysAddr>,
203        op: PageValidateOp,
204    ) -> Result<(), SvsmError>;
205
206    /// Marks a virtual range of pages as valid or invalid for use as private
207    /// pages.  Provided primarily for use in stage2 where validation by
208    /// physical address cannot be supported.
209    /// # Safety
210    /// The caller is required to ensure the safety of the validation operation
211    /// on this memory range.
212    unsafe fn validate_virtual_page_range(
213        &self,
214        region: MemoryRegion<VirtAddr>,
215        op: PageValidateOp,
216    ) -> Result<(), SvsmError>;
217
218    /// Performs a system-wide TLB flush.
219    fn flush_tlb(&self, flush_scope: &TlbFlushScope) {
220        flush_tlb(flush_scope);
221    }
222
223    /// Configures the use of alternate injection as requested.
224    fn configure_alternate_injection(&mut self, alt_inj_requested: bool) -> Result<(), SvsmError>;
225
226    /// Changes the state of APIC registration on this system, returning either
227    /// the current registration state or an error.
228    fn change_apic_registration_state(&self, incr: bool) -> Result<bool, SvsmError>;
229
230    /// Queries the state of APIC registration on this system.
231    fn query_apic_registration_state(&self) -> bool;
232
233    /// Determines whether the platform supports interrupts to the SVSM.
234    fn use_interrupts(&self) -> bool;
235
236    /// Determines whether a given interrupt vector was invoked as an external
237    /// interrupt.
238    fn is_external_interrupt(&self, vector: usize) -> bool;
239
240    /// Start an additional processor.
241    fn start_cpu(&self, cpu: &PerCpu, start_rip: u64) -> Result<(), SvsmError>;
242
243    /// Indicates whether this platform should invoke the SVSM request loop.
244    fn start_svsm_request_loop(&self) -> bool {
245        false
246    }
247
248    /// Perfrom a write to a memory-mapped IO area
249    ///
250    /// # Safety
251    ///
252    /// Caller must ensure that `vaddr` points to a properly aligned memory location and the
253    /// memory accessed is part of a valid MMIO range.
254    unsafe fn mmio_write(&self, vaddr: VirtAddr, data: &[u8]) -> Result<(), SvsmError>;
255
256    /// Perfrom a read from a memory-mapped IO area
257    ///
258    /// # Safety
259    ///
260    /// Caller must ensure that `vaddr` points to a properly aligned memory location and the
261    /// memory accessed is part of a valid MMIO range.
262    unsafe fn mmio_read(
263        &self,
264        vaddr: VirtAddr,
265        data: &mut [MaybeUninit<u8>],
266    ) -> Result<(), SvsmError>;
267
268    /// Terminates the guest.
269    fn terminate() -> !
270    where
271        Self: Sized;
272}

275#[derive(Clone, Copy, Debug)]
276pub enum SvsmPlatformCell {
277    Snp(SnpPlatform),
278    Tdp(TdpPlatform),
279    Native(NativePlatform),
280}
281
282impl SvsmPlatformCell {
283    pub fn new(suppress_svsm_interrupts: bool) -> Self {
284        match *SVSM_PLATFORM_TYPE {
285            SvsmPlatformType::Native => {
286                SvsmPlatformCell::Native(NativePlatform::new(suppress_svsm_interrupts))
287            }
288            SvsmPlatformType::Snp => {
289                SvsmPlatformCell::Snp(SnpPlatform::new(suppress_svsm_interrupts))
290            }
291            SvsmPlatformType::Tdp => {
292                SvsmPlatformCell::Tdp(TdpPlatform::new(suppress_svsm_interrupts))
293            }
294        }
295    }
296
297    pub fn global_init(self) {
298        SVSM_PLATFORM_CELL
299            .init(self)
300            .expect("Failed to initialize SVSM platform cell");
301        SVSM_PLATFORM
302            .init(SVSM_PLATFORM_CELL.platform())
303            .expect("Failed to initialize SVSM platform");
304    }
305
306    pub fn platform(&self) -> &dyn SvsmPlatform {
307        match self {
308            SvsmPlatformCell::Native(platform) => platform,
309            SvsmPlatformCell::Snp(platform) => platform,
310            SvsmPlatformCell::Tdp(platform) => platform,
311        }
312    }
313
314    pub fn platform_mut(&mut self) -> &mut dyn SvsmPlatform {
315        match self {
316            SvsmPlatformCell::Native(platform) => platform,
317            SvsmPlatformCell::Snp(platform) => platform,
318            SvsmPlatformCell::Tdp(platform) => platform,
319        }
320    }
321}

329        None
330    }
331
332    /// Obtains the virtual address of the secrets page, if any.

334        None

338#[derive(Debug)]
339pub enum Stage2PlatformCell {
340    Snp(SnpStage2Platform),
341    Tdp(TdpStage2Platform),
342    Native(NativeStage2Platform),
343}
344
345impl Stage2PlatformCell {
346    pub fn new(platform_type: SvsmPlatformType) -> Self {
347        match platform_type {
348            SvsmPlatformType::Native => Stage2PlatformCell::Native(NativeStage2Platform::new()),
349            SvsmPlatformType::Snp => Stage2PlatformCell::Snp(SnpStage2Platform::new()),
350            SvsmPlatformType::Tdp => Stage2PlatformCell::Tdp(TdpStage2Platform::new()),
351        }
352    }
353
354    pub fn platform(&self) -> &dyn Stage2Platform {
355        match self {
356            Stage2PlatformCell::Native(platform) => platform,
357            Stage2PlatformCell::Snp(platform) => platform,
358            Stage2PlatformCell::Tdp(platform) => platform,
359        }
360    }
361}

363pub fn init_platform_type(platform_type: SvsmPlatformType) {
364    SVSM_PLATFORM_TYPE.init(platform_type).unwrap();
365}
366
367pub fn init_capabilities() {
368    let caps = SVSM_PLATFORM.capabilities();
369    CAPS.init(caps).unwrap();
370}
371
372pub fn halt() {
373    // Use a platform-specific halt.  However, the SVSM_PLATFORM global may not
374    // yet be initialized, so go choose the halt implementation based on the
375    // platform-specific halt instead.
376    match *SVSM_PLATFORM_TYPE {
377        SvsmPlatformType::Native => NativePlatform::halt(),
378        SvsmPlatformType::Snp => SnpPlatform::halt(),
379        SvsmPlatformType::Tdp => TdpPlatform::halt(),
380    }
381}
382
383/// Terminates the guest with a platform-specific mechanism
384pub fn terminate() -> ! {
385    match *SVSM_PLATFORM_TYPE {
386        SvsmPlatformType::Native => NativePlatform::terminate(),
387        SvsmPlatformType::Snp => SnpPlatform::terminate(),
388        SvsmPlatformType::Tdp => TdpPlatform::terminate(),
389    }
