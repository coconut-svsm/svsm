76/// This defines a platform abstraction to permit the SVSM to run on different
77/// underlying architectures.
78pub trait SvsmPlatform: Sync {
79    #[cfg(test)]
80    fn platform_type(&self) -> SvsmPlatformType;
81
82    /// Halts the system required by the platform.  Interrupt state is under
83    /// control of the caller.
84    fn halt()
85    where
86        Self: Sized,
87    {
88        // SAFETY: executing HLT in assembly is always safe.
89        unsafe {
90            asm!("hlt");
91        }
92    }
93
94    /// Halts the system with interrupts enabled as required by the platform.
95    /// # Arguments
96    /// _irq_guard: an IRQ guard structure.  This is not actually used, but
97    /// serves as proof that interrupts have been correctly disabled so that
98    /// interrupt state can be correctly manipulated as required by the idle
99    /// halt action.
100    fn idle_halt(&self, _irq_guard: &IrqGuard);
101
102    /// Performs basic early initialization of the runtime environment.
103    fn env_setup(&mut self, debug_serial_port: u16, vtom: usize) -> Result<(), SvsmError>;
104
105    /// Initializes a platform-specific page.
106    /// # Safety
107    /// The caller must specify a valid virtual address for the specified type
108    /// of page.
109    unsafe fn initialize_platform_page(&self, _page_type: PlatformPageType, _vaddr: VirtAddr) {
110        // By default, no action is required.
111    }
112
113    /// Performs initialization of the platform runtime environment after
114    /// the core system environment has been initialized.
115    fn env_setup_late(&mut self, debug_serial_port: u16) -> Result<(), SvsmError>;
116
117    /// Performs initialiation of the environment specfic to the SVSM kernel
118    /// (for services not used by stage2).
119    fn env_setup_svsm(&self) -> Result<(), SvsmError>;
120
121    /// Frees a platforms-specific page if it is not used by the underlying
122    /// platform.
123    /// # Safety
124    /// The caller must specify a valid virtual address for the specified type
125    /// of page.
126    unsafe fn free_unused_platform_page(&self, _page_type: PlatformPageType, vaddr: VirtAddr) {
127        free_page(vaddr);
128    }
129
130    /// Performs the necessary preparations for launching guest boot firmware.
131    fn prepare_fw(
132        &self,
133        _boot_params: &BootParams<'_>,
134        _kernel_region: MemoryRegion<PhysAddr>,
135    ) -> Result<(), SvsmError> {
136        Ok(())
137    }
138
139    /// Launches guest boot firmware.
140    fn launch_fw(&self, _boot_params: &BootParams<'_>) -> Result<(), SvsmError> {
141        Ok(())
142    }
143
144    /// Completes initialization of a per-CPU object during construction.
145    fn setup_percpu(&self, cpu: &PerCpu) -> Result<(), SvsmError>;
146
147    /// Completes initialization of a per-CPU object on the target CPU.
148    fn setup_percpu_current(&self, cpu: &PerCpu) -> Result<(), SvsmError>;
149
150    /// Determines the paging encryption masks for the current architecture.
151    fn get_page_encryption_masks(&self) -> PageEncryptionMasks;
152
153    /// Determine whether shadow stacks are supported.
154    fn determine_cet_support(&self) -> bool {
155        determine_cet_support_from_cpuid()
156    }
157
158    /// Get the features and the capabilities of the platform.
159    fn capabilities(&self) -> Caps;
160
161    /// Enable platform-specific Hyper-V hypercall operations.
162    fn setup_hyperv_hypercalls(&self) -> Result<(), SvsmError> {
163        Ok(())
164    }
165
166    /// Perform a hypercall.
167    /// # Safety
168    /// Hypercalls may have side-effects that affect the integrity of the
169    /// system, and the caller must take responsibility for ensuring that the
170    /// hypercall operation is safe.
171    unsafe fn hypercall(
172        &self,
173        input_control: hyperv::HvHypercallInput,
174        hypercall_pages: &hyperv::HypercallPagesGuard<'_>,
175    ) -> hyperv::HvHypercallOutput;
176
177    /// Obtain CPUID using platform-specific tables.
178    fn cpuid(&self, eax: u32, ecx: u32) -> Option<CpuidResult>;
179
180    /// Write a host-owned MSR.
181    /// # Safety
182    /// The caller must ensure that the requested MSR modification does mot
183    /// affect memory safety.
184    unsafe fn write_host_msr(&self, msr: u32, value: u64);
185
186    /// Establishes state required for guest/host communication.
187    fn setup_guest_host_comm(&mut self, _cpu: &PerCpu, _is_bsp: bool) {}
188
189    /// Obtains a reference to an I/O port implemetation appropriate to the
190    /// platform.
191    fn get_io_port(&self) -> &'static dyn IOPort;
192
193    /// Validates low memory below the specified physical address, with the
194    /// exception of addresses reserved for use by the platform object which
195    /// may pre-validated.  Intended only for use during early boot.
196    /// # Safety
197    /// The caller is required to ensure that it is safe to validate low
198    /// memory.
199    unsafe fn validate_low_memory(&self, addr: u64) -> Result<(), SvsmError> {
200        // SAFETY: the caller takes responsibility for the safety of the
201        // validation operation.
202        unsafe {
203            self.validate_virtual_page_range(
204                MemoryRegion::new(VirtAddr::from(0u64), addr as usize),
205                PageValidateOp::Validate,
206            )
207        }
208    }
209
210    /// Performs a page state change between private and shared states.
211    fn page_state_change(
212        &self,
213        region: MemoryRegion<PhysAddr>,
214        op: PageStateChangeOp,
215    ) -> Result<(), SvsmError>;
216
217    /// Marks a physical range of pages as valid or invalid for use as private
218    /// pages.  Not usable in stage2.
219    ///
220    /// # Safety
221    ///
222    /// See the safety considerations for [SvsmPlatform::validate_virtual_page_range].
223    unsafe fn validate_physical_page_range(
224        &self,
225        region: MemoryRegion<PhysAddr>,
226        op: PageValidateOp,
227    ) -> Result<(), SvsmError>;
228
229    /// Marks a virtual range of pages as valid or invalid for use as private
230    /// pages.  Provided primarily for use in stage2 where validation by
231    /// physical address cannot be supported.
232    /// # Safety
233    /// The caller is required to ensure the safety of the validation operation
234    /// on this memory range.
235    unsafe fn validate_virtual_page_range(
236        &self,
237        region: MemoryRegion<VirtAddr>,
238        op: PageValidateOp,
239    ) -> Result<(), SvsmError>;
240
241    /// Performs a system-wide TLB flush.
242    fn flush_tlb(&self, flush_scope: &TlbFlushScope) {
243        flush_tlb(flush_scope);
244    }
245
246    /// Configures the use of alternate injection as requested.
247    fn configure_alternate_injection(&mut self, alt_inj_requested: bool) -> Result<(), SvsmError>;
248
249    /// Changes the state of APIC registration on this system, returning either
250    /// the current registration state or an error.
251    fn change_apic_registration_state(&self, incr: bool) -> Result<bool, SvsmError>;
252
253    /// Queries the state of APIC registration on this system.
254    fn query_apic_registration_state(&self) -> bool;
255
256    /// Determines whether the platform supports interrupts to the SVSM.
257    fn use_interrupts(&self) -> bool;
258
259    /// Determines whether a given interrupt vector was invoked as an external
260    /// interrupt.
261    fn is_external_interrupt(&self, vector: usize) -> bool;
262
263    /// Start an additional processor.
264    fn start_cpu(
265        &self,
266        cpu: &PerCpu,
267        start_rip: u64,
268        transition_page_table: &TransitionPageTable,
269    ) -> Result<(), SvsmError>;
270
271    /// Indicates whether this platform should invoke the SVSM request loop.
272    fn start_svsm_request_loop(&self) -> bool {
273        false
274    }
275
276    /// Perfrom a write to a memory-mapped IO area
277    ///
278    /// # Safety
279    ///
280    /// Caller must ensure that `vaddr` points to a properly aligned memory location and the
281    /// memory accessed is part of a valid MMIO range.
282    unsafe fn mmio_write(&self, vaddr: VirtAddr, data: &[u8]) -> Result<(), SvsmError>;
283
284    /// Perfrom a read from a memory-mapped IO area
285    ///
286    /// # Safety
287    ///
288    /// Caller must ensure that `vaddr` points to a properly aligned memory location and the
289    /// memory accessed is part of a valid MMIO range.
290    unsafe fn mmio_read(
291        &self,
292        vaddr: VirtAddr,
293        data: &mut [MaybeUninit<u8>],
294    ) -> Result<(), SvsmError>;
295
296    /// Terminates the guest.
297    fn terminate() -> !
298    where
299        Self: Sized;
300}

302//FIXME - remove Copy trait
303#[derive(Clone, Copy, Debug)]
304pub enum SvsmPlatformCell {
305    Snp(SnpPlatform),
306    Tdp(TdpPlatform),
307    Native(NativePlatform),
308}
309
310impl SvsmPlatformCell {
311    pub fn new(suppress_svsm_interrupts: bool) -> Self {
312        match *SVSM_PLATFORM_TYPE {
313            SvsmPlatformType::Native => {
314                SvsmPlatformCell::Native(NativePlatform::new(suppress_svsm_interrupts))
315            }
316            SvsmPlatformType::Snp => {
317                SvsmPlatformCell::Snp(SnpPlatform::new(suppress_svsm_interrupts))
318            }
319            SvsmPlatformType::Tdp => {
320                SvsmPlatformCell::Tdp(TdpPlatform::new(suppress_svsm_interrupts))
321            }
322        }
323    }
324
325    pub fn global_init(self) {
326        SVSM_PLATFORM_CELL
327            .init(self)
328            .expect("Failed to initialize SVSM platform cell");
329        SVSM_PLATFORM
330            .init(SVSM_PLATFORM_CELL.platform())
331            .expect("Failed to initialize SVSM platform");
332    }
333
334    pub fn platform(&self) -> &dyn SvsmPlatform {
335        match self {
336            SvsmPlatformCell::Native(platform) => platform,
337            SvsmPlatformCell::Snp(platform) => platform,
338            SvsmPlatformCell::Tdp(platform) => platform,
339        }
340    }
341
342    pub fn platform_mut(&mut self) -> &mut dyn SvsmPlatform {
343        match self {
344            SvsmPlatformCell::Native(platform) => platform,
345            SvsmPlatformCell::Snp(platform) => platform,
346            SvsmPlatformCell::Tdp(platform) => platform,
347        }
348    }
349
350    pub fn platform_type(&self) -> SvsmPlatformType {
351        match self {
352            SvsmPlatformCell::Native(_) => SvsmPlatformType::Native,
353            SvsmPlatformCell::Snp(_) => SvsmPlatformType::Snp,
354            SvsmPlatformCell::Tdp(_) => SvsmPlatformType::Tdp,
355        }
356    }
357}

359/// This defines a platform abstraction to permit stage2 to run on different
360/// underlying architectures.  It includes only functionality that is used
361/// only in stage2.
362pub trait Stage2Platform {
363    /// Obtains the virtual address of the CPUID page, if any.
364    fn get_cpuid_page(&self, _launch_info: &Stage2LaunchInfo) -> Option<VirtAddr> {
365        None
366    }
367
368    /// Obtains the virtual address of the secrets page, if any.
369    fn get_secrets_page(&self, _launch_info: &Stage2LaunchInfo) -> Option<VirtAddr> {
370        None
371    }
372}

374#[derive(Debug)]
375pub enum Stage2PlatformCell {
376    Snp(SnpStage2Platform),
377    Tdp(TdpStage2Platform),
378    Native(NativeStage2Platform),
379}
380
381impl Stage2PlatformCell {
382    pub fn new(platform_type: SvsmPlatformType) -> Self {
383        match platform_type {
384            SvsmPlatformType::Native => Stage2PlatformCell::Native(NativeStage2Platform::new()),
385            SvsmPlatformType::Snp => Stage2PlatformCell::Snp(SnpStage2Platform::new()),
386            SvsmPlatformType::Tdp => Stage2PlatformCell::Tdp(TdpStage2Platform::new()),
387        }
388    }
389
390    pub fn platform(&self) -> &dyn Stage2Platform {
391        match self {
392            Stage2PlatformCell::Native(platform) => platform,
393            Stage2PlatformCell::Snp(platform) => platform,
394            Stage2PlatformCell::Tdp(platform) => platform,
395        }
396    }
397}
398
399pub fn init_platform_type(platform_type: SvsmPlatformType) {
400    SVSM_PLATFORM_TYPE.init(platform_type).unwrap();
401}
402
403pub fn init_capabilities() {
404    let caps = SVSM_PLATFORM.capabilities();
405    CAPS.init(caps).unwrap();
406}
407
408pub fn halt() {
409    // Use a platform-specific halt.  However, the SVSM_PLATFORM global may not
410    // yet be initialized, so go choose the halt implementation based on the
411    // platform-specific halt instead.
412    match *SVSM_PLATFORM_TYPE {
413        SvsmPlatformType::Native => NativePlatform::halt(),
414        SvsmPlatformType::Snp => SnpPlatform::halt(),
415        SvsmPlatformType::Tdp => TdpPlatform::halt(),
416    }
417}
418
419/// Terminates the guest with a platform-specific mechanism
420pub fn terminate() -> ! {
421    match *SVSM_PLATFORM_TYPE {
422        SvsmPlatformType::Native => NativePlatform::terminate(),
423        SvsmPlatformType::Snp => SnpPlatform::terminate(),
424        SvsmPlatformType::Tdp => TdpPlatform::terminate(),
425    }
