68#[bitfield(u64)]
69struct EptMappingInfo {
70    #[bits(12)]
71    pub flags: u64,
72    #[bits(52)]
73    pub page_frame_number: u64,
74}

76impl From<PageFrame> for EptMappingInfo {
77    fn from(frame: PageFrame) -> Self {
78        let (gpa, flags) = match frame {
79            PageFrame::Size4K(gpa) => (u64::from(gpa), 0),
80            PageFrame::Size2M(gpa) => (u64::from(gpa), 1),
81            PageFrame::Size1G(gpa) => (u64::from(gpa), 2),
82        };
83        Self::new()
84            .with_flags(flags)
85            .with_page_frame_number(gpa >> PAGE_SHIFT)
86    }
87}

89/// # Safety
90/// This function has the potential to zero arbitrary memory, so the caller is
91/// required to ensure that the supplied physical address range is appropriate
92/// for acceptance.
93unsafe fn tdg_mem_page_accept(frame: PageFrame) -> u64 {
94    loop {
95        // SAFETY: executing TDCALL requires the use of assembly.  The caller
96        // takes responsibility for correctness of the parameters.
97        let err = unsafe {
98            let mut ret: u64;
99            asm!("tdcall",
100                 in("rax") TDG_MEM_PAGE_ACCEPT,
101                 in("rcx") EptMappingInfo::from(frame).into_bits(),
102                 lateout("rax") ret,
103                 options(att_syntax));
104            ret
105        };
106        if !tdx_recoverable_error(err) {
107            return err;
108        }
109    }
110}

112/// # Safety
113/// This function will zero arbitrary memory, so the caller is required to
114/// ensure that the supplied physical address range is appropriate for
115/// acceptance.  The caller is additionally required to ensure that the address
116/// range is appropriate aligned to 4 KB boundaries.
117pub unsafe fn td_accept_physical_memory(region: MemoryRegion<PhysAddr>) -> Result<(), SvsmError> {
118    let mut addr = region.start();
119    if !addr.is_aligned(PAGE_SIZE) {
120        return Err(SvsmError::InvalidAddress);
121    }
122
123    let end = region.end();
124
125    while addr < end {
126        if addr.is_aligned(PAGE_SIZE_2M) && addr + PAGE_SIZE_2M <= end {
127            // SAFETY: the caller takes responsibility for the correct usage of
128            // the physical address.
129            let ret = unsafe { tdx_result(tdg_mem_page_accept(PageFrame::Size2M(addr))) };
130            match ret {
131                Ok(s) => {
132                    if s == TdxSuccess::PageAlreadyAccepted {
133                        // The caller is expected not to accept a page twice
134                        // unless doing so is known to be safe.  If the TDX
135                        // module indicates that the page was already accepted,
136                        // it must mean that the page was not removed after a
137                        // previous attempt to accept.  In this case, the page
138                        // must be zeroed now because the caller expects every
139                        // accepted page to be zeroed.
140                        // SAFETY: the caller takes responsibility for the
141                        // correct usage of the physical address.
142                        unsafe {
143                            let mapping =
144                                PerCPUPageMappingGuard::create(addr, addr + PAGE_SIZE_2M, 0)?;
145                            mapping
146                                .virt_addr()
147                                .as_mut_ptr::<u8>()
148                                .write_bytes(0, PAGE_SIZE_2M);
149                        }
150                    }
151                    addr = addr + PAGE_SIZE_2M;
152                    continue;
153                }
154                Err(TdxError::PageSizeMismatch) => {
155                    // Fall through to the 4 KB path below.
156                }
157                Err(e) => {
158                    return Err(e.into());
159                }
160            }
161        }
162
163        // SAFETY: the caller takes responsibility for the correct usage of the
164        // physical address.
165        let ret = unsafe { tdx_result(tdg_mem_page_accept(PageFrame::Size4K(addr))) };
166        match ret {
167            Ok(s) => {
168                if s == TdxSuccess::PageAlreadyAccepted {
169                    // Zero the 4 KB page.
170                    // SAFETY: the caller takes responsibility for the correct
171                    // usage of the physical address.
172                    unsafe {
173                        let mapping = PerCPUPageMappingGuard::create(addr, addr + PAGE_SIZE, 0)?;
174                        mapping
175                            .virt_addr()
176                            .as_mut_ptr::<u8>()
177                            .write_bytes(0, PAGE_SIZE);
178                    }
179                }
180                addr = addr + PAGE_SIZE;
181            }
182            Err(e) => return Err(e.into()),
183        }
184    }
185
186    Ok(())
187}

189/// # Safety
190/// This function will zero arbitrary memory, so the caller is required to
191/// ensure that the supplied virtual address range is appropriate for
192/// acceptance.  The caller is additionally required to ensure that the address
193/// range is appropriate aligned to 4 KB boundaries.
194unsafe fn td_accept_virtual_4k(vaddr: VirtAddr, paddr: PhysAddr) -> Result<(), SvsmError> {
195    // SAFETY: the caller takes responsibility for the correct usage of the
196    // physical address.
197    let ret = unsafe { tdx_result(tdg_mem_page_accept(PageFrame::Size4K(paddr))) };
198    match ret {
199        Ok(s) => {
200            if s == TdxSuccess::PageAlreadyAccepted {
201                // Zero the 4 KB page.
202                // SAFETY: the caller takes responsibility for the correct
203                // usage of the virtual address.
204                unsafe {
205                    vaddr.as_mut_ptr::<u8>().write_bytes(0, PAGE_SIZE);
206                }
207            }
208            Ok(())
209        }
210        Err(e) => Err(e.into()),
211    }
212}

214/// # Safety
215/// This function will zero arbitrary memory, so the caller is required to
216/// ensure that the supplied virtual address range is appropriate for
217/// acceptance.  The caller is additionally required to ensure that the address
218/// range is appropriate aligned to 4 KB boundaries.
219unsafe fn td_accept_virtual_2m(vaddr: VirtAddr, paddr: PhysAddr) -> Result<(), SvsmError> {
220    // SAFETY: the caller takes responsibility for the correct usage of the
221    // physical address.
222    let ret = unsafe { tdx_result(tdg_mem_page_accept(PageFrame::Size2M(paddr))) };
223    match ret {
224        Ok(s) => {
225            if s == TdxSuccess::PageAlreadyAccepted {
226                // Zero the 2M page.
227                // SAFETY: the caller takes responsibility for the correct
228                // usage of the virtual address.
229                unsafe {
230                    vaddr.as_mut_ptr::<u8>().write_bytes(0, PAGE_SIZE_2M);
231                }
232            }
233            Ok(())
234        }
235        Err(TdxError::PageSizeMismatch) => {
236            // Process this 2 MB page as a series of 4 KB pages.
237            for offset in 0usize..512usize {
238                // SAFETY: the caller takes responsibility for the correct
239                // usage of the physical address.
240                unsafe {
241                    td_accept_virtual_4k(
242                        vaddr + (offset << PAGE_SHIFT),
243                        paddr + (offset << PAGE_SHIFT),
244                    )?;
245                }
246            }
247            Ok(())
248        }
249        Err(e) => Err(e.into()),
250    }
251}

253/// # Safety
254/// This function will zero arbitrary memory, so the caller is required to
255/// ensure that the supplied virtual address range is appropriate for
256/// acceptance.  The caller is additionally required to ensure that the address
257/// range is appropriate aligned to 4 KB boundaries.
258pub unsafe fn td_accept_virtual_memory(region: MemoryRegion<VirtAddr>) -> Result<(), SvsmError> {
259    let mut vaddr = region.start();
260    if !vaddr.is_aligned(PAGE_SIZE) {
261        return Err(SvsmError::InvalidAddress);
262    }
263
264    let vaddr_end = region.end();
265    while vaddr < vaddr_end {
266        let frame = virt_to_frame(vaddr);
267        let size = if vaddr.is_aligned(PAGE_SIZE_2M)
268            && vaddr + PAGE_SIZE_2M <= vaddr_end
269            && frame.size() >= PAGE_SIZE_2M
270        {
271            // SAFETY: the caller takes responsibility for the correct usage of
272            // the virtual address.
273            unsafe {
274                td_accept_virtual_2m(vaddr, frame.address())?;
275            }
276            PAGE_SIZE_2M
277        } else {
278            // SAFETY: the caller takes responsibility for the correct usage of
279            // the virtual address.
280            unsafe {
281                td_accept_virtual_4k(vaddr, frame.address())?;
282            }
283            PAGE_SIZE
284        };
285
286        vaddr = vaddr + size;
287    }
288    Ok(())
289}

291pub fn tdcall_get_ve_info() -> Option<TdVeInfo> {
292    let mut out_rcx: u64;
293    let mut out_rdx: u64;
294    let mut out_r8: u64;
295    let mut out_r9: u64;
296    let mut out_r10: u64;
297    // SAFETY: executing TDCALL requires the use of assembly.
298    let err = unsafe {
299        let mut ret: u64;
300        asm!("tdcall",
301             in("rax") TDG_VP_VEINFO_GET,
302             lateout("rax") ret,
303             out("rcx") out_rcx,
304             out("rdx") out_rdx,
305             out("r8") out_r8,
306             out("r9") out_r9,
307             out("r10") out_r10,
308             options(att_syntax));
309        ret
310    };
311    match tdx_result(err) {
312        Ok(_) => Some(TdVeInfo {
313            exit_reason: out_rcx as u32,
314            exit_qualification: out_rdx,
315            gla: out_r8,
316            gpa: out_r9,
317            exit_instruction_length: out_r10 as u32,
318            exit_instruction_info: ((out_r10 >> 32) as u32),
319        }),
320        Err(TdxError::NoVeInfo) => None,
321        Err(e) => panic!("Unknown TD error: {e:?}"),
322    }
323}

325pub fn tdcall_vm_read(field: u64) -> u64 {
326    let (val, err) = loop {
327        let mut val: u64;
328        // SAFETY: executing TDCALL requires the use of assembly.
329        let err = unsafe {
330            let mut ret: u64;
331            asm!("tdcall",
332                 in("rax") TDG_VM_RD,
333                 in("rcx") 0,
334                 in("rdx") field,
335                 lateout("rax") ret,
336                 lateout("rdx") _,
337                 out("r8") val,
338                 options(att_syntax));
339            ret
340        };
341        if !tdx_recoverable_error(err) {
342            break (val, err);
343        }
344    };
345    // Ignore errors here since the caller cannot handle them.
346    debug_assert!(tdx_result(err).is_ok());
347    // val = 0 in case of no success.
348    val
349}

351pub fn tdvmcall_map_gpa(mut gpa: u64, size: u64) -> Result<(), TdxError> {
352    let pass_regs = TdxRegs::R10 | TdxRegs::R11 | TdxRegs::R12 | TdxRegs::R13;
353    let end = gpa + size;
354    loop {
355        let mut ret: u64;
356        let mut vmcall_ret: u64;
357        let mut retry_gpa: u64;
358        // SAFETY: executing TDCALL requires the use of assembly.
359        unsafe {
360            asm!("tdcall",
361                 in("rax") TDG_VP_TDVMCALL,
362                 in("rcx") pass_regs.bits(),
363                 in("r10") 0,
364                 in("r11") TDVMCALL_MAP_GPA,
365                 in("r12") gpa,
366                 in("r13") end - gpa,
367                 lateout("rax") ret,
368                 lateout("r10") vmcall_ret,
369                 lateout("r11") retry_gpa,
370                 options(att_syntax));
371        }
372
373        debug_assert!(tdx_result(ret).is_ok());
374        let err = tdvmcall_result(vmcall_ret);
375
376        // If a retry was requested, then reissue the call as requested by the
377        // host.  No validation is performed on this value because it is
378        // being passed to an untrusted host.
379        if err != Err(TdxError::Vmcall(Retry)) {
380            return err;
381        }
382
383        gpa = retry_gpa;
384    }
385}

388    let pass_regs =
389        TdxRegs::R10 | TdxRegs::R11 | TdxRegs::R12 | TdxRegs::R13 | TdxRegs::R14 | TdxRegs::R15;
390    let mut ret: u64;
391    let mut vmcall_ret: u64;
392    let mut result_eax: u32;
393    let mut result_ebx: u32;
394    let mut result_ecx: u32;
395    let mut result_edx: u32;
396    // SAFETY: executing TDCALL requires the use of assembly.
397    unsafe {
398        asm!("tdcall",
399             in("rax") TDG_VP_TDVMCALL,
400             in("rcx") pass_regs.bits(),
401             in("r10") 0,
402             in("r11") TDVMCALL_CPUID,
403             in("r12") cpuid_fn,
404             in("r13") cpuid_subfn,
405             lateout("rax") ret,
406             lateout("r10") vmcall_ret,
407             lateout("r11") _,
408             lateout("r12") result_eax,
409             lateout("r13") result_ebx,
410             lateout("r14") result_ecx,
411             lateout("r15") result_edx,
412             options(att_syntax));
413    }
414    // r10 is expected to be TDG.VP.VMCALL_SUCCESS per the GHCI spec
415    // Make sure the result matches the expectation
416    debug_assert!(tdvmcall_result(vmcall_ret).is_ok());
417    debug_assert!(tdx_result(ret).is_ok());
418
419    CpuidResult {
420        eax: result_eax,
421        ebx: result_ebx,
422        ecx: result_ecx,
423        edx: result_edx,
424    }
425}

478#[derive(Clone, Copy, Debug)]
479pub enum TdpHaltInterruptState {
480    Enabled,
481    Disabled,
482}
483
484pub fn tdvmcall_halt(interrupt_state: TdpHaltInterruptState) {
485    let pass_regs = TdxRegs::R10 | TdxRegs::R11 | TdxRegs::R12;
486    let interrupts_blocked = match interrupt_state {
487        TdpHaltInterruptState::Enabled => 0,
488        TdpHaltInterruptState::Disabled => 1,
489    };
490    let mut ret: u64;
491    let mut vmcall_ret: u64;
492    // SAFETY: executing TDCALL requires the use of assembly.
493    unsafe {
494        asm!("pushfq",
495             "test %r12d, %r12d",
496             "jnz 1f",
497             "sti",
498             "1:",
499             "tdcall",
500             "popfq",
501             in("rax") TDG_VP_TDVMCALL,
502             in("rcx") pass_regs.bits(),
503             in("r10") 0,
504             in("r11") TDVMCALL_HLT,
505             in("r12") interrupts_blocked,
506             lateout("rax") ret,
507             lateout("r10") vmcall_ret,
508             lateout("r11") _,
509             lateout("r12") _,
510             options(att_syntax));
511    }
512    // r10 is expected to be TDG.VP.VMCALL_SUCCESS per the GHCI spec
513    // Make sure the result matches the expectation
514    debug_assert!(tdvmcall_result(vmcall_ret).is_ok());
515    debug_assert!(tdx_result(ret).is_ok());
516}

518/// R12 bitfield for `TDVMCALL_REPORT_FATAL_ERROR`.
519#[bitfield(u64)]
520struct FatalErrorR12 {
521    error_code: u32,
522    #[bits(31)]
523    ext_error_code: u32,
524    gpa_valid: bool,
525}
526
527pub fn tdvmcall_report_fatal_error() -> ! {
528    let pass_regs = TdxRegs::R10 | TdxRegs::R11 | TdxRegs::R12;
529    // Default error code (0)
530    let r12 = FatalErrorR12::new();
531
532    // This TDCALL is not supposed to return, but guard against
533    // a non-cooperating host with an infinite loop, ensuring
534    // execution never leaves this function.
535    loop {
536        // SAFETY: executing TDCALL requires the use of assembly.
537        unsafe {
538            asm!("tdcall",
539                 in("rax") TDG_VP_TDVMCALL,
540                 in("rcx") pass_regs.bits(),
541                 in("r10") 0,
542                 in("r11") TDVMCALL_REPORT_FATAL_ERROR,
543                 in("r12") r12.into_bits(),
544            );
545        }
546    }
547}

549fn tdvmcall_io(port: u16, data: u32, size: usize, write: bool) -> u32 {
550    let pass_regs =
551        TdxRegs::R10 | TdxRegs::R11 | TdxRegs::R12 | TdxRegs::R13 | TdxRegs::R14 | TdxRegs::R15;
552    let mut ret: u64;
553    let mut vmcall_ret: u64;
554    let mut output: u32;
555    // SAFETY: executing TDCALL requires the use of assembly.
556    unsafe {
557        asm!("tdcall",
558             in("rax") TDG_VP_TDVMCALL,
559             in("rcx") pass_regs.bits(),
560             in("r10") 0,
561             in("r11") TDVMCALL_IO,
562             in("r12") size,
563             in("r13") write as u32,
564             in("r14") port as u32,
565             in("r15") data,
566             lateout("rax") ret,
567             lateout("r10") vmcall_ret,
568             lateout("r11") output,
569             lateout("r12") _,
570             lateout("r13") _,
571             lateout("r14") _,
572             lateout("r15") _,
573             options(att_syntax));
574    }
575
576    // Ignore errors here.  The caller cannot handle them, and since the
577    // I/O operation was performed by an untrusted source, the error
578    // information is not meaningfully different than a maliciously unreliable
579    // operation.
580    debug_assert!(tdvmcall_result(vmcall_ret).is_ok());
581    debug_assert!(tdx_result(ret).is_ok());
582
583    output
584}

597pub fn tdvmcall_hyperv_hypercall(regs: &mut X86GeneralRegs) {
598    let pass_regs = TdxRegs::RDX | TdxRegs::R8 | TdxRegs::R10 | TdxRegs::R11;
599    let mut ret: u64;
600    let mut vmcall_ret: u64;
601    let mut hypercall_ret: u64;
602    // A Hyper-V hypercall uses VMCALL but does not use the standard GHCI
603    // convention.  The distinction is encoded in R10, which always passes
604    // zero for GHCI calls, and which passes the Hyper-V hypercall code (which
605    // is never zero) in the case of hypercalls.
606    debug_assert_ne!({ regs.rcx }, 0);
607    // SAFETY: executing TDCALL requires the use of assembly.
608    unsafe {
609        asm!("tdcall",
610             in("rax") TDG_VP_TDVMCALL,
611             in("rcx") pass_regs.bits(),
612             in("r10") regs.rcx,
613             in("rdx") regs.rdx,
614             in("r8") regs.r8,
615             lateout("rax") ret,
616             lateout("r10") vmcall_ret,
617             lateout("r11") hypercall_ret,
618             options(att_syntax));
619    }
620
621    // Ignore errors here.  The caller cannot handle them, and the final
622    // status is not trustworthy anyway.
623    debug_assert!(tdx_result(ret).is_ok());
624    debug_assert!(tdvmcall_result(vmcall_ret).is_ok());
625
626    regs.rax = hypercall_ret as usize;
627}
