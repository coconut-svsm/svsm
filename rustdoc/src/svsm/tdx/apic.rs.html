7use crate::cpu::apic::{ApicIcr, IcrDestFmt, IcrMessageType};
8use crate::cpu::msr::{read_msr, write_msr};
9use crate::cpu::percpu::this_cpu;
10use crate::cpu::x86::apic::{
11    APIC_OFFSET_ICR, APIC_OFFSET_ID, APIC_OFFSET_SELF_IPI, APIC_OFFSET_SPIV,
12};
13use crate::cpu::x86::x2apic::{MSR_X2APIC_BASE, MSR_X2APIC_SELF_IPI};
14use crate::cpu::x86::{ApicAccess, MSR_APIC_BASE};
15use crate::tdx::tdcall;
16
17#[derive(Debug)]
18pub struct TdxApicAccessor {}
19
20impl TdxApicAccessor {
21    fn is_self_ipi(offset: usize, value: u64) -> Option<u64> {
22        match offset {
23            // Preserve hw behaviors (e.g. reserved-bit checking)
24            APIC_OFFSET_SELF_IPI => Some(value),
25            // Convert to a self-IPI if applicable
26            APIC_OFFSET_ICR => {
27                let icr = ApicIcr::from(value);
28                match icr.destination_shorthand() {
29                    // Logical destination mode can be supported if needed;
30                    // defer to GHCI for now.
31                    // The same goes for lowest priority delivery mode.
32                    IcrDestFmt::Dest => {
33                        if icr.message_type() == IcrMessageType::Fixed
34                            && !icr.destination_mode()
35                            && icr.destination() == this_cpu().get_apic_id()
36                        {
37                            Some(icr.vector() as u64)
38                        } else {
39                            None
40                        }
41                    }
42                    IcrDestFmt::OnlySelf => {
43                        debug_assert!(icr.message_type() == IcrMessageType::Fixed);
44                        Some(icr.vector() as u64)
45                    }
46                    _ => None,
47                }
48            }
49            _ => None,
50        }
51    }
52
53    // NOTE:
54    // Needs to be updated when new GHCI APIC registers are used.
55    fn is_ghci_msr(offset: usize) -> bool {
56        matches!(offset, APIC_OFFSET_ID | APIC_OFFSET_SPIV | APIC_OFFSET_ICR)
57    }
58}
59
60impl ApicAccess for TdxApicAccessor {
61    fn update_apic_base(&self, and_mask: u64, or_mask: u64) {
62        let current_value = tdcall::tdvmcall_rdmsr(MSR_APIC_BASE);
63        let new_value = (current_value & and_mask) | or_mask;
64
65        if current_value != new_value {
66            tdcall::tdvmcall_wrmsr(MSR_APIC_BASE, new_value);
67        }
68    }
69
70    fn apic_write(&self, offset: usize, value: u64) {
71        let msr = MSR_X2APIC_BASE + u32::try_from(offset).unwrap();
72
73        if let Some(v) = Self::is_self_ipi(offset, value) {
74            // SAFETY: Writes to X2APIC MSRs never harm memory safety.
75            unsafe { write_msr(MSR_X2APIC_SELF_IPI, v) };
76        } else if Self::is_ghci_msr(offset) {
77            tdcall::tdvmcall_wrmsr(msr, value);
78        } else {
79            // SAFETY: Writes to X2APIC MSRs never harm memory safety.
80            unsafe { write_msr(msr, value) };
81        }
82    }
83
84    fn apic_read(&self, offset: usize) -> u64 {
85        let msr = MSR_X2APIC_BASE + u32::try_from(offset).unwrap();
86
87        if Self::is_ghci_msr(offset) {
88            tdcall::tdvmcall_rdmsr(msr)
89        } else {
90            read_msr(msr)
91        }
92    }
93}

95pub static TDX_APIC_ACCESSOR: TdxApicAccessor = TdxApicAccessor {};
