7//! SVSM kernel crypto API
8
9pub mod aead {
10    //! API for authentication encryption with associated data
11
12    use crate::{protocols::errors::SvsmReqError, sev::secrets_page::VMPCK_SIZE};
13
14    // Message Header Format (AMD SEV-SNP spec. table 98)
15
16    /// Authenticated tag size (128 bits)
17    pub const AUTHTAG_SIZE: usize = 16;
18    /// Initialization vector size (96 bits)
19    pub const IV_SIZE: usize = 12;
20    /// Key size
21    pub const KEY_SIZE: usize = VMPCK_SIZE;
22
23    /// AES-256 GCM
24    pub trait Aes256GcmTrait {
25        /// Encrypt the provided buffer using AES-256 GCM
26        ///
27        /// # Arguments
28        ///
29        /// * `iv`: Initialization vector
30        /// * `key`: 256-bit key
31        /// * `aad`: Additional authenticated data
32        /// * `inbuf`: Cleartext buffer to be encrypted
33        /// * `outbuf`: Buffer to store the encrypted data, it must be large enough to also
34        ///   hold the authenticated tag.
35        ///
36        /// # Returns
37        ///
38        /// * Success
39        ///     * `usize`: Number of bytes written to `outbuf`
40        /// * Error
41        ///     * [SvsmReqError]
42        fn encrypt(
43            iv: &[u8; IV_SIZE],
44            key: &[u8; KEY_SIZE],
45            aad: &[u8],
46            inbuf: &[u8],
47            outbuf: &mut [u8],
48        ) -> Result<usize, SvsmReqError>;
49
50        /// Decrypt the provided buffer using AES-256 GCM
51        ///
52        /// # Returns
53        ///
54        /// * `iv`: Initialization vector
55        /// * `key`: 256-bit key
56        /// * `aad`: Additional authenticated data
57        /// * `inbuf`: Cleartext buffer to be decrypted, followed by the authenticated tag
58        /// * `outbuf`: Buffer to store the decrypted data
59        ///
60        /// # Returns
61        ///
62        /// * Success
63        ///     * `usize`: Number of bytes written to `outbuf`
64        /// * Error
65        ///     * [SvsmReqError]
66        fn decrypt(
67            iv: &[u8; IV_SIZE],
68            key: &[u8; KEY_SIZE],
69            aad: &[u8],
70            inbuf: &[u8],
71            outbuf: &mut [u8],
72        ) -> Result<usize, SvsmReqError>;
73    }
74
75    /// Aes256Gcm type
76    #[derive(Copy, Clone, Debug)]
77    pub struct Aes256Gcm;
78}
79
80pub mod digest {
81    //! API for message digests.
82
83    extern crate alloc;
84
85    use alloc::vec::Vec;
86
87    // Note the typical new/update/finish API is not initially supported due
88    // to our inability to hide an abstract implementation's state representation
89    // behind a type without any fields.
90
91    pub trait Algorithm {
92        /// Digests `input` into an output vector of size `OUTPUT_LEN`.
93        fn digest(input: &[u8]) -> Vec<u8>;
94    }
95
96    /// Sha512 type
97    #[derive(Copy, Clone, Debug)]
98    pub struct Sha512;
99}
