7//! High level error typing for the public SVSM APIs.
8//!
9//! This module contains the generic [`SvsmError`] type, which may be returned
10//! from any public API in this codebase to signal an error during SVSM
11//! operation. Each variant of the type may give more specific information
12//! about the source of the error.
13//!
14//! As a general rule, functions private to a given module may directly return
15//! leaf error types, which are contained in [`SvsmError`] variants. Public
16//! functions should return an [`SvsmError`] containing a leaf error type,
17//! usually the one corresponding to that module. Each module should provide
18//! a way to convert a leaf error into a SvsmError via the [`From`] trait.
19

39/// Errors related to APIC handling.  These may originate from multiple
40/// layers in the system.
41#[derive(Clone, Copy, Debug)]
42pub enum ApicError {
43    /// An error arising because APIC emulation is disabled.
44    Disabled,
45
46    /// An error related to APIC emulation.
47    Emulation,
48
49    /// An error related to APIC registration.
50    Registration,
51}

53/// Errors related to Attestation handling. These may originate from multiple
54/// layers in the system. Added to protocol base for protocol error.
55#[repr(u64)]
56#[derive(Clone, Copy, Debug)]
57pub enum AttestError {
58    /// An error related to attestation report.
59    Report = 0,
60
61    /// An error related to attestation manifest.
62    Manifest = 1,
63}

65/// A generic error during SVSM operation.
66#[derive(Clone, Copy, Debug)]
67pub enum SvsmError {
68    /// Errors related to platform initialization.
69    PlatformInit,
70    /// Errors during ELF parsing and loading.
71    Elf(ElfError),
72    /// Errors related to GHCB
73    Ghcb(GhcbError),
74    /// Errors related to MSR protocol
75    GhcbMsr(GhcbMsrError),
76    /// Errors related to SEV-SNP operations, like PVALIDATE or RMPUPDATE
77    SevSnp(SevSnpError),
78    /// Errors related to TDX operations
79    Tdx(TdxError),
80    /// Generic errors related to memory management
81    Mem,
82    /// Errors related to the memory allocator
83    Alloc(AllocError),
84    /// Error reported when there is no VMSA set up.
85    MissingVMSA,
86    /// Error reported when there is no CAA (Calling Area Address) set up.
87    MissingCAA,
88    /// Error reported when there is no secrets page set up.
89    MissingSecrets,
90    /// Instruction decode related errors
91    Insn(InsnError),
92    /// Invalid address, usually provided by the guest
93    InvalidAddress,
94    /// Error reported when convert a usize to Bytes
95    InvalidBytes,
96    /// Error reported when converting to UTF-8
97    InvalidUtf8,
98    /// A fault occurred
99    Fault,
100    /// Errors related to firmware parsing
101    Firmware,
102    /// Errors related to console operation
103    Console,
104    /// Errors related to firmware configuration contents
105    FwCfg(FwCfgError),
106    /// Errors related to ACPI parsing.
107    Acpi,
108    /// Errors from the filesystem.
109    FileSystem(FsError),
110    /// Obj related error
111    Obj(ObjError),
112    /// Task management errors,
113    Task(TaskError),
114    /// Errors from #VC handler
115    Vc(VcError),
116    /// The operation is not supported.
117    NotSupported,
118    /// Generic errors related to APIC emulation.
119    Apic(ApicError),
120    /// Generic errors related to attestation handling.
121    Attestation(AttestError),
122    /// Errors related to Hyper-V.
123    HyperV(u16),
124    /// Errors related to Virtio drivers.
125    #[cfg(feature = "virtio-drivers")]
126    Virtio(VirtioError),
127    /// Errors related to block devices.
128    Block(BlockDeviceError),
129    /// Errors related to attesting SVSM's launch evidence.
130    #[cfg(feature = "attest")]
131    TeeAttestation(AttestationError),
132}

171impl From<SvsmError> for SysCallError {
172    fn from(err: SvsmError) -> Self {
173        match err {
174            SvsmError::Alloc(AllocError::OutOfMemory) => SysCallError::ENOMEM,
175            SvsmError::FileSystem(FsError::FileExists) => SysCallError::EEXIST,
176            SvsmError::FileSystem(FsError::WriteOnly) => SysCallError::EWRONLY,
177            SvsmError::FileSystem(FsError::ReadOnly) => SysCallError::ERDONLY,
178
179            SvsmError::FileSystem(FsError::FileNotFound) | SvsmError::Obj(ObjError::NotFound) => {
180                SysCallError::ENOTFOUND
181            }
182
183            SvsmError::NotSupported => SysCallError::ENOTSUPP,
184
185            SvsmError::FileSystem(FsError::Inval)
186            | SvsmError::Obj(ObjError::InvalidHandle)
187            | SvsmError::Mem
188            | SvsmError::InvalidAddress
189            | SvsmError::InvalidBytes
190            | SvsmError::InvalidUtf8 => SysCallError::EINVAL,
191
192            _ => SysCallError::UNKNOWN,
193        }
194    }
