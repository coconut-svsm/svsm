14#[derive(Clone, Copy, Debug)]
15pub enum ObjError {
16    InvalidHandle,
17    NotFound,
18    Busy,
19}
20
21/// An object represents the type of resource like file, VM, vCPU in the
22/// COCONUT-SVSM kernel which can be accessible by the user mode. The Obj
23/// trait is defined for such type of resource, which can be used to define
24/// the common functionalities of the objects. With the trait bounds of Send
25/// and Sync, the objects implementing Obj trait could be sent to another
26/// thread and shared between threads safely.
27pub trait Obj: Send + Sync + core::fmt::Debug {
28    fn as_fs(&self) -> Option<&FsObj> {
29        None
30    }
31}
32
33/// ObjHandle is a unique identifier for an object in the current process.
34/// An ObjHandle can be converted to a u32 id which can be used by the user
35/// mode to access this object. The passed id from the user mode by syscalls
36/// can be converted to an `ObjHandle`, which is used to access the object in
37/// the COCONUT-SVSM kernel.
38#[derive(Clone, Copy, Debug, Eq, Ord, PartialEq, PartialOrd)]
39pub struct ObjHandle(u32);
40
41impl ObjHandle {
42    pub fn new(id: u32) -> Self {
43        Self(id)
44    }
45}
46
47impl From<u32> for ObjHandle {
48    #[inline]
49    fn from(id: u32) -> Self {
50        Self(id)
51    }
52}
53
54impl From<ObjHandle> for u32 {
55    #[inline]
56    fn from(obj_handle: ObjHandle) -> Self {
57        obj_handle.0
58    }
59}
60
61/// Add an object to the current process and assigns it an `ObjHandle`.
62///
63/// # Arguments
64///
65/// * `obj` - An `Arc<dyn Obj>` representing the object to be added.
66///
67/// # Returns
68///
69/// * `Result<ObjHandle, SvsmError>` - Returns the object handle of the
70///   added object if successful, or an `SvsmError` on failure.
71///
72/// # Errors
73///
74/// This function will return an error if adding the object to the
75/// current task fails.
76pub fn obj_add(obj: Arc<dyn Obj>) -> Result<ObjHandle, SvsmError> {
77    current_task().add_obj(obj)
78}
79
80/// Closes an object identified by its ObjHandle.
81///
82/// # Arguments
83///
84/// * `id` - The ObjHandle for the object to be closed.
85///
86/// # Returns
87///
88/// * `Result<Arc<dyn Obj>>, SvsmError>` - Returns the `Arc<dyn Obj>`
89///   on success, or an `SvsmError` on failure.
90///
91/// # Errors
92///
93/// This function will return an error if removing the object from the
94/// current task fails.
95pub fn obj_close(id: ObjHandle) -> Result<Arc<dyn Obj>, SvsmError> {
96    current_task().remove_obj(id)
97}
98
99/// Retrieves an object by its ObjHandle.
100///
101/// # Arguments
102///
103/// * `id` - The ObjHandle for the object to be retrieved.
104///
105/// # Returns
106///
107/// * `Result<Arc<dyn Obj>>, SvsmError>` - Returns the `Arc<dyn Obj>` on
108///   success, or an `SvsmError` on failure.
109///
110/// # Errors
111///
112/// This function will return an error if retrieving the object from the
113/// current task fails.
114pub fn obj_get(id: ObjHandle) -> Result<Arc<dyn Obj>, SvsmError> {
115    current_task().get_obj(id)
116}
