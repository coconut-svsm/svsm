28#[derive(Debug)]
29pub struct HypercallPagesGuard<'a> {
30    // The page reference is never actually read; it exists here to bind the
31    // lifetime of the `Ref` to the lifetime of the `HypercallPagesGuard`.
32    // The input/output pages are unwrapped during construction.
33    _page_ref: RefMut<'a, (VirtPhysPair, VirtPhysPair)>,
34    pub input: VirtPhysPair,
35    pub output: VirtPhysPair,
36    _irq_guard: IrqGuard,
37}
38
39/// A structure that binds a reference to hypercall input/output pages.
40/// The pages remain usable until the structure is dropped.
41impl<'a> HypercallPagesGuard<'a> {
42    /// Creates a new `HypercallPagesGuard` structure to describe a pair of
43    /// input/output pages.
44    ///
45    /// # Safety
46    ///
47    /// No validation is performed to verify correct ownership of the specified
48    /// virtual addresses, and no validation is performed to verify the
49    /// correct association of virtual to physical address.  The caller is
50    /// responsible for ensuring that both virtual and physical addresses are
51    /// correct and usable.
52    pub unsafe fn new(page_ref: RefMut<'a, (VirtPhysPair, VirtPhysPair)>) -> Self {
53        let (input, output) = *page_ref;
54        Self {
55            _page_ref: page_ref,
56            input,
57            output,
58            _irq_guard: IrqGuard::new(),
59        }
60    }
61
62    /// Casts a hypercall input page into a header of type `H` and returns a
63    /// reference to that header object.
64    ///
65    /// This function requires the use of `mut self` because it generates
66    /// mutable references to the addresses described by the hypercall pages,
67    /// but the compiler cannot understand that relationship.  Therefore,
68    /// suppress the warning for needless mut because it is actually required
69    /// in this case.
70    #[allow(clippy::needless_pass_by_ref_mut)]
71    fn hypercall_input<H>(&mut self) -> &mut H {
72        let header = self.input.vaddr.as_mut_ptr::<H>();
73        assert!(size_of::<H>() <= PAGE_SIZE);
74        // SAFETY: the virtual address represents an entire page which is
75        // exclusively owned by the `HypercallPagesGuard` and can safely be
76        // cast to a header of type `H`.
77        unsafe { &mut *header }
78    }
79
80    /// Divides a hypercall input page into a header of type `H` and a slice
81    /// of repeated elements of type `T` and returns a reference to each
82    /// portion.
83    ///
84    /// This function requires the use of `mut self` because it generates
85    /// mutable references to the addresses described by the hypercall pages,
86    /// but the compiler cannot understand that relationship.  Therefore,
87    /// suppress the warning for needless mut because it is actually required
88    /// in this case.
89    #[allow(clippy::needless_pass_by_ref_mut)]
90    fn hypercall_rep_input<H, T>(&mut self) -> (&mut H, &mut [T]) {
91        let header_size = size_of::<H>();
92        assert!(header_size <= PAGE_SIZE);
93        let rep_count = (PAGE_SIZE - header_size) / size_of::<T>();
94        let header = self.input.vaddr.as_mut_ptr::<H>();
95        let rep_slice = self.input.vaddr.const_add(header_size).as_mut_ptr::<T>();
96        // SAFETY: the virtual address represents an entire page which is
97        // exclusively owned by the `HypercallPagesGuard` and can safely be
98        // cast to a header of type `H` followed by an array of `T` up to the
99        // size of one page.
100        unsafe {
101            (
102                &mut *header,
103                slice::from_raw_parts_mut(rep_slice, rep_count),
104            )
105        }
106    }
107
108    /// Casts a hypercall output page into a slice of repeated elements of
109    /// type `T` and returns a reference to that slice.
110    fn hypercall_output<T>(&self, output: HvHypercallOutput) -> &[T] {
111        // A non-REP hypercall is assumed to have a single output element.
112        let output_count = output.count();
113        let count: usize = if output_count != 0 {
114            output_count as usize
115        } else {
116            1
117        };
118        assert!(count * size_of::<T>() <= PAGE_SIZE);
119        // SAFETY: the virtual address represents an entire page which is
120        // exclusively owned by the `HypercallPagesGuard` and can safely be
121        // cast to an array of `T` up to the size of one page.
122        unsafe { slice::from_raw_parts(self.output.vaddr.as_ptr::<T>(), count) }
123    }
124}

165static HYPERV_HYPERCALL_CODE_PAGE: ImmutAfterInitCell<VirtAddr> = ImmutAfterInitCell::uninit();
166static CURRENT_VTL: ImmutAfterInitCell<u8> = ImmutAfterInitCell::uninit();

168pub fn is_hyperv_hypervisor() -> bool {
169    // Check if any hypervisor is present.
170    if (CpuidResult::get(1, 0).ecx & 0x80000000) == 0 {
171        return false;
172    }
173
174    // Get the hypervisor interface signature.
175    CpuidResult::get(0x40000001, 0).eax == 0x31237648
176}

178pub fn hyperv_setup_hypercalls() -> Result<(), SvsmError> {
179    // Allocate a page to use as the hypercall code page.
180    let page = allocate_pages(1)?;
181
182    // Map the page as executable at a known address.
183    let hypercall_va = SVSM_HYPERCALL_CODE_PAGE;
184    this_cpu()
185        .get_pgtable()
186        .map_4k(hypercall_va, virt_to_phys(page), PTEntryFlags::exec())?;
187
188    HYPERV_HYPERCALL_CODE_PAGE
189        .init(hypercall_va)
190        .expect("Hypercall code page already allocated");
191
192    // Set the guest OS ID.  The value is arbitrary.
193    // SAFETY: writing to HV Guest OS ID doesn't break safety.
194    unsafe { write_msr(HyperVMsr::GuestOSID.into(), 0xC0C0C0C0) };
195
196    // Set the hypercall code page address to the physical address of the
197    // allocated page, and mark it enabled.
198    let pa = virt_to_phys(page);
199    // SAFETY: we trust the page allocator to allocate a valid page to which pa
200    // points.
201    unsafe { write_msr(HyperVMsr::Hypercall.into(), u64::from(pa) | 1) };
202
203    // Obtain the current VTL for use in future hypercalls.
204    let vsm_status_value = get_vp_register(hyperv::HvRegisterName::VsmVpStatus)?;
205    let vsm_status = hyperv::HvRegisterVsmVpStatus::from(vsm_status_value);
206    let current_vtl = vsm_status.active_vtl();
207    CURRENT_VTL
208        .init(current_vtl)
209        .expect("Current VTL already initialized");
210
211    Ok(())
212}

214/// # Safety
215///
216/// Hypercalls can have side-effects that include modifying memory, so
217/// callers must be certain that any preconditions required for the safety
218/// of a hypercall are met.
219unsafe fn hypercall(
220    input_control: HvHypercallInput,
221    hypercall_pages: &HypercallPagesGuard<'_>,
222) -> HvHypercallOutput {
223    let hypercall_va = u64::from(*HYPERV_HYPERCALL_CODE_PAGE);
224    let mut output: u64;
225    // SAFETY: inline assembly is required to invoke the hypercall.
226    unsafe {
227        asm!("callq *%rax",
228             in("rax") hypercall_va,
229             in("rcx") input_control.into_bits(),
230             in("rdx") u64::from(hypercall_pages.input.paddr),
231             in("r8") u64::from(hypercall_pages.output.paddr),
232             lateout("rax") output,
233             options(att_syntax));
234    }
235    HvHypercallOutput::from(output)
236}

238/// Dead code is allowed because the hypercall input logic doesn't recognize
239/// that all fields are read when they are copied into the hypercall input
240/// page.
241#[allow(dead_code)]
242#[derive(Clone, Copy, Debug, Default)]
243struct HvInputGetVpRegister {
244    partition_id: u64,
245    vp_index: u32,
246    input_vtl: hyperv::HvInputVtl,
247    _rsvd: [u8; 3],
248}
249
250pub fn get_vp_register(name: hyperv::HvRegisterName) -> Result<u64, SvsmError> {
251    let input_control = HvHypercallInput::new()
252        .with_call_code(HvCallCode::GetVpRegister as u16)
253        .with_element_count(1);
254
255    let input_header = HvInputGetVpRegister {
256        partition_id: HV_PARTITION_ID_SELF,
257        vp_index: HV_VP_INDEX_SELF,
258        input_vtl: hyperv::HvInputVtl::use_self(),
259        ..Default::default()
260    };
261
262    let mut hypercall_pages = this_cpu().get_hypercall_pages();
263    let (header, slice) = hypercall_pages.hypercall_rep_input::<HvInputGetVpRegister, u32>();
264    *header = input_header;
265    slice[0] = name as u32;
266
267    // SAFETY: the GetVpRegisters hypercall does not write to any memory other
268    // than the hypercall page, and does not consume memory that is not
269    // included in the hypercall input.
270    let call_output = unsafe { hypercall(input_control, &hypercall_pages) };
271    let status = call_output.status();
272    if status != 0 {
273        return Err(HyperV(status));
274    }
275
276    let output = hypercall_pages.hypercall_output::<u64>(call_output);
277    let reg = output[0];
278
279    drop(hypercall_pages);
280
281    Ok(reg)
282}

294fn enable_vp_vtl_hypercall(
295    cpu: &PerCpu,
296    vtl: u8,
297    context: &HvInitialVpContext,
298) -> Result<(), SvsmError> {
299    let input_header = HvInputEnableVpVtl {
300        partition_id: HV_PARTITION_ID_SELF,
301        vtl,
302        vp_index: cpu.get_apic_id(),
303        context: *context,
304        ..Default::default()
305    };
306
307    let input_control = HvHypercallInput::new().with_call_code(HvCallCode::EnableVpVtl as u16);
308
309    let mut hypercall_pages = this_cpu().get_hypercall_pages();
310    let header = hypercall_pages.hypercall_input::<HvInputEnableVpVtl>();
311    *header = input_header;
312
313    // SAFETY: the EnableVpVtl hypercall does not write to any memory and
314    // does not consume memory that is not included in the hypercall input.
315    let call_output = unsafe { hypercall(input_control, &hypercall_pages) };
316    let status = call_output.status();
317
318    if status != 0 {
319        Err(HyperV(status))
320    } else {
321        Ok(())
322    }
323}

366pub fn hyperv_start_cpu(cpu: &PerCpu, context: &HvInitialVpContext) -> Result<(), SvsmError> {
367    // Enable the current VTL on the target CPU if the current VTL is not
368    // VTL 0.
369    let vtl = *CURRENT_VTL;
370    if vtl != 0 {
371        enable_vp_vtl_hypercall(cpu, vtl, context)?;
372    }
373    start_vp_hypercall(cpu, vtl, context)
374}
