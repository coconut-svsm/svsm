22struct PageStore {
23    pages: Vec<(PhysAddr, SharedBox<[u8; PAGE_SIZE]>)>,
24}
25
26impl PageStore {
27    pub fn new() -> Self {
28        PageStore { pages: Vec::new() }
29    }
30
31    pub fn push(&mut self, pa: PhysAddr, shared_page: SharedBox<[u8; PAGE_SIZE]>) {
32        self.pages.push((pa, shared_page));
33    }
34
35    pub fn pop(&mut self, pa: PhysAddr) -> Option<SharedBox<[u8; PAGE_SIZE]>> {
36        if let Some(p) = self.pages.iter().position(|e| e.0 == pa) {
37            Some(self.pages.remove(p).1)
38        } else {
39            None
40        }
41    }
42}
43
44static SHARED_MEM: SpinLock<OnceCell<PageStore>> = SpinLock::new(OnceCell::new());
45
46pub fn virtio_init() {
47    SHARED_MEM.lock().get_or_init(PageStore::new);
48}

53/// Implementation of virtio-drivers MMIO hardware abstraction for AMD SEV-SNP
54/// in the Coconut-SVSM context. Due to missing #VC handler for MMIO, use ghcb exits
55/// instead.
56///
57/// SAFETY: Complies with the safety requirements of the virtio_drivers::Hal trait.
58unsafe impl virtio_drivers::Hal for SvsmHal {
59    /// Allocates and zeroes the given number of contiguous physical pages of DMA memory for VirtIO
60    /// use.
61    ///
62    /// Limitation: `pages == 1` required.
63    fn dma_alloc(
64        pages: usize,
65        _direction: virtio_drivers::BufferDirection,
66    ) -> (virtio_drivers::PhysAddr, NonNull<u8>) {
67        // TODO: allow more than one page.
68        //       This currently works, because in "modern" virtio mode the crate only allocates
69        //       one page at a time.
70        assert!(pages == 1);
71
72        let shared_page = SharedBox::<[u8; PAGE_SIZE]>::try_new_zeroed().unwrap();
73        let pa = virt_to_phys(shared_page.addr());
74        let p = NonNull::<u8>::new(shared_page.addr().as_mut_ptr()).unwrap();
75
76        SHARED_MEM.lock().get_mut().unwrap().push(pa, shared_page);
77
78        (pa.into(), p)
79    }
80
81    /// Deallocates the given contiguous physical DMA memory pages.
82    ///
83    /// # Safety
84    ///
85    /// The memory must have been allocated by `dma_alloc` on the same `Hal` implementation, and not
86    /// yet deallocated. `pages` must be the same number passed to `dma_alloc` originally, and both
87    /// `paddr` and `vaddr` must be the values returned by `dma_alloc`.
88    ///
89    /// Limitation: `pages == 1` required.
90    unsafe fn dma_dealloc(
91        paddr: virtio_drivers::PhysAddr,
92        _vaddr: NonNull<u8>,
93        pages: usize,
94    ) -> i32 {
95        //TODO: allow more than one page
96        assert!(pages == 1);
97
98        SHARED_MEM
99            .lock()
100            .get_mut()
101            .unwrap()
102            .pop(paddr.into())
103            .unwrap();
104
105        0
106    }
107
108    /// Converts a physical address used for MMIO to a virtual address.
109    /// NOT IMPLEMENTED - the cut-down virtio driver for SVSM does not use this.
110    unsafe fn mmio_phys_to_virt(_paddr: virtio_drivers::PhysAddr, _size: usize) -> NonNull<u8> {
111        unimplemented!()
112    }
113
114    /// Shares the given memory range with the device, and returns the physical address that the
115    /// device can use to access it.
116    ///
117    /// # Safety
118    ///
119    /// The buffer must be a valid pointer to a non-empty memory range which will not be accessed by
120    /// any other thread for the duration of this method call.
121    ///
122    /// Limitation: `buffer.len() <= PAGE_SIZE`
123    unsafe fn share(
124        buffer: NonNull<[u8]>,
125        direction: virtio_drivers::BufferDirection,
126    ) -> virtio_drivers::PhysAddr {
127        // TODO: allow more than one page
128        assert!(buffer.len() <= PAGE_SIZE);
129
130        let shared_page = SharedBox::<[u8; PAGE_SIZE]>::try_new_zeroed().unwrap();
131
132        if direction == virtio_drivers::BufferDirection::DriverToDevice {
133            let src = buffer.as_ptr().cast::<u8>();
134            let dst = shared_page.addr().as_mut_ptr::<u8>();
135
136            // SAFETY: We demand a valid `buffer` from the caller (virtio-drivers crate).
137            //         We assterted that `dst` can hold at least `buffer.len()`.
138            unsafe {
139                core::ptr::copy_nonoverlapping(src, dst, buffer.len());
140            }
141        }
142
143        let pa = virt_to_phys(shared_page.addr());
144        SHARED_MEM.lock().get_mut().unwrap().push(pa, shared_page);
145
146        // return pa of shared page
147        pa.into()
148    }
149
150    /// Unshares the given memory range from the device and (if necessary) copies it back to the
151    /// original buffer.
152    ///
153    /// # Safety
154    ///
155    /// The buffer must be a valid pointer to a non-empty memory range which will not be accessed by
156    /// any other thread for the duration of this method call. The `paddr` must be the value
157    /// previously returned by the corresponding `share` call.
158    ///
159    /// Limitation: `buffer.len() <= PAGE_SIZE`
160    unsafe fn unshare(
161        paddr: virtio_drivers::PhysAddr,
162        buffer: NonNull<[u8]>,
163        direction: virtio_drivers::BufferDirection,
164    ) {
165        assert!(buffer.len() <= PAGE_SIZE);
166
167        if let Some(shared_page) = SHARED_MEM.lock().get_mut().unwrap().pop(paddr.into()) {
168            let vaddr = phys_to_virt(paddr.into());
169            let va_from_shared = shared_page.addr();
170            assert!(vaddr == va_from_shared);
171
172            if direction == virtio_drivers::BufferDirection::DeviceToDriver {
173                let dst = buffer.as_ptr().cast::<u8>();
174                let src = vaddr.as_mut_ptr::<u8>();
175
176                // SAFETY: `src` is valid, since it is returned by `phys_to_virt ()`
177                //         We rely on the caller (=virtio driver) to supply a valid `buffer` and
178                //         a matchting `paddr`, which was returned by a prevous call to `share ()`.
179                //         Thus we can assume that `dst` is valid and that `src` holds `buffer.len()` bytes.
180                unsafe {
181                    core::ptr::copy_nonoverlapping(src, dst, buffer.len());
182                }
183            }
184        } else {
185            panic!("unshare: No shared page found at given pa");
186        }
187        // implicit drop of share_page here.
188    }
189
190    /// Performs memory mapped read from location of `src`. `src` itself is not modified,
191    /// the value is returned instead.
192    ///
193    /// The default implementation performs a regular volatile_read. This method is intended
194    /// to be overwritten in case MMIO memory needs to be accessed in a special way (for example AMD SEV-SNP).
195    ///
196    /// # Safety
197    ///
198    /// `src` must be properly aligned and reside at a readable memory address.
199    unsafe fn mmio_read<T: FromBytes + Immutable>(src: &T) -> T {
200        let mut b = MaybeUninit::<T>::uninit();
201        // SAFETY: We are trusting the caller (the virtio driver) to ensure `src` is a valid MMIO
202        // address and that it is aligned properly. If SVSM_PLATFORM.mmio_read() doesn't fail
203        // we can assume that all the bytes are read from the device.
204        unsafe {
205            // MaybeUninit::as_bytes_mut() can avoid this, but it's still
206            // unstable. When it will be stabilized, we can simply use
207            // `b.as_bytes_mut()` instead of creating `b_slice`.
208            let b_slice = core::slice::from_raw_parts_mut(
209                b.as_mut_ptr().cast::<MaybeUninit<u8>>(),
210                size_of::<T>(),
211            );
212            SVSM_PLATFORM
213                .mmio_read(VirtAddr::from(addr_of!(*src)), b_slice)
214                .unwrap();
215            b.assume_init()
216        }
217    }
218
219    /// Performs memory mapped write of `value` to the location of `dst`.
220    ///
221    /// The default implementation performs a regular volatile_write. This method is intended
222    /// to be overwritten in case MMIO memory needs to be accessed in a special way (for example AMD SEV-SNP).
223    ///
224    /// # Safety
225    ///
226    /// `dst` must be properly aligned and reside at a writable memory address.
227    unsafe fn mmio_write<T: IntoBytes + Immutable>(dst: &mut T, v: T) {
228        // SAFETY: We are trusting the caller (the virtio driver) to ensure validity of `paddr` and alignment of data.
229        unsafe {
230            SVSM_PLATFORM
231                .mmio_write(VirtAddr::from(addr_of!(*dst)), v.as_bytes())
232                .unwrap();
233        }
234    }
