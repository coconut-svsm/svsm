27#[derive(Debug, Default)]
28pub struct MmioSlots {
29    slots: Vec<MmioSlot>,
30}
31
32/// Probes and enumerates all virtio-MMIO devices available in the system.
33///
34/// This function queries the fw_cfg interface to discover virtio-MMIO device
35/// addresses and maps their MMIO regions.
36///
37/// # Usage
38///
39/// This function is typically called once during early system initialization
40/// to discover all available virtio devices. Each slot in the returned collection
41/// should be consumed exactly once by calling [`MmioSlots::pop_slot`], as each
42/// slot represents exclusive ownership of a device's MMIO region.
43///
44/// # Returns
45///
46/// Returns an [`MmioSlots`] collection containing all discovered virtio-MMIO devices.
47/// Returns an empty collection if no devices are found or if the fw_cfg interface
48/// is unavailable.
49pub fn probe_mmio_slots(boot_params: &BootParams<'_>) -> MmioSlots {
50    // Virtio MMIO addresses are discovered via fw_cfg, so skip probing
51    // if it is not present.
52    if !boot_params.has_fw_cfg_port() {
53        return MmioSlots::default();
54    }
55
56    virtio_init();
57
58    let cfg = FwCfg::new(SVSM_PLATFORM.get_io_port());
59    let Ok(dev) = cfg.get_virtio_mmio_addresses() else {
60        return MmioSlots::default();
61    };
62
63    let mut slots = Vec::with_capacity(dev.len());
64
65    for addr in dev {
66        let phys_addr = PhysAddr::from(addr);
67
68        let Ok(mem) = map_global_range_4k_shared(phys_addr, PAGE_SIZE, PTEntryFlags::data()) else {
69            log::warn!("MmioSlots: Failed to map MMIO region at {addr:x}");
70            continue;
71        };
72
73        // Not expected to fail, because mem exists.
74        let header = NonNull::new(mem.addr().as_mut_ptr()).unwrap();
75
76        // SAFETY: `map_global_range_4k_shared` guarantees us proper address alignment.
77        // The memory region has the same lifetime of the MmioSlot structure which will be consumed by the driver.
78        let Ok(transport) = (unsafe { MmioTransport::<SvsmHal>::new(header) }) else {
79            // Currently QEMU advertises _all_ slots, regardless they are empty or not.
80            log::debug!("MmioSlots: {addr:x} empty");
81            continue;
82        };
83
84        log::info!("MmioSlots: Found {:?} at {addr:x}", transport.device_type());
85
86        let slot_type = MmioSlot {
87            mmio_range: mem,
88            transport,
89        };
90
91        slots.push(slot_type);
92    }
93
94    MmioSlots { slots }
95}
96
97impl MmioSlots {
98    /// Retrieves and removes the first available MMIO slot matching the specified device type.
99    ///
100    /// This method consumes the slot. Once retrieved, the slot is removed
101    /// from the array and cannot be obtained again. This ensures each
102    /// virtio device is initialized exactly once and forbids driver unloading.
103    ///
104    /// # Arguments
105    ///
106    /// * `virtio_type` - The virtio device type to search for
107    ///
108    /// # Returns
109    ///
110    /// * `Some(MmioSlot)` - The first matching slot
111    /// * `None` - No slot matching the requested device type was found
112    pub fn pop_slot(&mut self, virtio_type: DeviceType) -> Option<MmioSlot> {
113        let pos = self
114            .slots
115            .iter()
116            .position(|slot| slot.transport.device_type() == virtio_type)?;
117        Some(self.slots.remove(pos))
118    }
119}
