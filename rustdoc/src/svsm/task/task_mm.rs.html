20static KTASK_VADDR_BITMAP: SpinLock<BitmapAllocator1024> =
21    SpinLock::new(BitmapAllocator1024::new_empty());
22
23// The task virtual range guard manages the allocation of a task virtual
24// address range within the task address space.  The address range is reserved
25// as long as the guard continues to exist.
26#[derive(Debug)]
27struct TaskVirtualRegionGuard {
28    index: usize,
29}
30
31impl TaskVirtualRegionGuard {
32    fn alloc() -> Result<Self, SvsmError> {
33        let index = KTASK_VADDR_BITMAP
34            .lock()
35            .alloc(1, 0)
36            .ok_or(SvsmError::Alloc(AllocError::OutOfMemory))?;
37        Ok(Self { index })
38    }
39
40    fn vaddr_region(&self) -> MemoryRegion<VirtAddr> {
41        const SPAN: usize = SIZE_LEVEL3 / BitmapAllocator1024::CAPACITY;
42        let base = SVSM_PERTASK_BASE + (self.index * SPAN);
43        MemoryRegion::<VirtAddr>::new(base, SPAN)
44    }
45}
46
47impl Drop for TaskVirtualRegionGuard {
48    fn drop(&mut self) {
49        KTASK_VADDR_BITMAP.lock().free(self.index, 1);
50    }
51}

53pub struct TaskMM {
54    /// Virtual address region that has been allocated for this task.
55    /// This is not referenced but must be stored so that it is dropped when
56    /// the Task is dropped.
57    _ktask_region: TaskVirtualRegionGuard,
58
59    /// Task virtual memory range for use at CPL 0
60    vm_kernel_range: VMR,
61
62    /// Task virtual memory range for use at CPL 3 - None for kernel tasks
63    vm_user_range: Option<VMR>,
64}
65
66impl TaskMM {
67    /// Creates and initializes a new `TaskMM` structure.
68    ///
69    /// # Arguments
70    ///
71    /// * `user_vmr` - Optional `[VMR]` for the user-mode portion of the tasks address space.
72    ///
73    /// # Returns
74    ///
75    /// `Ok(TaskMM)` on success, `Err(SvsmError)` on failure.
76    pub fn create(user_vmr: Option<VMR>) -> Result<Self, SvsmError> {
77        let ktask_region = TaskVirtualRegionGuard::alloc()?;
78        let vaddr_region = ktask_region.vaddr_region();
79        let vm_kernel_range = VMR::new(
80            vaddr_region.start(),
81            vaddr_region.end(),
82            PTEntryFlags::empty(),
83        );
84        // SAFETY: The selected kernel mode task address range is the only
85        // range that will live within the top-level entry associated with the
86        // task address space.
87        unsafe {
88            vm_kernel_range.initialize()?;
89        }
90
91        Ok(TaskMM {
92            _ktask_region: ktask_region,
93            vm_kernel_range,
94            vm_user_range: user_vmr,
95        })
96    }
97
98    /// Return a reference to the `[VMR]` for the per-task kernel region.
99    ///
100    /// # Returns
101    ///
102    /// Reference to the kernel region `[VMR]`.
103    pub fn kernel_range(&self) -> &VMR {
104        &self.vm_kernel_range
105    }
106
107    /// Return an otional reference to the `[VMR]` for the per-task user region.
108    ///
109    /// # Returns
110    ///
111    /// `Some(&VMR)` referencing the user-mode `[VMR]` for a user-task, `None` otherwise.
112    pub fn user_range(&self) -> Option<&VMR> {
113        self.vm_user_range.as_ref()
114    }
115
116    /// Checks whether the task has a user-mode `[VMR]`.
117    ///
118    /// # Returns
119    ///
120    /// `True` if user-mode `[VMR]` is present, `False` otherwise.
121    pub fn has_user(&self) -> bool {
122        self.vm_user_range.is_some()
123    }
124}

126/// Guard a per-task kernel mapping and unmap it when going out of scope.
127pub struct TaskKernelMapping {
128    /// Pointer the struct `[TaskMM]` which contains the mapping.
129    mm: Arc<TaskMM>,
130    /// Virtual address of the mapping.
131    va: VirtAddr,
132}
133
134impl TaskKernelMapping {
135    /// Insert a given `[Mapping]` into the kernel address part of the task and
136    /// return a new `[TaskKernelMapping]`.
137    ///
138    /// # Arguments
139    ///
140    /// * `mm` - Pointer to struct `[TaskMM]` to map into.
141    /// * `mapping` - Pointer to `[Mapping]` to put into the kernel-part of the `TaskMM`.
142    ///
143    /// # Returns
144    ///
145    /// `Some(TaskKernelMapping)` on success, `Err(SvsmError)` on failure.
146    pub fn new(mm: Arc<TaskMM>, mapping: Arc<Mapping>) -> Result<Self, SvsmError> {
147        let va = mm.kernel_range().insert(mapping)?;
148        Ok(Self { mm, va })
149    }
150
151    /// Get the virtual address the guarded mapping starts at.
152    ///
153    /// # Returns
154    ///
155    /// Virtual start address of contained mapping.
156    pub fn virt_addr(&self) -> VirtAddr {
157        self.va
158    }
159}
160
161impl Drop for TaskKernelMapping {
162    fn drop(&mut self) {
163        self.mm
164            .kernel_range()
165            .remove(self.va)
166            .expect("Error removing TaskKernelMapping");
167    }
168}
