22fn convert_elf_phdr_flags(flags: Elf64PhdrFlags) -> VMFileMappingFlags {
23    let mut vm_flags = VMFileMappingFlags::Fixed;
24
25    if flags.contains(Elf64PhdrFlags::WRITE) {
26        vm_flags |= VMFileMappingFlags::Write | VMFileMappingFlags::Private;
27    }
28
29    if flags.contains(Elf64PhdrFlags::EXECUTE) {
30        vm_flags |= VMFileMappingFlags::Execute;
31    }
32
33    vm_flags
34}

36/// Returns the name of the binary file without preceeding directories. This is
37/// used as the official task name.
38fn task_name(binary: &str) -> String {
39    let mut items = binary.split('/').filter(|x| !x.is_empty());
40    match items.nth_back(0) {
41        Some(p) => String::from(p),
42        None => String::from("unknown"),
43    }
44}

46/// Loads and executes an ELF binary in user-mode.
47///
48/// # Arguments
49///
50/// * binary: Path to file in the file-system
51///
52/// # Returns
53///
54/// [`Ok(tid)`] on success, [`Err(SvsmError)`] on failure.
55pub fn exec_user(binary: &str, root: Arc<dyn Directory>) -> Result<u32, SvsmError> {
56    let fh = open_read(binary)?;
57    let file_size = fh.size();
58
59    let current_task = current_task();
60    let vstart = current_task.mmap_kernel_guard(
61        VirtAddr::new(0),
62        Some(&fh),
63        0,
64        file_size,
65        VMFileMappingFlags::Read,
66    )?;
67    // SAFETY: `vstart` has just been mapped using `file_size` as the size,
68    // so it is safe to create a slice of the same size.
69    let buf = unsafe { vstart.to_slice::<u8>(file_size) };
70    let elf_bin = Elf64File::read(buf).map_err(|_| SvsmError::Mem)?;
71
72    let alloc_info = elf_bin.image_load_vaddr_alloc_info();
73    let virt_base = alloc_info.range.vaddr_begin;
74    let entry = elf_bin.get_entry(virt_base);
75
76    let new_task = create_user_task(entry.try_into().unwrap(), root, task_name(binary))?;
77
78    for seg in elf_bin.image_load_segment_iter(virt_base) {
79        let virt_start = VirtAddr::from(seg.vaddr_range.vaddr_begin);
80        let virt_end = VirtAddr::from(seg.vaddr_range.vaddr_end).align_up(PAGE_SIZE);
81        let file_offset = seg.file_range.offset_begin;
82        let len = virt_end - virt_start;
83        let file_size = seg.file_range.offset_end - seg.file_range.offset_begin;
84        let flags = convert_elf_phdr_flags(seg.flags);
85
86        if file_offset > 0 {
87            if file_size > len {
88                return Err(SvsmError::Elf(elf::ElfError::InvalidFileRange));
89            }
90
91            // Handle unaligned VirtAddr and Offset
92            let start_aligned = virt_start.page_align();
93            let offset = file_offset - virt_start.page_offset();
94            let size = file_size + virt_start.page_offset();
95            new_task.mmap_user(start_aligned, Some(&fh), offset, size, flags)?;
96
97            let size_aligned = align_up(size, PAGE_SIZE);
98            if size_aligned < len {
99                let start_anon = start_aligned.const_add(size_aligned);
100                let remaining_len = len - size_aligned;
101                new_task.mmap_user(start_anon, None, 0, remaining_len, flags)?;
102            }
103        } else {
104            new_task.mmap_user(virt_start, None, 0, len, flags)?;
105        }
106    }
107
108    // Make sure the mapping is gone before calling schedule
109    drop(vstart);
110
111    // Setup 64k of task stack
112    let user_stack_size: usize = 64 * 1024;
113    let stack_flags: VMFileMappingFlags = VMFileMappingFlags::Fixed | VMFileMappingFlags::Write;
114    let stack_addr = USER_MEM_END - user_stack_size;
115    new_task.mmap_user(stack_addr, None, 0, user_stack_size, stack_flags)?;
116
117    finish_user_task(new_task.clone());
118    schedule();
119
120    Ok(new_task.get_task_id())
121}
