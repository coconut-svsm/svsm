36fn convert_elf_phdr_flags(flags: Elf64PhdrFlags) -> VMFileMappingFlags {
37    let mut vm_flags = VMFileMappingFlags::Fixed;
38
39    if flags.contains(Elf64PhdrFlags::WRITE) {
40        vm_flags |= VMFileMappingFlags::Write | VMFileMappingFlags::Private;
41    }
42
43    if flags.contains(Elf64PhdrFlags::EXECUTE) {
44        vm_flags |= VMFileMappingFlags::Execute;
45    }
46
47    vm_flags
48}

50/// Returns the name of the binary file without preceeding directories. This is
51/// used as the official task name.
52fn task_name(binary: &str) -> &str {
53    let mut items = binary.split('/').filter(|x| !x.is_empty());
54    match items.nth_back(0) {
55        Some(p) => p,
56        None => "unknown",
57    }
58}

60/// Loads and executes an user-mode ELF binary into the current tasks address
61/// space.
62///
63/// # Arguments
64///
65/// * info: Instance of [`UserExecInfo`] describing the user-mode task.
66///
67/// # Returns
68///
69/// [`Ok(TaskPointer)`] on success, [`Err(SvsmError)`] on failure.
70pub fn exec(info: UserExecInfo) -> Result<u64, SvsmError> {
71    let fh = open_read(&info.binary)?;
72    let file_size = fh.size();
73
74    let current_task = current_task();
75
76    let vstart = current_task.mmap_kernel_guard(
77        VirtAddr::new(0),
78        Some(&fh),
79        0,
80        file_size,
81        VMFileMappingFlags::Read,
82    )?;
83
84    // SAFETY: `vstart` has just been mapped using `file_size` as the size,
85    // so it is safe to create a slice of the same size.
86    let buf = unsafe { vstart.to_slice::<u8>(file_size) };
87    let elf_bin = Elf64File::read(buf).map_err(|_| SvsmError::Mem)?;
88
89    let alloc_info = elf_bin.image_load_vaddr_alloc_info();
90    let virt_base = alloc_info.range.vaddr_begin;
91    let entry = elf_bin.get_entry(virt_base);
92
93    for seg in elf_bin.image_load_segment_iter(virt_base) {
94        let virt_start = VirtAddr::from(seg.vaddr_range.vaddr_begin);
95        let virt_end = VirtAddr::from(seg.vaddr_range.vaddr_end);
96        let file_offset = seg.file_range.offset_begin;
97        let mem_size = virt_end - virt_start;
98        let file_size = seg.file_range.offset_end - seg.file_range.offset_begin;
99        let flags = convert_elf_phdr_flags(seg.flags);
100
101        // Handle unaligned VirtAddr and Offset
102        let map_start = virt_start.page_align();
103        let mem_map_size = mem_size + virt_start.page_offset();
104
105        if file_size == 0 {
106            // No file backing - Map the segment with anonymous memory
107            mmap_user(map_start, None, 0, mem_map_size, flags)?;
108            continue;
109        }
110
111        if file_size > mem_size {
112            return Err(SvsmError::Elf(elf::ElfError::InvalidFileRange));
113        }
114
115        let page_offset = virt_start.page_offset();
116        if file_offset % PAGE_SIZE != page_offset {
117            return Err(SvsmError::Elf(elf::ElfError::UnalignedSegmentAddress));
118        }
119
120        let map_offset = file_offset - page_offset;
121        let file_map_size = file_size + page_offset;
122        mmap_user(map_start, Some(&fh), map_offset, file_map_size, flags)?;
123
124        let file_map_size_aligned = align_up(file_map_size, PAGE_SIZE);
125        let zero_end = file_map_size_aligned.min(mem_map_size);
126
127        // Zero the part of the last file-backed page covered by p_memsz but not
128        // by p_filesz.  mmap_user() maps whole pages, so this would otherwise
129        // expose bytes following the segment in the ELF file.
130        if zero_end > file_map_size && seg.flags.contains(Elf64PhdrFlags::WRITE) {
131            zero_user_mem(map_start + file_map_size, zero_end - file_map_size)?;
132        }
133
134        if file_map_size_aligned < mem_map_size {
135            let remaining_len = mem_map_size - file_map_size_aligned;
136            mmap_user(
137                map_start + file_map_size_aligned,
138                None,
139                0,
140                remaining_len,
141                flags,
142            )?;
143        }
144    }
145
146    // Make sure the mapping is gone before calling schedule
147    drop(vstart);
148
149    // Setup 64k of task stack
150    let user_stack_size: usize = 64 * 1024;
151    let stack_flags: VMFileMappingFlags = VMFileMappingFlags::Fixed | VMFileMappingFlags::Write;
152    let stack_addr = USER_MEM_END - user_stack_size;
153    mmap_user(stack_addr, None, 0, user_stack_size, stack_flags)?;
154
155    Ok(entry)
156}

158/// Starts a new task and sets it up for loading and executing a user-mode
159/// binary.
160///
161/// # Arguments
162///
163/// * binary: Path to file in the file-system
164/// * root: Root directory associated with the new task
165///
166/// # Returns
167///
168/// [`Ok(TaskPointer)`] on success, [`Err(SvsmError)`] on failure.
169pub fn exec_user<S: Into<String>>(
170    binary: S,
171    root: Arc<dyn Directory>,
172) -> Result<TaskPointer, SvsmError> {
173    let info = Box::new(UserExecInfo::new(binary.into()));
174    let name = Arc::from(task_name(&info.binary));
175    let new_task = create_user_task(info, root, name)?;
176
177    finish_user_task(new_task.clone());
178    schedule();
179
180    Ok(new_task)
181}
