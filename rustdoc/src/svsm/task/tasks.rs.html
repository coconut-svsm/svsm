44/// This trait is used to describe a type that can be used as the start
45/// parameter for a kernel thread.  The thread start path requires that all
46/// start paremeters be able to fit into a single register, which constrains
47/// them to `usize`.  Other types can be used as long as there is an
48/// appropriate mechanism to covert those types to and from the `usize` that
49/// must be passed via the parameter register.  This trait defines such a
50/// conversion mechanism.
51///
52/// # Safety
53///
54/// Implementations of this thread must ensure that the `to_usize` and
55/// `from_usize` methods correctly preserve the safety of the parameter as it
56/// transitions from its native type to and from a temporary `usize`
57/// representation.
58pub unsafe trait KernelThreadStartParameter {
59    fn to_usize(parameter: Self) -> usize;
60
61    /// # Safety
62    /// This function must only be called with a `usize` parameter that was
63    /// previously returned from a call to `to_usize()`.
64    unsafe fn from_usize(parameter: usize) -> Self;
65}
66
67// SAFETY: types that implement `From` for conversions to and from `usize` are
68// trivially convertible to and from `usize` and thus will always preserve
69// safety of the parameter during conversion.
70unsafe impl<T> KernelThreadStartParameter for T
71where
72    T: From<usize>,
73    usize: From<T>,
74{
75    fn to_usize(parameter: Self) -> usize {
76        usize::from(parameter)
77    }
78
79    unsafe fn from_usize(parameter: usize) -> Self {
80        Self::from(parameter)
81    }
82}

143#[derive(Debug, Default)]
144struct TaskIDAllocator {
145    next_id: AtomicU32,
146}
147
148impl TaskIDAllocator {
149    const fn new() -> Self {
150        Self {
151            next_id: AtomicU32::new(INITIAL_TASK_ID + 1),
152        }
153    }
154
155    fn next_id(&self) -> u32 {
156        let mut id = self.next_id.fetch_add(1, Ordering::Relaxed);
157        // Reserve IDs of 0 and 1
158        while (id == 0_u32) || (id == INITIAL_TASK_ID) {
159            id = self.next_id.fetch_add(1, Ordering::Relaxed);
160        }
161        id
162    }
163}

273struct CreateTaskArguments {
274    // The thread start information.
275    start_info: ThreadStartInfo,
276
277    // The name of the task.
278    name: String,
279
280    // For a user task, supplies the `VMR` that will represent the user-mode
281    // address space.
282    vm_user_range: Option<VMR>,
283
284    // The root directory that will be associated with this task.
285    rootdir: Arc<dyn Directory>,
286
287    // Share state with another task (aka create a thread)
288    thread_of: Option<TaskPointer>,
289}

291impl Task {
292    fn create_common(cpu: &PerCpu, args: CreateTaskArguments) -> Result<TaskPointer, SvsmError> {
293        let mut pgtable = cpu.get_pgtable().clone_shared()?;
294
295        cpu.populate_page_table(&mut pgtable);
296
297        let (task_mm, objtree) = {
298            if let Some(parent_thread) = args.thread_of {
299                (parent_thread.mm.clone(), parent_thread.objs.clone())
300            } else {
301                (
302                    Arc::new(TaskMM::create(args.vm_user_range)?),
303                    Arc::new(RWLock::new(BTreeMap::new())),
304                )
305            }
306        };
307
308        let xsa = Self::allocate_xsave_area();
309        let xsa_addr = u64::from(xsa.vaddr()) as usize;
310
311        // Determine which kernel-mode entry/exit routines will be used for
312        // this task.
313        let (entry_return, exit_return) = match args.start_info {
314            ThreadStartInfo::User(_) => (return_new_task as *const () as usize, None),
315            ThreadStartInfo::Kernel(ref info) => {
316                (info.start_routine, Some(task_exit as *const () as usize))
317            }
318        };
319
320        let mut shadow_stack_offset = VirtAddr::null();
321        let mut shadow_stack_base = VirtAddr::null();
322        let shadow_stack_mapping = if is_cet_ss_supported() {
323            // Allocate shadow stack and safe top_of_stack offset
324            let shadow_stack = VMKernelStack::new_shadow()?;
325            let offset = shadow_stack.top_of_stack();
326            let shadow_stack_page = shadow_stack.shadow_page();
327
328            let base_token_addr;
329
330            // Map shadow stack into virtual address range
331            let mapping =
332                TaskKernelMapping::new(task_mm.clone(), Arc::new(Mapping::new(shadow_stack)))?;
333            let stack_base = mapping.virt_addr();
334
335            // Initialize shadow stack
336            (base_token_addr, shadow_stack_offset) = init_shadow_stack(
337                &shadow_stack_page,
338                stack_base + offset,
339                ShadowStackInit::Normal {
340                    entry_return,
341                    exit_return,
342                },
343            );
344
345            if let Some(base_addr) = base_token_addr {
346                shadow_stack_base = base_addr;
347            }
348
349            Some(mapping)
350        } else {
351            None
352        };
353
354        // Call the correct stack creation routine for this task.
355        let (stack, raw_bounds, rsp_offset) = match args.start_info {
356            ThreadStartInfo::User(entry) => Self::allocate_utask_stack(cpu, entry, xsa_addr)?,
357            ThreadStartInfo::Kernel(ref info) => Self::allocate_ktask_stack(
358                cpu,
359                info.entry,
360                info.start_routine,
361                xsa_addr,
362                info.start_parameter,
363            )?,
364        };
365        let kernel_stack_mapping = TaskKernelMapping::new(task_mm.clone(), stack)?;
366        let stack_start = kernel_stack_mapping.virt_addr();
367
368        task_mm.kernel_range().populate(&mut pgtable);
369
370        // Remap at the per-task offset
371        let bounds = MemoryRegion::new(stack_start + raw_bounds.start().into(), raw_bounds.len());
372        // Stack frames should be 16b-aligned
373        debug_assert!(bounds.end().is_aligned(16));
374
375        Ok(Arc::new(Task {
376            rsp: bounds
377                .end()
378                .checked_sub(rsp_offset)
379                .expect("Invalid stack offset from task stack allocator")
380                .bits() as u64,
381            ssp: shadow_stack_offset,
382            xsa,
383            stack_bounds: bounds,
384            shadow_stack_base,
385            page_table: SpinLock::new(pgtable),
386            _kernel_stack: kernel_stack_mapping,
387            _shadow_stack: shadow_stack_mapping,
388            mm: task_mm,
389            sched_state: RWLock::new(TaskSchedState {
390                idle_task: false,
391                state: TaskState::RUNNING,
392                cpu_index: cpu.get_cpu_index(),
393            }),
394            name: args.name,
395            id: TASK_ID_ALLOCATOR.next_id(),
396            rootdir: args.rootdir,
397            list_link: LinkedListAtomicLink::default(),
398            runlist_link: LinkedListAtomicLink::default(),
399            objs: objtree,
400        }))
401    }

435            thread_of: None,

531    pub fn fault(&self, vaddr: VirtAddr, write: bool) -> Result<(), SvsmError> {
532        if vaddr >= USER_MEM_START && vaddr < USER_MEM_END && self.mm.has_user() {
533            let vmr = self.mm.user_range().unwrap();
534            let mut pgtbl = self.page_table.lock();
535            vmr.populate_addr(&mut pgtbl, vaddr);
536            vmr.handle_page_fault(vaddr, write)?;
537            Ok(())
538        } else {
539            Err(SvsmError::Mem)
540        }

552    fn allocate_ktask_stack(
553        cpu: &PerCpu,
554        entry: usize,
555        start_routine: usize,
556        xsa_addr: usize,
557        start_parameter: usize,
558    ) -> Result<(Arc<Mapping>, MemoryRegion<VirtAddr>, usize), SvsmError> {
559        let (mapping, bounds) = Task::allocate_stack_common()?;
560
561        let percpu_mapping = cpu.new_mapping(mapping.clone())?;
562
563        // We need to setup a context on the stack that matches the stack layout
564        // defined in switch_context below.
565        let stack_tos = percpu_mapping.virt_addr() + bounds.end().bits();
566        // Make space for the task termination handler
567        let stack_offset = size_of::<u64>();
568        let stack_ptr = stack_tos
569            .checked_sub(stack_offset)
570            .unwrap()
571            .as_mut_ptr::<u8>();
572        // To ensure stack frames are 16b-aligned, ret_addr must be 16b-aligned
573        // so that (%rsp + 8) is 16b-aligned after the ret instruction in
574        // switch_context
575        debug_assert!(
576            VirtAddr::from(stack_ptr)
577                .checked_sub(8)
578                .unwrap()
579                .is_aligned(16)
580        );
581        // Make sure there is room for TaskContext
582        debug_assert!(
583            (percpu_mapping.virt_addr() + bounds.start().bits())
584                <= VirtAddr::from(stack_ptr)
585                    .checked_sub(size_of::<TaskContext>())
586                    .unwrap()
587        );
588
589        // 'Push' the task frame onto the stack
590        //
591        // SAFETY: we ensure that both `TaskContext` and the function pointer
592        // can be written to valid memory. The address storing the function
593        // pointer is always 8b-aligned.
594        unsafe {
595            let task_context = stack_ptr
596                .sub(size_of::<TaskContext>())
597                .cast::<TaskContext>();
598            // The processor flags must always be in a default state, unrelated
599            // to the flags of the caller.  In particular, interrupts must be
600            // disabled because the task switch code expects to execute a new
601            // task with interrupts disabled.
602            (*task_context).flags = 2;
603            // ret_addr
604            (*task_context).regs.rdi = entry;
605            // xsave area addr
606            (*task_context).regs.rsi = xsa_addr;
607            // start argument parameter.
608            (*task_context).regs.rdx = start_parameter;
609            (*task_context).ret_addr = start_routine as u64;
610            // Task termination handler for when entry point returns
611            stack_ptr.cast::<u64>().write(task_exit as *const () as u64);
612        }
613
614        Ok((mapping, bounds, stack_offset + size_of::<TaskContext>()))
615    }

617    fn allocate_utask_stack(
618        cpu: &PerCpu,
619        user_entry: usize,
620        xsa_addr: usize,
621    ) -> Result<(Arc<Mapping>, MemoryRegion<VirtAddr>, usize), SvsmError> {
622        let (mapping, bounds) = Task::allocate_stack_common()?;
623        // Do not run user-mode with IRQs enabled on platforms which are not
624        // ready for it.
625        let iret_rflags: usize = if SVSM_PLATFORM.use_interrupts() {
626            2 | EFLAGS_IF
627        } else {
628            2
629        };
630
631        let percpu_mapping = cpu.new_mapping(mapping.clone())?;
632
633        // We need to setup a context on the stack that matches the stack layout
634        // defined in switch_context below.
635        let stack_tos = percpu_mapping.virt_addr() + bounds.end().bits();
636        // Make space for the IRET frame
637        let stack_offset = size_of::<X86ExceptionContext>();
638        let stack_ptr = stack_tos
639            .checked_sub(stack_offset)
640            .unwrap()
641            .as_mut_ptr::<u8>();
642        // return_new_task is asm code, so %rsp must be 16b-aligned after the
643        // ret instruction in switch_context to ensure stack frames are
644        // 16b-aligned
645        debug_assert!(VirtAddr::from(stack_ptr).is_aligned(16));
646        // Make sure there is room for TaskContext
647        debug_assert!(
648            (percpu_mapping.virt_addr() + bounds.start().bits())
649                <= VirtAddr::from(stack_ptr)
650                    .checked_sub(size_of::<TaskContext>())
651                    .unwrap()
652        );
653
654        // 'Push' the task frame onto the stack
655        //
656        // SAFETY: we ensure that both `X86ExceptionContext` and `TaskContext`
657        // can be written to valid memory.
658        unsafe {
659            // Setup IRQ return frame.  User-mode tasks always run with
660            // interrupts enabled.
661            let mut iret_frame = X86ExceptionContext::default();
662            iret_frame.frame.rip = user_entry;
663            iret_frame.frame.cs = (SVSM_USER_CS | 3).into();
664            iret_frame.frame.flags = iret_rflags;
665            iret_frame.frame.rsp = (USER_MEM_END - 8).into();
666            iret_frame.frame.ss = (SVSM_USER_DS | 3).into();
667            debug_assert!(is_aligned(iret_frame.frame.rsp + 8, 16));
668
669            // Copy IRET frame to stack
670            let stack_iret_frame = stack_ptr.cast::<X86ExceptionContext>();
671            *stack_iret_frame = iret_frame;
672
673            let task_context = TaskContext {
674                regs: X86GeneralRegs {
675                    rdi: xsa_addr, // XSAVE area addr
676                    ..Default::default()
677                },
678                ret_addr: VirtAddr::from(return_new_task as *const ())
679                    .bits()
680                    .try_into()
681                    .unwrap(),
682                ..Default::default()
683            };
684
685            let stack_task_context = stack_ptr
686                .sub(size_of::<TaskContext>())
687                .cast::<TaskContext>();
688            *stack_task_context = task_context;
689        }
690
691        Ok((mapping, bounds, stack_offset + size_of::<TaskContext>()))
692    }

769    /// Adds an object to the current task.
770    ///
771    /// # Arguments
772    ///
773    /// * `obj` - The object to be added.
774    ///
775    /// # Returns
776    ///
777    /// * `Result<ObjHandle, SvsmError>` - Returns the object handle for the object
778    ///   to be added if successful, or an `SvsmError` on failure.
779    ///
780    /// # Errors
781    ///
782    /// This function will return an error if allocating the object handle fails.
783    pub fn add_obj(&self, obj: Arc<dyn Obj>) -> Result<ObjHandle, SvsmError> {
784        let mut objs = self.objs.lock_write();
785        let last_key = objs
786            .keys()
787            .last()
788            .map_or(Some(0), |k| u32::from(*k).checked_add(1))
789            .ok_or(SvsmError::from(ObjError::InvalidHandle))?;
790        let id = ObjHandle::new(if last_key != objs.len() as u32 {
791            objs.keys()
792                .enumerate()
793                .find(|&(i, key)| i as u32 != u32::from(*key))
794                .unwrap()
795                .0 as u32
796        } else {
797            last_key
798        });
799
800        objs.insert(id, obj);
801
802        Ok(id)
803    }

805    /// Adds an object to the current task and maps it to a given object-id.
806    ///
807    /// # Arguments
808    ///
809    /// * `obj` - The object to be added.
810    /// * `handle` - Object handle to reference the object.
811    ///
812    /// # Returns
813    ///
814    /// * `Result<ObjHandle, SvsmError>` - Returns the object handle for the object
815    ///   to be added if successful, or an `SvsmError` on failure.
816    ///
817    /// # Errors
818    ///
819    /// This function will return an error if allocating the object handle
820    /// fails or the object id is already in use.
821    pub fn add_obj_at(&self, obj: Arc<dyn Obj>, handle: ObjHandle) -> Result<ObjHandle, SvsmError> {
822        let mut objs = self.objs.lock_write();
823
824        if objs.get(&handle).is_some() {
825            return Err(SvsmError::from(ObjError::Busy));
826        }
827
828        objs.insert(handle, obj);
829
830        Ok(handle)
831    }

833    /// Removes an object from the current task.
834    ///
835    /// # Arguments
836    ///
837    /// * `id` - The ObjHandle for the object to be removed.
838    ///
839    /// # Returns
840    ///
841    /// * `Result<Arc<dyn Obj>>, SvsmError>` - Returns the removed `Arc<dyn Obj>`
842    ///   on success, or an `SvsmError` on failure.
843    ///
844    /// # Errors
845    ///
846    /// This function will return an error if the object handle id does not
847    /// exist in the current task.
848    pub fn remove_obj(&self, id: ObjHandle) -> Result<Arc<dyn Obj>, SvsmError> {
849        self.objs
850            .lock_write()
851            .remove(&id)
852            .ok_or(ObjError::NotFound.into())
853    }

856    ///
857    /// # Arguments
858    ///
859    /// * `id` - The ObjHandle for the object to be retrieved.
860    ///
861    /// # Returns
862    ///
863    /// * `Result<Arc<dyn Obj>>, SvsmError>` - Returns the `Arc<dyn Obj>` on
864    ///   success, or an `SvsmError` on failure.
865    ///
866    /// # Errors
867    ///
868    /// This function will return an error if the object handle id does not exist

871        self.objs
872            .lock_read()
873            .get(&id)
874            .cloned()
875            .ok_or(ObjError::NotFound.into())

879pub fn is_task_fault(vaddr: VirtAddr) -> bool {
880    (vaddr >= USER_MEM_START && vaddr < USER_MEM_END)
881        || (vaddr >= SVSM_PERTASK_BASE && vaddr < SVSM_PERTASK_END)
882}

892/// Runs the first time a new task is scheduled, in the context of the new
893/// task. Any first-time initialization and setup work for a new task that
894/// needs to happen in its context must be done here.
895/// # Safety
896/// The caller is required to verify the correctness of the save area address.
897#[unsafe(no_mangle)]
898unsafe fn setup_user_task(xsa_addr: u64) {
899    // SAFETY: caller needs to make sure xsa_addr is valid and points to a
900    // memory region of sufficient size.
901    unsafe {
902        // Needs to be the first function called here.
903        setup_new_task_common(xsa_addr);
904    }
905    task_attach_console();
906}
907
908unsafe fn setup_new_task_common(xsa_addr: u64) {
909    // Re-enable IRQs here, as they are still disabled from the
910    // schedule()/sched_init() functions. After the context switch the IrqGuard
911    // from the previous task is not dropped, which causes IRQs to stay
912    // disabled in the new task.
913    // This only needs to be done for the first time a task runs. Any
914    // subsequent task switches will go through schedule() and there the guard
915    // is dropped, re-enabling IRQs.
916
917    irqs_enable();
918
919    // Perform housekeeping actions following a task switch.
920    after_task_switch();
921
922    // SAFETY: The caller takes responsibility for the correctness of the save
923    // area address.
924    unsafe {
925        sse_restore_context(xsa_addr);
926    }
927}
928
929/// # Safety
930/// This function must only be invoked as a thread start function, and must
931/// be invoked according to the start parameter data type expected by the entry
932/// point.
933unsafe fn run_kernel_task<T: KernelThreadStartParameter>(
934    entry: fn(T),
935    xsa_addr: u64,
936    start_parameter: usize,
937) {
938    // SAFETY: the save area address is provided by the context switch assembly
939    // code.
940    unsafe {
941        setup_new_task_common(xsa_addr);
942    }
943
944    // SAFETY: the `usize` start parameter was generated from an earlier call
945    // to `KernelThreadStartParameter::to_usize` when the start argument was
946    // prepared for use in the initial stack context, and therefore the safety
947    // requirements of `from_usize` are met.  The exception is if this task
948    // starts via an unsafe start function, in which case safety is the
949    // responsibility of whoever created the thread, and is not required to be
950    // assured here.
951    entry(unsafe { T::from_usize(start_parameter) });
952}

954fn task_exit() {
955    current_task_terminated();
956    schedule();
957}
