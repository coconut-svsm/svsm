7//! Round-Robin scheduler implementation for COCONUT-SVSM
8//!
9//! This module implements a round-robin scheduler for cooperative multi-tasking.
10//! It works by assigning a single owner for each struct [`Task`]. The owner
11//! depends on the state of the task:
12//!
13//! * [`RUNNING`] A task in running state is owned by the [`RunQueue`] and either
14//!   stored in the `run_list` (when the task is not actively running) or in
15//!   `current_task` when it is scheduled on the CPU.
16//! * [`BLOCKED`] A task in this state is waiting for an event to become runnable
17//!   again. It is owned by a wait object when in this state.
18//! * [`TERMINATED`] The task is about to be destroyed and owned by the [`RunQueue`].
19//!
20//! The scheduler is cooperative. A task runs until it voluntarily calls the
21//! [`schedule()`] function.
22//!
23//! Only when a task is in [`RUNNING`] or [`TERMINATED`] state it is assigned to a
24//! specific CPU. Tasks in the [`BLOCKED`] state have no CPU assigned and will run
25//! on the CPU where their event is triggered that makes them [`RUNNING`] again.
26//!
27//! [`RUNNING`]: super::tasks::TaskState::RUNNING
28//! [`BLOCKED`]: super::tasks::TaskState::BLOCKED
29//! [`TERMINATED`]: super::tasks::TaskState::TERMINATED
30

56/// A RunQueue implementation that uses an RBTree to efficiently sort the priority
57/// of tasks within the queue.
58#[derive(Debug, Default)]
59pub struct RunQueue {
60    /// Linked list with runable tasks
61    run_list: LinkedList<TaskRunListAdapter>,
62
63    /// Pointer to currently running task
64    current_task: Option<TaskPointer>,
65
66    /// Idle task - runs when there is no other runnable task
67    idle_task: Option<TaskPointer>,
68
69    /// Temporary storage for tasks which are about to be terminated
70    terminated_task: Option<TaskPointer>,
71
72    /// Pointer to a task that should be woken when returning from idle
73    wake_from_idle: Option<TaskPointer>,
74
75    /// Pointer to a task that is requesting an affinity change to another
76    /// processor, along with the CPU index describing the new affinity..
77    set_affinity: Option<(TaskPointer, usize)>,
78}
79
80impl RunQueue {
81    /// Create a new runqueue for an id. The id would normally be set
82    /// to the APIC ID of the CPU that owns the runqueue and is used to
83    /// determine the affinity of tasks.
84    pub fn new() -> Self {
85        Self {
86            run_list: LinkedList::new(TaskRunListAdapter::new()),
87            current_task: None,
88            idle_task: None,
89            terminated_task: None,
90            wake_from_idle: None,
91            set_affinity: None,
92        }
93    }
94
95    /// Find the next task to run, which is either the task at the front of the
96    /// run_list or the idle task, if the run_list is empty.
97    ///
98    /// # Returns
99    ///
100    /// Pointer to next task to run
101    ///
102    /// # Panics
103    ///
104    /// Panics if there are no tasks to run and no idle task has been
105    /// allocated via [`set_idle_task()`](Self::set_idle_task).
106    fn get_next_task(&mut self) -> TaskPointer {
107        self.run_list
108            .pop_front()
109            .unwrap_or_else(|| self.idle_task.clone().unwrap())
110    }
111
112    /// Update state before a task is scheduled out. Non-idle tasks in RUNNING
113    /// state will be put at the end of the run_list. Terminated tasks will be
114    /// stored in the terminated_task field of the RunQueue and be destroyed
115    /// after the task-switch.
116    fn handle_task(&mut self, task: TaskPointer) {
117        if task.is_running() && !task.is_idle_task() {
118            self.run_list.push_back(task);
119        } else if task.is_terminated() {
120            self.terminated_task = Some(task);
121        }
122    }
123
124    /// Initialized the scheduler for this (RunQueue)[RunQueue]. This method is
125    /// called on the very first scheduling event when there is no current task
126    /// yet.
127    ///
128    /// # Returns
129    ///
130    /// [TaskPointer] to the first task to run
131    pub fn schedule_init(&mut self) -> TaskPointer {
132        let task = self.get_next_task();
133        self.current_task = Some(task.clone());
134        task
135    }
136
137    /// Prepares a task switch. The function checks if a task switch needs to
138    /// be done and return pointers to the current and next task. It will
139    /// also call `handle_task()` on the current task in case a task-switch
140    /// is requested.
141    ///
142    /// # Returns
143    ///
144    /// `None` when no task-switch is needed.
145    /// `Some` with current and next task in case a task-switch is required.
146    ///
147    /// # Panics
148    ///
149    /// Panics if there is no current task.
150    pub fn schedule_prepare(&mut self) -> Option<(TaskPointer, TaskPointer)> {
151        // Remove current and put it back into the RunQueue in case it is still
152        // runnable. This is important to make sure the last runnable task
153        // keeps running, even if it calls schedule()
154        let current = self.current_task.take().unwrap();
155        self.handle_task(current.clone());
156
157        // Get next task and update current_task state
158        let next = self.get_next_task();
159        self.current_task = Some(next.clone());
160
161        // Check if task switch is needed
162        if current != next {
163            Some((current, next))
164        } else {
165            None
166        }
167    }
168
169    pub fn current_task_id(&self) -> u32 {
170        self.current_task
171            .as_ref()
172            .map_or(INITIAL_TASK_ID, |t| t.get_task_id())
173    }
174
175    /// Sets the idle task for this RunQueue. This function sets a
176    /// OnceCell at the end and can thus be only called once.
177    ///
178    /// # Returns
179    ///
180    /// Ok(()) on success, SvsmError on failure
181    ///
182    /// # Panics
183    ///
184    /// Panics if the idle task was already set.
185    pub fn set_idle_task(&mut self, task: TaskPointer) {
186        task.set_idle_task();
187
188        // Add idle task to global task list
189        TASKLIST.lock().list().push_front(task.clone());
190
191        self.idle_task.replace(task);
192    }
193
194    /// Gets a pointer to the current task
195    ///
196    /// # Panics
197    ///
198    /// Panics if there is no current task.
199    pub fn current_task(&self) -> TaskPointer {
200        self.current_task.as_ref().unwrap().clone()
201    }
202
203    /// Wakes a task from idle if required.
204    ///
205    /// # Returns
206    ///
207    /// An `Option<TaskPointer>` indicating which task should be woken, if
208    /// any.
209    pub fn wake_from_idle(&mut self) -> Option<TaskPointer> {
210        self.wake_from_idle.take()
211    }
212}

214/// Global task list
215/// This contains every task regardless of affinity or run state.
216#[derive(Debug, Default)]
217pub struct TaskList {
218    list: Option<LinkedList<TaskListAdapter>>,
219}
220
221impl TaskList {
222    pub const fn new() -> Self {
223        Self { list: None }
224    }
225
226    pub fn list(&mut self) -> &mut LinkedList<TaskListAdapter> {
227        self.list
228            .get_or_insert_with(|| LinkedList::new(TaskListAdapter::new()))
229    }
230
231    pub fn get_task(&self, id: u32) -> Option<TaskPointer> {
232        let task_list = &self.list.as_ref()?;
233        let mut cursor = task_list.front();
234        while let Some(task) = cursor.get() {
235            if task.get_task_id() == id {
236                return cursor.clone_pointer();
237            }
238            cursor.move_next();
239        }
240        None
241    }
242
243    /// # Safety
244    /// The caller must ensure that `task` is already a member of this task
245    /// list.
246    unsafe fn terminate(&mut self, task: TaskPointer) {
247        // Set the task state as terminated. If the task being terminated is the
248        // current task then the task context will still need to be in scope until
249        // the next schedule() has completed. Schedule will keep a reference to this
250        // task until some time after the context switch.
251        task.set_task_terminated();
252        // SAFETY: `task` must be a task pointer that is part of this list.
253        let mut cursor = unsafe { self.list().cursor_mut_from_ptr(task.as_ref()) };
254        cursor.remove();
255    }
256}

260/// Creates, initializes and starts a new kernel task. Note that the task has
261/// already started to run before this function returns.
262///
263/// # Arguments
264///
265/// * entry: The function to run as the new tasks main function
266///
267/// # Returns
268///
269/// A new instance of [`TaskPointer`] on success, [`SvsmError`] on failure.
270pub fn start_kernel_task(
271    start_info: KernelThreadStartInfo,
272    name: String,
273) -> Result<TaskPointer, SvsmError> {
274    let cpu = this_cpu();
275    let task = Task::create(cpu, start_info, name)?;
276    TASKLIST.lock().list().push_back(task.clone());
277
278    // Put task on the runqueue of this CPU
279    cpu.runqueue().lock_write().handle_task(task.clone());
280
281    schedule();
282
283    Ok(task)
284}

286/// Creates, initializes and starts a new kernel thread of the currently
287/// running kernel task. Note that the thread has already started to run before
288/// this function returns.
289///
290/// # Arguments
291///
292/// * `entry` -  The function to run as the new task's main function
293/// * `start_parameter` - Parameter of type `usize` to pass to the new threads main function.
294///
295/// # Returns
296///
297/// A new instance of [`TaskPointer`] on success, [`SvsmError`] on failure.
298pub fn start_kernel_thread(start_info: KernelThreadStartInfo) -> Result<TaskPointer, SvsmError> {
299    let current_task = current_task();
300    let cpu = this_cpu();
301    let task = Task::create_thread(
302        cpu,
303        start_info,
304        current_task.get_task_name().clone(),
305        current_task,
306    )?;
307    TASKLIST.lock().list().push_back(task.clone());
308
309    // Put task on the runqueue of this CPU
310    cpu.runqueue().lock_write().handle_task(task.clone());
311
312    schedule();
313
314    Ok(task)
315}

362/// Terminates the current task.
363///
364/// # Panic
365///
366/// This function must only be called after scheduling is initialized, otherwise it will panic.
367pub fn current_task_terminated() {
368    let cpu = this_cpu();
369    let mut rq = cpu.runqueue().lock_write();
370    let task_node = rq
371        .current_task
372        .as_mut()
373        .expect("Task termination handler called when there is no current task");
374    // SAFETY: the scheduler guarantees that `current_task` always points to a
375    // valid task, and every task has its pointer pushed into the global task
376    // list during its creation.
377    unsafe { TASKLIST.lock().terminate(task_node.clone()) }
378}

380pub fn terminate() {
381    current_task_terminated();
382    schedule();
383}

385pub fn go_idle() {
386    // Mark this task as blocked and indicate that it is waiting for wake after
387    // idle.  Only one task on each CPU can be in the wake-from-idle state at
388    // one time.
389    let task = this_cpu().current_task();
390    task.set_task_blocked();
391    let mut runqueue = this_cpu().runqueue().lock_write();
392    assert!(runqueue.wake_from_idle.is_none());
393    runqueue.wake_from_idle = Some(task);
394    drop(runqueue);
395
396    // Find another task to run.  If no other task is runnable, then the idle
397    // thread will execute.
398    schedule();
399}

401pub fn set_affinity(cpu_index: usize) {
402    // Mark the current task as blocked so it is not scheduled again.
403    let task = this_cpu().current_task();
404    task.set_task_blocked();
405
406    // Mark this task as the task pending an affinity change.
407    let mut runqueue = this_cpu().runqueue().lock_write();
408    assert!(runqueue.set_affinity.is_none());
409    runqueue.set_affinity = Some((task, cpu_index));
410    drop(runqueue);
411
412    // Find another task to run.  The scheduler will complete the affinity
413    // change once a new task has been selected on this processor.
414    schedule();
415}

417// SAFETY: This function returns a raw pointer to a task. It is safe
418// because this function is only used in the task switch code, which also only
419// takes a single reference to the next and previous tasks. Also, this
420// function works on an Arc, which ensures that only a single mutable reference
421// can exist.
422fn task_pointer(taskptr: TaskPointer) -> *const Task {
423    Arc::as_ptr(&taskptr)
424}

426// SAFETY: The caller is required to provide correct pointers for the previous
427// and current tasks.
428#[inline(always)]
429unsafe fn switch_to(prev: *const Task, next: *const Task) {
430    // SAFETY: Assuming the caller has provided the correct task pointers,
431    // the page table and stack information in those tasks are correct and
432    // can be used to switch to the correct page table and execution stack.
433    unsafe {
434        let cr3 = (*next).page_table.lock().cr3_value().bits() as u64;
435
436        // The location of a cpu-local stack that's mapped into every set of
437        // page tables for use during context switches.
438        //
439        // If an IRQ is raised after switching the page tables but before
440        // switching to the new stack, the CPU will try to access the old stack
441        // in the new page tables. To protect against this, we switch to another
442        // stack that's mapped into both the old and the new set of page tables.
443        // That way we always have a valid stack to handle exceptions on.
444        let tos_cs: u64 = this_cpu().get_top_of_context_switch_stack().unwrap().into();
445
446        // Switch to new task
447        asm!(
448            r#"
449            call switch_context
450            "#,
451            in("r12") prev as u64,
452            in("r13") next as u64,
453            in("r14") tos_cs,
454            in("r15") cr3,
455            options(att_syntax));
456    }
457}

459/// Initializes the [RunQueue] on the current CPU. It will switch to the idle
460/// task and initialize the current_task field of the RunQueue. After this
461/// function has ran it is safe to call [`schedule()`] on the current CPU.
462///
463/// # Safety
464///
465/// This function can only be called when it is known that there is no current
466/// task.  Otherwise, the run state can become corrupted, and thus future
467/// calculation of task pointers can be incorrect.
468pub unsafe fn schedule_init() {
469    let guard = IrqGuard::new();
470    // SAFETY: The caller guarantees that there is no current task, and the
471    // pointer obtained for the next task will always be correct, thus
472    // providing a guarantee that the task switch will be safe.
473    unsafe {
474        let next = task_pointer(this_cpu().schedule_init());
475        switch_to(null_mut(), next);
476    }
477    drop(guard);
478}

480fn preemption_checks() {
481    assert!(irq_nesting_count() == 0);
482    assert!(raw_get_tpr() == 0 || !SVSM_PLATFORM.use_interrupts());
483}

485/// Perform a task switch and hand the CPU over to the next task on the
486/// run-list. In case the current task is terminated, it will be destroyed after
487/// the switch to the next task.
488pub fn schedule() {
489    // check if preemption is safe
490    preemption_checks();
491
492    let guard = IrqGuard::new();
493
494    let work = this_cpu().schedule_prepare();
495
496    // !!! Runqueue lock must be release here !!!
497    if let Some((current, next)) = work {
498        // Update per-cpu mappings if needed
499        let cpu_index = this_cpu().get_cpu_index();
500
501        if next.update_cpu(cpu_index) != cpu_index {
502            // Task has changed CPU, update per-cpu mappings
503            let mut pt = next.page_table.lock();
504            this_cpu().populate_page_table(&mut pt);
505        }
506
507        // SAFETY: ths stack pointer is known to be correct.
508        unsafe {
509            this_cpu().set_tss_rsp0(next.stack_bounds.end());
510        }
511        if is_cet_ss_supported() {
512            // SAFETY: Task::exception_shadow_stack is always initialized when
513            // creating a new Task.
514            unsafe {
515                write_msr(PL0_SSP, next.shadow_stack_base.bits() as u64);
516            }
517        }
518
519        // Get task-pointers, consuming the Arcs and release their reference
520        //
521        // SAFETY: the scheduler guarantees that both `current` and `next`
522        // always point to valid tasks. The XSAVE area in each task must be
523        // valid and not aliased.
524        unsafe {
525            let a = task_pointer(current);
526            let b = task_pointer(next);
527            sse_save_context(u64::from((*a).xsa.vaddr()));
528
529            // Switch tasks
530            switch_to(a, b);
531
532            // We're now in the context of task pointed to by 'a'
533            // which was previously scheduled out.
534            sse_restore_context(u64::from((*a).xsa.vaddr()));
535        }
536    }
537
538    drop(guard);
539
540    // Perform housekeeping actions following a task switch.
541    after_task_switch();
542
543    // If the previous task had terminated then we can release
544    // it's reference here.
545    let _ = this_cpu().runqueue().lock_write().terminated_task.take();
546}

548struct SetAffinityMessage {
549    task: TaskPointer,
550}
551
552// SAFETY: The SetAffinityMessage structure contains no references other than
553// global references, and can safely rely on the default implementation of the
554// IPI message copy routines.
555unsafe impl IpiMessage for SetAffinityMessage {
556    fn invoke(&self) {
557        // Mark the task as runnable on the current CPU.  It will be selected
558        // to run at the next scheduling interval.  If the CPU is currently
559        // idle, then the idle task will wake and invoke the scheduler to
560        // invoke this task.
561        assert!(!self.task.is_running());
562        enqueue_task(self.task.clone());
563    }
564}
565
566pub fn after_task_switch() {
567    // Determine whether any task is pending an affinity change.  This must be
568    // done with the run queue locked, but the actual affinity change must
569    // happen without holding the run queue lock.
570    let mut runqueue = this_cpu().runqueue().lock_write();
571    let set_affinity = runqueue.set_affinity.take();
572    drop(runqueue);
573
574    if let Some((task, cpu_index)) = set_affinity {
575        // Send an IPI to the target processor indicating which task it should
576        // take.
577        let set_affinity_message = SetAffinityMessage { task };
578        send_multicast_ipi(IpiTarget::Single(cpu_index), &set_affinity_message);
579    }
580}
581
582fn enqueue_task(task: TaskPointer) {
583    task.set_task_running();
584    this_cpu().runqueue().lock_write().handle_task(task);
585}

587pub fn schedule_task(task: TaskPointer) {
588    enqueue_task(task);
589    schedule();

592global_asm!(
593    r#"
594        .text
595
596    switch_context:
597        // Save the current context. The layout must match the TaskContext structure.
598        pushfq
599        pushq   %rax
600        pushq   %rbx
601        pushq   %rcx
602        pushq   %rdx
603        pushq   %rsi
604        pushq   %rdi
605        pushq   %rbp
606        pushq   %r8
607        pushq   %r9
608        pushq   %r10
609        pushq   %r11
610        pushq   %r12
611        pushq   %r13
612        pushq   %r14
613        pushq   %r15
614        pushq   %rsp
615
616        // If `prev` is not null...
617        testq   %r12, %r12
618        // The initial stack is always mapped in the new page table.
619        jz      1f
620
621        // Save the current stack pointer
622        movq    %rsp, {TASK_RSP_OFFSET}(%r12)
623
624        // Switch to a stack pointer that's valid in both the old and new page tables.
625        mov     %r14, %rsp
626
627        cmpb    $0, {IS_CET_SUPPORTED}(%rip)
628        je      1f
629        // Save the current shadow stack pointer
630        rdssp   %rax
631        sub     $8, %rax
632        movq    %rax, {TASK_SSP_OFFSET}(%r12)
633        // Switch to a shadow stack that's valid in both page tables and move
634        // the "shadow stack restore token" to the old shadow stack.
635        mov     ${CONTEXT_SWITCH_RESTORE_TOKEN}, %rax
636        rstorssp (%rax)
637        saveprevssp
638
639    1:
640        // Switch to the new task state
641
642        // Switch to the new task page tables
643        mov     %r15, %cr3
644
645        cmpb    $0, {IS_CET_SUPPORTED}(%rip)
646        je      2f
647        // Switch to the new task shadow stack and move the "shadow stack
648        // restore token" back.
649        mov     {TASK_SSP_OFFSET}(%r13), %rdx
650        rstorssp (%rdx)
651        saveprevssp
652    2:
653
654        // Switch to the new task stack
655        movq    {TASK_RSP_OFFSET}(%r13), %rsp
656
657        // We've already restored rsp
658        addq    $8, %rsp
659
660        // Restore the task context
661        popq    %r15
662        popq    %r14
663        popq    %r13
664        popq    %r12
665        popq    %r11
666        popq    %r10
667        popq    %r9
668        popq    %r8
669        popq    %rbp
670        popq    %rdi
671        popq    %rsi
672        popq    %rdx
673        popq    %rcx
674        popq    %rbx
675        popq    %rax
676        popfq
677
678        ret
679    "#,
680    TASK_RSP_OFFSET = const offset_of!(Task, rsp),
681    TASK_SSP_OFFSET = const offset_of!(Task, ssp),
682    IS_CET_SUPPORTED = sym IS_CET_SUPPORTED,
683    CONTEXT_SWITCH_RESTORE_TOKEN = const CONTEXT_SWITCH_RESTORE_TOKEN.as_usize(),
684    options(att_syntax)
685);

687/// The location of a cpu-local shadow stack restore token that's mapped into
688/// every set of page tables for use during context switches.
689///
690/// One interesting difference between the normal stack pointer and the shadow
691/// stack pointer is how they can be switched: For the normal stack pointer we
692/// can just move a new value into the RSP register. This doesn't work for the
693/// SSP register (the shadow stack pointer) because there's no way to directly
694/// move a value into it. Instead we have to use the `rstorssp` instruction.
695/// The key difference between this instruction and a regular `mov` is that
696/// `rstorssp` expects a "shadow stack restore token" to be at the top of the
697/// new shadow stack (this is just a special value that marks the top of a
698/// inactive shadow stack). After switching to a new shadow stack, the previous
699/// shadow stack is now inactive, and so the `saveprevssp` instruction can be
700/// used to transfer the shadow stack restore token from the new shadow stack
701/// to the previous one: `saveprevssp` atomically pops the stack token of the
702/// new shadow stack and pushes it on the previous shadow stack. This means
703/// that we have to execute both `rstorssp` and `saveprevssp` every time we
704/// want to switch the shadow stacks.
705///
706/// There's one major problem though: `saveprevssp` needs to access both the
707/// previous and the new shadow stack, but we only map each shadow stack into a
708/// single task's page tables. If each set of page tables only has access to
709/// either the previous or the new shadow stack, but not both, we can't execute
710/// `saveprevssp` and so we we can't move the shadow stack restore token to the
711/// previous shadow stack. If there's no shadow stack restore token on the
712/// previous shadow stack that means we can't restore this shadow stack at a
713/// later point. To work around this, we map another shadow stack into each
714/// CPU's set of pagetables. This allows us to do the following:
715///
716/// 1. Switch to the context-switch shadow stack using `rstorssp`.
717/// 2. Transfer the shadow stack restore token from the context switch shadow
718///    stack to the previous shadow stack by executing `saveprevssp`.
719/// 3. Switch the page tables. This doesn't lead to problems with the context
720///    switch shadow stack because it's mapped into both page tables.
721/// 4. Switch to the new shadow stack using `rstorssp`.
722/// 5. Transfer the shadow stack restore token from the new shadow stack back
723///    to the context switch shadow stacks by executing `saveprevssp`.
724///
725/// We just switched between two shadow stack tables in different page tables :)
726///
727/// Stack offset calculation:
728///
729/// 0x1ff8 = Size(GuardPage) + Size(ShadowStack) - 8; where Size(GuardPage) == Size(ShadowStack) == PAGE_SIZE.
730const CONTEXT_SWITCH_RESTORE_TOKEN: VirtAddr = SVSM_CONTEXT_SWITCH_SHADOW_STACK.const_add(0x1ff8);
