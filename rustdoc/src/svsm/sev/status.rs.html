7use crate::cpu::msr::read_msr;
8use crate::utils::immut_after_init::ImmutAfterInitCell;
9use cpuarch::sev_status::MSR_SEV_STATUS;
10use cpuarch::sev_status::SEVStatusFlags;
11
12static SEV_FLAGS: ImmutAfterInitCell<SEVStatusFlags> = ImmutAfterInitCell::uninit();
13
14fn read_sev_status() -> SEVStatusFlags {
15    SEVStatusFlags::from_bits_truncate(read_msr(MSR_SEV_STATUS))
16}
17
18pub fn sev_flags() -> SEVStatusFlags {
19    *SEV_FLAGS
20}
21
22pub fn sev_status_init() {
23    let status: SEVStatusFlags = read_sev_status();
24    SEV_FLAGS
25        .init(status)
26        .expect("Already initialized SEV flags");
27}
28
29pub fn vtom_enabled() -> bool {
30    sev_flags().contains(SEVStatusFlags::VTOM)
31}
32
33pub fn sev_status_verify() {
34    let required = SEVStatusFlags::SEV
35        | SEVStatusFlags::SEV_ES
36        | SEVStatusFlags::SEV_SNP
37        | SEVStatusFlags::REST_INJ
38        | SEVStatusFlags::DBGSWP;
39    let supported = SEVStatusFlags::VTOM
40        | SEVStatusFlags::PREV_HOST_IBS
41        | SEVStatusFlags::BTB_ISOLATION
42        | SEVStatusFlags::SMT_PROT;
43
44    let status = sev_flags();
45    let required_check = status & required;
46    let not_supported_check = status & !(supported | required);
47
48    if required_check != required {
49        log::error!(
50            "Required features not available: {}",
51            required & !required_check
52        );
53        panic!("Required SEV features not available");
54    }
55
56    if !not_supported_check.is_empty() {
57        log::error!("Unsupported features enabled: {not_supported_check}");
58        panic!("Unsupported SEV features enabled");
59    }
60}
