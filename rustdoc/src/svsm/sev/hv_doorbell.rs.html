52/// Allocates a new HV doorbell page and registers it on the hypervisor
53/// using the given GHCB.
54pub fn allocate_hv_doorbell_page(ghcb: &GHCB) -> Result<SharedBox<HVDoorbell>, SvsmError> {
55    let page = SharedBox::<HVDoorbell>::try_new_zeroed()?;
56
57    let vaddr = page.addr();
58    let paddr = virt_to_phys(vaddr);
59    ghcb.register_hv_doorbell(paddr)?;
60
61    Ok(page)
62}

64#[repr(C)]
65#[derive(Debug, FromBytes)]
66pub struct HVDoorbell {
67    pub vector: AtomicU8,
68    pub flags: AtomicU8,
69    pub no_eoi_required: AtomicU8,
70    pub per_vmpl_events: AtomicU8,
71    reserved_63_4: [AtomicU8; 60],
72    pub per_vmpl: [HVExtIntInfo; 3],
73}
74
75impl HVDoorbell {
76    pub fn process_pending_events(&self) {
77        // Clear the NoFurtherSignal bit before processing.  If any additional
78        // signal comes in after processing has commenced, it may be missed by
79        // this loop, but it will be detected when interrupts are processed
80        // again.  Also clear the NMI bit, since NMIs are not expected.
81        let no_further_signal_mask: u8 = HVDoorbellFlags::new()
82            .with_no_further_signal(true)
83            .with_nmi_pending(true)
84            .into();
85        let flags = HVDoorbellFlags::from(
86            self.flags
87                .fetch_and(!no_further_signal_mask, Ordering::Relaxed),
88        );
89
90        // #MC handling is not possible, so panic if a machine check has
91        // occurred.
92        if flags.mc_pending() {
93            panic!("#MC exception delivered via #HV");
94        }
95
96        // Consume interrupts as long as they are available and as long as
97        // they are appropriate for the current task priority.
98        let tpr = raw_get_tpr();
99        let mut vector = self.vector.load(Ordering::Relaxed);
100        loop {
101            // Check whether an interrupt is present.  If it is at or below
102            // the current task priority, then it will not be dispatched.
103            // If the interrupt is not dispatched, then the vector must remain
104            // in the #HV doorbell page so the hypervisor knows it has not been
105            // placed into service.
106            if (tpr_from_vector(vector as usize)) <= tpr {
107                break;
108            }
109            match self
110                .vector
111                .compare_exchange_weak(vector, 0, Ordering::Relaxed, Ordering::Relaxed)
112            {
113                Ok(_) => common_isr_handler(vector as usize),
114                Err(new) => vector = new,
115            }
116        }
117
118        // Ignore per-VMPL events; these will be consumed when APIC emulation
119        // is performed.
120    }
121
122    fn events_to_process(&self) -> bool {
123        // If NoFurtherSignal is set, it means the hypervisor has posted a
124        // new event, and there must be an event that requires processing.  If
125        // NoFurtherSignal is not set, then processing is required if the
126        // pending vector is a higher priority than the current TPR.
127        let flags = HVDoorbellFlags::from(self.flags.load(Ordering::Relaxed));
128        if flags.no_further_signal() {
129            true
130        } else {
131            let min_vector = (raw_get_tpr() + 1) << 4;
132            self.vector.load(Ordering::Relaxed) as usize >= min_vector
133        }
134    }
135
136    /// This function must always be called with interrupts enabled.
137    pub fn process_if_required(&self, irq_state: &IrqState) {
138        while self.events_to_process() {
139            // #HV event processing must always be performed with interrupts
140            // disabled.
141            irq_state.disable();
142            self.process_pending_events();
143
144            // Do not call the standard enable routine, because that could
145            // recursively call to process new events, resulting in unbounded
146            // nesting.  Intead, decrement the nesting count and directly
147            // enable interrupts here before continuing the loop to see
148            // whether additional events have arrived.
149            assert_eq!(irq_state.pop_nesting(), 0);
150            // SAFETY: interrupts can now be enabled directly.
151            unsafe {
152                asm!("sti");
153            }
154        }
155    }
156
157    pub fn no_eoi_required(&self) -> bool {
158        // Check to see if the "no EOI required" flag is set to determine
159        // whether an explicit EOI can be avoided.
160        let mut no_eoi_required = self.no_eoi_required.load(Ordering::Relaxed);
161        loop {
162            // If the flag is not set, then an explicit EOI is required.
163            if (no_eoi_required & 1) == 0 {
164                return false;
165            }
166            // Attempt to atomically clear the flag.
167            match self.no_eoi_required.compare_exchange_weak(
168                no_eoi_required,
169                no_eoi_required & !1,
170                Ordering::Relaxed,
171                Ordering::Relaxed,
172            ) {
173                Ok(_) => break,
174                Err(new) => no_eoi_required = new,
175            }
176        }
177
178        // If the flag was successfully cleared, then no explicit EOI is
179        // required.
180        true
181    }
182}

184/// Gets the HV doorbell page configured for this CPU.
185///
186/// # Panics
187///
188/// Panics if te HV doorbell page has not been set up beforehand.
189pub fn current_hv_doorbell() -> &'static HVDoorbell {
190    this_cpu()
191        .hv_doorbell()
192        .expect("HV doorbell page dereferenced before allocating")
193}

195/// # Safety
196/// This function takes a raw pointer to the #HV doorbell page because it is
197/// called directly from assembly, and should not be invoked directly from
198/// Rust code.
199#[unsafe(no_mangle)]
200pub unsafe extern "C" fn process_hv_events(hv_doorbell: *const HVDoorbell) {
201    // Update the IRQ nesting state of the current CPU so calls to common
202    // code recognize that interrupts have been disabled.  Proceed as if
203    // interrupts were previously enabled, so that any code that deals with
204    // maskable interrupts knows that interrupts were enabled prior to reaching
205    // this point.
206    let cpu = this_cpu();
207    cpu.irqs_push_nesting(true);
208    // SAFETY: the correctness of #HV doorbell page has been guaranteed by the
209    // caller.
210    unsafe {
211        (*hv_doorbell).process_pending_events();
212    }
213    cpu.irqs_pop_nesting();
214}
