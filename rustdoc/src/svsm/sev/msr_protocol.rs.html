21#[derive(Clone, Copy, Debug)]
22pub enum GhcbMsrError {
23    // The info section of the response did not match our request
24    InfoMismatch,
25    // The data section of the response did not match our request,
26    // or it was malformed altogether.
27    DataMismatch,
28}
29
30impl From<GhcbMsrError> for SvsmError {
31    fn from(e: GhcbMsrError) -> Self {
32        Self::GhcbMsr(e)
33    }
34}

37#[non_exhaustive]
38pub enum GHCBMsr {}

40impl GHCBMsr {
41    pub const SEV_INFO_REQ: u64 = 0x02;
42    pub const SEV_INFO_RESP: u64 = 0x01;
43    pub const SNP_REG_GHCB_GPA_REQ: u64 = 0x12;
44    pub const SNP_REG_GHCB_GPA_RESP: u64 = 0x13;
45    pub const SNP_STATE_CHANGE_REQ: u64 = 0x14;
46    pub const SNP_STATE_CHANGE_RESP: u64 = 0x15;
47    pub const SNP_HV_FEATURES_REQ: u64 = 0x80;
48    pub const SNP_HV_FEATURES_RESP: u64 = 0x81;
49    pub const TERM_REQ: u64 = 0x100;
50}

52bitflags! {
53    #[derive(Clone, Copy, Debug)]
54    pub struct GHCBHvFeatures: u64 {
55        const SEV_SNP                 = 1 << 0;
56        const SEV_SNP_AP_CREATION     = 1 << 1;
57        const SEV_SNP_RESTR_INJ       = 1 << 2;

59        const APIC_ID_LIST            = 1 << 4;
60        const SEV_SNP_MULTI_VMPL      = 1 << 5;
61        const SEV_PAGE_STATE_CHANGE   = 1 << 6;
62        const SEV_SNP_EXT_INTERRUPTS  = 1 << 9;
63    }
64}

74/// Check that we support the hypervisor's advertised GHCB versions.
75pub fn verify_ghcb_version() {
76    // This function is normally only called early during initializtion before
77    // interrupts have been enabled, and before interrupt guards can safely be
78    // used.
79    assert!(!irqs_enabled());
80    // Request SEV information.
81    // SAFETY: Requesting info through the GHCB MSR protocol is safe.
82    unsafe { write_msr(SEV_GHCB, GHCBMsr::SEV_INFO_REQ) };
83    raw_vmgexit();
84    let sev_info = read_msr(SEV_GHCB);
85
86    // Parse the results.
87
88    let response_ty = sev_info & 0xfff;
89    assert_eq!(
90        response_ty,
91        GHCBMsr::SEV_INFO_RESP,
92        "unexpected response type: {response_ty:#05x}"
93    );
94
95    // Compare announced supported GHCB MSR protocol version range
96    // for compatibility.
97    let min_version = (sev_info >> 32) & 0xffff;
98    let max_version = (sev_info >> 48) & 0xffff;
99    assert!(
100        (min_version..=max_version).contains(&2),
101        "the hypervisor doesn't support GHCB version 2 (min: {min_version}, max: {max_version})"
102    );
103}

105pub fn hypervisor_ghcb_features() -> GHCBHvFeatures {
106    *GHCB_HV_FEATURES

109pub fn init_hypervisor_ghcb_features() -> Result<(), GhcbMsrError> {
110    let guard = IrqGuard::new();
111    // SAFETY: Requesting HV features through the GHCB MSR protocol is safe.
112    unsafe { write_msr(SEV_GHCB, GHCBMsr::SNP_HV_FEATURES_REQ) };
113    raw_vmgexit();
114    let result = read_msr(SEV_GHCB);
115    drop(guard);
116    if (result & 0xFFF) == GHCBMsr::SNP_HV_FEATURES_RESP {
117        let features = GHCBHvFeatures::from_bits_truncate(result >> 12);
118
119        // Verify that the required features are supported.
120        let required = GHCBHvFeatures::SEV_SNP
121            | GHCBHvFeatures::SEV_SNP_AP_CREATION
122            | GHCBHvFeatures::SEV_SNP_MULTI_VMPL;
123        let missing = !features & required;
124        if !missing.is_empty() {
125            log::error!(
126                "Required hypervisor GHCB features not available: present={:#x}, required={:#x}, missing={:#x}",
127                features, required, missing
128            );
129            // FIXME - enforce this panic once KVM advertises the required
130            // features.
131            // panic!("Required hypervisor GHCB features not available");
132        }
133
134        GHCB_HV_FEATURES
135            .init(features)
136            .expect("Already initialized GHCB HV features");
137        Ok(())
138    } else {
139        Err(GhcbMsrError::InfoMismatch)
140    }
141}

143/// # Safety
144///
145/// Since this causes the GHCB to be remapped to a different physical address
146/// (allowing leaking and modifying its content), `addr` should be validated.
147pub unsafe fn register_ghcb_gpa_msr(addr: PhysAddr) -> Result<(), GhcbMsrError> {
148    let mut info = addr.bits() as u64;
149
150    info |= GHCBMsr::SNP_REG_GHCB_GPA_REQ;
151    let guard = IrqGuard::new();
152    // SAFETY: safety requirements should be checked by the caller
153    unsafe { write_msr(SEV_GHCB, info) };
154    raw_vmgexit();
155    info = read_msr(SEV_GHCB);
156    drop(guard);
157
158    if (info & 0xfff) != GHCBMsr::SNP_REG_GHCB_GPA_RESP {
159        return Err(GhcbMsrError::InfoMismatch);
160    }
161
162    if (info & !0xfff) != (addr.bits() as u64) {
163        return Err(GhcbMsrError::DataMismatch);
164    }
165
166    Ok(())
167}

169/// # Safety
170///
171/// See [`validate_page_msr`] or [`invalidate_page_msr`] safety requirements.
172unsafe fn set_page_valid_status_msr(addr: PhysAddr, valid: bool) -> Result<(), GhcbMsrError> {
173    let mut info: u64 = (addr.bits() as u64) & 0x000f_ffff_ffff_f000;
174
175    if valid {
176        info |= 1u64 << 52;
177    } else {
178        info |= 2u64 << 52;
179    }
180
181    info |= GHCBMsr::SNP_STATE_CHANGE_REQ;
182    let guard = IrqGuard::new();
183    // SAFETY: safety requirements are delegated to the caller.
184    unsafe { write_msr(SEV_GHCB, info) };
185    raw_vmgexit();
186    let response = read_msr(SEV_GHCB);
187    drop(guard);
188
189    if (response & 0xfff) != GHCBMsr::SNP_STATE_CHANGE_RESP {
190        return Err(GhcbMsrError::InfoMismatch);
191    }
192
193    if (response & !0xfff) != 0 {
194        return Err(GhcbMsrError::DataMismatch);
195    }
196
197    Ok(())
198}

200/// # Safety
201///
202/// Since this causes a page to be remmaped with a different encryption
203/// attribute, `addr` should be validated.
204pub unsafe fn validate_page_msr(addr: PhysAddr) -> Result<(), GhcbMsrError> {
205    // SAFETY: safety requirements are delegated to the caller.
206    unsafe { set_page_valid_status_msr(addr, true) }
207}
208
209/// # Safety
210///
211/// Since this causes a page to be remmaped with a different encryption
212/// attribute, `addr` should be validated.
213pub unsafe fn invalidate_page_msr(addr: PhysAddr) -> Result<(), GhcbMsrError> {
214    // SAFETY: safety requirements are delegated to the caller.
215    unsafe { set_page_valid_status_msr(addr, false) }
216}

218pub fn request_termination_msr() -> ! {
219    let info: u64 = GHCBMsr::TERM_REQ;
220
221    // Since this processor is destined for a fatal termination, there is
222    // no reason to preserve interrupt state.  Interrupts can be disabled
223    // outright prior to shutdown.
224    raw_irqs_disable();
225    // SAFETY: Requesting termination doesn't break memory safety.
226    unsafe { write_msr(SEV_GHCB, info) };
227    raw_vmgexit();
228    loop {
229        halt();
230    }
