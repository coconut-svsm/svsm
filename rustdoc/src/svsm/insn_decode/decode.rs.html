52/// Represents the raw bytes of an instruction and
53/// tracks the number of bytes being processed.
54#[derive(Clone, Copy, Debug, Default, PartialEq)]
55pub struct InsnBytes {
56    /// Raw instruction bytes
57    bytes: [u8; MAX_INSN_SIZE],
58    /// Number of instruction bytes being processed
59    nr_processed: usize,
60}
61
62impl InsnBytes {
63    /// Creates a new `OpCodeBytes` instance with the provided instruction bytes.
64    ///
65    /// # Arguments
66    ///
67    /// * `bytes` - An array of raw instruction bytes
68    ///
69    /// # Returns
70    ///
71    /// A new instance of `OpCodeBytes` with the `bytes` set to the provided
72    /// array and the `nr_processed` field initialized to zero.
73    pub const fn new(bytes: [u8; MAX_INSN_SIZE]) -> Self {
74        Self {
75            bytes,
76            nr_processed: 0,
77        }
78    }
79
80    /// Retrieves a single unprocessed instruction byte.
81    ///
82    /// # Returns
83    ///
84    /// An instruction byte if success or an [`InsnError`] otherwise.
85    pub fn peek(&self) -> Result<u8, InsnError> {
86        self.bytes
87            .get(self.nr_processed)
88            .copied()
89            .ok_or(InsnError::InsnPeek)
90    }
91
92    /// Increases the count by one after a peeked byte being processed.
93    pub fn advance(&mut self) {
94        self.nr_processed += 1
95    }
96
97    /// Retrieves the number of processed instruction bytes.
98    ///
99    /// # Returns
100    ///
101    /// Returns the number of processed bytes as a `usize`.
102    pub fn processed(&self) -> usize {
103        self.nr_processed
104    }
105}

133/// This trait provides the necessary context for an instruction decoder
134/// to decode instructions based on the current state of the machine
135/// that executed them. It abstracts the interfaces through which an
136/// instruction decoder can access specific registers and state that may
137/// influence the decoding from the machine (such as a CPU or VMM).
138pub trait InsnMachineCtx: core::fmt::Debug {
139    type Ptr<T: FromBytes + IntoBytes>: InsnMachineMem<T>;
140
141    /// Read EFER register
142    fn read_efer(&self) -> u64;
143    /// Read a code segment register
144    fn read_seg(&self, seg: SegRegister) -> u64;
145    /// Read CR0 register
146    fn read_cr0(&self) -> u64;
147    /// Read CR4 register
148    fn read_cr4(&self) -> u64;
149
150    /// Read a register
151    fn read_reg(&self, _reg: Register) -> usize {
152        unimplemented!("Reading register is not implemented");
153    }
154
155    /// Read rflags register
156    fn read_flags(&self) -> usize {
157        unimplemented!("Reading flags is not implemented");
158    }
159
160    /// Write a register
161    fn write_reg(&mut self, _reg: Register, _val: usize) {
162        unimplemented!("Writing register is not implemented");
163    }
164
165    /// Read the current privilege level
166    fn read_cpl(&self) -> usize {
167        unimplemented!("Reading CPL is not implemented");
168    }
169
170    /// Map the given linear address region to a machine memory object
171    /// which provides access to the memory of this linear address region.
172    ///
173    /// # Arguments
174    ///
175    /// * `la` - The linear address of the region to map.
176    /// * `write` - Whether write access is allowed to the mapped region.
177    /// * `fetch` - Whether fetch access is allowed to the mapped region.
178    ///
179    /// # Returns
180    ///
181    /// A pointer-like type that allows reading and writing via [`InsnMachineMem`],
182    /// or an `InsnError` if mapping fails.
183    fn map_linear_addr<T: FromBytes + IntoBytes>(
184        &self,
185        _la: usize,
186        _write: bool,
187        _fetch: bool,
188    ) -> Result<Self::Ptr<T>, InsnError> {
189        Err(InsnError::MapLinearAddr)
190    }
191
192    /// Check IO permission bitmap.
193    ///
194    /// # Arguments
195    ///
196    /// * `port` - The I/O port to check.
197    /// * `size` - The size of the I/O operation.
198    /// * `io_read` - Whether the I/O operation is a read operation.
199    ///
200    /// # Returns
201    ///
202    /// A `Result` containing true if the port is permitted otherwise false.
203    fn ioio_perm(&self, _port: u16, _size: Bytes, _io_read: bool) -> bool {
204        unimplemented!("Checking IO permission bitmap is not implemented");
205    }

207    /// Handle an I/O in operation.
208    ///
209    /// # Arguments
210    ///
211    /// * `port` - The I/O port to read from.
212    /// * `size` - The size of the data to read.
213    ///
214    /// # Returns
215    ///
216    /// A `Result` containing the read data if success or an `InsnError` if
217    /// the operation fails.
218    fn ioio_in(&self, _port: u16, _size: Bytes) -> Result<u64, InsnError> {
219        Err(InsnError::IoIoIn)
220    }
221
222    /// Handle an I/O out operation.
223    ///
224    /// # Arguments
225    ///
226    /// * `port` - The I/O port to write to.
227    /// * `size` - The size of the data to write.
228    /// * `data` - The data to write to the I/O port.
229    ///
230    /// # Returns
231    ///
232    /// A `Result` indicating success or an `InsnError` if the operation fails.
233    fn ioio_out(&mut self, _port: u16, _size: Bytes, _data: u64) -> Result<(), InsnError> {
234        Err(InsnError::IoIoOut)
235    }
236
237    /// Translate the given linear address to a physical address.
238    ///
239    /// # Arguments
240    ///
241    /// * `la` - The linear address to translate.
242    /// * `write` - Whether the translation is for a write operation.
243    /// * `fetch` - Whether the translation is for a fetch operation.
244    ///
245    /// # Returns
246    ///
247    /// A `Result` containing the translated physical address and a boolean
248    /// indicating whether the physical address is shared or an `InsnError` if
249    /// the translation fails.
250    fn translate_linear_addr(
251        &self,
252        _la: usize,
253        _write: bool,
254        _fetch: bool,

259    /// Handle a memory-mapped I/O read operation.

262    ///

264    /// * `shared` - Whether the MMIO address is shared.

268    ///

274    /// Caller must ensure that, when memory safety may be impacted, this

278        _pa: usize,
279        _shared: bool,

282        Err(InsnError::HandleMmioRead)
283    }
284
285    /// Handle a memory-mapped I/O write operation.
286    ///
287    /// # Arguments
288    ///
289    /// * `pa` - The MMIO physical address to write to.
290    /// * `shared` - Whether the MMIO address is shared.
291    /// * `size` - The size of the data to write.
292    /// * `data` - The data to write to the MMIO.
293    ///
294    /// # Returns
295    ///
296    /// A `Result` indicating success or an `InsnError` if the operation fails.
297    ///
298    /// # Safety
299    ///
300    /// Caller must ensure that, when memory safety may be impacted, this
301    /// operation does not cause Undefined Behavior.
302    unsafe fn handle_mmio_write(
303        &mut self,
304        _pa: usize,
305        _shared: bool,
306        _size: Bytes,
307        _data: u64,
308    ) -> Result<(), InsnError> {
309        Err(InsnError::HandleMmioWrite)
310    }

313/// Trait representing a machine memory for instruction decoding.
314pub trait InsnMachineMem<T: FromBytes + IntoBytes> {
315    /// Read data from the memory at the specified offset.
316    ///
317    /// # Safety
318    ///
319    /// The caller must verify not to read data from arbitrary memory. The object implements this
320    /// trait should guarantee the memory region is readable.
321    ///
322    /// # Returns
323    ///
324    /// Returns the read data on success, or an `InsnError` if the read
325    /// operation fails.
326    unsafe fn mem_read(&self) -> Result<T, InsnError> {
327        Err(InsnError::MemRead)
328    }
329
330    /// Write data to the memory at the specified offset.
331    ///
332    /// # Safety
333    ///
334    /// The caller must verify not to write data to corrupt arbitrary memory. The object implements
335    /// this trait should guarantee the memory region is writable.
336    ///
337    /// # Arguments
338    ///
339    /// * `data` - The data to write to the memory.
340    ///
341    /// # Returns
342    ///
343    /// Returns `Ok`on success, or an `InsnError` if the write operation fails.
344    unsafe fn mem_write(&mut self, _data: T) -> Result<(), InsnError> {
345        Err(InsnError::MemWrite)
346    }
347}

349#[derive(Clone, Copy, Debug, PartialEq)]
350enum PagingLevel {
351    Level4,
352    Level5,
353}

355#[derive(Clone, Copy, Debug, Default, PartialEq)]
356enum CpuMode {
357    #[default]
358    Real,
359    Protected,
360    Compatibility,
361    Bit64(PagingLevel),
362}
363
364impl CpuMode {
365    fn is_bit64(&self) -> bool {
366        matches!(self, CpuMode::Bit64(_))
367    }
368}
369
370fn get_cpu_mode<I: InsnMachineCtx>(mctx: &I) -> CpuMode {
371    if (mctx.read_efer() & EFERFlags::LMA.bits()) != 0 {
372        // EFER.LMA = 1
373        if (mctx.read_seg(SegRegister::CS) & SegDescAttrFlags::L.bits()) != 0 {
374            // CS.L = 1 represents 64bit mode.
375            // While this sub-mode produces 64-bit linear addresses, the processor
376            // enforces canonicality, meaning that the upper bits of such an address
377            // are identical: bits 63:47 for 4-level paging and bits 63:56 for
378            // 5-level paging. 4-level paging (respectively, 5-level paging) does not
379            // use bits 63:48 (respectively, bits 63:57) of such addresses
380            let level = if (mctx.read_cr4() & CR4Flags::LA57.bits()) != 0 {
381                PagingLevel::Level5
382            } else {
383                PagingLevel::Level4
384            };
385            CpuMode::Bit64(level)
386        } else {
387            CpuMode::Compatibility
388        }
389    } else if (mctx.read_cr0() & CR0Flags::PE.bits()) != 0 {
390        // CR0.PE = 1
391        CpuMode::Protected
392    } else {
393        CpuMode::Real
394    }
395}

397// Translate the decoded number from the instruction ModR/M
398// or SIB to the corresponding register
399struct RegCode(u8);
400impl TryFrom<RegCode> for Register {
401    type Error = InsnError;
402
403    fn try_from(val: RegCode) -> Result<Register, Self::Error> {
404        match val.0 {
405            0 => Ok(Register::Rax),
406            1 => Ok(Register::Rcx),
407            2 => Ok(Register::Rdx),
408            3 => Ok(Register::Rbx),
409            4 => Ok(Register::Rsp),
410            5 => Ok(Register::Rbp),
411            6 => Ok(Register::Rsi),
412            7 => Ok(Register::Rdi),
413            8 => Ok(Register::R8),
414            9 => Ok(Register::R9),
415            10 => Ok(Register::R10),
416            11 => Ok(Register::R11),
417            12 => Ok(Register::R12),
418            13 => Ok(Register::R13),
419            14 => Ok(Register::R14),
420            15 => Ok(Register::R15),
421            // Rip is not represented by ModR/M or SIB
422            _ => Err(InsnError::InvalidRegister),
423        }
424    }
425}

457const MOD_INDIRECT: u8 = 0;
458const MOD_INDIRECT_DISP8: u8 = 1;
459const MOD_INDIRECT_DISP32: u8 = 2;
460const MOD_DIRECT: u8 = 3;
461const RM_SIB: u8 = 4;

467    }

470#[derive(Copy, Clone, Debug, PartialEq)]
471enum RM {
472    Reg(Register),
473    Sib,
474    Disp32,
475}
476
477#[derive(Copy, Clone, Debug, PartialEq)]
478enum Mod {
479    Indirect,
480    IndirectDisp8,
481    IndirectDisp32,
482    Direct,
483}
484
485impl ModRM {
486    fn get_mod(&self) -> Mod {
487        let v = (self.0 >> 6) & 0x3;
488
489        match v {
490            MOD_INDIRECT => Mod::Indirect,
491            MOD_INDIRECT_DISP8 => Mod::IndirectDisp8,
492            MOD_INDIRECT_DISP32 => Mod::IndirectDisp32,
493            MOD_DIRECT => Mod::Direct,
494            _ => {
495                unreachable!("Mod has only two bits, so its value is always 0 ~ 3");
496            }
497        }
498    }
499
500    fn get_reg(&self) -> u8 {
501        (self.0 >> 3) & 0x7
502    }
503
504    fn get_rm(&self) -> RM {
505        let rm = self.0 & 0x7;
506        let r#mod = self.get_mod();
507
508        // RM depends on the Mod value
509        if r#mod == Mod::Indirect && rm == RM_DISP32 {
510            RM::Disp32
511        } else if r#mod != Mod::Direct && rm == RM_SIB {
512            RM::Sib
513        } else {
514            RM::Reg(Register::try_from(RegCode(rm)).unwrap())
515        }
516    }
517}

542#[inline]
543fn read_reg<I: InsnMachineCtx>(mctx: &I, reg: Register, size: Bytes) -> usize {
544    mctx.read_reg(reg) & size.mask() as usize
545}
546
547#[inline]
548fn write_reg<I: InsnMachineCtx>(mctx: &mut I, reg: Register, data: usize, size: Bytes) {
549    mctx.write_reg(
550        reg,
551        match size {
552            Bytes::Zero => return,
553            // Writing 8bit or 16bit register will not affect the upper bits.
554            Bytes::One | Bytes::Two => {
555                let old = mctx.read_reg(reg);
556                (data & size.mask() as usize) | (old & !size.mask() as usize)
557            }
558            // Writing 32bit register will zero out the upper bits.
559            Bytes::Four => data & size.mask() as usize,
560            Bytes::Eight => data,
561        },
562    );
563}

565#[inline]
566fn segment_base(segment: u64) -> u32 {
567    // Segment base bits 0 ~ 23: raw value bits 16 ~ 39
568    // Segment base bits 24 ~ 31: raw value bits 56 ~ 63
569    (((segment >> 16) & 0xffffff) | ((segment >> 56) << 24)) as u32
570}
571
572#[inline]
573fn segment_limit(segment: u64) -> u32 {
574    // Segment limit bits 0 ~ 15: raw value bits 0 ~ 15
575    // Segment limit bits 16 ~ 19: raw value bits 48 ~ 51
576    let limit = ((segment & 0xffff) | ((segment >> 32) & 0xf0000)) as u32;
577
578    if SegDescAttrFlags::from_bits_truncate(segment).contains(SegDescAttrFlags::G) {
579        (limit << 12) | 0xfff
580    } else {
581        limit
582    }
583}

585fn ioio_perm<I: InsnMachineCtx>(mctx: &I, port: u16, size: Bytes, io_read: bool) -> bool {
586    if mctx.read_cr0() & CR0Flags::PE.bits() != 0
587        && (mctx.read_cpl() > ((mctx.read_flags() >> 12) & 3)
588            || mctx.read_cr4() & CR4Flags::VME.bits() != 0)
589    {
590        // In protected mode with CPL > IOPL or virtual-8086 mode, if
591        // any I/O Permission Bit for I/O port being accessed = 1, the I/O
592        // operation is not allowed.
593        mctx.ioio_perm(port, size, io_read)
594    } else {
595        true
596    }
597}

599#[inline]
600fn read_bytereg<I: InsnMachineCtx>(mctx: &I, reg: Register, lhbr: bool) -> u8 {
601    let data = mctx.read_reg(reg);
602    // To obtain the value of a legacy high byte register shift the
603    // base register right by 8 bits (%ah = %rax >> 8).
604    (if lhbr { data >> 8 } else { data }) as u8
605}
606
607#[inline]
608fn write_bytereg<I: InsnMachineCtx>(mctx: &mut I, reg: Register, lhbr: bool, data: u8) {
609    let old = mctx.read_reg(reg);
610    let mask = (Bytes::One).mask() as usize;
611
612    let new = if lhbr {
613        (data as usize) << 8 | (old & !(mask << 8))
614    } else {
615        (data as usize) | (old & !mask)
616    };
617
618    mctx.write_reg(reg, new);
619}

660impl DecodedInsnCtx {
661    /// Constructs a new `DecodedInsnCtx` by decoding the given
662    /// instruction bytes using the provided machine context.
663    ///
664    /// # Arguments
665    ///
666    /// * `bytes` - The raw bytes of the instruction to be decoded.
667    /// * `mctx` - A reference to an object implementing the
668    ///   `InsnMachineCtx` trait to provide the necessary machine context
669    ///   for decoding.
670    ///
671    ///  # Returns
672    ///
673    ///  A `DecodedInsnCtx` if decoding is successful or an `InsnError`
674    ///  otherwise.
675    pub(super) fn new<I: InsnMachineCtx>(
676        bytes: &[u8; MAX_INSN_SIZE],
677        mctx: &I,
678    ) -> Result<Self, InsnError> {
679        let mut insn_ctx = Self {
680            cpu_mode: get_cpu_mode(mctx),
681            ..Default::default()
682        };
683
684        insn_ctx.decode(bytes, mctx).map(|_| insn_ctx)
685    }
686
687    /// Retrieves the decoded instruction, if available.
688    ///
689    /// # Returns
690    ///
691    /// An `Option<DecodedInsn>` containing the DecodedInsn.
692    pub fn insn(&self) -> Option<DecodedInsn> {
693        self.insn
694    }
695
696    /// Retrieves the length of the decoded instruction in bytes.
697    ///
698    /// # Returns
699    ///
700    /// The length of the decoded instruction as a `usize`. If the
701    /// repeat count is greater than 1, then return 0 to indicate not to
702    /// skip this instruction. If the repeat count is less than 1, then
703    /// return instruction len to indicate this instruction can be skipped.
704    pub fn size(&self) -> usize {
705        if self.repeat > 1 { 0 } else { self.insn_len }
706    }

708    /// Emulates the decoded instruction using the provided machine context.
709    ///
710    /// # Arguments
711    ///
712    /// * `mctx` - A mutable reference to an object implementing the
713    ///   `InsnMachineCtx` trait to provide the necessary machine context
714    ///   for emulation.
715    ///
716    /// # Returns
717    ///
718    /// An `Ok(())` if emulation is successful or an `InsnError` otherwise.
719    pub fn emulate<I: InsnMachineCtx>(&self, mctx: &mut I) -> Result<(), InsnError> {
720        self.insn
721            .ok_or(InsnError::UnSupportedInsn)
722            .and_then(|insn| match insn {
723                DecodedInsn::In(port, opsize) => self.emulate_in_out(port, opsize, mctx, true),
724                DecodedInsn::Out(port, opsize) => self.emulate_in_out(port, opsize, mctx, false),
725                DecodedInsn::Ins => self.emulate_ins_outs(mctx, true),
726                DecodedInsn::Outs => self.emulate_ins_outs(mctx, false),
727                DecodedInsn::Mov => self.emulate_mov(mctx),
728                _ => Err(InsnError::UnSupportedInsn),
729            })
730    }

732    /// Emulates IOIO instructions using the provided machine context.
733    ///
734    /// # Arguments
735    ///
736    /// * `mctx` - A mutable reference to an object implementing the
737    ///   `InsnMachineCtx` trait to provide the necessary machine context
738    ///   for emulation.
739    ///
740    /// # Returns
741    ///
742    /// An `Ok(())` if emulation is successful or an `InsnError` otherwise.
743    pub fn emulate_ioio<I: InsnMachineCtx>(&self, mctx: &mut I) -> Result<(), InsnError> {
744        self.insn
745            .ok_or(InsnError::UnSupportedInsn)
746            .and_then(|insn| match insn {
747                DecodedInsn::In(_, _)
748                | DecodedInsn::Out(_, _)
749                | DecodedInsn::Ins
750                | DecodedInsn::Outs => self.emulate(mctx),
751                _ => Err(InsnError::UnSupportedInsn),
752            })
753    }
754
755    fn decode<I: InsnMachineCtx>(
756        &mut self,
757        bytes: &[u8; MAX_INSN_SIZE],
758        mctx: &I,
759    ) -> Result<(), InsnError> {
760        self.decode_prefixes(bytes, mctx)
761            .and_then(|insn| self.decode_opcode(insn))
762            .and_then(|insn| self.decode_modrm_sib(insn))
763            .and_then(|(insn, disp_bytes)| self.decode_displacement(insn, disp_bytes))
764            .and_then(|insn| self.decode_immediate(insn))
765            .and_then(|insn| self.decode_moffset(insn))
766            .and_then(|insn| self.complete_decode(insn, mctx))
767    }

774    fn decode_rex_prefix(&mut self, code: u8) -> bool {
775        if !self.cpu_mode.is_bit64() {
776            return false;
777        }
778
779        match code {
780            0x40..=0x4F => {
781                let rex = RexPrefix::from_bits_truncate(code);
782                self.prefix.insert(PrefixFlags::REX_P);
783                if rex.contains(RexPrefix::W) {
784                    self.prefix.insert(PrefixFlags::REX_W);
785                }
786                if rex.contains(RexPrefix::R) {
787                    self.prefix.insert(PrefixFlags::REX_R);
788                }
789                if rex.contains(RexPrefix::X) {
790                    self.prefix.insert(PrefixFlags::REX_X);
791                }
792                if rex.contains(RexPrefix::B) {
793                    self.prefix.insert(PrefixFlags::REX_B);
794                }
795                true
796            }
797            _ => false,
798        }
799    }

801    fn decode_op_addr_size(&mut self, cs: u64) {
802        (self.addrsize, self.opsize) = if self.cpu_mode.is_bit64() {
803            (
804                if self.prefix.contains(PrefixFlags::ADDRSIZE_OVERRIDE) {
805                    Bytes::Four
806                } else {
807                    Bytes::Eight
808                },
809                if self.prefix.contains(PrefixFlags::REX_W) {
810                    Bytes::Eight
811                } else if self.prefix.contains(PrefixFlags::OPSIZE_OVERRIDE) {
812                    Bytes::Two
813                } else {
814                    Bytes::Four
815                },
816            )
817        } else if (cs & SegDescAttrFlags::DB.bits()) != 0 {
818            // Default address and operand sizes are 32-bits
819            (
820                if self.prefix.contains(PrefixFlags::ADDRSIZE_OVERRIDE) {
821                    Bytes::Two
822                } else {
823                    Bytes::Four
824                },
825                if self.prefix.contains(PrefixFlags::OPSIZE_OVERRIDE) {
826                    Bytes::Two
827                } else {
828                    Bytes::Four
829                },
830            )
831        } else {
832            // Default address and operand sizes are 16-bits
833            (
834                if self.prefix.contains(PrefixFlags::ADDRSIZE_OVERRIDE) {
835                    Bytes::Four
836                } else {
837                    Bytes::Two
838                },
839                if self.prefix.contains(PrefixFlags::OPSIZE_OVERRIDE) {
840                    Bytes::Four
841                } else {
842                    Bytes::Two
843                },
844            )
845        };
846    }

848    fn decode_prefixes<I: InsnMachineCtx>(
849        &mut self,
850        bytes: &[u8; MAX_INSN_SIZE],
851        mctx: &I,
852    ) -> Result<OpCodeBytes, InsnError> {
853        let mut insn = PrefixBytes(InsnBytes::new(*bytes));
854        for _ in 0..PREFIX_SIZE {
855            match insn.0.peek()? {
856                0x66 => self.prefix.insert(PrefixFlags::OPSIZE_OVERRIDE),
857                0x67 => self.prefix.insert(PrefixFlags::ADDRSIZE_OVERRIDE),
858                0xF3 => self.prefix.insert(PrefixFlags::REPZ_P),
859                0xF2 => self.prefix.insert(PrefixFlags::REPNZ_P),
860                0x2E => self.override_seg = Some(SegRegister::CS),
861                0x36 => self.override_seg = Some(SegRegister::SS),
862                0x3E => self.override_seg = Some(SegRegister::DS),
863                0x26 => self.override_seg = Some(SegRegister::ES),
864                0x64 => self.override_seg = Some(SegRegister::FS),
865                0x65 => self.override_seg = Some(SegRegister::GS),
866                _ => break,
867            }
868            insn.0.advance();
869        }
870
871        // From section 2.2.1, "REX Prefixes", Intel SDM Vol 2:
872        // - Only one REX prefix is allowed per instruction.
873        // - The REX prefix must immediately precede the opcode byte or the
874        //   escape opcode byte.
875        // - If an instruction has a mandatory prefix (0x66, 0xF2 or 0xF3)
876        //   the mandatory prefix must come before the REX prefix.
877        if self.decode_rex_prefix(insn.0.peek()?) {
878            insn.0.advance();
879        }
880
881        self.decode_op_addr_size(mctx.read_seg(SegRegister::CS));
882
883        Ok(OpCodeBytes(insn.0))
884    }

886    fn decode_opcode(&mut self, mut insn: OpCodeBytes) -> Result<ModRmBytes, InsnError> {
887        let opdesc = OpCodeDesc::decode(&mut insn).ok_or(InsnError::DecodeOpCode)?;
888
889        if opdesc.flags.contains(OpCodeFlags::BYTE_OP) {
890            self.opsize = Bytes::One;
891        } else if opdesc.flags.contains(OpCodeFlags::WORD_OP) {
892            self.opsize = Bytes::Two;
893        }
894
895        self.opdesc = Some(opdesc);
896
897        Ok(ModRmBytes(insn.0))
898    }

900    fn decode_modrm_sib(&mut self, mut insn: ModRmBytes) -> Result<(DisBytes, Bytes), InsnError> {
901        if self.get_opdesc()?.flags.contains(OpCodeFlags::NO_MODRM) {
902            return Ok((DisBytes(insn.0), Bytes::Zero));
903        }
904
905        if self.cpu_mode == CpuMode::Real {
906            return Err(InsnError::DecodeModRM);
907        }
908
909        self.modrm = ModRM::from(insn.0.peek()?);
910
911        if self.get_opdesc()?.flags.contains(OpCodeFlags::OP_NONE) {
912            insn.0.advance();
913            return Ok((DisBytes(insn.0), Bytes::Zero));
914        }
915
916        let r#mod = self.modrm.get_mod();
917        let reg = self.modrm.get_reg() | ((self.prefix.contains(PrefixFlags::REX_R) as u8) << 3);
918        self.modrm_reg = Some(Register::try_from(RegCode(reg))?);
919
920        // As the modrm decoding is majorly for MMIO instructions which requires
921        // a memory access, a direct addressing mode makes no sense in the context.
922        // There has to be a memory access involved to trap the MMIO instruction.
923        if r#mod == Mod::Direct {
924            return Err(InsnError::DecodeModRM);
925        }
926
927        // SDM Vol2 Table 2-5: Special Cases of REX Encodings
928        // For mod=0 r/m=5 and mod!=3 r/m=4, the 'b' bit in the REX
929        // prefix is 'don't care' in these two cases.
930        //
931        // RM::Disp32 represent mod=0 r/m=5
932        // RM::Sib represent mod!=3 r/m=4
933        // RM::Reg(r) represent the other cases.
934        let disp_bytes = match self.modrm.get_rm() {
935            RM::Reg(r) => {
936                let ext_r = Register::try_from(RegCode(
937                    r as u8 | ((self.prefix.contains(PrefixFlags::REX_B) as u8) << 3),
938                ))?;
939                self.base_reg = Some(ext_r);
940                match r#mod {
941                    Mod::IndirectDisp8 => Bytes::One,
942                    Mod::IndirectDisp32 => Bytes::Four,
943                    Mod::Indirect | Mod::Direct => Bytes::Zero,
944                }
945            }
946            RM::Disp32 => {
947                // SDM Vol2 Table 2-7: RIP-Relative Addressing
948                // In 64bit mode, mod=0 r/m=5 implies [rip] + disp32
949                // whereas in compatibility mode it just implies disp32.
950                self.base_reg = if self.cpu_mode.is_bit64() {
951                    Some(Register::Rip)
952                } else {
953                    None
954                };
955                Bytes::Four
956            }
957            RM::Sib => {
958                insn.0.advance();
959                return self.decode_sib(SibBytes(insn.0));
960            }
961        };
962
963        insn.0.advance();
964        Ok((DisBytes(insn.0), disp_bytes))
965    }

967    fn decode_sib(&mut self, mut insn: SibBytes) -> Result<(DisBytes, Bytes), InsnError> {
968        // Process only if SIB byte is present
969        if self.modrm.get_rm() != RM::Sib {
970            return Err(InsnError::DecodeSib);
971        }
972
973        self.sib = Sib::from(insn.0.peek()?);
974        let index = self.sib.get_index() | ((self.prefix.contains(PrefixFlags::REX_X) as u8) << 3);
975        let base = self.sib.get_base() | ((self.prefix.contains(PrefixFlags::REX_B) as u8) << 3);
976
977        let r#mod = self.modrm.get_mod();
978        let disp_bytes = match r#mod {
979            Mod::IndirectDisp8 => {
980                self.base_reg = Some(Register::try_from(RegCode(base))?);
981                Bytes::One
982            }
983            Mod::IndirectDisp32 => {
984                self.base_reg = Some(Register::try_from(RegCode(base))?);
985                Bytes::Four
986            }
987            Mod::Indirect => {
988                let mut disp_bytes = Bytes::Zero;
989                // SMD Vol 2 Table 2-5 Special Cases of REX Encoding
990                // Base register is unused if mod=0 base=RBP/R13.
991                self.base_reg = if base == Register::Rbp as u8 || base == Register::R13 as u8 {
992                    disp_bytes = Bytes::Four;
993                    None
994                } else {
995                    Some(Register::try_from(RegCode(base))?)
996                };
997                disp_bytes
998            }
999            Mod::Direct => Bytes::Zero,
1000        };
1001
1002        // SMD Vol 2 Table 2-5 Special Cases of REX Encoding
1003        // Index register not used when index=RSP
1004        if index != Register::Rsp as u8 {
1005            self.index_reg = Some(Register::try_from(RegCode(index))?);
1006            // 'scale' makes sense only in the context of an index register
1007            self.scale = 1 << self.sib.get_scale();
1008        }
1009
1010        insn.0.advance();
1011        Ok((DisBytes(insn.0), disp_bytes))
1012    }

1014    fn decode_displacement(
1015        &mut self,
1016        mut insn: DisBytes,
1017        disp_bytes: Bytes,
1018    ) -> Result<ImmBytes, InsnError> {
1019        match disp_bytes {
1020            Bytes::Zero => Ok(ImmBytes(insn.0)),
1021            Bytes::One | Bytes::Four => {
1022                let mut buf = [0; 4];
1023
1024                for v in buf.iter_mut().take(disp_bytes as usize) {
1025                    *v = insn.0.peek()?;
1026                    insn.0.advance();
1027                }
1028
1029                self.displacement = if disp_bytes == Bytes::One {
1030                    buf[0] as i8 as i64
1031                } else {
1032                    i32::from_le_bytes(buf) as i64
1033                };
1034
1035                Ok(ImmBytes(insn.0))
1036            }
1037            _ => Err(InsnError::DecodeDisp),
1038        }
1039    }

1041    fn decode_immediate(&mut self, mut insn: ImmBytes) -> Result<MoffBytes, InsnError> {
1042        // Figure out immediate operand size (if any)
1043        let imm_bytes = if self.get_opdesc()?.flags.contains(OpCodeFlags::IMM) {
1044            match self.opsize {
1045                // SDM Vol 2 2.2.1.5 "Immediates"
1046                // In 64-bit mode the typical size of immediate operands
1047                // remains 32-bits. When the operand size if 64-bits, the
1048                // processor sign-extends all immediates to 64-bits prior
1049                // to their use.
1050                Bytes::Four | Bytes::Eight => Bytes::Four,
1051                _ => Bytes::Two,
1052            }
1053        } else if self.get_opdesc()?.flags.contains(OpCodeFlags::IMM8) {
1054            Bytes::One
1055        } else {
1056            // No flags on immediate operand size
1057            return Ok(MoffBytes(insn.0));
1058        };
1059
1060        let mut buf = [0; 4];
1061
1062        for v in buf.iter_mut().take(imm_bytes as usize) {
1063            *v = insn.0.peek()?;
1064            insn.0.advance();
1065        }
1066
1067        self.immediate = match imm_bytes {
1068            Bytes::One => buf[0] as i8 as i64,
1069            Bytes::Two => i16::from_le_bytes([buf[0], buf[1]]) as i64,
1070            Bytes::Four => i32::from_le_bytes(buf) as i64,
1071            _ => return Err(InsnError::DecodeImm),
1072        };
1073
1074        Ok(MoffBytes(insn.0))
1075    }

1077    fn decode_moffset(&mut self, mut insn: MoffBytes) -> Result<DecodedBytes, InsnError> {
1078        if !self.get_opdesc()?.flags.contains(OpCodeFlags::MOFFSET) {
1079            return Ok(DecodedBytes(insn.0));
1080        }
1081
1082        match self.addrsize {
1083            Bytes::Zero | Bytes::One => Err(InsnError::DecodeMOffset),
1084            _ => {
1085                // SDM Vol 2 Section 2.2.1.4, "Direct Memory-Offset MOVs"
1086                // In 64-bit mode, direct memory-offset forms of the MOV
1087                // instruction are extended to specify a 64-bit immediate
1088                // absolute address.
1089                //
1090                // The memory offset size follows the address-size of the instruction.
1091                let mut buf = [0; 8];
1092                for v in buf.iter_mut().take(self.addrsize as usize) {
1093                    *v = insn.0.peek()?;
1094                    insn.0.advance();
1095                }
1096                self.displacement = i64::from_le_bytes(buf);
1097                Ok(DecodedBytes(insn.0))
1098            }
1099        }
1100    }

1102    fn complete_decode<I: InsnMachineCtx>(
1103        &mut self,
1104        insn: DecodedBytes,
1105        mctx: &I,
1106    ) -> Result<(), InsnError> {
1107        self.insn_len = insn.0.processed();
1108        self.decoded_insn(mctx)
1109            .map(|decoded_insn| self.insn = Some(decoded_insn))
1110    }

1112    fn decoded_insn<I: InsnMachineCtx>(&mut self, mctx: &I) -> Result<DecodedInsn, InsnError> {
1113        let opdesc = self.get_opdesc()?;
1114        Ok(match opdesc.class {
1115            OpCodeClass::Cpuid => DecodedInsn::Cpuid,
1116            OpCodeClass::In => {
1117                if opdesc.flags.contains(OpCodeFlags::IMM8) {
1118                    DecodedInsn::In(
1119                        Operand::Imm(Immediate::U8(self.immediate as u8)),
1120                        self.opsize,
1121                    )
1122                } else {
1123                    DecodedInsn::In(Operand::rdx(), self.opsize)
1124                }
1125            }
1126            OpCodeClass::Out => {
1127                if opdesc.flags.contains(OpCodeFlags::IMM8) {
1128                    DecodedInsn::Out(
1129                        Operand::Imm(Immediate::U8(self.immediate as u8)),
1130                        self.opsize,
1131                    )
1132                } else {
1133                    DecodedInsn::Out(Operand::rdx(), self.opsize)
1134                }
1135            }
1136            OpCodeClass::Ins | OpCodeClass::Outs => {
1137                if self.prefix.contains(PrefixFlags::REPZ_P) {
1138                    // The prefix REPZ(F3h) actually represents REP for ins/outs.
1139                    // The count register is depending on the address size of the
1140                    // instruction.
1141                    self.repeat = read_reg(mctx, Register::Rcx, self.addrsize);
1142                };
1143
1144                if opdesc.class == OpCodeClass::Ins {
1145                    DecodedInsn::Ins
1146                } else {
1147                    DecodedInsn::Outs
1148                }
1149            }
1150            OpCodeClass::Rdmsr => DecodedInsn::Rdmsr,
1151            OpCodeClass::Rdtsc => DecodedInsn::Rdtsc,
1152            OpCodeClass::Rdtscp => DecodedInsn::Rdtscp,
1153            OpCodeClass::Wrmsr => DecodedInsn::Wrmsr,
1154            OpCodeClass::Mov => DecodedInsn::Mov,
1155            _ => return Err(InsnError::UnSupportedInsn),
1156        })
1157    }

1164    fn cal_modrm_bytereg(&self) -> Result<(Register, bool), InsnError> {
1165        let reg = self.get_modrm_reg()?;
1166        // 64-bit mode imposes limitations on accessing legacy high byte
1167        // registers (lhbr).
1168        //
1169        // The legacy high-byte registers cannot be addressed if the REX
1170        // prefix is present. In this case the values 4, 5, 6 and 7 of the
1171        // 'ModRM:reg' field address %spl, %bpl, %sil and %dil respectively.
1172        //
1173        // If the REX prefix is not present then the values 4, 5, 6 and 7
1174        // of the 'ModRM:reg' field address the legacy high-byte registers,
1175        // %ah, %ch, %dh and %bh respectively.
1176        Ok(
1177            if !self.prefix.contains(PrefixFlags::REX_P) && (reg as u8 & 0x4) != 0 {
1178                (Register::try_from(RegCode(reg as u8 & 0x3))?, true)
1179            } else {
1180                (reg, false)
1181            },
1182        )
1183    }

1185    fn canonical_check(&self, la: usize) -> Option<usize> {
1186        if match self.cpu_mode {
1187            CpuMode::Bit64(level) => {
1188                let virtaddr_bits = if level == PagingLevel::Level4 { 48 } else { 57 };
1189                let mask = !((1 << virtaddr_bits) - 1);
1190                if la & (1 << (virtaddr_bits - 1)) != 0 {
1191                    la & mask == mask
1192                } else {
1193                    la & mask == 0
1194                }
1195            }
1196            _ => true,
1197        } {
1198            Some(la)
1199        } else {
1200            None
1201        }
1202    }

1204    fn alignment_check(&self, la: usize, size: Bytes) -> Option<usize> {
1205        match size {
1206            // Zero size is not allowed
1207            Bytes::Zero => None,
1208            // One byte is always aligned
1209            Bytes::One => Some(la),
1210            // Two/Four/Eight bytes must be aligned on a boundary
1211            _ => {
1212                if la & (size as usize - 1) != 0 {
1213                    None
1214                } else {
1215                    Some(la)
1216                }
1217            }
1218        }
1219    }

1221    fn cal_linear_addr<I: InsnMachineCtx>(
1222        &self,
1223        mctx: &I,
1224        seg: SegRegister,
1225        ea: usize,
1226        writable: bool,
1227    ) -> Option<usize> {
1228        let segment = mctx.read_seg(seg);
1229
1230        let addrsize = if self.cpu_mode.is_bit64() {
1231            Bytes::Eight
1232        } else {
1233            let attr = SegDescAttrFlags::from_bits_truncate(segment);
1234            // Invalid if is system segment
1235            if !attr.contains(SegDescAttrFlags::S) {
1236                return None;
1237            }
1238
1239            if writable {
1240                // Writing to a code segment, or writing to a read-only
1241                // data segment is not allowed.
1242                if attr.contains(SegDescAttrFlags::C_D) || !attr.contains(SegDescAttrFlags::R_W) {
1243                    return None;
1244                }
1245            } else {
1246                // Data segment is always read-able, but code segment
1247                // may be execute only. Invalid if read an execute only
1248                // code segment.
1249                if attr.contains(SegDescAttrFlags::C_D) && !attr.contains(SegDescAttrFlags::R_W) {
1250                    return None;
1251                }
1252            }
1253
1254            let mut limit = segment_limit(segment) as usize;
1255
1256            if !attr.contains(SegDescAttrFlags::C_D) && attr.contains(SegDescAttrFlags::C_E) {
1257                // Expand-down segment, check low limit
1258                if ea <= limit {
1259                    return None;
1260                }
1261
1262                limit = if attr.contains(SegDescAttrFlags::DB) {
1263                    u32::MAX as usize
1264                } else {
1265                    u16::MAX as usize
1266                }
1267            }
1268
1269            // Check high limit for each byte
1270            for i in 0..self.opsize as usize {
1271                if ea + i > limit {
1272                    return None;
1273                }
1274            }
1275
1276            Bytes::Four
1277        };
1278
1279        self.canonical_check(
1280            if self.cpu_mode.is_bit64() && seg != SegRegister::FS && seg != SegRegister::GS {
1281                ea & (addrsize.mask() as usize)
1282            } else {
1283                (segment_base(segment) as usize + ea) & addrsize.mask() as usize
1284            },
1285        )
1286    }

1288    fn get_linear_addr<I: InsnMachineCtx>(
1289        &self,
1290        mctx: &I,
1291        seg: SegRegister,
1292        ea: usize,
1293        writable: bool,
1294    ) -> Result<usize, InsnError> {
1295        self.cal_linear_addr(mctx, seg, ea, writable)
1296            .ok_or(if seg == SegRegister::SS {
1297                InsnError::ExceptionSS
1298            } else {
1299                InsnError::ExceptionGP(0)
1300            })
1301            .and_then(|la| {
1302                if (mctx.read_cpl() == 3)
1303                    && (mctx.read_cr0() & CR0Flags::AM.bits()) != 0
1304                    && (mctx.read_flags() & RFlags::AC.bits()) != 0
1305                {
1306                    self.alignment_check(la, self.opsize)
1307                        .ok_or(InsnError::ExceptionAC)
1308                } else {
1309                    Ok(la)
1310                }
1311            })
1312    }

1314    fn emulate_ins_outs<I: InsnMachineCtx>(
1315        &self,
1316        mctx: &mut I,
1317        io_read: bool,
1318    ) -> Result<(), InsnError> {
1319        // I/O port number is stored in DX.
1320        let port = mctx.read_reg(Register::Rdx) as u16;
1321
1322        // Check the IO permission bit map.
1323        if !ioio_perm(mctx, port, self.opsize, io_read) {
1324            return Err(InsnError::ExceptionGP(0));
1325        }
1326
1327        let (seg, reg) = if io_read {
1328            // Input byte from I/O port specified in DX into
1329            // memory location specified with ES:(E)DI or
1330            // RDI.
1331            (SegRegister::ES, Register::Rdi)
1332        } else {
1333            // Output byte/word/doubleword from memory location specified in
1334            // DS:(E)SI (The DS segment may be overridden with a segment
1335            // override prefix.) or RSI to I/O port specified in DX.
1336            (
1337                self.override_seg.map_or(SegRegister::DS, |s| s),
1338                Register::Rsi,
1339            )
1340        };
1341
1342        // Decode the linear addresses and map as a memory object
1343        // which allows accessing to the memory represented by the
1344        // linear addresses.
1345        let linear_addr =
1346            self.get_linear_addr(mctx, seg, read_reg(mctx, reg, self.addrsize), io_read)?;
1347        if io_read {
1348            // Read data from IO port and then write to the memory location.
1349            let data = mctx.ioio_in(port, self.opsize)?;
1350            // SAFETY: The linear address is decoded from the instruction and
1351            // verified. It can be successfully remapped to a memory object
1352            // with write permission, and the remapped memory size matches the
1353            // operand size of the instruction.
1354            unsafe {
1355                match self.opsize {
1356                    Bytes::One => mctx
1357                        .map_linear_addr::<u8>(linear_addr, io_read, false)?
1358                        .mem_write(data as u8)?,
1359                    Bytes::Two => mctx
1360                        .map_linear_addr::<u16>(linear_addr, io_read, false)?
1361                        .mem_write(data as u16)?,
1362                    Bytes::Four => mctx
1363                        .map_linear_addr::<u32>(linear_addr, io_read, false)?
1364                        .mem_write(data as u32)?,
1365                    _ => return Err(InsnError::IoIoIn),
1366                };
1367            }
1368        } else {
1369            // Read data from memory location and then write to the IO port
1370            //
1371            // SAFETY: The linear address is decoded from the instruction and
1372            // verified. It can be successfully remapped to a memory object
1373            // with read permission, and the remapped memory size matches the
1374            // operand size of the instruction.
1375            let data = unsafe {
1376                match self.opsize {
1377                    Bytes::One => mctx
1378                        .map_linear_addr::<u8>(linear_addr, io_read, false)?
1379                        .mem_read()? as u64,
1380                    Bytes::Two => mctx
1381                        .map_linear_addr::<u16>(linear_addr, io_read, false)?
1382                        .mem_read()? as u64,
1383                    Bytes::Four => mctx
1384                        .map_linear_addr::<u32>(linear_addr, io_read, false)?
1385                        .mem_read()? as u64,
1386                    _ => return Err(InsnError::IoIoOut),
1387                }
1388            };
1389            mctx.ioio_out(port, self.opsize, data)?;
1390        }
1391
1392        let rflags = RFlags::from_bits_truncate(mctx.read_flags());
1393        if rflags.contains(RFlags::DF) {
1394            // The DF flag is 1, the (E)SI/DI register is decremented.
1395            write_reg(
1396                mctx,
1397                reg,
1398                read_reg(mctx, reg, self.addrsize)
1399                    .checked_sub(self.opsize as usize)
1400                    .ok_or(InsnError::IoIoOut)?,
1401                self.addrsize,
1402            );
1403        } else {
1404            // The DF flag is 0, the (E)SI/DI register is incremented.
1405            write_reg(
1406                mctx,
1407                reg,
1408                read_reg(mctx, reg, self.addrsize)
1409                    .checked_add(self.opsize as usize)
1410                    .ok_or(InsnError::IoIoOut)?,
1411                self.addrsize,
1412            );
1413        }
1414
1415        if self.repeat != 0 {
1416            // Update the count register with the left count which are not
1417            // emulated yet.
1418            write_reg(mctx, Register::Rcx, self.repeat - 1, self.addrsize);
1419        }
1420
1421        Ok(())
1422    }

1424    fn emulate_in_out<I: InsnMachineCtx>(
1425        &self,
1426        port: Operand,
1427        opsize: Bytes,
1428        mctx: &mut I,
1429        io_read: bool,
1430    ) -> Result<(), InsnError> {
1431        let port = match port {
1432            Operand::Reg(Register::Rdx) => mctx.read_reg(Register::Rdx) as u16,
1433            Operand::Reg(..) => unreachable!("Port value is always in DX"),
1434            Operand::Imm(imm) => match imm {
1435                Immediate::U8(val) => val as u16,
1436                _ => unreachable!("Port value in immediate is always 1 byte"),
1437            },
1438        };
1439
1440        // Check the IO permission bit map
1441        if !ioio_perm(mctx, port, opsize, io_read) {
1442            return Err(InsnError::ExceptionGP(0));
1443        }
1444
1445        if io_read {
1446            // Read data from IO port and then write to AL/AX/EAX.
1447            write_reg(
1448                mctx,
1449                Register::Rax,
1450                mctx.ioio_in(port, opsize)? as usize,
1451                opsize,
1452            );
1453        } else {
1454            // Read data from AL/AX/EAX and then write to the IO port.
1455            mctx.ioio_out(port, opsize, read_reg(mctx, Register::Rax, opsize) as u64)?;
1456        }
1457
1458        Ok(())
1459    }

1461    fn cal_effective_addr<I: InsnMachineCtx>(&self, mctx: &I) -> Result<usize, InsnError> {
1462        let base = if let Some(reg) = self.base_reg {
1463            match reg {
1464                Register::Rip => {
1465                    // RIP relative addressing is used in 64bit mode and
1466                    // starts from the following instruction
1467                    mctx.read_reg(reg) + self.insn_len
1468                }
1469                _ => mctx.read_reg(reg),
1470            }
1471        } else {
1472            0
1473        };
1474
1475        let index = if let Some(reg) = self.index_reg {
1476            mctx.read_reg(reg)
1477        } else {
1478            0
1479        };
1480
1481        Ok(base
1482            .checked_add(index << (self.scale as usize))
1483            .and_then(|v| v.checked_add(self.displacement as usize))
1484            .ok_or(InsnError::InvalidDecode)?
1485            & self.addrsize.mask() as usize)
1486    }

1488    #[inline]
1489    fn emulate_mmio_read<I: InsnMachineCtx>(
1490        &self,
1491        mctx: &I,
1492        seg: SegRegister,
1493        ea: usize,
1494    ) -> Result<u64, InsnError> {
1495        mctx.translate_linear_addr(self.get_linear_addr(mctx, seg, ea, false)?, false, false)
1496            .and_then(|(addr, shared)| {
1497                // SAFETY: The linear address is decoded from the instruction
1498                // and verified. It can be successfully translated to a
1499                // physical address with read permission.
1500                unsafe { mctx.handle_mmio_read(addr, shared, self.opsize) }
1501            })
1502    }

1521    fn emulate_mov<I: InsnMachineCtx>(&self, mctx: &mut I) -> Result<(), InsnError> {
1522        if self.prefix.contains(PrefixFlags::REPZ_P) {
1523            return Err(InsnError::UnSupportedInsn);
1524        }
1525
1526        let seg = if let Some(s) = self.override_seg {
1527            s
1528        } else if self.base_reg == Some(Register::Rsp) || self.base_reg == Some(Register::Rbp) {
1529            SegRegister::SS
1530        } else {
1531            SegRegister::DS
1532        };
1533        let ea = self.cal_effective_addr(mctx)?;
1534
1535        match self.get_opdesc()?.code {
1536            0x88 => {
1537                // Mov byte from reg (ModRM:reg) to mem (ModRM:r/m)
1538                // 88/r:	mov r/m8, r8
1539                // REX + 88/r:	mov r/m8, r8 (%ah, %ch, %dh, %bh not available)
1540                let (reg, lhbr) = self.cal_modrm_bytereg()?;
1541                let data = read_bytereg(mctx, reg, lhbr);
1542                self.emulate_mmio_write(mctx, seg, ea, data as u64)?;
1543            }
1544            0x89 => {
1545                // MOV from reg (ModRM:reg) to mem (ModRM:r/m)
1546                // 89/r:	mov r/m16, r16
1547                // 89/r:	mov r/m32, r32
1548                // REX.W + 89/r	mov r/m64, r64
1549                let data = read_reg(mctx, self.get_modrm_reg()?, self.opsize);
1550                self.emulate_mmio_write(mctx, seg, ea, data as u64)?;
1551            }
1552            0x8A => {
1553                // MOV byte from mem (ModRM:r/m) to reg (ModRM:reg)
1554                // 8A/r:	mov r8, r/m8
1555                // REX + 8A/r:	mov r8, r/m8
1556                let data = self.emulate_mmio_read(mctx, seg, ea)?;
1557                let (reg, lhbr) = self.cal_modrm_bytereg()?;
1558                write_bytereg(mctx, reg, lhbr, data as u8);
1559            }
1560            0x8B => {
1561                // MOV from mem (ModRM:r/m) to reg (ModRM:reg)
1562                // 8B/r:	mov r16, r/m16
1563                // 8B/r:	mov r32, r/m32
1564                // REX.W 8B/r:	mov r64, r/m64
1565                let data = self.emulate_mmio_read(mctx, seg, ea)?;
1566                write_reg(mctx, self.get_modrm_reg()?, data as usize, self.opsize);
1567            }
1568            0xA1 => {
1569                // MOV from seg:moffset to AX/EAX/RAX
1570                // A1:		mov AX, moffs16
1571                // A1:		mov EAX, moffs32
1572                // REX.W + A1:	mov RAX, moffs64
1573                let data = self.emulate_mmio_read(mctx, seg, ea)?;
1574                write_reg(mctx, Register::Rax, data as usize, self.opsize);
1575            }
1576            0xA3 => {
1577                // MOV from AX/EAX/RAX to seg:moffset
1578                // A3:		mov moffs16, AX
1579                // A3:		mov moffs32, EAX
1580                // REX.W + A3:	mov moffs64, RAX
1581                let data = read_reg(mctx, Register::Rax, self.opsize);
1582                self.emulate_mmio_write(mctx, seg, ea, data as u64)?;
1583            }
1584            0xC6 | 0xC7 => {
1585                // MOV from imm8 to mem (ModRM:r/m)
1586                // C6/0		mov r/m8, imm8
1587                // REX + C6/0	mov r/m8, imm8
1588                // MOV from imm16/imm32 to mem (ModRM:r/m)
1589                // C7/0		mov r/m16, imm16
1590                // C7/0		mov r/m32, imm32
1591                // REX.W + C7/0	mov r/m64, imm32 (sign-extended to 64-bits)
1592                self.emulate_mmio_write(mctx, seg, ea, self.immediate as u64 & self.opsize.mask())?;
1593            }
1594            _ => return Err(InsnError::UnSupportedInsn),
1595        }
1596
1597        Ok(())
1598    }
