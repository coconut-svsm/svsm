7use core::iter;
8
9use crate::address::PhysAddr;
10use crate::error::SvsmError;
11use crate::mm::alloc::PageRef;
12use crate::types::{PAGE_SHIFT, PAGE_SIZE};
13use crate::utils::align_up;
14
15extern crate alloc;
16use alloc::vec::Vec;
17
18/// Contains base functionality for all [`VirtualMapping`](super::api::VirtualMapping)
19/// types which use self-allocated PageFile pages.
20#[derive(Default, Debug)]
21pub struct RawAllocMapping {
22    /// A vec containing references to PageFile allocations
23    pages: Vec<Option<PageRef>>,
24
25    /// Number of pages required in `pages`
26    count: usize,
27}
28
29impl RawAllocMapping {
30    /// Creates a new instance of RawAllocMapping
31    ///
32    /// # Arguments
33    ///
34    /// * `size` - Size of the mapping in bytes
35    ///
36    /// # Returns
37    ///
38    /// New instance of RawAllocMapping. Still needs to call `alloc_pages()` on it before it can be used.
39    pub fn new(size: usize) -> Self {
40        let count = align_up(size, PAGE_SIZE) >> PAGE_SHIFT;
41        let pages: Vec<Option<PageRef>> = iter::repeat_n(None, count).collect();
42        RawAllocMapping { pages, count }
43    }
44
45    /// Allocates a single backing page of type PageFile if the page has not already
46    /// been allocated
47    ///
48    /// # Argument
49    ///
50    /// * 'offset' - The offset in bytes from the start of the mapping
51    ///
52    /// # Returns
53    ///
54    /// `Ok(())` if the page has been allocated, `Err(SvsmError::Mem)` otherwise
55    pub fn alloc_page(&mut self, offset: usize) -> Result<(), SvsmError> {
56        let index = offset >> PAGE_SHIFT;
57        if index < self.count {
58            let entry = self.pages.get_mut(index).ok_or(SvsmError::Mem)?;
59            entry.get_or_insert(PageRef::new()?);
60        }
61        Ok(())
62    }
63
64    /// Allocates a full set of backing pages of type PageFile
65    ///
66    /// # Returns
67    ///
68    /// `Ok(())` when all pages could be allocated, `Err(SvsmError::Mem)` otherwise
69    pub fn alloc_pages(&mut self) -> Result<(), SvsmError> {
70        for index in 0..self.count {
71            self.alloc_page(index * PAGE_SIZE)?;
72        }
73        Ok(())
74    }
75
76    /// Returns a reference to a page at a given index. The page must already
77    /// been allocated.
78    ///
79    /// # Arguments
80    ///
81    /// * `index` - Page index to reference
82    ///
83    /// # Returns
84    ///
85    /// A reference to the requested page.
86    ///
87    /// # Panics
88    ///
89    /// This function panics if an invalid index is accessed or the page at the
90    /// index has not been allocated.
91    pub fn page(&self, index: usize) -> PageRef {
92        self.pages[index].as_ref().unwrap().clone()
93    }
94
95    /// Request size of the mapping in bytes
96    ///
97    /// # Returns
98    ///
99    /// The size of the mapping in bytes as `usize`.
100    pub fn mapping_size(&self) -> usize {
101        self.count * PAGE_SIZE
102    }
103
104    /// Request physical address to map for a given offset
105    ///
106    /// # Arguments
107    ///
108    /// * `offset` - Byte offset into the memory mapping
109    ///
110    /// # Returns
111    ///
112    /// Physical address to map for the given offset.
113    pub fn map(&self, offset: usize) -> Option<PhysAddr> {
114        let pfn = offset >> PAGE_SHIFT;
115        self.pages
116            .get(pfn)
117            .and_then(|r| r.as_ref().map(|r| r.phys_addr()))
118    }
119
120    /// Unmap call-back - currently nothing to do in this function
121    ///
122    /// # Arguments
123    ///
124    /// * `_offset` - Byte offset into the mapping
125    pub fn unmap(&self, _offset: usize) {
126        // Nothing to do for now
127    }
128
129    /// Check if a page has been allocated
130    ///
131    /// # Arguments
132    ///
133    /// * 'offset' - Byte offset into the mapping
134    ///
135    /// # Returns
136    ///
137    /// 'true' if the page containing the offset has been allocated
138    /// otherwise 'false'.
139    pub fn present(&self, offset: usize) -> bool {
140        let pfn = offset >> PAGE_SHIFT;
141        self.pages.get(pfn).and_then(|r| r.as_ref()).is_some()
142    }
143}
