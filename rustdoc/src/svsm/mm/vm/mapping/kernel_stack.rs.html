22/// Mapping to be used as a kernel stack. This maps a stack including guard
23/// pages at the top and bottom.
24#[derive(Default, Debug)]
25pub struct VMKernelStack {
26    /// Allocation for stack pages
27    alloc: RawAllocMapping,
28    /// Number of guard pages to reserve address space for
29    guard_pages: usize,
30    /// `True` for shadow stacks
31    shadow: bool,
32}
33
34impl VMKernelStack {
35    /// Returns the virtual address for the top of this kernel stack
36    ///
37    /// # Arguments
38    ///
39    /// * `base` - Virtual base address this stack is mapped at (including
40    ///   guard pages).
41    ///
42    /// # Returns
43    ///
44    /// Offset from start of virtual mapping to top-of-stack.
45    pub fn top_of_stack(&self) -> usize {
46        let guard_size = self.guard_pages * PAGE_SIZE;
47        let tos = guard_size + self.alloc.mapping_size();
48        debug_assert!(tos > guard_size);
49        tos
50    }
51
52    /// Returns the stack bounds of this kernel stack
53    ///
54    /// # Arguments
55    ///
56    /// * `base` - Virtual base address this stack is mapped at (including
57    ///   guard pages).
58    ///
59    /// # Returns
60    ///
61    /// A [`MemoryRegion`] object containing the bottom and top addresses for
62    /// the stack
63    pub fn bounds(&self, base: VirtAddr) -> MemoryRegion<VirtAddr> {
64        let mapping_size = self.alloc.mapping_size();
65        let guard_size = self.guard_pages * PAGE_SIZE;
66        MemoryRegion::new(base + guard_size, mapping_size)
67    }
68
69    /// Create a new [`VMKernelStack`] with a given size. This function will
70    /// already allocate the backing pages for the stack.
71    ///
72    /// # Arguments
73    ///
74    /// * `size` - Size of the kernel stack, without guard pages
75    /// * `shadow` - Whether this mapping is for a shadow stack
76    ///
77    /// # Returns
78    ///
79    /// Initialized stack on success, Err(SvsmError::Mem) on error
80    fn new_size(size: usize, shadow: bool) -> Result<Self, SvsmError> {
81        // Make sure size is page-aligned
82        let size = page_align_up(size);
83        // At least two guard-pages needed
84        let total_size = (size + 2 * PAGE_SIZE).next_power_of_two();
85        let guard_pages = ((total_size - size) >> PAGE_SHIFT) / 2;
86        let mut stack = VMKernelStack {
87            alloc: RawAllocMapping::new(size),
88            guard_pages,
89            shadow,
90        };
91        stack.alloc_pages()?;
92
93        Ok(stack)
94    }
95
96    /// Create a new [`VMKernelStack`] with the default size. This function
97    /// will already allocate the backing pages for the stack.
98    ///
99    /// # Returns
100    ///
101    /// Initialized stack on success, Err(SvsmError::Mem) on error
102    pub fn new() -> Result<Self, SvsmError> {
103        VMKernelStack::new_size(STACK_SIZE, false)
104    }
105
106    /// Create a new [`VMKernelStack`] with one page in size for use as a
107    /// shadow stack.
108    ///
109    /// # Returns
110    ///
111    /// Shadow stack on success, Err(SvsmError::Mem) on error
112    pub fn new_shadow() -> Result<Self, SvsmError> {
113        VMKernelStack::new_size(PAGE_SIZE, true)
114    }
115
116    /// Get a reference to the shadow stack page. This is required because
117    /// shadow stacks are mapped read-only and a separate mapping needs to be
118    /// created to initialize it.
119    ///
120    /// # Returns
121    ///
122    /// A [`PageRef`] reference to the shadow stack page
123    ///
124    /// # Panics
125    ///
126    /// The function panics if called on a non-shadow-stack object.
127    pub fn shadow_page(&self) -> PageRef {
128        assert!(self.shadow);
129        self.alloc.page(0)
130    }
131
132    /// Create a new [`VMKernelStack`] with the default size, packed into a
133    /// [`Mapping`]. This function will allocate the backing pages for the
134    /// stack.
135    ///
136    /// # Returns
137    ///
138    /// Initialized Mapping to stack on success, Err(SvsmError::Mem) on error
139    pub fn new_mapping() -> Result<Mapping, SvsmError> {
140        Ok(Arc::new(Self::new()?))
141    }
142
143    fn alloc_pages(&mut self) -> Result<(), SvsmError> {
144        self.alloc.alloc_pages()
145    }
146}
147
148impl VirtualMapping for VMKernelStack {
149    fn mapping_size(&self) -> usize {
150        self.alloc.mapping_size() + ((self.guard_pages * 2) << PAGE_SHIFT)
151    }
152
153    fn map(&self, offset: usize) -> Option<PhysAddr> {
154        let pfn = offset >> PAGE_SHIFT;
155        let guard_offset = self.guard_pages << PAGE_SHIFT;
156
157        if pfn >= self.guard_pages {
158            self.alloc.map(offset - guard_offset)
159        } else {
160            None
161        }
162    }
163
164    fn unmap(&self, offset: usize) {
165        let pfn = offset >> PAGE_SHIFT;
166
167        if pfn >= self.guard_pages {
168            self.alloc.unmap(pfn - self.guard_pages);
169        }
170    }
171
172    fn pt_flags(&self, _offset: usize) -> PTEntryFlags {
173        if self.shadow {
174            // The CPU requires shadow stacks to be dirty and not writable.
175            PTEntryFlags::NX | PTEntryFlags::ACCESSED | PTEntryFlags::DIRTY
176        } else {
177            PTEntryFlags::WRITABLE | PTEntryFlags::NX | PTEntryFlags::ACCESSED | PTEntryFlags::DIRTY
178        }
179    }
180}
