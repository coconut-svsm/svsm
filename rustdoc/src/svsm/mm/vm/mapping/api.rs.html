23/// Information required to resolve a page fault within a virtual mapping
24#[derive(Debug, Copy, Clone)]
25pub struct VMPageFaultResolution {
26    /// The physical address of a page that must be mapped to the page fault
27    /// virtual address to resolve the page fault.
28    pub paddr: PhysAddr,
29
30    /// The flags to use to map the virtual memory page.
31    pub flags: PTEntryFlags,
32}
33
34pub trait VirtualMapping: core::fmt::Debug + Send + Sync {
35    /// Request the size of the virtual memory mapping
36    ///
37    /// # Returns
38    ///
39    /// Mapping size. Will always be a multiple of `VirtualMapping::page_size()`
40    fn mapping_size(&self) -> usize;
41
42    /// Indicates whether the mapping has any associated data.
43    ///
44    /// # Returns
45    ///
46    /// `true' if there is associated physical data, or `false' if there is
47    /// none.
48    fn has_data(&self) -> bool {
49        // Defaults to true
50        true
51    }
52
53    /// Request physical address to map for a given offset
54    ///
55    /// # Arguments
56    ///
57    /// * `offset` - Offset into the virtual memory mapping
58    ///
59    /// # Returns
60    ///
61    /// Physical address to map for the given offset, if any. None is also a
62    /// valid return value and does not indicate an error.
63    fn map(&self, offset: usize) -> Option<PhysAddr>;
64
65    /// Inform the virtual memory mapping about an offset being unmapped.
66    /// Implementing `unmap()` is optional.
67    ///
68    /// # Arguments
69    ///
70    /// * `_offset`
71    fn unmap(&self, _offset: usize) {
72        // Provide default in case there is nothing to do
73    }
74
75    /// Request the PTEntryFlags used for this virtual memory mapping.
76    ///
77    /// # Arguments
78    ///
79    /// * 'offset' -> The offset in bytes into the `VirtualMapping`. The flags
80    ///   returned from this function relate to the page at the
81    ///   given offset
82    ///
83    /// # Returns
84    ///
85    /// A combination of:
86    ///
87    /// * PTEntryFlags::WRITABLE
88    /// * PTEntryFlags::NX,
89    /// * PTEntryFlags::ACCESSED
90    /// * PTEntryFlags::DIRTY
91    fn pt_flags(&self, offset: usize) -> PTEntryFlags;
92
93    /// Request the page size used for mappings
94    ///
95    /// # Returns
96    ///
97    /// Either PAGE_SIZE or PAGE_SIZE_2M
98    fn page_size(&self) -> PageSize {
99        // Default to system page-size
100        PageSize::Regular
101    }
102
103    /// Request whether the mapping is shared or private. Defaults to private
104    /// unless overwritten by the specific type.
105    ///
106    /// # Returns
107    ///
108    /// * `True` - When mapping is shared
109    /// * `False` - When mapping is private
110    fn shared(&self) -> bool {
111        // Shared with the HV - defaults not No
112        false
113    }
114
115    /// Handle a page fault that occurred on a virtual memory address within
116    /// this mapping.
117    ///
118    /// # Arguments
119    ///
120    /// * 'vmr' - Virtual memory range that contains the mapping. This
121    ///   [`VirtualMapping`] can use this to insert/remove regions
122    ///   as necessary to handle the page fault.
123    ///
124    /// * `offset` - Offset into the virtual mapping that was the subject of
125    ///   the page fault.
126    ///
127    /// * 'write' - `true` if the fault was due to a write to the memory
128    ///   location, or 'false' if the fault was due to a read.
129    fn handle_page_fault(
130        &mut self,
131        _vmr: &VMR,
132        _offset: usize,
133        _write: bool,
134    ) -> Result<VMPageFaultResolution, SvsmError> {
135        Err(SvsmError::Mem)
136    }
137}

139#[derive(Debug)]
140pub struct Mapping {
141    mapping: RWLock<Box<dyn VirtualMapping>>,
142}
143
144impl Mapping {
145    pub fn new<T>(mapping: T) -> Self
146    where
147        T: VirtualMapping + 'static,
148    {
149        Mapping {
150            mapping: RWLock::new(Box::new(mapping)),
151        }
152    }
153
154    pub fn get(&self) -> ReadLockGuard<'_, Box<dyn VirtualMapping>> {
155        self.mapping.lock_read()
156    }
157
158    pub fn get_mut(&self) -> WriteLockGuard<'_, Box<dyn VirtualMapping>> {
159        self.mapping.lock_write()
160    }
161}

163/// A single mapping of virtual memory in a virtual memory range
164#[derive(Debug)]
165pub struct VMM {
166    /// Link for storing this instance in an RBTree
167    link: AtomicLink,
168
169    /// The virtual memory range covered by this mapping
170    /// It is stored in a RefCell to check borrowing rules at runtime.
171    /// This is safe as any modification to `range` is protected by a lock in
172    /// the parent data structure. This is required because changes here also
173    /// need changes in the parent data structure.
174    range: Range<usize>,
175
176    /// Pointer to the actual mapping
177    /// It is protected by an RWLock to serialize concurent accesses.
178    mapping: Arc<Mapping>,
179}
180
181intrusive_adapter!(pub VMMAdapter = Box<VMM>: VMM { link: AtomicLink });
182
183impl<'a> KeyAdapter<'a> for VMMAdapter {
184    type Key = usize;
185    fn get_key(&self, node: &'a VMM) -> Self::Key {
186        node.range.start
187    }
188}
189
190impl VMM {
191    /// Create a new VMM instance with at a given address and backing struct
192    ///
193    /// # Arguments
194    ///
195    /// * `start_pfn` - Virtual start pfn to store in the mapping
196    /// * `mapping` - `Arc<Mapping>` pointer to the backing struct
197    ///
198    /// # Returns
199    ///
200    /// New instance of VMM
201    pub fn new(start_pfn: usize, mapping: Arc<Mapping>) -> Self {
202        let size = mapping.get().mapping_size() >> PAGE_SHIFT;
203        VMM {
204            link: AtomicLink::new(),
205            range: Range {
206                start: start_pfn,
207                end: start_pfn + size,
208            },
209            mapping,
210        }
211    }
212
213    /// Request the mapped range as page frame numbers
214    ///
215    /// # Returns
216    ///
217    /// The start and end (non-inclusive) virtual address for this virtual
218    /// mapping, right-shifted by `PAGE_SHIFT`.
219    pub fn range_pfn(&self) -> (usize, usize) {
220        (self.range.start, self.range.end)
221    }
222
223    /// Request the mapped range
224    ///
225    /// # Returns
226    ///
227    /// The start and end virtual address for this virtual mapping.
228    pub fn range(&self) -> (VirtAddr, VirtAddr) {
229        (
230            VirtAddr::from(self.range.start << PAGE_SHIFT),
231            VirtAddr::from(self.range.end << PAGE_SHIFT),
232        )
233    }
234
235    pub fn get_mapping(&self) -> ReadLockGuard<'_, Box<dyn VirtualMapping>> {
236        self.mapping.get()
237    }
238
239    pub fn get_mapping_mut(&self) -> WriteLockGuard<'_, Box<dyn VirtualMapping>> {
240        self.mapping.get_mut()
241    }
242
243    pub fn get_mapping_clone(&self) -> Arc<Mapping> {
244        self.mapping.clone()
245    }
246}
