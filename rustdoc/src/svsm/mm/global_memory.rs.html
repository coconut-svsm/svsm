18struct GlobalRanges {
19    range_4k: VirtualRange,
20    range_2m: VirtualRange,
21}
22
23impl GlobalRanges {
24    const fn new() -> Self {
25        Self {
26            range_4k: VirtualRange::new(),
27            range_2m: VirtualRange::new(),
28        }
29    }
30
31    fn init(&mut self) {
32        let region_4k_start = SVSM_GLOBAL_MAPPING_BASE;
33        let region_2m_start = SVSM_GLOBAL_MAPPING_BASE + SIZE_LEVEL1;
34        let page_count_4k = (region_2m_start - SVSM_GLOBAL_MAPPING_BASE) / PAGE_SIZE;
35        let page_count_2m = (SVSM_GLOBAL_MAPPING_END - region_2m_start) / PAGE_SIZE_2M;
36
37        self.range_4k
38            .init(region_4k_start, page_count_4k, PAGE_SHIFT);
39        self.range_2m
40            .init(region_2m_start, page_count_2m, PAGE_SHIFT_2M);
41    }
42
43    fn alloc(
44        &mut self,
45        page_count: usize,
46        huge: bool,
47        shared: bool,
48    ) -> Result<GlobalRangeGuard, SvsmError> {
49        let vstart = if huge {
50            self.range_2m.alloc(page_count, 0)?
51        } else {
52            self.range_4k.alloc(page_count, 0)?
53        };
54
55        Ok(GlobalRangeGuard::new(vstart, page_count, huge, shared))
56    }
57
58    fn free(&mut self, vaddr: VirtAddr, page_count: usize, huge: bool) {
59        if huge {
60            self.range_2m.free(vaddr, page_count);
61        } else {
62            self.range_4k.free(vaddr, page_count);
63        }
64    }
65}

67#[derive(Debug)]
68pub struct GlobalRangeGuard {
69    vstart: VirtAddr,
70    pages: usize,
71    huge: bool,
72    shared: bool,
73}
74
75impl GlobalRangeGuard {
76    /// Create a [`GlobalRangeGuard`] with the given parameters.
77    ///
78    /// # Arguments
79    ///
80    /// * `vstart`: Start virtual address.
81    /// * `pages`: Number pages mapped.
82    /// * `huge`: Whether to use normal or huge pages.
83    /// * `shared`: Whether mapping is private or shared.
84    ///
85    /// # Returns
86    ///
87    /// A new instance of [`GlobalRangeGuard`] set up with the requested
88    /// parameters.
89    fn new(vstart: VirtAddr, pages: usize, huge: bool, shared: bool) -> Self {
90        Self {
91            vstart,
92            pages,
93            huge,
94            shared,
95        }
96    }
97
98    /// Request the virtual start address of the global mapping.
99    ///
100    /// # Returns
101    ///
102    /// Virtual start address of the global mapping.
103    pub fn addr(&self) -> VirtAddr {
104        self.vstart
105    }
106
107    /// Request the length in bytes of the global mapping.
108    ///
109    /// # Returns
110    ///
111    /// Length of the global mapping in bytes.
112    pub fn size(&self) -> usize {
113        let page_size = if self.huge { PAGE_SIZE_2M } else { PAGE_SIZE };
114        self.pages * page_size
115    }
116
117    fn map(&self, paddr: PhysAddr, flags: PTEntryFlags) -> Result<(), SvsmError> {
118        if self.huge {
119            this_cpu()
120                .get_pgtable()
121                .map_region_2m(self.region(), paddr, flags, self.shared)
122        } else {
123            this_cpu()
124                .get_pgtable()
125                .map_region_4k(self.region(), paddr, flags, self.shared)
126        }
127    }
128
129    fn unmap(&self) {
130        if self.huge {
131            this_cpu().get_pgtable().unmap_region_2m(self.region());
132        } else {
133            this_cpu().get_pgtable().unmap_region_4k(self.region());
134        }
135    }
136
137    /// Request the mapped region as a [`MemoryRegion`].
138    ///
139    /// # Returns
140    ///
141    /// The global mapped region as an instance of [`MemoryRegion`].
142    pub fn region(&self) -> MemoryRegion<VirtAddr> {
143        let page_size = if self.huge { PAGE_SIZE_2M } else { PAGE_SIZE };
144        MemoryRegion::new(self.vstart, self.pages * page_size)
145    }
146}
147
148impl Drop for GlobalRangeGuard {
149    fn drop(&mut self) {
150        self.unmap();
151        // Flush TLB before allowing to re-use addresses
152        flush_tlb_global_sync();
153        GLOBAL_RANGES
154            .lock()
155            .free(self.vstart, self.pages, self.huge);
156    }
157}
158
159static GLOBAL_RANGES: SpinLock<GlobalRanges> = SpinLock::new(GlobalRanges::new());
160
161/// Initialize global allocatable virtual address ranges.
162pub fn init_global_ranges() {
163    GLOBAL_RANGES.lock().init();
164}

166/// Map physical addresses into the global shared address range.
167///
168/// # Arguments
169///
170/// * `pstart`: Start physical to map, must be aligned to requested page-size.
171/// * `size`: Number of bytes to map. Will be aligned up to requested page-size.
172/// * `flages`: Page-table flags to use for mapping.
173/// * `huge`: Request normal or huge pages for the mapping.
174/// * `shared`: Request a shared or private mapping.
175///
176/// # Returns
177///
178/// A Result with a [`GlobalRangeGuard`] on success or [`SvsmError`] on failure.
179pub fn map_global_range(
180    pstart: PhysAddr,
181    size: usize,
182    flags: PTEntryFlags,
183    huge: bool,
184    shared: bool,
185) -> Result<GlobalRangeGuard, SvsmError> {
186    assert!(pstart.is_page_aligned());
187
188    let page_size = if huge { PAGE_SIZE_2M } else { PAGE_SIZE };
189    let size_aligned = align_up(size, page_size);
190
191    if size_aligned == 0 {
192        return Err(SvsmError::Mem);
193    }
194
195    let pages = size_aligned / page_size;
196
197    let guard = GLOBAL_RANGES.lock().alloc(pages, huge, shared)?;
198    guard.map(pstart, flags)?;
199
200    Ok(guard)
201}

203/// Create a private mapping using of physical addresses into the global shared
204/// address range using 4KiB pages.
205///
206/// # Arguments

208/// * `pstart`: Start physical to map, must be aligned to 4KiB.
209/// * `size`: Number of bytes to map. Will be aligned up to 4KiB.
210/// * `flages`: Page-table flags to use for mapping.
211///
212/// # Returns
213///
214/// A Result with a [`GlobalRangeGuard`] on success or [`SvsmError`] on failure.
215pub fn map_global_range_4k_private(
216    pstart: PhysAddr,
217    size: usize,
218    flags: PTEntryFlags,
219) -> Result<GlobalRangeGuard, SvsmError> {
