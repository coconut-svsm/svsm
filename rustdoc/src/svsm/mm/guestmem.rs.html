22#[inline]
23pub fn read_u8(v: VirtAddr) -> Result<u8, SvsmError> {
24    let mut rcx: u64;
25    let mut val: u64;
26
27    unsafe {
28        asm!("1: movb ({0}), %al",
29             "   xorq %rcx, %rcx",
30             "2:",
31             ".pushsection \"__exception_table\",\"a\"",
32             ".balign 16",
33             ".quad (1b)",
34             ".quad (2b)",
35             ".popsection",
36                in(reg) v.bits(),
37                out("rax") val,
38                out("rcx") rcx,
39                options(att_syntax, nostack));
40    }
41
42    let ret: u8 = (val & 0xff) as u8;
43    if rcx == 0 {
44        Ok(ret)
45    } else {
46        Err(SvsmError::Fault)
47    }
48}

50/// Writes 1 byte at a virtual address.
51///
52/// # Safety
53///
54/// The caller must verify not to corrupt arbitrary memory, as this function
55/// doesn't make any checks in that regard.
56///
57/// # Returns
58///
59/// Returns an error if the specified address is not mapped or is not mapped
60/// with the appropriate write permissions.
61#[inline]
62pub unsafe fn write_u8(v: VirtAddr, val: u8) -> Result<(), SvsmError> {
63    let mut rcx: u64;
64
65    unsafe {
66        asm!("1: movb %al, ({0})",
67             "   xorq %rcx, %rcx",
68             "2:",
69             ".pushsection \"__exception_table\",\"a\"",
70             ".balign 16",
71             ".quad (1b)",
72             ".quad (2b)",
73             ".popsection",
74                in(reg) v.bits(),
75                in("rax") val as u64,
76                out("rcx") rcx,
77                options(att_syntax, nostack));
78    }
79
80    if rcx == 0 {
81        Ok(())
82    } else {
83        Err(SvsmError::Fault)
84    }
85}

172#[inline]
173unsafe fn copy_bytes(src: usize, dst: usize, size: usize) -> Result<(), SvsmError> {
174    let mut rcx: u64;
175
176    unsafe {
177        asm!("1:cld
178                rep movsb
179              2:
180             .pushsection \"__exception_table\",\"a\"
181             .balign 16
182             .quad (1b)
183             .quad (2b)
184             .popsection",
185                inout("rsi") src => _,
186                inout("rdi") dst => _,
187                inout("rcx") size => rcx,
188                options(att_syntax, nostack));
189    }
190
191    if rcx == 0 {
192        Ok(())
193    } else {
194        Err(SvsmError::Fault)
195    }
196}

198#[inline]
199unsafe fn do_movsb<T>(src: *const T, dst: *mut T) -> Result<(), SvsmError> {
200    let size: usize = size_of::<T>();
201    let s = src as usize;
202    let d = dst as usize;
203
204    // SAFETY: Only safe when safety requirements for do_movsb() are fulfilled.
205    unsafe { copy_bytes(s, d, size) }
206}

208#[derive(Debug)]
209pub struct GuestPtr<T: Copy> {
210    ptr: *mut T,
211}
212
213impl<T: Copy> GuestPtr<T> {
214    #[inline]
215    pub fn new(v: VirtAddr) -> Self {
216        Self {
217            ptr: v.as_mut_ptr::<T>(),
218        }
219    }
220
221    #[inline]
222    pub const fn from_ptr(p: *mut T) -> Self {
223        Self { ptr: p }
224    }
225
226    /// # Safety
227    ///
228    /// The caller must verify not to read arbitrary memory, as this function
229    /// doesn't make any checks in that regard.
230    ///
231    /// # Returns
232    ///
233    /// Returns an error if the specified address is not mapped.
234    #[inline]
235    pub unsafe fn read(&self) -> Result<T, SvsmError> {
236        let mut buf = MaybeUninit::<T>::uninit();
237
238        unsafe {
239            do_movsb(self.ptr, buf.as_mut_ptr())?;
240            Ok(buf.assume_init())
241        }
242    }
243
244    /// # Safety
245    ///
246    /// The caller must verify not to corrupt arbitrary memory, as this function
247    /// doesn't make any checks in that regard.
248    ///
249    /// # Returns
250    ///
251    /// Returns an error if the specified address is not mapped or is not mapped
252    /// with the appropriate write permissions.
253    #[inline]
254    pub unsafe fn write(&self, buf: T) -> Result<(), SvsmError> {
255        unsafe { do_movsb(&buf, self.ptr) }
256    }
257
258    /// # Safety
259    ///
260    /// The caller must verify not to corrupt arbitrary memory, as this function
261    /// doesn't make any checks in that regard.
262    ///
263    /// # Returns
264    ///
265    /// Returns an error if the specified address is not mapped or is not mapped
266    /// with the appropriate write permissions.
267    #[inline]
268    pub unsafe fn write_ref(&self, buf: &T) -> Result<(), SvsmError> {
269        unsafe { do_movsb(buf, self.ptr) }
270    }
271
272    #[inline]
273    pub const fn cast<N: Copy>(&self) -> GuestPtr<N> {
274        GuestPtr::from_ptr(self.ptr.cast())
275    }
276
277    #[inline]
278    pub fn offset(&self, count: isize) -> Self {
279        GuestPtr::from_ptr(self.ptr.wrapping_offset(count))
280    }
281}

283impl<T: Copy> InsnMachineMem for GuestPtr<T> {
284    type Item = T;
285
286    /// Safety: See the GuestPtr's read() method documentation for safety requirements.
287    unsafe fn mem_read(&self) -> Result<Self::Item, InsnError> {
288        unsafe { self.read().map_err(|_| InsnError::MemRead) }
289    }
290
291    /// Safety: See the GuestPtr's write() method documentation for safety requirements.
292    unsafe fn mem_write(&mut self, data: Self::Item) -> Result<(), InsnError> {
293        unsafe { self.write(data).map_err(|_| InsnError::MemWrite) }
294    }

299impl UserAccessGuard {

301        stac();

303    }

312#[derive(Debug)]
313pub struct UserPtr<T: Copy> {
314    guest_ptr: GuestPtr<T>,
315}
316
317impl<T: Copy> UserPtr<T> {
318    #[inline]
319    pub fn new(v: VirtAddr) -> Self {
320        Self {
321            guest_ptr: GuestPtr::new(v),
322        }
323    }
324
325    fn check_bounds(&self) -> bool {
326        let v = VirtAddr::from(self.guest_ptr.ptr);
327
328        (USER_MEM_START..USER_MEM_END).contains(&v)
329            && (USER_MEM_START..USER_MEM_END).contains(&(v + size_of::<T>()))
330    }
331
332    #[inline]
333    pub fn read(&self) -> Result<T, SvsmError>
334    where
335        T: FromBytes,
336    {
337        if !self.check_bounds() {
338            return Err(SvsmError::InvalidAddress);
339        }
340        let _guard = UserAccessGuard::new();
341        unsafe { self.guest_ptr.read() }
342    }
343
344    #[inline]
345    pub fn write(&self, buf: T) -> Result<(), SvsmError> {
346        self.write_ref(&buf)
347    }
348
349    #[inline]
350    pub fn write_ref(&self, buf: &T) -> Result<(), SvsmError> {
351        if !self.check_bounds() {
352            return Err(SvsmError::InvalidAddress);
353        }
354        let _guard = UserAccessGuard::new();
355        unsafe { self.guest_ptr.write_ref(buf) }
356    }
357
358    #[inline]
359    pub const fn cast<N: Copy>(&self) -> UserPtr<N> {
360        UserPtr {
361            guest_ptr: self.guest_ptr.cast(),
362        }
363    }
364
365    #[inline]
366    pub fn offset(&self, count: isize) -> UserPtr<T> {
367        UserPtr {
368            guest_ptr: self.guest_ptr.offset(count),
369        }
370    }
371}

373impl UserPtr<c_char> {
374    /// Reads a null-terminated C string from the user space.
375    /// Allocates memory for the string and returns a `String`.
376    pub fn read_c_string(&self) -> Result<String, SvsmError> {
377        let mut buffer = Vec::new();
378
379        for offset in 0..PATH_MAX {
380            let current_ptr = self.offset(offset as isize);
381            let char_result = current_ptr.read()?;
382            match char_result {
383                0 => return String::from_utf8(buffer).map_err(|_| SvsmError::InvalidUtf8),
384                c => buffer.push(c as u8),
385            }
386        }
387        Err(SvsmError::InvalidBytes)
388    }
389}

391fn check_bounds_user(start: usize, len: usize) -> Result<(), SvsmError> {
392    let end: usize = start.checked_add(len).ok_or(SvsmError::InvalidAddress)?;
393
394    if end > USER_MEM_END.bits() {
395        Err(SvsmError::InvalidAddress)
396    } else {
397        Ok(())
398    }
399}

401pub fn copy_from_user(src: VirtAddr, dst: &mut [u8]) -> Result<(), SvsmError> {
402    let source = src.bits();
403    let destination = dst.as_mut_ptr() as usize;
404    let size = dst.len();
405
406    check_bounds_user(source, size)?;
407
408    // SAFETY: Safe because the copy only happens to the memory belonging to
409    // the dst slice from user-mode memory.
410    unsafe {
411        let _guard = UserAccessGuard::new();
412        copy_bytes(source, destination, size)
413    }
414}
415
416pub fn copy_to_user(src: &[u8], dst: VirtAddr) -> Result<(), SvsmError> {
417    let source = src.as_ptr() as usize;
418    let destination = dst.bits();
419    let size = src.len();
420
421    check_bounds_user(destination, size)?;
422
423    // SAFETY: Only reads data from with the slice and copies to an address
424    // guaranteed to be in user-space.
425    unsafe {
426        let _guard = UserAccessGuard::new();
427        copy_bytes(source, destination, size)
428    }
429}

437    fn test_read_u8_valid_address() {
438        // Create a region to read from
439        let test_buffer: [u8; 6] = [0; 6];

441

448    #[cfg_attr(miri, ignore = "inline assembly")]
449    fn test_write_u8_valid_address() {
450        // Create a mutable region we can write into
451        let mut test_buffer: [u8; 6] = [0; 6];
452        let test_address = VirtAddr::from(test_buffer.as_mut_ptr());
453        let data_to_write = 0x42;
454
455        // SAFETY: test_address points to the virtual address of test_buffer.
456        unsafe {
457            write_u8(test_address, data_to_write).unwrap();
458        }
459
460        assert_eq!(test_buffer[0], data_to_write);
461    }

468        let ptr: GuestPtr<[u8; 15]> = GuestPtr::new(test_addr);
