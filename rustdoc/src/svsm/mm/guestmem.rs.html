26/// Read one byte from a virtual address.
27///
28/// # Arguments
29///
30/// - `v` - Virtual address to read.
31///
32/// # Returns
33///
34/// `Ok(u8)` with the value read on success, `Err(SvsmError)` on failure.
35///
36/// # Safety
37///
38/// Any safety requirements for accessing raw pointers apply here as well.
39#[inline]
40pub unsafe fn read_u8(v: VirtAddr) -> Result<u8, SvsmError> {
41    let mut rcx: u64;
42    let mut val: u64;
43
44    // SAFETY: Assembly dereferences the pointer, which is safe when the
45    // function's safety requirements are fulfilled.
46    unsafe {
47        asm!("1: movb ({0}), %al",
48             "   xorq %rcx, %rcx",
49             "2:",
50             ".pushsection \"__exception_table\",\"a\"",
51             ".balign 16",
52             ".quad (1b)",
53             ".quad (2b)",
54             ".popsection",
55                in(reg) v.bits(),
56                out("rax") val,
57                out("rcx") rcx,
58                options(att_syntax, nostack));
59    }
60
61    let ret: u8 = (val & 0xff) as u8;
62    if rcx == 0 {
63        Ok(ret)
64    } else {
65        Err(SvsmError::Fault)
66    }
67}

69/// Writes one byte at a virtual address.
70///
71/// # Safety
72///
73/// The caller must verify not to corrupt arbitrary memory, as this function
74/// doesn't make any checks in that regard.
75///
76/// # Returns
77///
78/// Returns an error if the specified address is not mapped or is not mapped
79/// with the appropriate write permissions.
80#[inline]
81pub unsafe fn write_u8(v: VirtAddr, val: u8) -> Result<(), SvsmError> {
82    let mut rcx: u64;
83
84    // SAFETY: Assembly writes to virtual address, safe when function's safety
85    // requirements are fulfilled.
86    unsafe {
87        asm!("1: movb %al, ({0})",
88             "   xorq %rcx, %rcx",
89             "2:",
90             ".pushsection \"__exception_table\",\"a\"",
91             ".balign 16",
92             ".quad (1b)",
93             ".quad (2b)",
94             ".popsection",
95                in(reg) v.bits(),
96                in("rax") val as u64,
97                out("rcx") rcx,
98                options(att_syntax, nostack));
99    }
100
101    if rcx == 0 {
102        Ok(())
103    } else {
104        Err(SvsmError::Fault)
105    }
106}

238/// Copies `size` number of bytes from `src` to `dst`, catching any fault that
239/// might happen during the operation.
240///
241/// # Safety
242///
243/// The caller must make sure that writing to `dst` does not harm memory safety.
244#[inline]
245unsafe fn copy_bytes(src: *const u8, dst: *mut u8, size: usize) -> Result<(), SvsmError> {
246    let mut rcx: u64;
247
248    // SAFETY: Safe as long as the function's safety requirements are met. Any
249    // fault that might happen is handled via the exception handlers.
250    unsafe {
251        asm!("1:cld
252                rep movsb
253              2:
254             .pushsection \"__exception_table\",\"a\"
255             .balign 16
256             .quad (1b)
257             .quad (2b)
258             .popsection",
259                inout("rsi") src.expose_provenance() => _,
260                inout("rdi") dst.expose_provenance() => _,
261                inout("rcx") size => rcx,
262                options(att_syntax, nostack));
263    }
264
265    if rcx == 0 {
266        Ok(())
267    } else {
268        Err(SvsmError::Fault)
269    }
270}

272/// Copies `src` to `dst`.
273///
274/// # Safety
275///
276/// The caller must make sure that writing to `dst` does not harm memory safety.
277#[inline]
278unsafe fn do_movsb<T>(src: *const T, dst: *mut T) -> Result<(), SvsmError> {
279    let size: usize = size_of::<T>();
280
281    // SAFETY: Only safe when safety requirements for do_movsb() are fulfilled.
282    unsafe { copy_bytes(src.cast(), dst.cast(), size) }
283}

285#[derive(Debug)]
286pub struct GuestPtr<T: Copy> {
287    ptr: *mut T,
288}
289
290impl<T: Copy> GuestPtr<T> {
291    #[inline]
292    pub fn new(v: VirtAddr) -> Self {
293        Self {
294            ptr: v.as_mut_ptr::<T>(),
295        }
296    }
297
298    #[inline]
299    pub const fn from_ptr(p: *mut T) -> Self {
300        Self { ptr: p }
301    }
302
303    /// # Safety
304    ///
305    /// The caller must verify not to read arbitrary memory, as this function
306    /// doesn't make any checks in that regard.
307    ///
308    /// # Returns
309    ///
310    /// Returns an error if the specified address is not mapped.
311    #[inline]
312    pub unsafe fn read(&self) -> Result<T, SvsmError> {
313        let mut buf = MaybeUninit::<T>::uninit();
314
315        // SAFETY: Safe because `dst` is on the local stack. The data is
316        // explicitly uninitialized and no one else has access to it.
317        unsafe {
318            do_movsb(self.ptr, buf.as_mut_ptr())?;
319            Ok(buf.assume_init())
320        }
321    }
322
323    /// # Safety
324    ///
325    /// The caller must verify not to corrupt arbitrary memory, as this function
326    /// doesn't make any checks in that regard.
327    ///
328    /// # Returns
329    ///
330    /// Returns an error if the specified address is not mapped or is not mapped
331    /// with the appropriate write permissions.
332    #[inline]
333    pub unsafe fn write(&self, buf: T) -> Result<(), SvsmError> {
334        // SAFETY: Safe when self.ptr does not point to SVSM memory because
335        // then the write can not harm memory safety.
336        unsafe { do_movsb(&buf, self.ptr) }
337    }
338
339    /// # Safety
340    ///
341    /// The caller must verify not to corrupt arbitrary memory, as this function
342    /// doesn't make any checks in that regard.
343    ///
344    /// # Returns
345    ///
346    /// Returns an error if the specified address is not mapped or is not mapped
347    /// with the appropriate write permissions.
348    #[inline]
349    pub unsafe fn write_ref(&self, buf: &T) -> Result<(), SvsmError> {
350        // SAFETY: Safe when self.ptr does not point to SVSM memory because
351        // then the write can not harm memory safety.
352        unsafe { do_movsb(buf, self.ptr) }
353    }
354
355    #[inline]
356    pub const fn cast<N: Copy>(&self) -> GuestPtr<N> {
357        GuestPtr::from_ptr(self.ptr.cast())
358    }
359
360    #[inline]
361    pub fn offset(&self, count: isize) -> Self {
362        // SAFETY: Safe when self.ptr does not point to SVSM memory because
363        // then the write can not harm memory safety.
364        GuestPtr::from_ptr(self.ptr.wrapping_offset(count))
365    }
366}

368impl<T: Copy> From<NonNull<T>> for GuestPtr<T> {
369    fn from(value: NonNull<T>) -> Self {
370        Self::from_ptr(value.as_ptr())
371    }
372}

409#[derive(Debug)]
410pub struct UserPtr<T: Copy> {
411    guest_ptr: GuestPtr<T>,
412}
413
414impl<T: Copy> UserPtr<T> {
415    #[inline]
416    pub fn new(v: VirtAddr) -> Self {
417        Self {
418            guest_ptr: GuestPtr::new(v),
419        }
420    }
421
422    fn check_bounds(&self) -> bool {
423        let v = VirtAddr::from(self.guest_ptr.ptr);
424
425        (USER_MEM_START..USER_MEM_END).contains(&v)
426            && (USER_MEM_START..USER_MEM_END).contains(&(v + size_of::<T>()))
427    }
428
429    #[inline]
430    pub fn read(&self) -> Result<T, SvsmError>
431    where
432        T: FromBytes,
433    {
434        if !self.check_bounds() {
435            return Err(SvsmError::InvalidAddress);
436        }
437        let _guard = UserAccessGuard::new();
438        // SAFETY: Target pointer is guaranteed to point to user memory.
439        unsafe { self.guest_ptr.read() }
440    }
441
442    #[inline]
443    pub fn write(&self, buf: T) -> Result<(), SvsmError> {
444        self.write_ref(&buf)
445    }
446
447    #[inline]
448    pub fn write_ref(&self, buf: &T) -> Result<(), SvsmError> {
449        if !self.check_bounds() {
450            return Err(SvsmError::InvalidAddress);
451        }
452        let _guard = UserAccessGuard::new();
453        // SAFETY: Target pointer is guaranteed to point to user memory.
454        unsafe { self.guest_ptr.write_ref(buf) }
455    }
456
457    #[inline]
458    pub const fn cast<N: Copy>(&self) -> UserPtr<N> {
459        UserPtr {
460            guest_ptr: self.guest_ptr.cast(),
461        }
462    }
463
464    #[inline]
465    pub fn offset(&self, count: isize) -> UserPtr<T> {
466        UserPtr {
467            guest_ptr: self.guest_ptr.offset(count),
468        }
469    }
470}

472impl UserPtr<c_char> {
473    /// Reads a null-terminated C string from the user space.
474    /// Allocates memory for the string and returns a `String`.
475    pub fn read_c_string(&self) -> Result<String, SvsmError> {
476        let mut buffer = Vec::new();
477
478        for offset in 0..PATH_MAX {
479            let current_ptr = self.offset(offset as isize);
480            let char_result = current_ptr.read()?;
481            match char_result {
482                0 => return String::from_utf8(buffer).map_err(|_| SvsmError::InvalidUtf8),
483                c => buffer.push(c as u8),
484            }
485        }
486        Err(SvsmError::InvalidBytes)
487    }
488}

490fn check_bounds_user(start: usize, len: usize) -> Result<(), SvsmError> {
491    let end: usize = start.checked_add(len).ok_or(SvsmError::InvalidAddress)?;
492
493    if end > USER_MEM_END.bits() {
494        Err(SvsmError::InvalidAddress)
495    } else {
496        Ok(())
497    }
498}

500pub fn copy_from_user(src: VirtAddr, dst: &mut [u8]) -> Result<(), SvsmError> {
501    let destination = dst.as_mut_ptr();
502    let size = dst.len();
503
504    check_bounds_user(src.bits(), size)?;
505
506    // SAFETY: Safe because the copy only happens to the memory belonging to
507    // the dst slice from user-mode memory.
508    unsafe {
509        let _guard = UserAccessGuard::new();
510        let source = with_exposed_provenance::<u8>(src.bits());
511        copy_bytes(source, destination, size)
512    }
513}
514
515pub fn copy_to_user(src: &[u8], dst: VirtAddr) -> Result<(), SvsmError> {
516    let source = src.as_ptr();
517    let size = src.len();
518
519    check_bounds_user(dst.bits(), size)?;
520
521    // SAFETY: Only reads data from with the slice and copies to an address
522    // guaranteed to be in user-space.
523    unsafe {
524        let _guard = UserAccessGuard::new();
525        let destination = with_exposed_provenance_mut::<u8>(dst.bits());
526        copy_bytes(source, destination, size)
527    }
528}

530fn checked_guest_region(start: PhysAddr, size: usize) -> Result<MemoryRegion<PhysAddr>, SvsmError> {
531    let region = MemoryRegion::checked_new(start, size).ok_or(SvsmError::Mem)?;
532    if !valid_phys_region(&region) {
533        return Err(SvsmError::InvalidAddress);
534    }
535    Ok(region)
536}

538/// Reads a slice of bytes from a physical address region outside of SVSM use.
539///
540/// # Safety
541///
542/// The caller must ensure that dst..dst+size is memory that it owns.
543///
544/// # Arguments
545///
546/// * `src`: The physical address designating the start of continguous physical
547///   memory to read from.
548/// * `dst`: The pointer to the beginning of the SVSM memory range to populate.
549/// * `size`: The number of bytes to write from src to dst.
550///
551/// # Returns
552///
553/// This function returns a `Result` that indicates the success or failure of the operation.
554/// If the physical address region cannot be mapped, it returns `Err(SvsmError::Mem)`.
555/// If the physical address region cannot be read, it returns `Err(SvsmError::Fault)`.
556/// If the physical address region is not allocated to the guest, it returns
557///   `Err(SvsmError::InvalidAddress)`.
558pub unsafe fn copy_from_guest(src: PhysAddr, dst: *mut u8, size: usize) -> Result<(), SvsmError> {
559    let region = checked_guest_region(src, size)?;
560    let start = region.start().page_align();
561    // Offset will always be 0..4K, so this is infallible.
562    let offset = region.start().page_offset() as isize;
563    let end = region.end().page_align_up();
564
565    // SAFETY: Only reads data from a region outside the SVSM.
566    unsafe {
567        let guard = PerCPUPageMappingGuard::create(start, end, 0)?;
568        let source = with_exposed_provenance::<u8>(guard.virt_addr().bits()).offset(offset);
569        copy_bytes(source, dst, size)
570    }
571}

574///

576///
577/// * `src`: The physical address designating the start of continguous physical
578///   memory to read from.
579/// * `dst`: A mutable slice of SVSM memory to populate from src.

586/// If the physical address region is not allocated to the guest, it returns
587///   `Err(SvsmError::InvalidAddress)`.
588pub fn copy_slice_from_guest(src: PhysAddr, dst: &mut [u8]) -> Result<(), SvsmError> {
589    // SAFETY: The safety property of slices ensures the memory is owned and mutable.
590    unsafe { copy_from_guest(src, dst.as_mut_ptr(), dst.len()) }

593/// Writes a slice of bytes to a physical address region outside of SVSM use.
594///
595/// # Arguments
596///
597/// * `src`: The byte slice to write to guest memory.
598/// * `dst`: The physical address designating the start of continguous physical
599///   memory to write to.
600///
601/// # Returns
602///
603/// This function returns a `Result` that indicates the success or failure of the operation.
604/// If the physical address region cannot be mapped, it returns `Err(SvsmError::Mem)`.
605/// If the physical address region cannot be read, it returns `Err(SvsmError::Fault)`.
606/// If the physical address region is not allocated to the guest, it returns
607///   `Err(SvsmError::InvalidAddress)`.
608pub fn copy_slice_to_guest(src: &[u8], dst: PhysAddr) -> Result<(), SvsmError> {
609    let size = src.len();
610    let region = checked_guest_region(dst, src.len())?;
611    let start = region.start().page_align();
612    // Offset will always be 0..4K so this is infallible.
613    let offset = region.start().page_offset() as isize;
614    let end = region.end().page_align_up();
615
616    // SAFETY: Only reads data from a region outside the SVSM.
617    unsafe {
618        let guard = PerCPUPageMappingGuard::create(start, end, 0)?;
619        let destination =
620            with_exposed_provenance_mut::<u8>(guard.virt_addr().bits()).offset(offset);
621        copy_bytes(src.as_ptr(), destination, size)
622    }
623}

626///
627/// # Arguments
628///
629/// * `src`: The physical address designating the start of continguous physical
630///   memory to read from.
631/// * `size`: The length of the physical address region to read into a vector.

633/// # Returns

635/// This function returns a `Result` that indicates the success or failure of the operation.
636/// On success, returns a vector of length `size`.
637/// If the physical address region cannot be mapped, it returns `Err(SvsmError::Mem)`.

641pub fn read_bytes_from_guest(src: PhysAddr, size: usize) -> Result<Vec<u8>, SvsmError> {
642    let mut result = Vec::with_capacity(size);
643    // SAFETY: The vector's capacity, `size` has been populated by copy_from_guest.
644    // The translation of bytes to T are checked.
645    unsafe {
646        copy_from_guest(src, result.as_mut_ptr(), size)?;
647        result.set_len(size);
648    }
649    Ok(result)

652/// Reads an instance of T from a physical address region outside of SVSM use.
653///
654/// # Arguments
655///
656/// * `src`: The physical address designating the start of continguous physical
657///   memory to read from.
658///
659/// # Returns
660///
661/// This function returns a `Result` that indicates the success or failure of the operation.
662/// On success, returns an instance of T.
663/// If the physical address region cannot be mapped, it returns `Err(SvsmError::Mem)`.
664/// If the physical address region cannot be read, it returns `Err(SvsmError::Fault)`.
665/// If the physical address region is not allocated to the guest, it returns
666///   `Err(SvsmError::InvalidAddress)`.
667pub fn read_from_guest<T: KnownLayout + Sized>(src: PhysAddr) -> Result<T, SvsmError> {
668    let mut t: MaybeUninit<T> = MaybeUninit::uninit();
669    // SAFETY: copy_from_guest does not read `t`, so it's safe to take a mutable pointer.
670    // The `t` layout is known, so populating through the casted *mut u8 is safe.
671    // The size of T is allocated on the SVSM stack, so it is fully owned.
672    // copy_from_guest populates the full contents of t so it is same to assume t is initialized.
673    unsafe {
674        copy_from_guest(src, t.as_mut_ptr().cast::<u8>(), size_of::<T>())?;
675        Ok(t.assume_init())
676    }
677}

691/// If the physical address region cannot be read, it returns `Err(SvsmError::Fault)`.
692/// If the physical address region is not allocated to the guest, it returns
693///   `Err(SvsmError::InvalidAddress)`.
694#[inline]
695pub fn write_to_guest<T: IntoBytes + Immutable>(v: &T, dst: PhysAddr) -> Result<(), SvsmError> {
696    copy_slice_to_guest(v.as_bytes(), dst)
697}
