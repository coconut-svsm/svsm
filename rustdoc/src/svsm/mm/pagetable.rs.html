50/// Initializes paging settings.
51pub fn paging_init(platform: &dyn SvsmPlatform, suppress_global: bool) -> ImmutAfterInitResult<()> {
52    init_encrypt_mask(platform)?;
53
54    let mut feature_mask = PTEntryFlags::all();
55    if suppress_global {
56        feature_mask.remove(PTEntryFlags::GLOBAL);
57    }
58    FEATURE_MASK.init(feature_mask)
59}
60
61/// Initializes the encrypt mask.
62fn init_encrypt_mask(platform: &dyn SvsmPlatform) -> ImmutAfterInitResult<()> {
63    let masks = platform.get_page_encryption_masks();
64
65    PRIVATE_PTE_MASK.init(masks.private_pte_mask)?;
66    SHARED_PTE_MASK.init(masks.shared_pte_mask)?;
67
68    let guest_phys_addr_size = (masks.phys_addr_sizes >> 16) & 0xff;
69    let host_phys_addr_size = masks.phys_addr_sizes & 0xff;
70    let phys_addr_size = if guest_phys_addr_size == 0 {
71        // When [GuestPhysAddrSize] is zero, refer to the PhysAddrSize field
72        // for the maximum guest physical address size.
73        // - APM3, E.4.7 Function 8000_0008h - Processor Capacity Parameters and Extended Feature Identification
74        host_phys_addr_size
75    } else {
76        guest_phys_addr_size
77    };
78
79    PHYS_ADDR_SIZE.init(phys_addr_size)?;
80
81    // If the C-bit is a physical address bit however, the guest physical
82    // address space is effectively reduced by 1 bit.
83    // - APM2, 15.34.6 Page Table Support
84    let effective_phys_addr_size = cmp::min(masks.addr_mask_width, phys_addr_size);
85
86    let max_addr = 1 << effective_phys_addr_size;
87    MAX_PHYS_ADDR.init(max_addr)
88}

110/// Set address as shared via mask.
111fn make_shared_address(paddr: PhysAddr) -> PhysAddr {
112    (strip_confidentiality_bits(paddr).bits() | shared_pte_mask()).into()
113}
114
115/// Set address as private via mask.
116fn make_private_address(paddr: PhysAddr) -> PhysAddr {
117    (strip_shared_address_bits(paddr).bits() | private_pte_mask()).into()
118}

120// Returns true if the address is shared.
121fn is_shared(paddr: PhysAddr) -> bool {
122    paddr == make_shared_address(paddr)
123}

125fn strip_confidentiality_bits(paddr: PhysAddr) -> PhysAddr {
126    (paddr.bits() & !private_pte_mask()).into()
127}

130    (paddr.bits() & !shared_pte_mask()).into()

173/// Represents paging mode.
174#[derive(Clone, Copy, Debug, Eq, PartialEq)]
175pub enum PagingMode {
176    // Paging mode is disabled
177    NoPaging,
178    // 32bit legacy paging mode
179    NonPAE,
180    // 32bit PAE paging mode
181    PAE,
182    // 4 level paging mode
183    PML4,
184    // 5 level paging mode
185    PML5,
186}
187
188impl PagingMode {
189    pub fn new(efer: EFERFlags, cr0: CR0Flags, cr4: CR4Flags) -> Self {
190        if !cr0.contains(CR0Flags::PG) {
191            // Paging is disabled
192            PagingMode::NoPaging
193        } else if efer.contains(EFERFlags::LMA) {
194            // Long mode is activated
195            if cr4.contains(CR4Flags::LA57) {
196                PagingMode::PML5
197            } else {
198                PagingMode::PML4
199            }
200        } else if cr4.contains(CR4Flags::PAE) {
201            // PAE mode
202            PagingMode::PAE
203        } else {
204            // Non PAE mode
205            PagingMode::NonPAE
206        }
207    }
208}

210/// Represents a page table entry.
211#[repr(C)]
212#[derive(Copy, Clone, Debug, FromZeros)]
213pub struct PTEntry(PhysAddr);
214
215impl PTEntry {
216    /// Check if the page table entry is clear (null).
217    pub fn is_clear(&self) -> bool {
218        self.0.is_null()
219    }
220
221    /// Clear the page table entry.
222    pub fn clear(&mut self) {
223        self.0 = PhysAddr::null();
224    }
225
226    /// Check if the page table entry is present.
227    pub fn present(&self) -> bool {
228        self.flags().contains(PTEntryFlags::PRESENT)
229    }
230
231    /// Check if the page table entry is huge.
232    pub fn huge(&self) -> bool {
233        self.flags().contains(PTEntryFlags::HUGE)
234    }
235
236    /// Check if the page table entry is writable.
237    pub fn writable(&self) -> bool {
238        self.flags().contains(PTEntryFlags::WRITABLE)
239    }
240
241    /// Check if the page table entry is NX (no-execute).
242    pub fn nx(&self) -> bool {
243        self.flags().contains(PTEntryFlags::NX)
244    }
245
246    /// Check if the page table entry is user-accessible.
247    pub fn user(&self) -> bool {
248        self.flags().contains(PTEntryFlags::USER)
249    }
250
251    /// Check if the page table entry is global.
252    pub fn global(&self) -> bool {
253        self.flags().contains(PTEntryFlags::GLOBAL)
254    }
255
256    /// Check if the page table entry has reserved bits set.
257    pub fn has_reserved_bits(&self, pm: PagingMode, level: usize) -> bool {
258        let reserved_mask = match pm {
259            PagingMode::NoPaging => unreachable!("NoPaging does not have page table"),
260            PagingMode::NonPAE => {
261                match level {
262                    // No reserved bits in 4k PTE.
263                    0 => 0,
264                    1 => {
265                        if self.huge() {
266                            // Bit21 is reserved in 4M PDE.
267                            BIT_MASK!(21, 21)
268                        } else {
269                            // No reserved bits in PDE.
270                            0
271                        }
272                    }
273                    _ => unreachable!("Invalid NonPAE page table level"),
274                }
275            }
276            PagingMode::PAE => {
277                // Bit62 ~ MAXPHYSADDR are reserved for each
278                // level in PAE page table.
279                BIT_MASK!(62, *PHYS_ADDR_SIZE)
280                    | match level {
281                        // No additional reserved bits in 4k PTE.
282                        0 => 0,
283                        1 => {
284                            if self.huge() {
285                                // Bit20 ~ Bit13 are reserved in 2M PDE.
286                                BIT_MASK!(20, 13)
287                            } else {
288                                // No additional reserved bits in PDE.
289                                0
290                            }
291                        }
292                        // Bit63 and Bit8 ~ Bit5 are reserved in PDPTE.
293                        2 => BIT_MASK!(63, 63) | BIT_MASK!(8, 5),
294                        _ => unreachable!("Invalid PAE page table level"),
295                    }
296            }
297            PagingMode::PML4 | PagingMode::PML5 => {
298                // Bit51 ~ MAXPHYSADDR are reserved for each level
299                // in PML4 and PML5 page table.
300                let common = if *PHYS_ADDR_SIZE > 51 {
301                    0
302                } else {
303                    // Remove the encryption mask bit as this bit is not reserved
304                    BIT_MASK!(51, *PHYS_ADDR_SIZE)
305                        & !((shared_pte_mask() | private_pte_mask()) as u64)
306                };
307
308                common
309                    | match level {
310                        // No additional reserved bits in 4k PTE.
311                        0 => 0,
312                        1 => {
313                            if self.huge() {
314                                // Bit20 ~ Bit13 are reserved in 2M PDE.
315                                BIT_MASK!(20, 13)
316                            } else {
317                                // No additional reserved bits in PDE.
318                                0
319                            }
320                        }
321                        2 => {
322                            if self.huge() {
323                                // Bit29 ~ Bit13 are reserved in 1G PDPTE.
324                                BIT_MASK!(29, 13)
325                            } else {
326                                // No additional reserved bits in PDPTE.
327                                0
328                            }
329                        }
330                        // Bit8 ~ Bit7 are reserved in PML4E.
331                        3 => BIT_MASK!(8, 7),
332                        4 => {
333                            if pm == PagingMode::PML4 {
334                                unreachable!("Invalid PML4 page table level");
335                            } else {
336                                // Bit8 ~ Bit7 are reserved in PML5E.
337                                BIT_MASK!(8, 7)
338                            }
339                        }
340                        _ => unreachable!("Invalid PML4/PML5 page table level"),
341                    }
342            }
343        };
344
345        self.raw() & reserved_mask != 0
346    }
347
348    /// Get the raw bits (`u64`) of the page table entry.
349    pub fn raw(&self) -> u64 {
350        self.0.bits() as u64
351    }
352
353    /// Get the flags of the page table entry.
354    pub fn flags(&self) -> PTEntryFlags {
355        PTEntryFlags::from_bits_truncate(self.0.bits() as u64)
356    }
357
358    /// Set the page table entry with the specified address and flags.
359    pub fn set(&mut self, addr: PhysAddr, flags: PTEntryFlags) {
360        let addr = addr.bits() as u64;
361        assert_eq!(addr & !0x000f_ffff_ffff_f000, 0);
362        self.0 = PhysAddr::from(addr | supported_flags(flags).bits());
363    }
364
365    /// Get the address from the page table entry, including the shared bit.
366    pub fn page_frame(&self) -> PhysAddr {
367        let addr = PhysAddr::from(self.0.bits() & 0x000f_ffff_ffff_f000);
368        strip_confidentiality_bits(addr)
369    }
370
371    /// Get the address from the page table entry, excluding the C/shared bit.
372    pub fn address(&self) -> PhysAddr {
373        strip_shared_address_bits(self.page_frame())
374    }
375
376    /// Read a page table entry from the specified virtual address.
377    ///
378    /// # Safety
379    ///
380    /// Reads from an arbitrary virtual address, making this essentially a
381    /// raw pointer read.  The caller must be certain to calculate the correct
382    /// address.
383    pub unsafe fn read_pte(vaddr: VirtAddr) -> Self {
384        // SAFETY: When the methods safety requirements are met, the raw
385        // pointer read is safe.
386        unsafe { *vaddr.as_ptr::<Self>() }
387    }
388}

390/// A pagetable page with multiple entries.
391#[repr(C)]
392#[derive(Debug, FromZeros)]
393pub struct PTPage {
394    entries: [PTEntry; ENTRY_COUNT],
395}
396
397impl PTPage {
398    /// Allocates a zeroed pagetable page and returns a mutable reference to
399    /// it, plus its physical address.
400    ///
401    /// # Errors
402    ///
403    /// Returns [`SvsmError`] if the page cannot be allocated.
404    fn alloc() -> Result<(&'static mut Self, PhysAddr), SvsmError> {
405        let page: PageBox<Self> = PageBox::try_new_zeroed()?;
406        let paddr = virt_to_phys(page.vaddr());
407        Ok((PageBox::leak(page), paddr))
408    }
409
410    /// Frees a pagetable page.
411    ///
412    /// # Safety
413    ///
414    /// The given reference must correspond to a valid previously allocated
415    /// page table page.
416    unsafe fn free(page: &'static Self) {
417        // SAFETY: The page put into the PageBox is a previously allocated
418        // page table page.
419        unsafe {
420            let _ = PageBox::from_raw(NonNull::from(page));
421        }
422    }
423
424    /// Converts a pagetable entry to a mutable reference to a [`PTPage`],
425    /// if the entry is present and not huge.
426    fn from_entry(entry: PTEntry) -> Option<&'static mut Self> {
427        let flags = entry.flags();
428        if !flags.contains(PTEntryFlags::PRESENT) || flags.contains(PTEntryFlags::HUGE) {
429            return None;
430        }
431
432        let address = phys_to_virt(entry.address());
433        // SAFETY: Every PTEntry points to a previously allocated page-table
434        // page, so this pointer dereference is safe.
435        Some(unsafe { &mut *address.as_mut_ptr::<PTPage>() })
436    }
437}

464/// A physical address within a page frame
465#[derive(Clone, Copy, Debug)]
466pub enum PageFrame {
467    Size4K(PhysAddr),
468    Size2M(PhysAddr),
469    Size1G(PhysAddr),
470}
471
472impl PageFrame {
473    /// Get the address from the page frame, including the shared bit.
474    pub fn page_frame(&self) -> PhysAddr {
475        let paddr = match *self {
476            Self::Size4K(pa) => pa,
477            Self::Size2M(pa) => pa,
478            Self::Size1G(pa) => pa,
479        };
480        strip_confidentiality_bits(paddr)
481    }
482
483    /// Get the address from the page frame, excluding the C/shared bit.
484    pub fn address(&self) -> PhysAddr {
485        strip_shared_address_bits(self.page_frame())
486    }
487
488    pub fn size(&self) -> usize {
489        match self {
490            Self::Size4K(_) => PAGE_SIZE,
491            Self::Size2M(_) => PAGE_SIZE_2M,
492            Self::Size1G(_) => PAGE_SIZE_1G,
493        }
494    }
495
496    pub fn start(&self) -> PhysAddr {
497        let end = self.address().bits() & !(self.size() - 1);
498        end.into()
499    }
500
501    pub fn end(&self) -> PhysAddr {
502        self.start() + self.size()
503    }
504}

506/// Page table structure containing a root page with multiple entries.
507#[repr(C)]
508#[derive(Debug, FromZeros)]
509pub struct PageTable {
510    root: PTPage,
511}
512
513impl PageTable {
514    /// Load the current page table into the CR3 register.
515    ///
516    /// # Safety
517    ///
518    /// The caller must ensure to take other actions to make sure a memory safe
519    /// execution state is warranted (e.g. changing the stack and register state)
520    pub unsafe fn load(&self) {
521        // SAFETY: demanded to the caller
522        unsafe {
523            write_cr3(self.cr3_value());
524        }
525    }
526
527    /// Get the CR3 register value for the current page table.
528    pub fn cr3_value(&self) -> PhysAddr {
529        let pgtable = VirtAddr::from(self as *const Self);
530        virt_to_phys(pgtable)
531    }
532
533    /// Allocate a new page table root.
534    ///
535    /// # Errors
536    /// Returns [`SvsmError`] if the page cannot be allocated.
537    pub fn allocate_new() -> Result<PageBox<Self>, SvsmError> {
538        let mut pgtable: PageBox<Self> = PageBox::try_new_zeroed()?;
539        let paddr = virt_to_phys(pgtable.vaddr());
540
541        // Set the self-map entry.
542        let entry = &mut pgtable.root[PGTABLE_LVL3_IDX_PTE_SELFMAP];
543        let flags = PTEntryFlags::PRESENT
544            | PTEntryFlags::WRITABLE
545            | PTEntryFlags::ACCESSED
546            | PTEntryFlags::DIRTY
547            | PTEntryFlags::NX;
548        entry.set(make_private_address(paddr), flags);
549
550        Ok(pgtable)
551    }
552
553    /// Clone the shared part of the page table; excluding the private
554    /// parts.
555    ///
556    /// # Errors
557    /// Returns [`SvsmError`] if the page cannot be allocated.
558    pub fn clone_shared(&self) -> Result<PageBox<PageTable>, SvsmError> {
559        let mut pgtable = Self::allocate_new()?;
560        pgtable.root.entries[PGTABLE_LVL3_IDX_SHARED] = self.root.entries[PGTABLE_LVL3_IDX_SHARED];
561        Ok(pgtable)
562    }
563
564    /// Copy an entry `entry` from another [`PageTable`].
565    pub fn copy_entry(&mut self, other: &Self, entry: usize) {
566        self.root.entries[entry] = other.root.entries[entry];
567    }
568
569    /// Computes the index within a page table at the given level for a
570    /// virtual address `vaddr`.
571    ///
572    /// # Parameters
573    /// - `vaddr`: The virtual address to compute the index for.
574    ///
575    /// # Returns
576    /// The index within the page table.
577    pub fn index<const L: usize>(vaddr: VirtAddr) -> usize {
578        vaddr.to_pgtbl_idx::<L>()
579        //vaddr.bits() >> (12 + L * 9) & 0x1ff
580    }
581
582    /// Walks a page table at level 0 to find a mapping.
583    ///
584    /// # Parameters
585    /// - `page`: A mutable reference to the root page table.
586    /// - `vaddr`: The virtual address to find a mapping for.
587    ///
588    /// # Returns
589    /// A `Mapping` representing the found mapping.
590    fn walk_addr_lvl0(page: &mut PTPage, vaddr: VirtAddr) -> Mapping<'_> {
591        let idx = Self::index::<0>(vaddr);
592        Mapping::Level0(&mut page[idx])
593    }
594
595    /// Walks a page table at level 1 to find a mapping.
596    ///
597    /// # Parameters
598    /// - `page`: A mutable reference to the root page table.
599    /// - `vaddr`: The virtual address to find a mapping for.
600    ///
601    /// # Returns
602    /// A `Mapping` representing the found mapping.
603    fn walk_addr_lvl1(page: &mut PTPage, vaddr: VirtAddr) -> Mapping<'_> {
604        let idx = Self::index::<1>(vaddr);
605        let entry = page[idx];
606        match PTPage::from_entry(entry) {
607            Some(page) => Self::walk_addr_lvl0(page, vaddr),
608            None => Mapping::Level1(&mut page[idx]),
609        }
610    }
611
612    /// Walks a page table at level 2 to find a mapping.
613    ///
614    /// # Parameters
615    /// - `page`: A mutable reference to the root page table.
616    /// - `vaddr`: The virtual address to find a mapping for.
617    ///
618    /// # Returns
619    /// A `Mapping` representing the found mapping.
620    fn walk_addr_lvl2(page: &mut PTPage, vaddr: VirtAddr) -> Mapping<'_> {
621        let idx = Self::index::<2>(vaddr);
622        let entry = page[idx];
623        match PTPage::from_entry(entry) {
624            Some(page) => Self::walk_addr_lvl1(page, vaddr),
625            None => Mapping::Level2(&mut page[idx]),
626        }
627    }
628
629    /// Walks the page table to find a mapping for a given virtual address.
630    ///
631    /// # Parameters
632    /// - `page`: A mutable reference to the root page table.
633    /// - `vaddr`: The virtual address to find a mapping for.
634    ///
635    /// # Returns
636    /// A `Mapping` representing the found mapping.
637    fn walk_addr_lvl3(page: &mut PTPage, vaddr: VirtAddr) -> Mapping<'_> {
638        let idx = Self::index::<3>(vaddr);
639        let entry = page[idx];
640        match PTPage::from_entry(entry) {
641            Some(page) => Self::walk_addr_lvl2(page, vaddr),
642            None => Mapping::Level3(&mut page[idx]),
643        }
644    }
645
646    /// Walk the virtual address and return the corresponding mapping.
647    ///
648    /// # Parameters
649    /// - `vaddr`: The virtual address to find a mapping for.
650    ///
651    /// # Returns
652    /// A `Mapping` representing the found mapping.
653    fn walk_addr(&mut self, vaddr: VirtAddr) -> Mapping<'_> {
654        Self::walk_addr_lvl3(&mut self.root, vaddr)
655    }
656
657    /// Calculate the virtual address of a PTE in the self-map, which maps a
658    /// specified virtual address.
659    ///
660    /// # Parameters
661    /// - `vaddr': The virtual address whose PTE should be located.
662    ///
663    /// # Returns
664    /// The virtual address of the PTE.
665    fn get_pte_address(vaddr: VirtAddr) -> VirtAddr {
666        SVSM_PTE_BASE + ((usize::from(vaddr) & 0x0000_FFFF_FFFF_F000) >> 9)
667    }
668
669    /// Perform a virtual to physical translation using the self-map.
670    ///
671    /// # Parameters
672    /// - `vaddr': The virtual address to translate.
673    ///
674    /// # Returns
675    /// Some(PageFrame) if the virtual address is valid.
676    /// None if the virtual address is not valid.
677    pub fn virt_to_frame(vaddr: VirtAddr) -> Option<PageFrame> {
678        // Calculate the virtual addresses of each level of the paging
679        // hierarchy in the self-map.
680        let pte_addr = Self::get_pte_address(vaddr);
681        let pde_addr = Self::get_pte_address(pte_addr);
682        let pdpe_addr = Self::get_pte_address(pde_addr);
683        let pml4e_addr = Self::get_pte_address(pdpe_addr);
684
685        // SAFETY: Check each entry in the paging hierarchy to determine
686        // whether this address is mapped.  Because the hierarchy is read from
687        // the top down using self-map addresses that were calculated
688        // correctly, the reads are safe to perform.
689        let pml4e = unsafe { PTEntry::read_pte(pml4e_addr) };
690        if !pml4e.present() {
691            return None;
692        }
693
694        // There is no need to check for a large page in the PML4E because
695        // the architecture does not support the large bit at the top-level
696        // entry.  If a large page is detected at a lower level of the
697        // hierarchy, the low bits from the virtual address must be combined
698        // with the physical address from the PDE/PDPE.
699
700        // SAFETY: The PML4E was checked to be present, so the PDPE exists and
701        // can be read safely.
702        let pdpe = unsafe { PTEntry::read_pte(pdpe_addr) };
703        if !pdpe.present() {
704            return None;
705        }
706        if pdpe.huge() {
707            let pa = pdpe.page_frame() + (usize::from(vaddr) & 0x3FFF_FFFF);
708            return Some(PageFrame::Size1G(pa));
709        }
710
711        // SAFETY: The PDPE was checked to be present and not to be a huge
712        // page. So the PDE exists and can be read safely.
713        let pde = unsafe { PTEntry::read_pte(pde_addr) };
714        if !pde.present() {
715            return None;
716        }
717        if pde.huge() {
718            let pa = pde.page_frame() + (usize::from(vaddr) & 0x001F_FFFF);
719            return Some(PageFrame::Size2M(pa));
720        }
721
722        // SAFETY: The PDE was checked to be present and not to be a huge
723        // page. So the PTE exists and can be read safely.
724        let pte = unsafe { PTEntry::read_pte(pte_addr) };
725        if pte.present() {
726            let pa = pte.page_frame() + (usize::from(vaddr) & 0xFFF);
727            Some(PageFrame::Size4K(pa))
728        } else {
729            None
730        }
731    }

832    /// Splits a 2MB page into 4KB pages.
833    ///
834    /// # Parameters
835    /// - `entry`: The 2M page table entry to split.
836    ///
837    /// # Returns
838    /// A result indicating success or an error [`SvsmError`] in failure.
839    fn do_split_4k(entry: &mut PTEntry) -> Result<(), SvsmError> {
840        let (page, paddr) = PTPage::alloc()?;
841        let mut flags = entry.flags();
842
843        assert!(flags.contains(PTEntryFlags::HUGE));
844
845        let addr_2m = PhysAddr::from(entry.address().bits() & 0x000f_ffff_fff0_0000);
846
847        flags.remove(PTEntryFlags::HUGE);
848
849        // Prepare PTE leaf page
850        for (i, e) in page.entries.iter_mut().enumerate() {
851            let addr_4k = addr_2m + (i * PAGE_SIZE);
852            e.clear();
853            e.set(make_private_address(addr_4k), flags);
854        }
855
856        entry.set(make_private_address(paddr), flags);
857
858        flush_tlb_global_sync();
859
860        Ok(())
861    }

1145    /// Maps a memory region to physical memory with specified flags.
1146    ///
1147    /// # Parameters
1148    /// - `region`: The virtual memory region to map.
1149    /// - `phys`: The starting physical address to map to.
1150    /// - `flags`: The flags to apply to the page table entries.
1151    ///
1152    /// # Returns
1153    /// A result indicating success (`Ok`) or failure (`Err`).
1154    pub fn map_region(
1155        &mut self,
1156        region: MemoryRegion<VirtAddr>,
1157        phys: PhysAddr,
1158        flags: PTEntryFlags,
1159    ) -> Result<(), SvsmError> {
1160        let mut vaddr = region.start();
1161        let end = region.end();
1162        let mut paddr = phys;
1163
1164        while vaddr < end {
1165            if vaddr.is_aligned(PAGE_SIZE_2M)
1166                && paddr.is_aligned(PAGE_SIZE_2M)
1167                && vaddr + PAGE_SIZE_2M <= end
1168                && self.map_2m(vaddr, paddr, flags).is_ok()
1169            {
1170                vaddr = vaddr + PAGE_SIZE_2M;
1171                paddr = paddr + PAGE_SIZE_2M;
1172                continue;
1173            }
1174
1175            self.map_4k(vaddr, paddr, flags)?;
1176            vaddr = vaddr + PAGE_SIZE;
1177            paddr = paddr + PAGE_SIZE;
1178        }
1179
1180        Ok(())
1181    }

1223    ///

1317        PageTable::walk_addr_lvl2(&mut self.page, vaddr)
1318    }

1320    /// Allocates a 4KB page table entry for a given virtual address.
1321    ///
1322    /// # Parameters
1323    /// - `vaddr`: The virtual address for which to allocate the PTE.
1324    ///
1325    /// # Returns
1326    /// The [`Mapping`] representing the allocated or existing PTE for the address.
1327    ///
1328    /// # Panics
1329    /// Panics if a level 3 mapping is attempted in a [`RawPageTablePart`].
1330    fn alloc_pte_4k(&mut self, vaddr: VirtAddr) -> Mapping<'_> {
1331        let m = self.walk_addr(vaddr);
1332
1333        match m {
1334            Mapping::Level0(entry) => Mapping::Level0(entry),
1335            Mapping::Level1(entry) => PageTable::alloc_pte_lvl1(entry, vaddr, PageSize::Regular),
1336            Mapping::Level2(entry) => PageTable::alloc_pte_lvl2(entry, vaddr, PageSize::Regular),
1337            Mapping::Level3(_) => panic!("PT level 3 not possible in PageTablePart"),
1338        }
1339    }

1360    /// Maps a 4KB page.
1361    ///
1362    /// # Parameters
1363    /// - `vaddr`: The virtual address to map.
1364    /// - `paddr`: The physical address to map to.
1365    /// - `flags`: The flags to apply to the mapping.
1366    /// - `shared`: Indicates whether the mapping is shared.
1367    ///
1368    /// # Returns
1369    /// A result indicating success (`Ok`) or failure (`Err`).
1370    fn map_4k(
1371        &mut self,
1372        vaddr: VirtAddr,
1373        paddr: PhysAddr,
1374        flags: PTEntryFlags,
1375        shared: bool,
1376    ) -> Result<(), SvsmError> {
1377        let mapping = self.alloc_pte_4k(vaddr);
1378
1379        let addr = if !shared {
1380            make_private_address(paddr)
1381        } else {
1382            make_shared_address(paddr)
1383        };
1384
1385        if let Mapping::Level0(entry) = mapping {
1386            entry.set(addr, flags);
1387            Ok(())
1388        } else {
1389            Err(SvsmError::Mem)
1390        }
1391    }

1393    /// Unmaps a 4KB page.
1394    ///
1395    /// # Parameters
1396    /// - `vaddr`: The virtual address of the mapping to unmap.
1397    ///
1398    /// # Returns
1399    /// An optional [`PTEntry`] representing the unmapped page table entry.
1400    fn unmap_4k(&mut self, vaddr: VirtAddr) -> Option<PTEntry> {
1401        let mapping = self.walk_addr(vaddr);
1402
1403        match mapping {
1404            Mapping::Level0(entry) => {
1405                let e = *entry;
1406                entry.clear();
1407                Some(e)
1408            }
1409            Mapping::Level1(entry) => {
1410                assert!(!entry.present());
1411                None
1412            }
1413            Mapping::Level2(entry) => {
1414                assert!(!entry.present());
1415                None
1416            }
1417            Mapping::Level3(entry) => {
1418                assert!(!entry.present());
1419                None
1420            }
1421        }
1422    }

1424    /// Maps a 2MB page.
1425    ///
1426    /// # Parameters
1427    /// - `vaddr`: The virtual address to map.
1428    /// - `paddr`: The physical address to map to.
1429    /// - `flags`: The flags to apply to the mapping.
1430    /// - `shared`: Indicates whether the mapping is shared
1431    ///
1432    /// # Returns
1433    /// A result indicating success (`Ok`) or failure (`Err`).
1434    ///
1435    /// # Panics
1436    ///
1437    /// Panics if `vaddr` or `paddr` are not 2MB-aligned
1438    fn map_2m(
1439        &mut self,
1440        vaddr: VirtAddr,
1441        paddr: PhysAddr,
1442        flags: PTEntryFlags,
1443        shared: bool,
1444    ) -> Result<(), SvsmError> {
1445        assert!(vaddr.is_aligned(PAGE_SIZE_2M));
1446        assert!(paddr.is_aligned(PAGE_SIZE_2M));
1447
1448        let mapping = self.alloc_pte_2m(vaddr);
1449        let addr = if !shared {
1450            make_private_address(paddr)
1451        } else {
1452            make_shared_address(paddr)
1453        };
1454
1455        if let Mapping::Level1(entry) = mapping {
1456            entry.set(addr, flags | PTEntryFlags::HUGE);
1457            Ok(())
1458        } else {
1459            Err(SvsmError::Mem)
1460        }
1461    }
1462
1463    /// Unmaps a 2MB page.
1464    ///
1465    /// # Parameters
1466    /// - `vaddr`: The virtual address of the mapping to unmap.
1467    ///
1468    /// # Returns
1469    /// An optional [`PTEntry`] representing the unmapped page table entry.
1470    ///
1471    /// # Panics
1472    ///
1473    /// Panics if `vaddr` is not memory aligned.
1474    fn unmap_2m(&mut self, vaddr: VirtAddr) -> Option<PTEntry> {
1475        assert!(vaddr.is_aligned(PAGE_SIZE_2M));
1476
1477        let mapping = self.walk_addr(vaddr);
1478
1479        match mapping {
1480            Mapping::Level0(_) => None,
1481            Mapping::Level1(entry) => {
1482                entry.clear();
1483                Some(*entry)
1484            }
1485            Mapping::Level2(entry) => {
1486                assert!(!entry.present());
1487                None
1488            }
1489            Mapping::Level3(entry) => {
1490                assert!(!entry.present());
1491                None
1492            }
1493        }
1494    }

1666bitflags! {
1667    /// Flags to represent how memory is accessed, e.g. write data to the
1668    /// memory or fetch code from the memory.
1669    #[derive(Clone, Copy, Debug)]
1670    pub struct MemAccessMode: u32 {
1671        const WRITE     = 1 << 0;
1672        const FETCH     = 1 << 1;
1673    }
1674}
1675
1676/// Attributes to determin Whether a memory access (write/fetch) is permitted
1677/// by a translation which includes the paging-mode modifiers in CR0, CR4 and
1678/// EFER; EFLAGS.AC; and the supervisor/user mode access.
1679#[derive(Clone, Copy, Debug)]
1680pub struct PTWalkAttr {
1681    cr0: CR0Flags,
1682    cr4: CR4Flags,
1683    efer: EFERFlags,
1684    flags: RFlags,
1685    user_mode_access: bool,
1686    pm: PagingMode,
1687}
1688
1689impl PTWalkAttr {
1690    /// Creates a new `PTWalkAttr` instance with the specified attributes.
1691    ///
1692    /// # Arguments
1693    ///
1694    /// * `cr0`, `cr4`, and `efer` - Represent the control register
1695    ///   flags for CR0, CR4, and EFER respectively.
1696    /// * `flags` - Represents the CPU Flags.
1697    /// * `user_mode_access` - Indicates whether the access is in user mode.
1698    ///
1699    /// Returns a new `PTWalkAttr` instance.
1700    pub fn new(
1701        cr0: CR0Flags,
1702        cr4: CR4Flags,
1703        efer: EFERFlags,
1704        flags: RFlags,
1705        user_mode_access: bool,
1706    ) -> Self {
1707        Self {
1708            cr0,
1709            cr4,
1710            efer,
1711            flags,
1712            user_mode_access,
1713            pm: PagingMode::new(efer, cr0, cr4),
1714        }
1715    }
1716
1717    /// Checks the access rights for a page table entry.
1718    ///
1719    /// # Arguments
1720    ///
1721    /// * `entry` - The page table entry to check.
1722    /// * `mem_am` - Indicates how to access the memory.
1723    /// * `last_level` - Indicates whether the entry is at the last level
1724    ///   of the page table.
1725    /// * `pteflags` - The PTE flags to indicate if the corresponding page
1726    ///   table entry allows the access rights.
1727    ///
1728    /// # Returns
1729    ///
1730    /// Returns `Ok((entry, leaf))` if the access rights are valid, where
1731    /// `entry` is the modified page table entry and `leaf` is a boolean
1732    /// indicating whether the entry is a leaf node, or `Err(PageFaultError)`
1733    /// to indicate the page fault error code if the access rights are invalid.
1734    pub fn check_access_rights(
1735        &self,
1736        entry: PTEntry,
1737        mem_am: MemAccessMode,
1738        level: usize,
1739        pteflags: &mut PTEntryFlags,
1740    ) -> Result<(PTEntry, bool), PageFaultError> {
1741        let pf_err = self.default_pf_err(mem_am) | PageFaultError::P;
1742
1743        if !entry.present() {
1744            // Entry is not present.
1745            return Err(pf_err & !PageFaultError::P);
1746        }
1747
1748        if entry.has_reserved_bits(self.pm, level) {
1749            // Reserved bits have been set.
1750            return Err(pf_err | PageFaultError::R);
1751        }
1752
1753        // SDM 4.6.1 Determination of Access Rights:
1754        // If the U/S flag (bit 2) is 0 in at least one of the
1755        // paging-structure entries, the address is a supervisor-mode
1756        // address. Otherwise, the address is a user-mode address.
1757        // So by-default assume the address is user mode address.
1758        if !entry.user() {
1759            *pteflags &= !PTEntryFlags::USER;
1760        }
1761
1762        // SDM 4.6.1 Determination of Access Rights:
1763        // R/W flag (bit 1) is 1 in every paging-structure entry controlling
1764        // the translation and with a protection key for which write access is
1765        // permitted; data may not be written to any supervisor-mode
1766        // address with a translation for which the R/W flag is 0 in any
1767        // paging-structure entry controlling the translation.
1768        // The same for user mode address
1769        if !entry.writable() {
1770            *pteflags &= !PTEntryFlags::WRITABLE;
1771        }
1772
1773        // SDM 4.6.1 Determination of Access Rights:
1774        // For non 32-bit paging modes with IA32_EFER.NXE = 1, instructions
1775        // may be fetched from any supervisormode address with a translation
1776        // for which the XD flag (bit 63) is 0 in every paging-structure entry
1777        // controlling the translation; instructions may not be fetched from
1778        // any supervisor-mode address with a translation for which the XD flag
1779        // is 1 in any paging-structure entry controlling the translation
1780        if self.efer.contains(EFERFlags::NXE) && entry.nx() {
1781            *pteflags |= PTEntryFlags::NX;
1782        } else if !self.efer.contains(EFERFlags::NXE) && entry.nx() {
1783            // XD bit must be 0 if efer.NXE = 0
1784            return Err(pf_err | PageFaultError::R);
1785        }
1786
1787        let leaf = if level == 0 || entry.huge() {
1788            // User mode cannot access any supervisor mode addresses
1789            if self.user_mode_access && !pteflags.contains(PTEntryFlags::USER) {
1790                return Err(pf_err);
1791            }
1792
1793            // Always check for reading. For the case of supervisor mode read user
1794            // mode addresses, do special checking. For other cases, read is allowed.
1795            if !self.user_mode_access && pteflags.contains(PTEntryFlags::USER) {
1796                // Read not allowed with SMAP = 1 && flags.ac = 0
1797                if self.cr4.contains(CR4Flags::SMAP) && !self.flags.contains(RFlags::AC) {
1798                    return Err(pf_err);
1799                }
1800            }
1801
1802            if mem_am.contains(MemAccessMode::WRITE) {
1803                if !self.user_mode_access && pteflags.contains(PTEntryFlags::USER) {
1804                    // Check supervisor mode write user mode addresses
1805                    if !self.cr0.contains(CR0Flags::WP) {
1806                        // Check write with CR0.WP = 0
1807                        if self.cr4.contains(CR4Flags::SMAP) && !self.flags.contains(RFlags::AC) {
1808                            // Write not allowed with SMAP = 1 && flags.ac = 0
1809                            return Err(pf_err);
1810                        }
1811                    } else {
1812                        // Check write with CR0.WP = 1
1813                        if !self.cr4.contains(CR4Flags::SMAP) {
1814                            // SMAP = 0
1815                            if !pteflags.contains(PTEntryFlags::WRITABLE) {
1816                                // Write not allowed R/W = 0
1817                                return Err(pf_err);
1818                            }
1819                        } else {
1820                            // SMAP = 1
1821                            if !self.flags.contains(RFlags::AC)
1822                                || !pteflags.contains(PTEntryFlags::WRITABLE)
1823                            {
1824                                // Write not allowed with flags.AC = 0 || R/W = 0
1825                                return Err(pf_err);
1826                            }
1827                        }
1828                    }
1829                } else if !self.user_mode_access && !pteflags.contains(PTEntryFlags::USER) {
1830                    // Check supervisor mode write supervisor mode addresses
1831                    if self.cr0.contains(CR0Flags::WP) && !pteflags.contains(PTEntryFlags::WRITABLE)
1832                    {
1833                        // Write not allowed with CR0.WP = 1 && R/W = 0
1834                        return Err(pf_err);
1835                    }
1836                } else if self.user_mode_access && pteflags.contains(PTEntryFlags::USER) {
1837                    // Check user mode write user mode addresses
1838                    if !pteflags.contains(PTEntryFlags::WRITABLE) {
1839                        // Write not allowed R/W = 0
1840                        return Err(pf_err);
1841                    }
1842                }
1843                // User mode write supervisor mode addresses is checked already
1844            }
1845
1846            if mem_am.contains(MemAccessMode::FETCH) {
1847                // For instruction fetch, the rule is the same except for the case of
1848                // supervisor mode fetch user mode addresses
1849                if !self.user_mode_access && pteflags.contains(PTEntryFlags::USER) {
1850                    // Fetch not allowed with SMEP = 1
1851                    if self.cr4.contains(CR4Flags::SMEP) {
1852                        return Err(pf_err);
1853                    }
1854                }
1855
1856                // For non-32bit paging mode, fetch not allowed with efer.NXE = 1 && XD = 1
1857                if self.cr4.contains(CR4Flags::PAE)
1858                    && self.efer.contains(EFERFlags::NXE)
1859                    && pteflags.contains(PTEntryFlags::NX)
1860                {
1861                    return Err(pf_err);
1862                }
1863            }
1864            true
1865        } else {
1866            false
1867        };
1868
1869        Ok((entry, leaf))
1870    }
1871
1872    fn default_pf_err(&self, mem_am: MemAccessMode) -> PageFaultError {
1873        let mut err = PageFaultError::empty();
1874
1875        if mem_am.contains(MemAccessMode::WRITE) {
1876            err |= PageFaultError::W;
1877        }
1878
1879        if mem_am.contains(MemAccessMode::FETCH) {
1880            err |= PageFaultError::I;
1881        }
1882
1883        if self.user_mode_access {
1884            err |= PageFaultError::U;
1885        }
1886
1887        err
1888    }
