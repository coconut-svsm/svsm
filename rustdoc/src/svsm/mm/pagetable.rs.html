51/// Initializes paging settings.
52pub fn paging_init(platform: &dyn SvsmPlatform, suppress_global: bool) -> ImmutAfterInitResult<()> {
53    init_encrypt_mask(platform)?;
54
55    let mut feature_mask = PTEntryFlags::all();
56    if suppress_global {
57        feature_mask.remove(PTEntryFlags::GLOBAL);
58    }
59    FEATURE_MASK.init(feature_mask)
60}
61
62/// Initializes the encrypt mask.
63fn init_encrypt_mask(platform: &dyn SvsmPlatform) -> ImmutAfterInitResult<()> {
64    let masks = platform.get_page_encryption_masks();
65
66    PRIVATE_PTE_MASK.init(masks.private_pte_mask)?;
67    SHARED_PTE_MASK.init(masks.shared_pte_mask)?;
68
69    let guest_phys_addr_size = (masks.phys_addr_sizes >> 16) & 0xff;
70    let host_phys_addr_size = masks.phys_addr_sizes & 0xff;
71    let phys_addr_size = if guest_phys_addr_size == 0 {
72        // When [GuestPhysAddrSize] is zero, refer to the PhysAddrSize field
73        // for the maximum guest physical address size.
74        // - APM3, E.4.7 Function 8000_0008h - Processor Capacity Parameters and Extended Feature Identification
75        host_phys_addr_size
76    } else {
77        guest_phys_addr_size
78    };
79
80    PHYS_ADDR_SIZE.init(phys_addr_size)?;
81
82    // If the C-bit is a physical address bit however, the guest physical
83    // address space is effectively reduced by 1 bit.
84    // - APM2, 15.34.6 Page Table Support
85    let effective_phys_addr_size = cmp::min(masks.addr_mask_width, phys_addr_size);
86
87    let max_addr = 1 << effective_phys_addr_size;
88    MAX_PHYS_ADDR.init(max_addr)
89}

111/// Set address as shared via mask.
112fn make_shared_address(paddr: PhysAddr) -> PhysAddr {
113    (strip_confidentiality_bits(paddr).bits() | shared_pte_mask()).into()
114}

116/// Set address as private via mask.
117pub fn make_private_address(paddr: PhysAddr) -> PhysAddr {
118    (strip_shared_address_bits(paddr).bits() | private_pte_mask()).into()
119}

121// Returns true if the address is shared.
122fn is_shared(paddr: PhysAddr) -> bool {
123    paddr == make_shared_address(paddr)
124}

127    (paddr.bits() & !private_pte_mask()).into()

131    (paddr.bits() & !shared_pte_mask()).into()

174/// Represents paging mode.
175#[derive(Clone, Copy, Debug, Eq, PartialEq)]
176pub enum PagingMode {
177    // Paging mode is disabled
178    NoPaging,
179    // 32bit legacy paging mode
180    NonPAE,
181    // 32bit PAE paging mode
182    PAE,
183    // 4 level paging mode
184    PML4,
185    // 5 level paging mode
186    PML5,
187}
188
189impl PagingMode {
190    pub fn new(efer: EFERFlags, cr0: CR0Flags, cr4: CR4Flags) -> Self {
191        if !cr0.contains(CR0Flags::PG) {
192            // Paging is disabled
193            PagingMode::NoPaging
194        } else if efer.contains(EFERFlags::LMA) {
195            // Long mode is activated
196            if cr4.contains(CR4Flags::LA57) {
197                PagingMode::PML5
198            } else {
199                PagingMode::PML4
200            }
201        } else if cr4.contains(CR4Flags::PAE) {
202            // PAE mode
203            PagingMode::PAE
204        } else {
205            // Non PAE mode
206            PagingMode::NonPAE
207        }
208    }
209}

211/// Represents a page table entry.
212#[repr(C)]
213#[derive(Copy, Clone, Debug, FromBytes)]
214pub struct PTEntry(PhysAddr);
215
216impl PTEntry {
217    /// Check if the page table entry is clear (null).
218    pub fn is_clear(&self) -> bool {
219        self.0.is_null()
220    }
221
222    /// Clear the page table entry.
223    pub fn clear(&mut self) {
224        self.0 = PhysAddr::null();
225    }
226
227    /// Check if the page table entry is present.
228    pub fn present(&self) -> bool {
229        self.flags().contains(PTEntryFlags::PRESENT)
230    }
231
232    /// Check if the page table entry is huge.
233    pub fn huge(&self) -> bool {
234        self.flags().contains(PTEntryFlags::HUGE)
235    }
236
237    /// Check if the page table entry is writable.
238    pub fn writable(&self) -> bool {
239        self.flags().contains(PTEntryFlags::WRITABLE)
240    }
241
242    /// Check if the page table entry is NX (no-execute).
243    pub fn nx(&self) -> bool {
244        self.flags().contains(PTEntryFlags::NX)
245    }
246
247    /// Check if the page table entry is user-accessible.
248    pub fn user(&self) -> bool {
249        self.flags().contains(PTEntryFlags::USER)
250    }
251
252    /// Check if the page table entry is global.
253    pub fn global(&self) -> bool {
254        self.flags().contains(PTEntryFlags::GLOBAL)
255    }
256
257    /// Check if the page table entry has reserved bits set.
258    pub fn has_reserved_bits(&self, pm: PagingMode, level: usize) -> bool {
259        let reserved_mask = match pm {
260            PagingMode::NoPaging => unreachable!("NoPaging does not have page table"),
261            PagingMode::NonPAE => {
262                match level {
263                    // No reserved bits in 4k PTE.
264                    0 => 0,
265                    1 => {
266                        if self.huge() {
267                            // Bit21 is reserved in 4M PDE.
268                            BIT_MASK!(21, 21)
269                        } else {
270                            // No reserved bits in PDE.
271                            0
272                        }
273                    }
274                    _ => unreachable!("Invalid NonPAE page table level"),
275                }
276            }
277            PagingMode::PAE => {
278                // Bit62 ~ MAXPHYSADDR are reserved for each
279                // level in PAE page table.
280                BIT_MASK!(62, *PHYS_ADDR_SIZE)
281                    | match level {
282                        // No additional reserved bits in 4k PTE.
283                        0 => 0,
284                        1 => {
285                            if self.huge() {
286                                // Bit20 ~ Bit13 are reserved in 2M PDE.
287                                BIT_MASK!(20, 13)
288                            } else {
289                                // No additional reserved bits in PDE.
290                                0
291                            }
292                        }
293                        // Bit63 and Bit8 ~ Bit5 are reserved in PDPTE.
294                        2 => BIT_MASK!(63, 63) | BIT_MASK!(8, 5),
295                        _ => unreachable!("Invalid PAE page table level"),
296                    }
297            }
298            PagingMode::PML4 | PagingMode::PML5 => {
299                // Bit51 ~ MAXPHYSADDR are reserved for each level
300                // in PML4 and PML5 page table.
301                let common = if *PHYS_ADDR_SIZE > 51 {
302                    0
303                } else {
304                    // Remove the encryption mask bit as this bit is not reserved
305                    BIT_MASK!(51, *PHYS_ADDR_SIZE)
306                        & !((shared_pte_mask() | private_pte_mask()) as u64)
307                };
308
309                common
310                    | match level {
311                        // No additional reserved bits in 4k PTE.
312                        0 => 0,
313                        1 => {
314                            if self.huge() {
315                                // Bit20 ~ Bit13 are reserved in 2M PDE.
316                                BIT_MASK!(20, 13)
317                            } else {
318                                // No additional reserved bits in PDE.
319                                0
320                            }
321                        }
322                        2 => {
323                            if self.huge() {
324                                // Bit29 ~ Bit13 are reserved in 1G PDPTE.
325                                BIT_MASK!(29, 13)
326                            } else {
327                                // No additional reserved bits in PDPTE.
328                                0
329                            }
330                        }
331                        // Bit8 ~ Bit7 are reserved in PML4E.
332                        3 => BIT_MASK!(8, 7),
333                        4 => {
334                            if pm == PagingMode::PML4 {
335                                unreachable!("Invalid PML4 page table level");
336                            } else {
337                                // Bit8 ~ Bit7 are reserved in PML5E.
338                                BIT_MASK!(8, 7)
339                            }
340                        }
341                        _ => unreachable!("Invalid PML4/PML5 page table level"),
342                    }
343            }
344        };
345
346        self.raw() & reserved_mask != 0
347    }
348
349    /// Get the raw bits (`u64`) of the page table entry.
350    pub fn raw(&self) -> u64 {
351        self.0.bits() as u64
352    }
353
354    /// Get the flags of the page table entry.
355    pub fn flags(&self) -> PTEntryFlags {
356        PTEntryFlags::from_bits_truncate(self.0.bits() as u64)
357    }
358
359    /// Set the page table entry with the specified address and flags.
360    pub fn set_unrestricted(&mut self, addr: PhysAddr, flags: PTEntryFlags) {
361        let addr = addr.bits() as u64;
362        assert_eq!(addr & !0x000f_ffff_ffff_f000, 0);
363        self.0 = PhysAddr::from(addr | flags.bits());
364    }
365
366    /// Set the page table entry with the specified address, with flags
367    /// constrained to the supported feature flags.
368    pub fn set(&mut self, addr: PhysAddr, flags: PTEntryFlags) {
369        self.set_unrestricted(addr, supported_flags(flags));
370    }
371
372    /// Inserts the private address mask if the page is present.
373    pub fn make_private_if_present(&mut self) {
374        if self.flags().contains(PTEntryFlags::PRESENT) {
375            self.0 = make_private_address(self.0);
376        }
377    }
378
379    /// Get the address from the page table entry, including the shared bit.
380    pub fn page_frame(&self) -> PhysAddr {
381        let addr = PhysAddr::from(self.0.bits() & 0x000f_ffff_ffff_f000);
382        strip_confidentiality_bits(addr)
383    }
384
385    /// Get the address from the page table entry, excluding the C/shared bit.
386    pub fn address(&self) -> PhysAddr {
387        strip_shared_address_bits(self.page_frame())
388    }
389
390    /// Read a page table entry from the specified virtual address.
391    ///
392    /// # Safety
393    ///
394    /// Reads from an arbitrary virtual address, making this essentially a
395    /// raw pointer read.  The caller must be certain to calculate the correct
396    /// address.
397    pub unsafe fn read_pte(vaddr: VirtAddr) -> Self {
398        // SAFETY: When the methods safety requirements are met, the raw
399        // pointer read is safe.
400        unsafe { *vaddr.as_ptr::<Self>() }
401    }
402}

404/// A pagetable page with multiple entries.
405#[repr(C)]
406#[derive(Debug, FromBytes)]
407pub struct PTPage {
408    entries: [PTEntry; ENTRY_COUNT],
409}
410
411impl PTPage {
412    /// Allocates a zeroed pagetable page and returns a `PageBox` containing
413    /// the allocation.
414    ///
415    /// # Errors
416    ///
417    /// Returns [`SvsmError`] if the page cannot be allocated.
418    pub fn alloc_box() -> Result<PageBox<Self>, SvsmError> {
419        PageBox::try_new_zeroed()
420    }
421
422    /// Allocates a zeroed pagetable page and returns a mutable reference to
423    /// it, plus its physical address.
424    ///
425    /// # Errors
426    ///
427    /// Returns [`SvsmError`] if the page cannot be allocated.
428    fn alloc() -> Result<(&'static mut Self, PhysAddr), SvsmError> {
429        let page = Self::alloc_box()?;
430        let paddr = virt_to_phys(page.vaddr());
431        Ok((PageBox::leak(page), paddr))
432    }
433
434    /// Frees a pagetable page.
435    ///
436    /// # Safety
437    ///
438    /// The given reference must correspond to a valid previously allocated
439    /// page table page.
440    unsafe fn free(page: &'static Self) {
441        // SAFETY: The page put into the PageBox is a previously allocated
442        // page table page.
443        unsafe {
444            let _ = PageBox::from_raw(NonNull::from(page));
445        }
446    }
447
448    /// Converts a pagetable entry to a mutable reference to a [`PTPage`],
449    /// if the entry is present and not huge.
450    fn from_entry(entry: PTEntry) -> Option<&'static mut Self> {
451        let flags = entry.flags();
452        if !flags.contains(PTEntryFlags::PRESENT) || flags.contains(PTEntryFlags::HUGE) {
453            return None;
454        }
455
456        let address = phys_to_virt(entry.address());
457        // SAFETY: Every PTEntry points to a previously allocated page-table
458        // page, so this pointer dereference is safe.
459        Some(unsafe { &mut *address.as_mut_ptr::<PTPage>() })
460    }
461}

488/// A physical address within a page frame
489#[derive(Clone, Copy, Debug)]
490pub enum PageFrame {
491    Size4K(PhysAddr),
492    Size2M(PhysAddr),
493    Size1G(PhysAddr),
494}
495
496impl PageFrame {
497    /// Get the address from the page frame, including the shared bit.
498    pub fn page_frame(&self) -> PhysAddr {
499        let paddr = match *self {
500            Self::Size4K(pa) => pa,
501            Self::Size2M(pa) => pa,
502            Self::Size1G(pa) => pa,
503        };
504        strip_confidentiality_bits(paddr)
505    }
506
507    /// Get the address from the page frame, excluding the C/shared bit.
508    pub fn address(&self) -> PhysAddr {
509        strip_shared_address_bits(self.page_frame())
510    }
511
512    pub fn size(&self) -> usize {
513        match self {
514            Self::Size4K(_) => PAGE_SIZE,
515            Self::Size2M(_) => PAGE_SIZE_2M,
516            Self::Size1G(_) => PAGE_SIZE_1G,
517        }
518    }
519
520    pub fn start(&self) -> PhysAddr {
521        let end = self.address().bits() & !(self.size() - 1);
522        end.into()
523    }
524
525    pub fn end(&self) -> PhysAddr {
526        self.start() + self.size()
527    }
528}

530/// Page table structure containing a root page with multiple entries.
531#[repr(C)]
532#[derive(Debug, FromZeros)]
533pub struct PageTable {
534    root: PTPage,
535}
536
537impl PageTable {
538    /// Load the current page table into the CR3 register.
539    ///
540    /// # Safety
541    ///
542    /// The caller must ensure to take other actions to make sure a memory safe
543    /// execution state is warranted (e.g. changing the stack and register state)
544    pub unsafe fn load(&self) {
545        // SAFETY: demanded to the caller
546        unsafe {
547            write_cr3(self.cr3_value());
548        }
549    }
550
551    /// Get the CR3 register value for the current page table.
552    pub fn cr3_value(&self) -> PhysAddr {
553        let pgtable = VirtAddr::from(self as *const Self);
554        virt_to_phys(pgtable)
555    }
556
557    /// Allocate a new page table root.
558    ///
559    /// # Errors
560    /// Returns [`SvsmError`] if the page cannot be allocated.
561    pub fn allocate_new() -> Result<PageBox<Self>, SvsmError> {
562        let mut pgtable: PageBox<Self> = PageBox::try_new_zeroed()?;
563        let paddr = virt_to_phys(pgtable.vaddr());
564
565        // Set the self-map entry.
566        let entry = &mut pgtable.root[PGTABLE_LVL3_IDX_PTE_SELFMAP];
567        let flags = PTEntryFlags::PRESENT
568            | PTEntryFlags::WRITABLE
569            | PTEntryFlags::ACCESSED
570            | PTEntryFlags::DIRTY
571            | PTEntryFlags::NX;
572        entry.set(make_private_address(paddr), flags);
573
574        Ok(pgtable)
575    }
576
577    /// Clone the shared part of the page table; excluding the private
578    /// parts.
579    ///
580    /// # Errors
581    /// Returns [`SvsmError`] if the page cannot be allocated.
582    pub fn clone_shared(&self) -> Result<PageBox<PageTable>, SvsmError> {
583        let mut pgtable = Self::allocate_new()?;
584        pgtable.root.entries[PGTABLE_LVL3_IDX_SHARED] = self.root.entries[PGTABLE_LVL3_IDX_SHARED];
585        Ok(pgtable)
586    }
587
588    /// Copy an entry `entry` from another [`PageTable`].
589    pub fn copy_entry(&mut self, other: &Self, entry: usize) {
590        self.root.entries[entry] = other.root.entries[entry];
591    }
592
593    /// Computes the index within a page table at the given level for a
594    /// virtual address `vaddr`.
595    ///
596    /// # Parameters
597    /// - `vaddr`: The virtual address to compute the index for.
598    ///
599    /// # Returns
600    /// The index within the page table.
601    pub fn index<const L: usize>(vaddr: VirtAddr) -> usize {
602        vaddr.to_pgtbl_idx::<L>()
603        //vaddr.bits() >> (12 + L * 9) & 0x1ff
604    }
605
606    /// Walks a page table at level 0 to find a mapping.
607    ///
608    /// # Parameters
609    /// - `page`: A mutable reference to the root page table.
610    /// - `vaddr`: The virtual address to find a mapping for.
611    ///
612    /// # Returns
613    /// A `Mapping` representing the found mapping.
614    fn walk_addr_lvl0(page: &mut PTPage, vaddr: VirtAddr) -> Mapping<'_> {
615        let idx = Self::index::<0>(vaddr);
616        Mapping::Level0(&mut page[idx])
617    }
618
619    /// Walks a page table at level 1 to find a mapping.
620    ///
621    /// # Parameters
622    /// - `page`: A mutable reference to the root page table.
623    /// - `vaddr`: The virtual address to find a mapping for.
624    ///
625    /// # Returns
626    /// A `Mapping` representing the found mapping.
627    fn walk_addr_lvl1(page: &mut PTPage, vaddr: VirtAddr) -> Mapping<'_> {
628        let idx = Self::index::<1>(vaddr);
629        let entry = page[idx];
630        match PTPage::from_entry(entry) {
631            Some(page) => Self::walk_addr_lvl0(page, vaddr),
632            None => Mapping::Level1(&mut page[idx]),
633        }
634    }
635
636    /// Walks a page table at level 2 to find a mapping.
637    ///
638    /// # Parameters
639    /// - `page`: A mutable reference to the root page table.
640    /// - `vaddr`: The virtual address to find a mapping for.
641    ///
642    /// # Returns
643    /// A `Mapping` representing the found mapping.
644    fn walk_addr_lvl2(page: &mut PTPage, vaddr: VirtAddr) -> Mapping<'_> {
645        let idx = Self::index::<2>(vaddr);
646        let entry = page[idx];
647        match PTPage::from_entry(entry) {
648            Some(page) => Self::walk_addr_lvl1(page, vaddr),
649            None => Mapping::Level2(&mut page[idx]),
650        }
651    }
652
653    /// Walks the page table to find a mapping for a given virtual address.
654    ///
655    /// # Parameters
656    /// - `page`: A mutable reference to the root page table.
657    /// - `vaddr`: The virtual address to find a mapping for.
658    ///
659    /// # Returns
660    /// A `Mapping` representing the found mapping.
661    fn walk_addr_lvl3(page: &mut PTPage, vaddr: VirtAddr) -> Mapping<'_> {
662        let idx = Self::index::<3>(vaddr);
663        let entry = page[idx];
664        match PTPage::from_entry(entry) {
665            Some(page) => Self::walk_addr_lvl2(page, vaddr),
666            None => Mapping::Level3(&mut page[idx]),
667        }
668    }
669
670    /// Walk the virtual address and return the corresponding mapping.
671    ///
672    /// # Parameters
673    /// - `vaddr`: The virtual address to find a mapping for.
674    ///
675    /// # Returns
676    /// A `Mapping` representing the found mapping.

693    /// Perform a virtual to physical translation using the self-map.
694    ///
695    /// # Parameters
696    /// - `vaddr': The virtual address to translate.
697    ///
698    /// # Returns
699    /// Some(PageFrame) if the virtual address is valid.
700    /// None if the virtual address is not valid.
701    pub fn virt_to_frame(vaddr: VirtAddr) -> Option<PageFrame> {
702        // Calculate the virtual addresses of each level of the paging
703        // hierarchy in the self-map.
704        let pte_addr = Self::get_pte_address(vaddr);
705        let pde_addr = Self::get_pte_address(pte_addr);
706        let pdpe_addr = Self::get_pte_address(pde_addr);
707        let pml4e_addr = Self::get_pte_address(pdpe_addr);
708
709        // SAFETY: Check each entry in the paging hierarchy to determine
710        // whether this address is mapped.  Because the hierarchy is read from
711        // the top down using self-map addresses that were calculated
712        // correctly, the reads are safe to perform.
713        let pml4e = unsafe { PTEntry::read_pte(pml4e_addr) };
714        if !pml4e.present() {
715            return None;
716        }
717
718        // There is no need to check for a large page in the PML4E because
719        // the architecture does not support the large bit at the top-level
720        // entry.  If a large page is detected at a lower level of the
721        // hierarchy, the low bits from the virtual address must be combined
722        // with the physical address from the PDE/PDPE.
723
724        // SAFETY: The PML4E was checked to be present, so the PDPE exists and
725        // can be read safely.
726        let pdpe = unsafe { PTEntry::read_pte(pdpe_addr) };
727        if !pdpe.present() {
728            return None;
729        }
730        if pdpe.huge() {
731            let pa = pdpe.page_frame() + (usize::from(vaddr) & 0x3FFF_FFFF);
732            return Some(PageFrame::Size1G(pa));
733        }
734
735        // SAFETY: The PDPE was checked to be present and not to be a huge
736        // page. So the PDE exists and can be read safely.
737        let pde = unsafe { PTEntry::read_pte(pde_addr) };
738        if !pde.present() {
739            return None;
740        }
741        if pde.huge() {
742            let pa = pde.page_frame() + (usize::from(vaddr) & 0x001F_FFFF);
743            return Some(PageFrame::Size2M(pa));
744        }
745
746        // SAFETY: The PDE was checked to be present and not to be a huge
747        // page. So the PTE exists and can be read safely.
748        let pte = unsafe { PTEntry::read_pte(pte_addr) };
749        if pte.present() {
750            let pa = pte.page_frame() + (usize::from(vaddr) & 0xFFF);
751            Some(PageFrame::Size4K(pa))
752        } else {
753            None
754        }
755    }

856    /// Splits a 2MB page into 4KB pages.
857    ///
858    /// # Parameters
859    /// - `entry`: The 2M page table entry to split.
860    ///
861    /// # Returns
862    /// A result indicating success or an error [`SvsmError`] in failure.
863    fn do_split_4k(entry: &mut PTEntry) -> Result<(), SvsmError> {
864        let (page, paddr) = PTPage::alloc()?;
865        let mut flags = entry.flags();
866
867        assert!(flags.contains(PTEntryFlags::HUGE));
868
869        let addr_2m = PhysAddr::from(entry.address().bits() & 0x000f_ffff_fff0_0000);
870
871        flags.remove(PTEntryFlags::HUGE);
872
873        // Prepare PTE leaf page
874        for (i, e) in page.entries.iter_mut().enumerate() {
875            let addr_4k = addr_2m + (i * PAGE_SIZE);
876            e.clear();
877            e.set(make_private_address(addr_4k), flags);
878        }
879
880        entry.set(make_private_address(paddr), flags);
881
882        flush_tlb_global_sync();
883
884        Ok(())
885    }

1083                    return Err(SvsmError::Mem);
1084                }

1087            }
1088            Mapping::Level2(_entry) => Err(SvsmError::Mem),
1089            Mapping::Level3(_entry) => Err(SvsmError::Mem),

1169    /// Maps a memory region to physical memory with specified flags.
1170    ///
1171    /// # Parameters
1172    /// - `region`: The virtual memory region to map.
1173    /// - `phys`: The starting physical address to map to.
1174    /// - `flags`: The flags to apply to the page table entries.
1175    ///
1176    /// # Returns
1177    /// A result indicating success (`Ok`) or failure (`Err`).
1178    pub fn map_region(
1179        &mut self,
1180        region: MemoryRegion<VirtAddr>,
1181        phys: PhysAddr,
1182        flags: PTEntryFlags,
1183    ) -> Result<(), SvsmError> {
1184        let mut vaddr = region.start();
1185        let end = region.end();
1186        let mut paddr = phys;
1187
1188        while vaddr < end {
1189            if vaddr.is_aligned(PAGE_SIZE_2M)
1190                && paddr.is_aligned(PAGE_SIZE_2M)
1191                && vaddr + PAGE_SIZE_2M <= end
1192                && self.map_2m(vaddr, paddr, flags).is_ok()
1193            {
1194                vaddr = vaddr + PAGE_SIZE_2M;
1195                paddr = paddr + PAGE_SIZE_2M;
1196                continue;
1197            }
1198
1199            self.map_4k(vaddr, paddr, flags)?;
1200            vaddr = vaddr + PAGE_SIZE;
1201            paddr = paddr + PAGE_SIZE;
1202        }
1203
1204        Ok(())
1205    }

1246    /// This method is meant for global pages only.
1247    ///
1248    /// # Safety
1249    ///
1250    /// The caller should verify that `region` can be made read-only, i.e. that
1251    /// no write can happen or that a #PF raised by any tentative write is

1263                        return Err(SvsmError::Mem);
1264                    }

1267

1269                        make_shared_address(entry.address())
1270                    } else {
1271                        make_private_address(entry.address())
1272                    };
1273
1274                    entry.set(paddr, flags);
1275                }
1276                Mapping::Level1(entry) | Mapping::Level2(entry) => {
1277                    // Ensure we never fell on a huge page while iterating over the region pages.
1278                    if entry.huge() {
1279                        return Err(SvsmError::Mem);
1280                    }
1281                }
1282                _ => {}
1283            }
1284        }

1691    /// Flags to represent how memory is accessed, e.g. write data to the
1692    /// memory or fetch code from the memory.
1693    #[derive(Clone, Copy, Debug)]
1694    pub struct MemAccessMode: u32 {
1695        const WRITE     = 1 << 0;
1696        const FETCH     = 1 << 1;
1697    }

1700/// Attributes to determin Whether a memory access (write/fetch) is permitted
1701/// by a translation which includes the paging-mode modifiers in CR0, CR4 and
1702/// EFER; EFLAGS.AC; and the supervisor/user mode access.
1703#[derive(Clone, Copy, Debug)]
1704pub struct PTWalkAttr {
1705    cr0: CR0Flags,
1706    cr4: CR4Flags,
1707    efer: EFERFlags,
1708    flags: RFlags,
1709    user_mode_access: bool,
1710    pm: PagingMode,
1711}
1712
1713impl PTWalkAttr {
1714    /// Creates a new `PTWalkAttr` instance with the specified attributes.
1715    ///
1716    /// # Arguments
1717    ///
1718    /// * `cr0`, `cr4`, and `efer` - Represent the control register
1719    ///   flags for CR0, CR4, and EFER respectively.
1720    /// * `flags` - Represents the CPU Flags.
1721    /// * `user_mode_access` - Indicates whether the access is in user mode.
1722    ///
1723    /// Returns a new `PTWalkAttr` instance.
1724    pub fn new(
1725        cr0: CR0Flags,
1726        cr4: CR4Flags,
1727        efer: EFERFlags,
1728        flags: RFlags,
1729        user_mode_access: bool,
1730    ) -> Self {
1731        Self {
1732            cr0,
1733            cr4,
1734            efer,
1735            flags,
1736            user_mode_access,
1737            pm: PagingMode::new(efer, cr0, cr4),
1738        }
1739    }
1740
1741    /// Checks the access rights for a page table entry.
1742    ///
1743    /// # Arguments
1744    ///
1745    /// * `entry` - The page table entry to check.
1746    /// * `mem_am` - Indicates how to access the memory.
1747    /// * `last_level` - Indicates whether the entry is at the last level
1748    ///   of the page table.
1749    /// * `pteflags` - The PTE flags to indicate if the corresponding page
1750    ///   table entry allows the access rights.
1751    ///
1752    /// # Returns
1753    ///
1754    /// Returns `Ok((entry, leaf))` if the access rights are valid, where
1755    /// `entry` is the modified page table entry and `leaf` is a boolean
1756    /// indicating whether the entry is a leaf node, or `Err(PageFaultError)`
1757    /// to indicate the page fault error code if the access rights are invalid.
1758    pub fn check_access_rights(
1759        &self,
1760        entry: PTEntry,
1761        mem_am: MemAccessMode,
1762        level: usize,
1763        pteflags: &mut PTEntryFlags,
1764    ) -> Result<(PTEntry, bool), PageFaultError> {
1765        let pf_err = self.default_pf_err(mem_am) | PageFaultError::P;
1766
1767        if !entry.present() {
1768            // Entry is not present.
1769            return Err(pf_err & !PageFaultError::P);
1770        }
1771
1772        if entry.has_reserved_bits(self.pm, level) {
1773            // Reserved bits have been set.
1774            return Err(pf_err | PageFaultError::R);
1775        }
1776
1777        // SDM 4.6.1 Determination of Access Rights:
1778        // If the U/S flag (bit 2) is 0 in at least one of the
1779        // paging-structure entries, the address is a supervisor-mode
1780        // address. Otherwise, the address is a user-mode address.
1781        // So by-default assume the address is user mode address.
1782        if !entry.user() {
1783            *pteflags &= !PTEntryFlags::USER;
1784        }
1785
1786        // SDM 4.6.1 Determination of Access Rights:
1787        // R/W flag (bit 1) is 1 in every paging-structure entry controlling
1788        // the translation and with a protection key for which write access is
1789        // permitted; data may not be written to any supervisor-mode
1790        // address with a translation for which the R/W flag is 0 in any
1791        // paging-structure entry controlling the translation.
1792        // The same for user mode address
1793        if !entry.writable() {
1794            *pteflags &= !PTEntryFlags::WRITABLE;
1795        }
1796
1797        // SDM 4.6.1 Determination of Access Rights:
1798        // For non 32-bit paging modes with IA32_EFER.NXE = 1, instructions
1799        // may be fetched from any supervisormode address with a translation
1800        // for which the XD flag (bit 63) is 0 in every paging-structure entry
1801        // controlling the translation; instructions may not be fetched from
1802        // any supervisor-mode address with a translation for which the XD flag
1803        // is 1 in any paging-structure entry controlling the translation
1804        if self.efer.contains(EFERFlags::NXE) && entry.nx() {
1805            *pteflags |= PTEntryFlags::NX;
1806        } else if !self.efer.contains(EFERFlags::NXE) && entry.nx() {
1807            // XD bit must be 0 if efer.NXE = 0
1808            return Err(pf_err | PageFaultError::R);
1809        }
1810
1811        let leaf = if level == 0 || entry.huge() {
1812            // User mode cannot access any supervisor mode addresses
1813            if self.user_mode_access && !pteflags.contains(PTEntryFlags::USER) {
1814                return Err(pf_err);
1815            }
1816
1817            // Always check for reading. For the case of supervisor mode read user
1818            // mode addresses, do special checking. For other cases, read is allowed.
1819            if !self.user_mode_access && pteflags.contains(PTEntryFlags::USER) {
1820                // Read not allowed with SMAP = 1 && flags.ac = 0
1821                if self.cr4.contains(CR4Flags::SMAP) && !self.flags.contains(RFlags::AC) {
1822                    return Err(pf_err);
1823                }
1824            }
1825
1826            if mem_am.contains(MemAccessMode::WRITE) {
1827                if !self.user_mode_access && pteflags.contains(PTEntryFlags::USER) {
1828                    // Check supervisor mode write user mode addresses
1829                    if !self.cr0.contains(CR0Flags::WP) {
1830                        // Check write with CR0.WP = 0
1831                        if self.cr4.contains(CR4Flags::SMAP) && !self.flags.contains(RFlags::AC) {
1832                            // Write not allowed with SMAP = 1 && flags.ac = 0
1833                            return Err(pf_err);
1834                        }
1835                    } else {
1836                        // Check write with CR0.WP = 1
1837                        if !self.cr4.contains(CR4Flags::SMAP) {
1838                            // SMAP = 0
1839                            if !pteflags.contains(PTEntryFlags::WRITABLE) {
1840                                // Write not allowed R/W = 0
1841                                return Err(pf_err);
1842                            }
1843                        } else {
1844                            // SMAP = 1
1845                            if !self.flags.contains(RFlags::AC)
1846                                || !pteflags.contains(PTEntryFlags::WRITABLE)
1847                            {
1848                                // Write not allowed with flags.AC = 0 || R/W = 0
1849                                return Err(pf_err);
1850                            }
1851                        }
1852                    }
1853                } else if !self.user_mode_access && !pteflags.contains(PTEntryFlags::USER) {
1854                    // Check supervisor mode write supervisor mode addresses
1855                    if self.cr0.contains(CR0Flags::WP) && !pteflags.contains(PTEntryFlags::WRITABLE)
1856                    {
1857                        // Write not allowed with CR0.WP = 1 && R/W = 0
1858                        return Err(pf_err);
1859                    }
1860                } else if self.user_mode_access && pteflags.contains(PTEntryFlags::USER) {
1861                    // Check user mode write user mode addresses
1862                    if !pteflags.contains(PTEntryFlags::WRITABLE) {
1863                        // Write not allowed R/W = 0
1864                        return Err(pf_err);
1865                    }
1866                }
1867                // User mode write supervisor mode addresses is checked already
1868            }
1869
1870            if mem_am.contains(MemAccessMode::FETCH) {
1871                // For instruction fetch, the rule is the same except for the case of
1872                // supervisor mode fetch user mode addresses
1873                if !self.user_mode_access && pteflags.contains(PTEntryFlags::USER) {
1874                    // Fetch not allowed with SMEP = 1
1875                    if self.cr4.contains(CR4Flags::SMEP) {
1876                        return Err(pf_err);
1877                    }
1878                }
1879
1880                // For non-32bit paging mode, fetch not allowed with efer.NXE = 1 && XD = 1
1881                if self.cr4.contains(CR4Flags::PAE)
1882                    && self.efer.contains(EFERFlags::NXE)
1883                    && pteflags.contains(PTEntryFlags::NX)
1884                {
1885                    return Err(pf_err);
1886                }
1887            }
1888            true
1889        } else {
1890            false
1891        };
1892
1893        Ok((entry, leaf))
1894    }
1895
1896    fn default_pf_err(&self, mem_am: MemAccessMode) -> PageFaultError {
1897        let mut err = PageFaultError::empty();
1898
1899        if mem_am.contains(MemAccessMode::WRITE) {
1900            err |= PageFaultError::W;
1901        }
1902
1903        if mem_am.contains(MemAccessMode::FETCH) {
1904            err |= PageFaultError::I;
1905        }
1906
1907        if self.user_mode_access {
1908            err |= PageFaultError::U;
1909        }
1910
1911        err
1912    }
